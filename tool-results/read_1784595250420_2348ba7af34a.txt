     1→/**
     2→ * CodeEditor — collaborative multi-file code surface for 'code' boards.
     3→ *
     4→ * CodeMirror 6 bound to one shared Y.Text per file via y-codemirror.next
     5→ * (yCollab): concurrent typing merges through Yjs, remote peers' carets and
     6→ * selections render inline with their presence color/name, and undo is a
     7→ * per-file Y.UndoManager scoped to YOUR edits (yUndoManagerKeymap — CM's own
     8→ * history is deliberately not installed, two histories would fight Ctrl+Z).
     9→ *
    10→ * Files live in the `code:files` Y.Map (id → { name }); each file's content
    11→ * is a top-level Y.Text (`code:text:<id>`) so every client resolves the same
    12→ * shared type. The creator seeds `main.js` on an empty board. Language is
    13→ * auto-detected from the filename via @codemirror/language-data, which
    14→ * dynamically imports only the grammars actually used.
    15→ *
    16→ * Polish (ROUND14-B):
    17→ *   - Fold gutter + indent-on-input + indent unit (2 spaces) so nested code
    18→ *     shows structure at a glance.
    19→ *   - Light/dark theme toggle. oneDark by default; the light theme is a CM
    20→ *     `EditorView.theme({ dark: false })` that lets the page tokens through.
    21→ *   - Tab strip above the editor: any file you click becomes a tab, middle-
    22→ *     click or the × on the tab closes it. The editor body always reflects
    23→ *     the active tab — clicking the same file in the rail is idempotent.
    24→ *   - A Find button in the toolbar that dispatches a synthetic Ctrl+F so
    25→ *     CM's built-in search panel (already bound via searchKeymap) opens.
    26→ *
    27→ * Split-view preview (ROUND18-A): an Eye toggle in the toolbar splits the
    28→ * editor 50/50 with a sandboxed <iframe srcdoc> that renders the project's
    29→ * HTML (via the shared buildPreview helper — local <link>/<script src> are
    30→ * inlined into the entry HTML, and a console bridge surfaces runtime logs
    31→ * inside the frame). The divider is drag-resizable (clamp 20%–80%); the
    32→ * preview auto-refreshes 400ms after any file change. The iframe runs with
    33→ * `allow-scripts` only (no `allow-same-origin`), so previewed code executes
    34→ * in a null origin and can't touch the Slate app or its sync state.
    35→ */

    70→/**
    71→ * Light theme — an empty rule set marked `{ dark: false }`. CM treats this as
    72→ * "light mode": it skips oneDark and lets the page tokens (via the
    73→ * `.slate-code-host .cm-editor { background: var(--bg) }` rules in
    74→ * codeEditor.css) carry the surface. Syntax tokens fall back to

    80→/**
    81→ * Indent guides — a tiny ViewPlugin that adds a `cm-indent-guide` line
    82→ * decoration to every visible line whose leading whitespace crosses one or
    83→ * more 2-space boundaries, and stashes the count in a `--cm-indent` CSS
    84→ * variable. codeEditor.css then paints vertical guides via a clipped
    85→ * `repeating-linear-gradient` background sized to that count.
    86→ *
    87→ * `@codemirror/view` shipped an official `indentationMarkers()` helper in a
    88→ * later 6.4x release; we're on 6.43.6 (latest published), so this is the
    89→ * local equivalent. Empty lines render with no guides — the trade-off is
    90→ * a few pixels of visual gap, but the implementation stays in one file.
    91→ */
    92→const indentGuides = ViewPlugin.fromClass(
    93→  class {
    94→    decos: DecorationSet;
    95→    constructor(view: EditorView) {
    96→      this.decos = this.build(view);
    97→    }
    98→    update(u: ViewUpdate) {
    99→      if (u.docChanged || u.viewportChanged || u.selectionSet) {
   100→        this.decos = this.build(u.view);
   101→      }
   102→    }
   103→    build(view: EditorView) {
   104→      const items: { from: number; value: ReturnType<typeof Decoration.line> }[] = [];
   105→      for (const { from, to } of view.visibleRanges) {
   106→        for (let pos = from; pos <= to; ) {
   107→          const line = view.state.doc.lineAt(pos);
   108→          // Count leading whitespace in 2-space units — a leading tab counts
   109→          // as one guide so mixed-indent files still get something useful.
   110→          const m = /^([ \t]+)/.exec(line.text);
   111→          const spaces = m?.[1] ? m[1].replace(/\t/g, '  ').length : 0;
   112→          const levels = Math.floor(spaces / 2);
   113→          if (levels > 0) {
   114→            items.push({
   115→              from: line.from,
   116→              value: Decoration.line({
   117→                class: 'cm-indent-guide',
   118→                attributes: { style: `--cm-indent: ${levels}` },
   119→              }),
   120→            });
   121→          }
   122→          pos = line.to + 1;
   123→        }
   124→      }
   125→      return Decoration.set(
   126→        items.map((i) => i.value.range(i.from)),
   127→        true,
   128→      );
   129→    }
   130→  },
   131→  { decorations: (v) => v.decos },
   132→);

   134→/** Default editor font size (matches the old hard-coded `.cm-editor { font-size: 13px }`
   135→ *  rule; the +/- buttons step in 1px increments and clamp to [10, 24]). */
   136→const DEFAULT_FONT_SIZE = 13;
   137→const MIN_FONT_SIZE = 10;
   138→const MAX_FONT_SIZE = 24;

   140→/** Build a CM theme extension that sets the editor + gutter font size. */

   144→    '.cm-gutters': { fontSize: `${px}px` },
   145→    '&': { fontSize: `${px}px` },
   146→  });
   147→}

   948→                  onKeyDown={(e) => {
   949→                    if (e.key === 'Escape') {
   950→                      e.preventDefault();
   951→                      closePalette();
   952→                    } else if (e.key === 'Enter') {
   953→                      e.preventDefault();
   954→                      const cmd = filteredCommands[paletteIndex];
   955→                      if (cmd) cmd.run();
   956→                    } else if (e.key === 'ArrowDown') {
   957→                      e.preventDefault();
   958→                      setPaletteIndex((i) => Math.min(filteredCommands.length - 1, i + 1));
   959→                    } else if (e.key === 'ArrowUp') {
   960→                      e.preventDefault();
   961→                      setPaletteIndex((i) => Math.max(0, i - 1));
   962→                    }
   963→                  }}
   964→                  placeholder="Type a command…"
   965→                  className="w-full bg-transparent text-sm text-text outline-none placeholder:text-text-dim"
   966→                />
   967→                <span className="shrink-0 rounded border border-border px-1.5 py-0.5 font-mono text-[10px] text-text-dim">Esc</span>
   968→              </div>
   969→              <ul className="max-h-72 overflow-y-auto py-1">
   970→                {filteredCommands.length === 0 && (
   971→                  <li className="px-3 py-3 text-center text-xs text-text-dim">No matching commands</li>
   972→                )}
   973→                {filteredCommands.map((cmd, i) => (
   974→                  <li key={cmd.id}>
   975→                    <button
   976→                      type="button"
   977→                      onMouseEnter={() => setPaletteIndex(i)}
   978→                      onClick={() => cmd.run()}
   979→                      className={`flex w-full items-center justify-between gap-2 px-3 py-1.5 text-left text-xs ${
   980→                        i === paletteIndex ? 'bg-accent/15 text-accent' : 'text-text-mid hover:bg-bg-3'
   981→                      }`}
   982→                    >
   983→                      <span>{cmd.label}</span>
   984→                      {cmd.hint && <span className="font-mono text-[10px] text-text-dim">{cmd.hint}</span>}
   985→                      {i === paletteIndex && <CornerDownLeft size={10} className="text-text-dim" />}
   986→                    </button>
   987→                  </li>
   988→                ))}
   989→              </ul>
   990→              <button
   991→                type="button"
   992→                aria-label="Close command palette"
   993→                onClick={closePalette}
   994→                className="block w-full border-t border-border bg-bg-3/40 py-1 text-center font-mono text-[10px] text-text-dim hover:bg-bg-3"
   995→              >
   996→                Ctrl+Shift+P to open · ↑↓ to navigate · Enter to run
   997→              </button>
   998→            </div>
   999→          )}
  1000→        </div>
  1001→        {/* Terminal strip — full-width along the bottom of the editor, above
  1002→            the status bar. Drag the top edge to resize. Reuses the same panel
  1003→            component as the dockable version; it shows the live Preview's
  1004→            console output. */}
  1005→        {showTerminal && (
  1006→          <div className="flex shrink-0 flex-col border-t border-border" style={{ height: terminalHeight }}>
  1007→            <div
  1008→              role="separator"
  1009→              aria-orientation="horizontal"
  1010→              aria-label="Resize terminal"
  1011→              onMouseDown={startTerminalDrag}
  1012→              className="h-1 shrink-0 cursor-row-resize bg-border hover:bg-accent/50"
  1013→            />
  1014→            <div className="min-h-0 flex-1">
  1015→              <CodeTerminalPanel />
  1016→            </div>
  1017→          </div>
  1018→        )}
  1019→        <div className="flex items-center gap-3 border-t border-border bg-bg-2 px-3 py-1 font-mono text-[10px] text-text-dim">
  1020→          <span>{langName}</span>
  1021→          <span>{lineCount} {lineCount === 1 ? 'line' : 'lines'}</span>
  1022→          <div className="flex-1" />
  1023→          <span>shared · every keystroke syncs</span>
  1024→        </div>
  1025→      </div>
  1026→    </div>
  1027→  );
  1028→}
