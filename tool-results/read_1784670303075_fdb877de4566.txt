    61→/** Nearest start position at which a clip of `duration` fits without
    62→ *  overlapping any blocker. Merges blockers into occupied intervals, then
    63→ *  picks the gap position closest to `desired`. This replaces the previous
    64→ *  "slide to the nearest side of the current blocker" iteration, which could
    65→ *  ping-pong between two adjacent blockers (each resolution landing on the
    66→ *  other) and exhaust its iteration budget while STILL overlapping —
    67→ *  committing a clip on top of another one. Scanning gaps directly is
    68→ *  deterministic: the open-ended gap after the last blocker always fits, so a
    69→ *  valid position always exists. */
    70→export function nearestFreeStart(desired: number, duration: number, blockersIn: { start: number; end: number }[]): number {
    71→  const eps = 1e-4;
    72→  desired = Math.max(0, desired);
    73→  if (blockersIn.length === 0) return desired;
    74→  const sorted = [...blockersIn].sort((a, b) => a.start - b.start);
    75→  const occupied: { start: number; end: number }[] = [];
    76→  for (const b of sorted) {
    77→    const last = occupied[occupied.length - 1];
    78→    if (last && b.start <= last.end + eps) last.end = Math.max(last.end, b.end);
    79→    else occupied.push({ start: b.start, end: b.end });
    80→  }
    81→  let best = desired;
    82→  let bestDist = Infinity;
    83→  const consider = (lo: number, hi: number) => {
    84→    if (hi - lo < duration - eps) return; // gap too small
    85→    const cand = Math.min(Math.max(desired, lo), hi - duration);
    86→    const dist = Math.abs(cand - desired);
    87→    if (dist < bestDist) { bestDist = dist; best = cand; }
    88→  };
    89→  let prevEnd = 0;
    90→  for (const o of occupied) {
    91→    consider(prevEnd, o.start);
    92→    prevEnd = Math.max(prevEnd, o.end);
    93→  }
    94→  consider(prevEnd, Infinity); // open-ended tail — always fits
    95→  return Math.max(0, best);
    96→}

    98→/** Clamp a multi-clip drag delta so the WHOLE group stays overlap-free.
    99→ *  Each selected clip moves by the same `dt`; for each clip we collect the
   100→ *  range of `dt` values that keep it off its non-selected neighbours on the
   101→ *  same track, then intersect those ranges across all clips. If the desired
   102→ *  `dt` falls outside the intersection, we clamp to the nearest bound — so a
   103→ *  group dragged into a wall stops flush against it instead of teleporting
   104→ *  past. The origin track per clip is used (multi-drag is horizontal-only —
   105→ *  no cross-track reshuffle, which would be ambiguous for a group). */
   106→function clampGroupDt(
   107→  dt: number,
   108→  origins: Map<string, { os: number; od: number; trackId: string }>,
   109→  byTrack: Map<string, { start: number; end: number }[]>,
   110→): number {
   111→  let minRight = Infinity; // upper bound on dt (blocker to the right of a clip)
   112→  let maxLeft = -Infinity; // lower bound on dt (blocker to the left of a clip)
   113→  for (const o of origins.values()) {
   114→    const blockers = byTrack.get(o.trackId) ?? [];
   115→    for (const b of blockers) {
   116→      // Blocker entirely to the LEFT of the clip's origin → moving LEFT past
   117→      // it would overlap. Constraint: dt >= b.end - o.os.
   118→      if (b.end <= o.os + 1e-4) {
   119→        maxLeft = Math.max(maxLeft, b.end - o.os);
   120→      }
   121→      // Blocker entirely to the RIGHT of the clip's origin → moving RIGHT past
   122→      // it would overlap. Constraint: dt <= b.start - o.os - o.od.
   123→      else if (b.start >= o.os + o.od - 1e-4) {
   124→        minRight = Math.min(minRight, b.start - o.os - o.od);
   125→      }
   126→      // A blocker that straddles the clip's original interval would mean the
   127→      // clip was ALREADY overlapping before the drag — ignore (shouldn't happen).
   128→    }
   129→  }
   130→  if (maxLeft > minRight) return 0; // conflicting constraints — no movement
   131→  return Math.max(maxLeft, Math.min(minRight, dt));
   132→}

   134→// ── Waveform cache: pre-computed PNG data URLs ───────────────────────────────
   135→// Key: `${clipId}:${sampleCount}:${width}` → data URL
   136→const waveformPNGCache = new Map<string, string>();
   137→
   138→// Draw only the window [startFrame, endFrame) of the sample across `width`
   139→// pixels. Rendering the *window* (not the whole sample stretched) is what makes
   140→// trimming look like cutting the audio away rather than squashing it.
   141→function computeWaveformPNG(samples: Float32Array, channels: number, width: number, color: string, height: number, startFrame: number, endFrame: number): string {
   142→  const canvas = document.createElement('canvas');
   143→  const dpr = window.devicePixelRatio || 1;
   144→  const w = Math.max(2, Math.floor(width));
   145→  canvas.width = w * dpr;
   146→  canvas.height = height * dpr;
   147→  const ctx = canvas.getContext('2d')!;
   148→  ctx.scale(dpr, dpr);
   149→  ctx.clearRect(0, 0, w, height);
   150→  const totalFrames = samples.length / channels;
   151→  const s = Math.max(0, Math.min(totalFrames, Math.floor(startFrame)));
   152→  const e = Math.max(s, Math.min(totalFrames, Math.floor(endFrame)));
   153→  const span = e - s;
   154→  if (span <= 0) return canvas.toDataURL();
   155→  const mid = height / 2;
   156→  ctx.fillStyle = color;
   157→  for (let x = 0; x < w; x++) {
   158→    const f0 = s + Math.floor((x / w) * span);
   159→    const f1 = Math.min(e, s + Math.floor(((x + 1) / w) * span) + 1);
   160→    let peak = 0;
   161→    for (let i = f0; i < f1; i++) { const v = Math.abs(samples[i * channels] ?? 0); if (v > peak) peak = v; }
   162→    const bh = Math.max(1, peak * mid * 0.85);
   163→    ctx.fillRect(x, mid - bh, 1, bh * 2);
   164→  }
   165→  return canvas.toDataURL();
   166→}

   168→/** Waveform image for the buffer window the clip actually plays. With speed s,
   169→ *  a clip of `duration` timeline seconds consumes `duration*s` buffer seconds.
   170→ *
   171→ *  Two robustness fixes:
   172→ *  1. If `loadSamples` returns an EMPTY Float32Array (length 0) — which happens
   173→ *     when a freshly-split/created clip's IndexedDB write hasn't landed yet —
   174→ *     we DON'T cache the resulting blank PNG. Instead we show the `···`
   175→ *     placeholder and retry a few times (500ms apart) so the real waveform
   176→ *     appears once the samples are available.
   177→ *  2. We listen for `slate:audio-clip-changed` (fired by Normalize/Reverse/split)
   178→ *     and invalidate our cache entry for the current clipId, then force a
   179→ *     recompute via a `bust` counter — otherwise the memoised component would
   180→ *     keep showing the stale PNG even after the cache was cleared. */
   181→const WaveformImg = memo(function WaveformImg({ clipId, sampleKey, channels, sampleRate, offset, duration, speed, width, color }: {
   182→  clipId: string; sampleKey: string; channels: number; sampleRate: number;
   183→  offset: number; duration: number; speed: number; width: number; color: string;
   184→}) {
   185→  const height = TRACK_H - 6;
   186→  const [imgUrl, setImgUrl] = useState<string>('');
   187→  const [bust, setBust] = useState(0);
   188→  const retryRef = useRef(0);
   189→
   190→  // Invalidate cached PNGs for this clip when its samples change (normalize /
   191→  // reverse / split). The parent AudioEditor also invalidates + bumps version,
   192→  // but the memoised WaveformImg wouldn't otherwise recompute (its primitive
   193→  // props are unchanged) — the `bust` counter forces the load effect to re-run.
   194→  useEffect(() => {
   195→    const onChanged = (e: Event) => {
   196→      const detail = (e as CustomEvent<string>).detail;
   197→      // Match by clipId (normalize/reverse/split events) OR by sampleKey
   198→      // (multiplayer sample-arrival events from registerSampleSyncMap).
   199→      if (detail !== clipId && detail !== sampleKey) return;
   200→      for (const key of waveformPNGCache.keys()) {
   201→        if (key.startsWith(`${clipId}:`)) waveformPNGCache.delete(key);
   202→      }
   203→      retryRef.current = 0;
   204→      setBust((n) => n + 1);
   205→    };
   206→    window.addEventListener('slate:audio-clip-changed', onChanged as EventListener);
   207→    return () => window.removeEventListener('slate:audio-clip-changed', onChanged as EventListener);
   208→  }, [clipId, sampleKey]);
   209→
   210→  useEffect(() => {
   211→    const startFrame = Math.round(offset * sampleRate);
   212→    const endFrame = Math.round((offset + duration * speed) * sampleRate);
   213→    const cacheKey = `${clipId}:${sampleKey}:${startFrame}:${endFrame}:${Math.floor(width)}`;
   214→    const cached = waveformPNGCache.get(cacheKey);
   215→    if (cached) { setImgUrl(cached); return; }
   216→    let cancelled = false;
   217→    let retryTimer: ReturnType<typeof setTimeout> | null = null;
   218→
   219→    const attempt = () => {
   220→      if (cancelled) return;
   221→      void loadSamples(sampleKey).then((samples) => {
   222→        if (cancelled) return;
   223→        // Empty samples = the IndexedDB write for this sampleKey hasn't landed
   224→        // yet (e.g. brand-new clip from a split, OR a remote peer's clip whose
   225→        // sample blob is still in flight via the Yjs sync map). DON'T cache a
   226→        // blank PNG — retry for up to 10s (20 × 500ms) so the real waveform
   227→        // appears once the samples arrive. The previous 5-retry / 2.5s budget
   228→        // was too short for cross-peer sample sync, which can take several
   229→        // seconds for a 5MB blob over a slow link. The `slate:audio-clip-changed`
   230→        // event (dispatched by onClipsAdded when a remote clip is added, and by
   231→        // registerSampleSyncMap when samples arrive) resets `retryRef` to 0 so
   232→        // the WaveformImg re-tries immediately on the signal — this retry
   233→        // budget is the safety net for the case where the event fires before
   234→        // the WaveformImg mounts, or doesn't fire at all.
   235→        if (samples.length === 0) {
   236→          if (retryRef.current < 20) {
   237→            retryRef.current += 1;
   238→            retryTimer = setTimeout(attempt, 500);
   239→          }
   240→          return;
   241→        }
   242→        retryRef.current = 0;
   243→        const url = computeWaveformPNG(samples, channels, width, color, height, startFrame, endFrame);
   244→        waveformPNGCache.set(cacheKey, url);
   245→        setImgUrl(url);
   246→      });
   247→    };
   248→    attempt();
   249→
   250→    return () => {
   251→      cancelled = true;
   252→      if (retryTimer) clearTimeout(retryTimer);
   253→    };
   254→  }, [clipId, sampleKey, channels, sampleRate, offset, duration, speed, width, color, height, bust]);
   255→
   256→  if (!imgUrl) return <div className="flex h-full items-center justify-center text-[7px] text-text-dim">···</div>;
   257→  return <img src={imgUrl} alt="" className="pointer-events-none h-full w-full" style={{ objectFit: 'fill' }} />;
   258→});
   259→
   260→/** Drop cached waveform PNGs for a clip (call when its samples change). */
   261→function invalidateWaveform(clipId: string): void {
   262→  for (const key of waveformPNGCache.keys()) if (key.startsWith(`${clipId}:`)) waveformPNGCache.delete(key);
   263→}

   431→    // attach() — registering here only on AudioEditor mount meant a peer on
   432→    // a 2D/3D board (audio panel closed) never registered the sync map and

   439→    // Any track edit (volume, pan, EQ, sends, mute/solo — local knob or remote
   440→    // peer) re-applies the Yjs values to the live audio graph, so the settings
   441→    // panel is audible mid-playback without a restart.
   442→    //
   443→    // REMOTE MUTE/SOLO FIX: this subscription is what makes "user A mutes a
   444→    // track → user B hears the mute" work. The mute state IS in Yjs (it's a
   445→    // field on the track Y.Map), so it syncs to every peer. But sync alone
   446→    // doesn't affect the audio graph — the local engine's per-track gain
   447→    // nodes still hold the old value until something re-applies them.
   448→    // `applyTracks` is that "something": it reads the fresh Yjs tracks and
   449→    // calls `setupTrackNodes`, which writes `gain.gain.value = audible ?
   450→    // volume : 0` per track (audible = solo ? track.solo : !track.muted).
   451→    // Without this subscription, remote mute/solo edits would land in Yjs
   452→    // (visible in the TrackHeader) but never reach the audio graph until the
   453→    // user pressed stop/play (which re-runs setupTrackNodes via play()).
   454→    const applyTracks = () => { engineRef.current?.updateTracks(slateRef.current); };
   455→    // Clip edits (position, trim, gain, pan, filters, speed/pitch — ours or a
   456→    // peer's) reschedule playback so they're audible mid-play. Debounced 500ms
   457→    // in scheduleRestart and a no-op while paused, so knob drags coalesce into

   463→    clips.observeDeep(restartOnClipEdit);

   465→    // restartPlayback so the new clip is picked up mid-playback without a
   466→    // manual stop/play. We watch the shallow clips map (not observeDeep) so
   467→    // we only react to key additions, not to every property edit on an
   468→    // existing clip (which would restart playback every time a peer nudged
   469→    // a clip's volume, fighting the user). The restart is debounced 500ms so
   470→    // a burst of additions (e.g. a peer imports 5 files) coalesces into one.
   471→    //
   472→    // LIVE WAVEFORM REFRESH: when a remote peer adds a clip, its metadata
   473→    // arrives via Yjs immediately but the sample blob travels separately via
   474→    // the audioSampleSync Y.Map (see sampleStore.ts). If we don't ping the
   475→    // WaveformImg, it might have already exhausted its empty-samples retry
   476→    // budget (now 10s, but a slow link can still exceed that) and given up
   477→    // showing a blank "···" placeholder until the user refreshes. So for
   478→    // each newly-added clip we:
   479→    //   1. clearCache(clipId) on the engine so the next play re-loads samples
   480→    //      (the cache might hold a null/empty entry from a prior failed getBuffer)
   481→    //   2. dispatch `slate:audio-clip-changed` with the clipId — the
   482→    //      AudioEditor's listener (below) re-invalidates the waveform cache +
   483→    //      clears the engine cache + bumps version, AND the WaveformImg's own
   484→    //      listener resets its retryRef to 0 and bumps `bust` to immediately
   485→    //      re-attempt loadSamples (which will now find the synced blob once
   486→    //      it lands).
   487→    //   3. scheduleRestart() so if we're currently playing, the new clip is
   488→    //      picked up atomically (debounced 500ms to coalesce bursts).
   489→    const onClipsAdded = (event: Y.YMapEvent<Y.Map<unknown>>) => {
   490→      const addedIds: string[] = [];

   492→        const change = event.changes.keys.get(key);
   493→        if (change?.action === 'add') addedIds.push(key);
   494→      }
   495→      if (addedIds.length === 0) return;
   496→      for (const id of addedIds) {
   497→        engineRef.current?.clearCache(id);
   498→        window.dispatchEvent(new CustomEvent('slate:audio-clip-changed', { detail: id }));
   499→      }
   500→      scheduleRestart();
   501→    };

   503→    const lateRead = setTimeout(bump, 200);
   504→    return () => { clearTimeout(lateRead); tracks.unobserveDeep(bump); tracks.unobserveDeep(applyTracks); clips.unobserveDeep(bump); clips.unobserveDeep(restartOnClipEdit); audioMap.unobserve(bump); clips.unobserve(onClipsAdded); };
   505→    // eslint-disable-next-line react-hooks/exhaustive-deps
   506→  }, [slate]);

   509→   *  Y.UndoManager only captures LOCAL transactions (origin null), so a
   510→   *  peer's edits are never undone from this client. Sample blobs stay in
   511→   *  IndexedDB when a clip is deleted (see scene.ts) so an undone delete
   512→   *  comes back playable. */

   516→      captureTimeout: 300,
   517→    });
   518→    undoRef.current = um;
   519→    return () => {
   520→      um.destroy();
   521→      undoRef.current = null;
   522→    };

   529→      // Cancel any pending restart so it doesn't fire after dispose().
   530→      if (restartTimerRef.current) { clearTimeout(restartTimerRef.current); restartTimerRef.current = null; }
   531→      engineRef.current?.dispose();

   569→  // Ctrl+scroll zoom — centred on the playhead so the screen position under
   570→  // the cursor/playhead stays put instead of zooming toward the left edge.
   571→  /** Zoom while keeping one timeline point at the same screen offset.
   572→   *  The anchor is the time under `anchorClientX` (mouse-wheel zoom: the spot
   573→   *  under the cursor stays put — how every DAW/map zoom feels), or the
   574→   *  playhead when no anchor is given (toolbar buttons, fit-to-window).
   575→   *  Keeping time t fixed on screen means newScroll = oldScroll + t·Δpps,
   576→   *  since screenX(t) = timelineLeft − scrollLeft + t·pps. The `scrollLeft`
   577→   *  write happens in the layout effect below (after React commits the new
   578→   *  `minWidth` on the timeline div — setting it earlier would be clamped by
   579→   *  the stale `scrollWidth`). */
   580→  const zoomAnchored = useCallback((newPxPerSec: number, anchorClientX?: number) => {
   581→    const el = scrollRef.current;
   582→    const oldPxPerSec = pxRef.current;
   583→    if (!el || oldPxPerSec === newPxPerSec) {
   584→      setPxPerSec(newPxPerSec);
   585→      return;
   586→    }
   587→    let t = positionRef.current; // default anchor: the playhead
   588→    if (anchorClientX !== undefined) {
   589→      const tl = el.querySelector('[data-timeline]') as HTMLElement | null;
   590→      if (tl) t = Math.max(0, (anchorClientX - tl.getBoundingClientRect().left) / oldPxPerSec);
   591→    }
   592→    pendingScrollRef.current = Math.max(0, el.scrollLeft + t * (newPxPerSec - oldPxPerSec));
   593→    setPxPerSec(newPxPerSec);
   594→  }, []);
   595→
   596→  // Apply any pending scroll correction after pxPerSec commits to the DOM.
   597→  useLayoutEffect(() => {
   598→    if (pendingScrollRef.current !== null && scrollRef.current) {
   599→      scrollRef.current.scrollLeft = pendingScrollRef.current;
   600→      pendingScrollRef.current = null;
   601→    }
   602→  }, [pxPerSec]);
   603→
   604→  useEffect(() => {
   605→    const el = scrollRef.current; if (!el) return;
   606→    const onWheel = (e: WheelEvent) => {
   607→      if (!e.ctrlKey && !e.metaKey) return;
   608→      e.preventDefault();
   609→      const next = Math.max(MIN_PX_PER_SEC, Math.min(MAX_PX_PER_SEC, pxRef.current * (e.deltaY < 0 ? 1.2 : 1 / 1.2)));
   610→      zoomAnchored(next, e.clientX); // zoom toward the cursor
   611→    };
   612→    el.addEventListener('wheel', onWheel, { passive: false });
   613→    return () => el.removeEventListener('wheel', onWheel);
   614→  }, [zoomAnchored]);
   615→
   616→  // Global pointermove/up — pure DOM, zero React state.
   617→  // rAF-throttled: pointermove fires at the hardware's report rate (often
   618→  // 60-120Hz), and the move handler does a neighbours-overlap scan on every
   619→  // event. Coalescing multiple moves into a single rAF callback caps the work
   620→  // at the display refresh rate (~60fps) so a fast pointer drag never stalls
   621→  // the main thread. A no-op skip (cursor hasn't moved since the last
   622→  // processed frame) avoids redundant DOM writes for stationary pointers.
   623→  useEffect(() => {
   624→    /** Last pointer position we actually applied to the DOM. Lets us skip
   625→     *  work when pointermove fires without real movement (e.g. a touchpad
   626→     *  reporting pressure changes, or sub-pixel jitter). */
   627→    let lastProcessedX: number | null = null;
   628→    let lastProcessedY: number | null = null;
   629→
   630→    /** Magnet: pull `t` (a clip START time) to the nearest beat line, other
   631→     *  clip edge, or the playhead — considering BOTH the clip's start and end
   632→     *  edges — if one is within SNAP_PX at the current zoom. `dur` null =
   633→     *  only the start edge matters (trims). Returns the adjusted start. */
   634→    const snapStart = (t: number, dur: number | null, snapTimes: number[]): number => {
   635→      const pps = pxRef.current;
   636→      const thr = SNAP_PX / pps;
   637→      const beat = 60 / bpmRef.current;
   638→      let best = t;
   639→      let bestDist = thr;
   640→      const consider = (target: number, edgeOffset: number) => {
   641→        const dist = Math.abs(t + edgeOffset - target);
   642→        if (dist < bestDist) { bestDist = dist; best = target - edgeOffset; }
   643→      };
   644→      consider(Math.round(t / beat) * beat, 0);
   645→      if (dur !== null) consider(Math.round((t + dur) / beat) * beat, dur);
   646→      for (const st of snapTimes) {
   647→        consider(st, 0);
   648→        if (dur !== null) consider(st, dur);
   649→      }
   650→      return best;
   651→    };
   652→
   653→    const applyMove = () => {
   654→      moveRafRef.current = 0;
   655→      const d = dragRef.current;
   656→      if (!d) return;
   657→      const clientX = moveXRef.current;
   658→      if (clientX === null) return;
   659→      const clientY = moveYRef.current;
   660→      // Skip no-op moves — cursor hasn't moved to a new pixel since the last
   661→      // frame we processed. Saves an overlap scan + 1-2 style writes.
   662→      if (lastProcessedX === clientX && lastProcessedY === clientY) return;
   663→      lastProcessedX = clientX;
   664→      lastProcessedY = clientY;
   665→
   666→      // Dead zone: a plain click (pointer never leaves a 3px box) must not
   667→      // nudge the clip. Once exceeded, the drag is live for good.
   668→      if (!d.moved) {
   669→        if (Math.abs(clientX - d.sx) < DRAG_DEADZONE_PX && Math.abs(clientY - d.sy) < DRAG_DEADZONE_PX) return;
   670→        d.moved = true;
   671→        if (d.mode === 'drag') {
   672→          d.el.style.zIndex = '40'; // ride above other rows while crossing tracks
   673→          // Elevate the whole group too so the user sees the multi-clip
   674→          // selection ride above non-selected neighbours while in transit.
   675→          if (d.origins) for (const o of d.origins.values()) if (o.el !== d.el) o.el.style.zIndex = '40';
   676→        }
   677→      }
   678→
   679→      const pps = pxRef.current;
   680→      const dt = (clientX - d.sx) / pps;
   681→      const oldEnd = d.os + d.od;
   682→      const wantSnap = snapRef.current && !moveAltRef.current;
   683→      if (d.mode === 'drag') {
   684→        // MULTI-SELECT DRAG: when `origins` is populated, every selected clip
   685→        // moves by the same horizontal delta. Vertical track-switching is
   686→        // disabled for the group (each clip stays on its origin track) since
   687→        // reshuffling every selected clip onto one target track would be
   688→        // ambiguous and likely overlap-prone. The dragged clip's snap result
   689→        // is applied to the whole group so the magnet still feels right.
   690→        if (d.origins && d.origins.size > 0) {
   691→          const dragged = d.origins.get(d.clipId);
   692→          let snappedDt = dt;
   693→          if (wantSnap && dragged) {
   694→            const snappedStart = snapStart(dragged.os + dt, dragged.od, d.snapTimes);
   695→            snappedDt = snappedStart - dragged.os;
   696→          }
   697→          for (const o of d.origins.values()) {
   698→            const newStart = Math.max(0, o.os + snappedDt);
   699→            o.el.style.left = `${newStart * pps}px`;
   700→          }
   701→          return;
   702→        }
   703→        // The clip follows the mouse 1:1 (snapped to the magnet grid). It's
   704→        // allowed to ride OVER other clips while the drag is live — it's
   705→        // elevated to z-40 so the overlap reads as "in transit", and the
   706→        // no-overlap invariant is enforced once, on drop (see onUp). The
   707→        // previous mid-drag resolution parked the clip flush against a
   708→        // blocker and then teleported it across when the cursor passed the
   709→        // blocker's midpoint, which made dragging feel sticky and jumpy.
   710→        const duration = d.od;
   711→        // Vertical: whole-row steps move the clip across tracks.
   712→        const rowDelta = Math.max(
   713→          -d.trackIndex,
   714→          Math.min(d.trackIds.length - 1 - d.trackIndex, Math.round((clientY - d.sy) / TRACK_H)),
   715→        );
   716→        d.rowDelta = rowDelta;
   717→        d.el.style.transform = rowDelta === 0 ? '' : `translateY(${rowDelta * TRACK_H}px)`;
   718→
   719→        const desired = wantSnap ? snapStart(d.os + dt, duration, d.snapTimes) : d.os + dt;
   720→        d.el.style.left = `${Math.max(0, desired) * pps}px`;
   721→      } else if (d.mode === 'trimL') {
   722→        // Cut from the left: never past the left neighbour, the source start
   723→        // (offset ≥ 0 → limited by the trimmed head in timeline seconds), or a
   724→        // minimum width.
   725→        const minStart = Math.max(d.leftLimit, d.os - d.oo / d.speed);
   726→        const desired = wantSnap ? snapStart(d.os + dt, null, d.snapTimes) : d.os + dt;
   727→        const start = Math.min(oldEnd - 0.1, Math.max(minStart, desired));
   728→        d.el.style.left = `${start * pps}px`;
   729→        d.el.style.width = `${(oldEnd - start) * pps}px`;
   730→        // Shift the (fixed-width) waveform the opposite way so the audio stays
   731→        // anchored — the newly-trimmed head is clipped, not squashed.
   732→        if (d.waveEl) d.waveEl.style.left = `${-(start - d.os) * pps}px`;
   733→      } else if (d.mode === 'trimR') {
   734→        // Cut from the right: never past the right neighbour or a min width.
   735→        const desired = wantSnap ? snapStart(oldEnd + dt, null, d.snapTimes) : oldEnd + dt;
   736→        const end = Math.min(d.rightLimit, Math.max(d.os + 0.1, desired));
   737→        d.el.style.width = `${(end - d.os) * pps}px`;
   738→      }
   739→    };
   740→
   741→    const onMove = (ev: PointerEvent) => {
   742→      const d = dragRef.current; if (!d) return;
   743→      // Stash the latest pointer state — the rAF callback reads it. Cheap
   744→      // (ref writes) so even 120Hz pointermove streams don't pile up work.
   745→      moveXRef.current = ev.clientX;
   746→      moveYRef.current = ev.clientY;
   747→      moveAltRef.current = ev.altKey;
   748→      // Schedule a frame if one isn't already pending. The guard is what
   749→      // coalesces multiple moves within the same frame into one DOM write.
   750→      if (!moveRafRef.current) {
   751→        moveRafRef.current = requestAnimationFrame(applyMove);
   752→      }
   753→    };
   754→    const onUp = () => {
   755→      const d = dragRef.current; if (!d) return;
   756→      // Cancel any pending rAF — we'll process the final position
   757→      // SYNCHRONOUSLY here so the committed Yjs value reflects the exact
   758→      // pointer-up location (otherwise the last pointermove before pointerup
   759→      // could be dropped, leaving the committed position ~1 frame behind the
   760→      // cursor). lastProcessed is reset to bypass the no-op skip in case the
   761→      // final pointermove matched the prior processed frame but we still need
   762→      // to read the just-written style to commit to Yjs.
   763→      if (moveRafRef.current) {
   764→        cancelAnimationFrame(moveRafRef.current);
   765→        moveRafRef.current = 0;
   766→      }
   767→      if (moveXRef.current !== null) {
   768→        lastProcessedX = null; // bypass the no-op skip on the final flush
   769→        lastProcessedY = null;
   770→        applyMove();
   771→      }
   772→      // Undo the drag-only visual state React doesn't manage. For a multi-clip
   773→      // drag, also clear the elevated z-index on every selected clip's element.
   774→      if (d.origins) for (const o of d.origins.values()) { o.el.style.zIndex = ''; o.el.style.transform = ''; }
   775→      d.el.style.zIndex = '';
   776→      d.el.style.transform = '';
   777→      // A click that never left the dead zone selects but must not write —
   778→      // committing a parseFloat round-trip of the style would drift the clip.
   779→      if (d.moved) {
   780→        const pps = pxRef.current;
   781→        const left = parseFloat(d.el.style.left) / pps;
   782→        const width = parseFloat(d.el.style.width) / pps;
   783→        if (d.mode === 'drag') {
   784→          // MULTI-SELECT COMMIT: every selected clip moves by the same delta,
   785→          // clamped so the group stays overlap-free against non-selected
   786→          // neighbours on each clip's origin track. The delta is recomputed
   787→          // from the dragged clip's pointer-up position (so the snapped /
   788→          // clamped value the user sees is what gets committed), then applied
   789→          // to every clip in the group via updateAudioClip inside a single
   790→          // Yjs transaction.
   791→          if (d.origins && d.origins.size > 0) {
   792→            const dragged = d.origins.get(d.clipId);
   793→            const dt = dragged ? left - dragged.os : 0;
   794→            const clampedDt = clampGroupDt(dt, d.origins, d.byTrack);
   795→            // Write the resolved positions to the DOM too so there's no
   796→            // flicker between pointer-up and the Yjs-driven re-render.
   797→            for (const o of d.origins.values()) {
   798→              const newStart = Math.max(0, o.os + clampedDt);
   799→              o.el.style.left = `${newStart * pps}px`;
   800→            }
   801→            // Single transaction so the whole group commits atomically to the
   802→            // Yjs doc — peers see one update, not N interleaved with their
   803→            // own edits.
   804→            slate.doc.transact(() => {
   805→              for (const [cid, o] of d.origins!) {
   806→                updateAudioClip(slate, cid, { start: Math.max(0, o.os + clampedDt) });
   807→              }
   808→            });
   809→          } else {
   810→            const targetTrackId = d.trackIds[d.trackIndex + d.rowDelta] ?? d.trackIds[d.trackIndex]!;
   811→            const originTrackId = d.trackIds[d.trackIndex]!;
   812→            // Enforce the no-overlap invariant HERE (the drag itself follows
   813→            // the mouse freely): place the clip in the free gap nearest to the
   814→            // dropped position.
   815→            const start = nearestFreeStart(left, d.od, d.byTrack.get(targetTrackId) ?? []);
   816→            // Write the resolved position to the DOM too so there's no flicker
   817→            // between pointer-up and the Yjs-driven re-render.
   818→            d.el.style.left = `${start * pps}px`;
   819→            updateAudioClip(slate, d.clipId, {
   820→              start,
   821→              ...(targetTrackId !== originTrackId ? { trackId: targetTrackId } : {}),
   822→            });
   823→          }
   824→        }
   825→        else if (d.mode === 'trimL') updateAudioClip(slate, d.clipId, { start: left, duration: width, offset: Math.max(0, d.oo + (left - d.os) * d.speed) });
   826→        else if (d.mode === 'trimR') updateAudioClip(slate, d.clipId, { duration: width });
   827→      }
   828→      moveXRef.current = null;
   829→      lastProcessedX = null;
   830→      lastProcessedY = null;
   831→      dragRef.current = null;
   832→    };
   833→    window.addEventListener('pointermove', onMove);
   834→    window.addEventListener('pointerup', onUp);
   835→    return () => {
   836→      window.removeEventListener('pointermove', onMove);
   837→      window.removeEventListener('pointerup', onUp);
   838→      // Don't leak a pending rAF if the component unmounts mid-drag.
   839→      if (moveRafRef.current) {
   840→        cancelAnimationFrame(moveRafRef.current);
   841→        moveRafRef.current = 0;
   842→      }
   843→    };
   844→  }, [slate]);

   906→  // When a clip's samples change (normalize/reverse from the settings panel,
   907→  // OR a remote peer's sample blob just landed via the Yjs sync map),
   908→  // drop the cached waveform + decoded buffer so both refresh, AND — if we're
   909→  // currently playing — schedule a debounced restartPlayback so the clip is
   910→  // picked up. This covers the case where a clip's metadata arrived but its
   911→  // samples were still in flight: getBuffer returned null on the first pass,
   912→  // and this restart re-schedules once the samples are usable.
   913→  useEffect(() => {
   914→    const onChanged = (e: Event) => {
   915→      const detail = (e as CustomEvent<string>).detail;
   916→      // The event's detail is a CLIP id for local edits, but a SAMPLE KEY for
   917→      // sample-sync arrivals (registerSampleSyncMap only knows the key).
   918→      // The engine's buffer cache and the waveform PNG cache are keyed by
   919→      // clip id — clearing them with a sampleKey was a silent no-op, so a
   920→      // peer whose clip was normalized/reversed/split by someone else got a
   921→      // fresh waveform but kept PLAYING the stale audio. Resolve the detail
   922→      // to every matching clip id (usually exactly one).
   923→      const clipIds: string[] = [];
   924→      const clipsMap = slateRef.current.audioClips();
   925→      if (clipsMap.get(detail)) clipIds.push(detail);
   926→      else clipsMap.forEach((m, cid) => { if (m.get('sampleKey') === detail) clipIds.push(cid); });
   927→      if (clipIds.length === 0) clipIds.push(detail); // unknown — invalidate as-is
   928→      for (const id of clipIds) {
   929→        invalidateWaveform(id);
   930→        engineRef.current?.clearCache(id);
   931→        // Pre-warm the engine's AudioBuffer cache even when not playing —
   932→        // otherwise a remote peer who's paused when samples land goes through
   933→        // the full getBuffer retry loop on their next play(), adding audible
   934→        // latency to the first playback. scheduleRestart() is a no-op when
   935→        // paused, so without this the cache stays cold until play. Safe to
   936→        // call when the engine has no AudioContext yet (preloadBuffer no-ops).
   937→        void engineRef.current?.preloadBuffer(slateRef.current, id);
   938→      }

   940→      // sampleKey — the engine's in-flight sample-retry loops key their
   941→      // budget reset on clip ids. Terminates: the re-dispatched event's
   942→      // detail IS a clip id, so this branch won't fire again.

   944→        if (id !== detail) window.dispatchEvent(new CustomEvent('slate:audio-clip-changed', { detail: id }));
   945→      }

   954→        title: 'Clip won’t sync to collaborators',
   955→        description: 'It plays for you, but this clip is too long or the board’s shared audio space is full. Collaborators won’t hear it.',
   956→        variant: 'error',
   957→      });
   958→    };
   959→    window.addEventListener('slate:audio-clip-changed', onChanged as EventListener);
   960→    window.addEventListener('slate:audio-sync-skipped', onSkipped);
   961→    return () => {
   962→      window.removeEventListener('slate:audio-clip-changed', onChanged as EventListener);
   963→      window.removeEventListener('slate:audio-sync-skipped', onSkipped);
   964→    };

   966→  }, []);

  1538→              setMarquee({ x1: s.startX, y1: s.startY, x2: ex, y2: ey });
  1539→              // Hit-test clips against the marquee rect
  1540→              const x1 = Math.min(s.startX, ex);
  1541→              const x2 = Math.max(s.startX, ex);
  1542→              const y1 = Math.min(s.startY, ey);
  1543→              const y2 = Math.max(s.startY, ey);
  1544→              const t1 = x1 / pxRef.current, t2 = x2 / pxRef.current;
  1545→              const startRowIdx = Math.floor(y1 / TRACK_H);
  1546→              const endRowIdx = Math.floor(y2 / TRACK_H);
  1547→              const next = new Set(s.origin);
  1548→              clips.forEach((c) => {
  1549→                const ti = tracks.findIndex((t) => t.id === c.trackId);
  1550→                if (ti < startRowIdx || ti > endRowIdx) return;
  1551→                const cEnd = c.start + c.duration;
  1552→                if (cEnd >= t1 && c.start <= t2) next.add(c.id);
  1553→              });
  1554→              setSelectedClipIds(next);
  1555→            }}
  1556→            onPointerUp={() => {
  1557→              const s = marqueeRef.current;
  1558→              marqueeRef.current = null;
  1559→              setMarquee(null);
  1560→              if (!s) return;
  1561→              if (!s.moved) {
  1562→                // Plain click → seek to the pointerdown position. Non-additive
  1563→                // clicks also clear the selection (clicking empty space
  1564→                // dismisses a multi-select); additive clicks (shift/cmd on
  1565→                // empty space) leave the selection alone.
  1566→                seek(s.seekTime);
  1567→                if (!s.additive) setSelectedClipIds(new Set());
  1568→              }
  1569→              // If `s.moved` is true the marquee selection was already
  1570→              // finalised incrementally in pointermove — nothing to do here.
  1571→            }}
  1572→          />
  1573→          {/* Marquee rect overlay */}
  1574→          {marquee && (
  1575→            <div className="pointer-events-none absolute z-30 border border-accent/70 bg-accent/15" style={{
  1576→              left: Math.min(marquee.x1, marquee.x2),
  1577→              top: 28 + Math.min(marquee.y1, marquee.y2),
  1578→              width: Math.abs(marquee.x2 - marquee.x1),
  1579→              height: Math.abs(marquee.y2 - marquee.y1),
  1580→            }} />
  1581→          )}
  1582→          {/* Grid background */}
  1583→          <div className="pointer-events-none absolute inset-0 top-7" style={gridStyle}>
  1584→            {looping && <div onPointerDown={(e) => startLoopDrag('move', e)} className="pointer-events-auto absolute top-0 bottom-0 cursor-grab bg-accent/8 border-x-2 border-accent/40" style={{ left: loopStart * pxPerSec, width: (loopEnd - loopStart) * pxPerSec }} />}
  1585→          </div>
  1586→          {/* Playhead */}
  1587→          <div ref={playheadRef} className="absolute top-0 bottom-0 z-20 w-0.5 bg-warn pointer-events-none" style={{ transform: 'translateX(0px)' }}>
  1588→            <div className="absolute top-0 -left-1 h-2 w-2 rounded-full bg-warn" />
  1589→          </div>
  1590→          {/* Clips */}
  1591→          {tracks.map((t) => (
  1592→            <div key={t.id} data-track-id={t.id} className="pointer-events-none relative border-b border-border/15" style={{ height: TRACK_H }}>
  1593→              {clips.filter((c) => c.trackId === t.id).map((c) => (
  1594→                <ClipBlock key={c.id} clip={c} pxPerSec={pxPerSec} selected={selectedClipIds.has(c.id)}
  1595→                  onDragStart={startDrag} onTrimStart={startTrim} />
  1596→              ))}
  1597→            </div>
  1598→          ))}
  1599→          {/* Remote playheads — overlay on top of clips, mirroring each peer's
  1600→           *  audio transport position. Rendered after clips so z-20 wins. */}
  1601→          <RemotePlayheads room={room} peerStates={peerStates} pxRef={pxRef} selfId={room.identity.peerId} />
  1602→        </div>
  1603→      </div>
  1604→
  1605→      {/* Status */}
  1606→      <div className="flex shrink-0 items-center gap-2 border-t border-border px-2 py-0.5 text-[8px] font-mono uppercase text-text-dim">
  1607→        <span>{tracks.length}T · {clips.length}C</span>
  1608→        {recording && <span className="text-danger">● Rec</span>}
  1609→        {playing && <span className="text-accent">▶ Play</span>}
  1610→        <span className="ml-auto">Space · Ctrl+Z=Undo · C=Split · D=Dup · Ctrl+C/V=Copy/Paste · Del · R=Rec · L=Loop · M=Met · ←→=Seek · Ctrl+Scroll=Zoom</span>
  1611→      </div>
  1612→    </div>
  1613→  );

  1618→const TrackHeader = memo(function TrackHeader({ track, hasSolo, slate, engineRef }: {
  1619→  track: AudioTrack; hasSolo: boolean;
  1620→  slate: ReturnType<typeof useRoom>['slate'];
  1621→  engineRef: React.RefObject<AudioEngine | null>;
  1622→}) {
  1623→  const [vol, setVol] = useState(track.volume);
  1624→  const [pan, setPan] = useState(track.pan);
  1625→  // Separate drag flags per slider — the previous single shared `isDraggingRef`
  1626→  // meant starting a volume drag also blocked the pan prop-sync effect (and
  1627→  // vice versa), which could cause one slider's committed Yjs value to be
  1628→  // momentarily ignored by the other's gating. Per-slider flags keep each
  1629→  // slider's prop-sync independent.
  1630→  const isDraggingVolRef = useRef(false);
  1631→  const isDraggingPanRef = useRef(false);
  1632→  useEffect(() => { if (!isDraggingVolRef.current) setVol(track.volume); }, [track.volume]);
  1633→  useEffect(() => { if (!isDraggingPanRef.current) setPan(track.pan); }, [track.pan]);
  1634→
  1635→  // Audibility — mirrors engine.setupTrackNodes so the live gain we write
  1636→  // during a volume drag respects mute/solo (don't briefly un-mute a muted
  1637→  // track just because the user is dragging its volume slider).
  1638→  const audible = hasSolo ? track.solo : !track.muted;
  1639→
  1640→  // `update` is for non-slider track edits (name/mute/solo/arm) — these DO
  1641→  // need a full updateTracks so the audio graph reflects the new state
  1642→  // (e.g. toggling solo rebalances every track's gain).
  1643→  const update = (patch: Partial<AudioTrack>) => { updateAudioTrack(slate, track.id, patch); engineRef.current?.updateTracks(slate); };
  1644→
  1645→  // Volume slider: update local state + the engine's gain node DIRECTLY for
  1646→  // immediate audio feedback. Do NOT call updateTracks (which re-reads every
  1647→  // track from Yjs and rebuilds the audio graph — far too expensive per
  1648→  // onChange event, and the source of the reported lag). Commit to Yjs once
  1649→  // on pointerup.
  1650→  const onVolDown = () => { isDraggingVolRef.current = true; };
  1651→  const onVol = (v: number) => { setVol(v); engineRef.current?.setTrackVolume(track.id, v, audible); };
  1652→  const onVolEnd = () => { isDraggingVolRef.current = false; updateAudioTrack(slate, track.id, { volume: vol }); };
  1653→
  1654→  // Pan slider: same pattern — set the panner node directly, commit on
  1655→  // pointerup. The pan slider is smaller than the volume slider (secondary
  1656→  // control) but still usable, with L/R labels so its function is obvious
  1657→  // (the user complaint was "two sliders, only one works" — partly because
  1658→  // neither slider was labelled, so it wasn't clear what each did).
  1659→  const onPanDown = () => { isDraggingPanRef.current = true; };
  1660→  const onPan = (p: number) => { setPan(p); engineRef.current?.setTrackPan(track.id, p); };
  1661→  const onPanEnd = () => { isDraggingPanRef.current = false; updateAudioTrack(slate, track.id, { pan: pan }); };
  1662→
  1663→  // ── Track kind: Audio ↔ MIDI ─────────────────────────────────────────────
  1664→  // A single icon button toggles the kind. On a MIDI track, an instrument
  1665→  // picker (compact <select>) sits next to the name so the user can choose
  1666→  // which synth preset or soundfont the track's MIDI clips play through.
  1667→  // Toggling kind to 'midi' defaults the instrument to the soundfont piano
  1668→  // (a sensible starting point for a fresh MIDI track); toggling back to
  1669→  // 'audio' leaves the instrumentId in place (harmless — it's only consulted
  1670→  // for MIDI clips).
  1671→  const isMidi = track.kind === 'midi';
  1672→  const toggleKind = () => {
  1673→    const nextKind: AudioTrack['kind'] = isMidi ? 'audio' : 'midi';
  1674→    // Switching to MIDI defaults the instrument to the soundfont piano if the
  1675→    // track doesn't already have one set; switching to audio clears the input
  1676→    // back to 'none' (a MIDI-armed audio track makes no sense).
  1677→    const patch: Partial<AudioTrack> = { kind: nextKind };
  1678→    if (nextKind === 'midi') {
  1679→      if (!track.instrumentId) patch.instrumentId = SOUNDFONT_PIANO_ID;
  1680→      // If the track was armed for mic, drop the arm — MIDI tracks arm for
  1681→      // instrument-take recording, not mic input.
  1682→      if (track.armed) { patch.armed = false; patch.input = 'none'; }
  1683→    } else if (track.input === 'midi') {
  1684→      patch.input = 'none';
  1685→      patch.armed = false;
  1686→    }
  1687→    update(patch);
  1688→  };
  1689→
  1690→  // Arm button: on an audio track → 'mic' input; on a MIDI track → 'midi'
  1691→  // input. The arm state still toggles; only the input source differs.
  1692→  const toggleArm = () => {
  1693→    if (track.armed) update({ armed: false, input: 'none' });
  1694→    else update({ armed: true, input: isMidi ? 'midi' : 'mic' });
  1695→  };
  1696→
  1697→  return (
  1698→    <div className="border-b border-border/15 px-2 py-1" style={{ height: TRACK_H }}>
  1699→      <div className="flex items-center gap-0.5">
  1700→        <span className="h-2 w-2 shrink-0 rounded-full" style={{ backgroundColor: track.color }} />
  1701→        <input type="text" value={track.name} onChange={(e) => update({ name: e.target.value })} className="min-w-0 flex-1 bg-transparent text-[11px] font-medium text-text outline-none" />
  1702→        {/* Kind toggle: Audio (Volume2 icon) ↔ MIDI (Piano icon). Highlighted
  1703→            when MIDI so the kind is visually distinct at a glance. */}
  1704→        <button onClick={toggleKind} className={`flex h-4 w-4 items-center justify-center rounded ${isMidi ? 'bg-accent/30 text-accent' : 'text-text-mid hover:bg-bg-3'}`} title={isMidi ? 'MIDI track (click for Audio)' : 'Audio track (click for MIDI)'}>{isMidi ? <Piano size={9} /> : <Volume2 size={9} />}</button>
  1705→        <button onClick={() => update({ muted: !track.muted })} className={`flex h-4 w-4 items-center justify-center rounded ${track.muted && !hasSolo ? 'bg-warn/30 text-warn' : 'text-text-mid hover:bg-bg-3'}`} title="M">{track.muted ? <VolumeX size={9} /> : <Volume2 size={9} />}</button>
  1706→        <button onClick={() => update({ solo: !track.solo })} className={`flex h-4 w-4 items-center justify-center rounded ${track.solo ? 'bg-accent/30 text-accent' : 'text-text-mid hover:bg-bg-3'}`} title="S"><Headphones size={9} /></button>
  1707→        <button onClick={toggleArm} className={`flex h-4 w-4 items-center justify-center rounded ${track.armed ? 'bg-danger/30 text-danger' : 'text-text-mid hover:bg-bg-3'}`} title={isMidi ? 'Arm for MIDI take' : 'Arm'}><div className="h-1.5 w-1.5 rounded-full" style={{ backgroundColor: track.armed ? 'currentColor' : 'transparent', border: '1px solid currentColor' }} /></button>
  1708→        <button onClick={() => deleteAudioTrack(slate, track.id)} className="flex h-4 w-4 items-center justify-center rounded text-text-mid hover:bg-bg-3 hover:text-danger" title="Del"><Trash2 size={9} /></button>
  1709→      </div>
  1710→      {/* MIDI tracks: instrument picker row replaces the pan slider row.
  1711→          Audio tracks keep the original volume + pan slider row. */}

  1713→        <div className="mt-0.5 flex items-center gap-1">
  1714→          <Piano size={9} className="shrink-0 text-accent" aria-hidden />
  1715→          <select
  1716→            value={track.instrumentId ?? SOUNDFONT_PIANO_ID}
  1717→            onChange={(e) => update({ instrumentId: e.target.value })}
  1718→            className="min-w-0 flex-1 rounded-sm border border-border bg-bg-3 px-1 py-0.5 text-[9px] text-text outline-none focus:border-accent"
  1719→            aria-label="Instrument"

  1751→const ClipBlock = memo(function ClipBlock({ clip, pxPerSec, selected, onDragStart, onTrimStart }: {
  1752→  clip: AudioClip; pxPerSec: number; selected: boolean;
  1753→  onDragStart: (clip: AudioClip, e: React.PointerEvent, el: HTMLElement, waveEl: HTMLElement | null) => void;
  1754→  onTrimStart: (clip: AudioClip, side: 'left' | 'right', e: React.PointerEvent, el: HTMLElement, waveEl: HTMLElement | null) => void;
  1755→}) {
  1756→  const left = clip.start * pxPerSec;
  1757→  const width = Math.max(4, clip.duration * pxPerSec);
  1758→  const elRef = useRef<HTMLDivElement | null>(null);
  1759→  const waveRef = useRef<HTMLDivElement | null>(null);
  1760→  const w = Math.max(2, Math.floor(width));
  1761→  // Fade overlay widths (clamped to the clip box so a too-long fade doesn't
  1762→  // spill past the opposite edge).
  1763→  const fadeInW = clip.fadeIn > 0 ? Math.min(width, clip.fadeIn * pxPerSec) : 0;
  1764→  const fadeOutW = clip.fadeOut > 0 ? Math.min(width, clip.fadeOut * pxPerSec) : 0;
  1765→
  1766→  // onDragStart handles selection (incl. Shift/Cmd additive) + the
  1767→  // clip-select event — selecting here too would clobber multi-select.
  1768→  // `data-clip-id` is set so the multi-drag startDrag can look up every
  1769→  // selected clip's DOM element via document.querySelector when assembling
  1770→  // the `origins` map (see startDrag's multiDrag branch).
  1771→  return (
  1772→    <div ref={elRef} data-clip-id={clip.id} onPointerDown={(e) => { if (elRef.current) onDragStart(clip, e, elRef.current, waveRef.current); }}
  1773→      className={`group pointer-events-auto absolute top-0.5 bottom-0.5 cursor-grab overflow-hidden rounded border ${selected ? 'border-warn' : 'border-black/30'} active:cursor-grabbing`} style={{ left, width, backgroundColor: `${clip.color}20` }}>
  1774→      {/* Fixed-width waveform layer — clipped by the box so trimming cuts the
  1775→          audio rather than squashing the whole wave into a smaller space. */}
  1776→      <div ref={waveRef} className="pointer-events-none absolute top-0 bottom-0 left-0" style={{ width }}>
  1777→        {clip.sampleKey && <WaveformImg clipId={clip.id} sampleKey={clip.sampleKey} channels={clip.channels} sampleRate={clip.sampleRate} offset={clip.offset} duration={clip.duration} speed={clip.speed ?? 1} width={w} color={clip.color} />}
  1778→      </div>

  1786→      {fadeOutW > 0 && (

  1790→      <div onPointerDown={(e) => { if (elRef.current) onTrimStart(clip, 'left', e, elRef.current, waveRef.current); }} className="absolute left-0 top-0 bottom-0 w-1.5 cursor-ew-resize bg-white/40 opacity-0 group-hover:opacity-100" />
  1791→      <div onPointerDown={(e) => { if (elRef.current) onTrimStart(clip, 'right', e, elRef.current, waveRef.current); }} className="absolute right-0 top-0 bottom-0 w-1.5 cursor-ew-resize bg-white/40 opacity-0 group-hover:opacity-100" />
