   190→   389→  // Global pointermove/up — pure DOM, zero React state.
   191→   390→  // rAF-throttled: pointermove fires at the hardware's report rate (often
   192→   391→  // 60-120Hz), and the move handler does a neighbours-overlap scan on every
   193→   392→  // event. Coalescing multiple moves into a single rAF callback caps the work
   194→   393→  // at the display refresh rate (~60fps) so a fast pointer drag never stalls
   195→   394→  // the main thread. A no-op skip (cursor hasn't moved since the last
   196→   395→  // processed frame) avoids redundant DOM writes for stationary pointers.
   197→   396→  useEffect(() => {
   198→   397→    /** Last clientX we actually applied to the DOM. Lets us skip work when
   199→   398→     *  pointermove fires with the same x (e.g. a touchpad reporting pressure
   200→   399→     *  changes without movement, or sub-pixel jitter that rounds to the same
   201→   400→     *  px). */
   202→   401→    let lastProcessedX: number | null = null;
   203→   402→
   204→   403→    const applyMove = () => {
   205→   404→      moveRafRef.current = 0;
   206→   405→      const d = dragRef.current;
   207→   406→      if (!d) return;
   208→   407→      const clientX = moveXRef.current;
   209→   408→      if (clientX === null) return;
   210→   409→      // Skip no-op moves — cursor hasn't moved to a new pixel since the last
   211→   410→      // frame we processed. Saves a neighbours scan + 1-2 style writes.
   212→   411→      if (lastProcessedX === clientX) return;
   213→   412→      lastProcessedX = clientX;
   214→   413→
   215→   414→      const pps = pxRef.current;
   216→   415→      const dt = (clientX - d.sx) / pps;
   217→   416→      const oldEnd = d.os + d.od;
   218→   417→      if (d.mode === 'drag') {
   219→   418→        // Snap-to-neighbour drag: the clip follows the mouse, but if the new
   220→   419→        // position would overlap another clip on the same track, instead of
   221→   420→        // stopping dead at the neighbour's edge we JUMP to the free side of
   222→   421→        // that neighbour (right of it when dragging right, left of it when
   223→   422→        // dragging left). Iterates so a chain of back-to-back clips resolves
   224→   423→        // to the next free slot. A small overlap threshold avoids accidental
   225→   424→        // snaps when the cursor barely crosses the boundary.
   226→   425→        const duration = d.od;
   227→   426→        const rawStart = d.os + dt;
   228→   427→        let start = rawStart;
   229→   428→        const overlapThreshold = 0.05; // s — ignore near-boundary jitter
   230→   429→        for (let iter = 0; iter < d.neighbours.length + 1; iter++) {
   231→   430→          const blocker = d.neighbours.find((n) => {
   232→   431→            const ovStart = Math.max(start, n.start);
   233→   432→            const ovEnd = Math.min(start + duration, n.end);
   234→   433→            return ovEnd - ovStart > overlapThreshold;
   235→   434→          });
   236→   435→          if (!blocker) break;
   237→   436→          if (dt >= 0) start = blocker.end;          // snap to RIGHT of blocker
   238→   437→          else start = blocker.start - duration;     // snap to LEFT of blocker
   239→   438→        }
   240→   439→        // Fallback clamp: never let the clip start before t=0.
   241→   440→        start = Math.max(0, start);
   242→   441→        d.el.style.left = `${start * pps}px`;
   243→   442→      } else if (d.mode === 'trimL') {
   244→   443→        // Cut from the left: never past the left neighbour, the source start
   245→   444→        // (offset ≥ 0 → limited by the trimmed head in timeline seconds), or a
   246→   445→        // minimum width.
   247→   446→        const minStart = Math.max(d.leftLimit, d.os - d.oo / d.speed);
   248→   447→        const start = Math.min(oldEnd - 0.1, Math.max(minStart, d.os + dt));
   249→   448→        d.el.style.left = `${start * pps}px`;
   250→   449→        d.el.style.width = `${(oldEnd - start) * pps}px`;
   251→   450→        // Shift the (fixed-width) waveform the opposite way so the audio stays
   252→   451→        // anchored — the newly-trimmed head is clipped, not squashed.
   253→   452→        if (d.waveEl) d.waveEl.style.left = `${-(start - d.os) * pps}px`;
   254→   453→      } else if (d.mode === 'trimR') {
   255→   454→        // Cut from the right: never past the right neighbour or a min width.
   256→   455→        const end = Math.min(d.rightLimit, Math.max(d.os + 0.1, oldEnd + dt));
   257→   456→        d.el.style.width = `${(end - d.os) * pps}px`;
   258→   457→      }
   259→   458→    };
   260→   459→
   261→   460→    const onMove = (ev: PointerEvent) => {
   262→   461→      const d = dragRef.current; if (!d) return;
   263→   462→      // Stash the latest clientX — the rAF callback reads it. Cheap (one ref
   264→   463→      // write) so even 120Hz pointermove streams don't pile up work.
   265→   464→      moveXRef.current = ev.clientX;
   266→   465→      // Schedule a frame if one isn't already pending. The guard is what
   267→   466→      // coalesces multiple moves within the same frame into one DOM write.
   268→   467→      if (!moveRafRef.current) {
   269→   468→        moveRafRef.current = requestAnimationFrame(applyMove);
   270→   469→      }
   271→   470→    };
   272→   471→    const onUp = () => {
   273→   472→      const d = dragRef.current; if (!d) return;
   274→   473→      // Cancel any pending rAF — we'll process the final position
   275→   474→      // SYNCHRONOUSLY here so the committed Yjs value reflects the exact
   276→   475→      // pointer-up location (otherwise the last pointermove before pointerup
   277→   476→      // could be dropped, leaving the committed position ~1 frame behind the
   278→   477→      // cursor). The `force` flag bypasses the no-op skip in case the final
   279→   478→      // pointermove had the same clientX as the prior processed frame but we
   280→   479→      // still need to read the just-written style to commit to Yjs.
   281→   480→      if (moveRafRef.current) {
   282→   481→        cancelAnimationFrame(moveRafRef.current);
   283→   482→        moveRafRef.current = 0;
   284→   483→      }
   285→   484→      if (moveXRef.current !== null) {
   286→   485→        lastProcessedX = null; // bypass the no-op skip on the final flush
   287→   486→        applyMove();
   288→   487→      }
   289→   488→      const pps = pxRef.current;
   290→   489→      const left = parseFloat(d.el.style.left) / pps;
   291→   490→      const width = parseFloat(d.el.style.width) / pps;
   292→   491→      if (d.mode === 'drag') updateAudioClip(slate, d.clipId, { start: Math.max(0, left) });
   293→   492→      else if (d.mode === 'trimL') updateAudioClip(slate, d.clipId, { start: left, duration: width, offset: Math.max(0, d.oo + (left - d.os) * d.speed) });
   294→   493→      else if (d.mode === 'trimR') updateAudioClip(slate, d.clipId, { duration: width });
   295→   494→      moveXRef.current = null;
   296→   495→      lastProcessedX = null;
   297→   496→      dragRef.current = null;
   298→   497→    };
   299→   498→    window.addEventListener('pointermove', onMove);
   300→   499→    window.addEventListener('pointerup', onUp);

   309→   508→    };
