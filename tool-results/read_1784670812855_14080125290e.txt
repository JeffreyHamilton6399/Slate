    62→/** Nearest start position at which a clip of `duration` fits without
    63→ *  overlapping any blocker. Merges blockers into occupied intervals, then
    64→ *  picks the gap position closest to `desired`. This replaces the previous
    65→ *  "slide to the nearest side of the current blocker" iteration, which could
    66→ *  ping-pong between two adjacent blockers (each resolution landing on the
    67→ *  other) and exhaust its iteration budget while STILL overlapping —
    68→ *  committing a clip on top of another one. Scanning gaps directly is
    69→ *  deterministic: the open-ended gap after the last blocker always fits, so a
    70→ *  valid position always exists. */
    71→export function nearestFreeStart(desired: number, duration: number, blockersIn: { start: number; end: number }[]): number {
    72→  const eps = 1e-4;
    73→  desired = Math.max(0, desired);
    74→  if (blockersIn.length === 0) return desired;
    75→  const sorted = [...blockersIn].sort((a, b) => a.start - b.start);
    76→  const occupied: { start: number; end: number }[] = [];
    77→  for (const b of sorted) {
    78→    const last = occupied[occupied.length - 1];
    79→    if (last && b.start <= last.end + eps) last.end = Math.max(last.end, b.end);
    80→    else occupied.push({ start: b.start, end: b.end });
    81→  }
    82→  let best = desired;
    83→  let bestDist = Infinity;
    84→  const consider = (lo: number, hi: number) => {
    85→    if (hi - lo < duration - eps) return; // gap too small
    86→    const cand = Math.min(Math.max(desired, lo), hi - duration);
    87→    const dist = Math.abs(cand - desired);
    88→    if (dist < bestDist) { bestDist = dist; best = cand; }
    89→  };
    90→  let prevEnd = 0;
    91→  for (const o of occupied) {
    92→    consider(prevEnd, o.start);
    93→    prevEnd = Math.max(prevEnd, o.end);
    94→  }
    95→  consider(prevEnd, Infinity); // open-ended tail — always fits
    96→  return Math.max(0, best);
    97→}

    99→/** Clamp a multi-clip drag delta so the WHOLE group stays overlap-free.
   100→ *  Each selected clip moves by the same `dt`; for each clip we collect the
   101→ *  range of `dt` values that keep it off its non-selected neighbours on the
   102→ *  same track, then intersect those ranges across all clips. If the desired
   103→ *  `dt` falls outside the intersection, we clamp to the nearest bound — so a
   104→ *  group dragged into a wall stops flush against it instead of teleporting
   105→ *  past. The origin track per clip is used (multi-drag is horizontal-only —
   106→ *  no cross-track reshuffle, which would be ambiguous for a group). */
   107→function clampGroupDt(
   108→  dt: number,
   109→  origins: Map<string, { os: number; od: number; trackId: string }>,
   110→  byTrack: Map<string, { start: number; end: number }[]>,
   111→): number {
   112→  let minRight = Infinity; // upper bound on dt (blocker to the right of a clip)
   113→  let maxLeft = -Infinity; // lower bound on dt (blocker to the left of a clip)
   114→  for (const o of origins.values()) {
   115→    const blockers = byTrack.get(o.trackId) ?? [];
   116→    for (const b of blockers) {
   117→      // Blocker entirely to the LEFT of the clip's origin → moving LEFT past
   118→      // it would overlap. Constraint: dt >= b.end - o.os.
   119→      if (b.end <= o.os + 1e-4) {
   120→        maxLeft = Math.max(maxLeft, b.end - o.os);
   121→      }
   122→      // Blocker entirely to the RIGHT of the clip's origin → moving RIGHT past
   123→      // it would overlap. Constraint: dt <= b.start - o.os - o.od.
   124→      else if (b.start >= o.os + o.od - 1e-4) {
   125→        minRight = Math.min(minRight, b.start - o.os - o.od);
   126→      }
   127→      // A blocker that straddles the clip's original interval would mean the
   128→      // clip was ALREADY overlapping before the drag — ignore (shouldn't happen).
   129→    }
   130→  }
   131→  if (maxLeft > minRight) return 0; // conflicting constraints — no movement
   132→  return Math.max(maxLeft, Math.min(minRight, dt));
   133→}

   135→// ── Waveform cache: pre-computed PNG data URLs ───────────────────────────────
   136→// Key: `${clipId}:${sampleCount}:${width}` → data URL
   137→const waveformPNGCache = new Map<string, string>();
   138→
   139→// Draw only the window [startFrame, endFrame) of the sample across `width`
   140→// pixels. Rendering the *window* (not the whole sample stretched) is what makes
   141→// trimming look like cutting the audio away rather than squashing it.
   142→function computeWaveformPNG(samples: Float32Array, channels: number, width: number, color: string, height: number, startFrame: number, endFrame: number): string {
   143→  const canvas = document.createElement('canvas');
   144→  const dpr = window.devicePixelRatio || 1;
   145→  const w = Math.max(2, Math.floor(width));
   146→  canvas.width = w * dpr;
   147→  canvas.height = height * dpr;
   148→  const ctx = canvas.getContext('2d')!;
   149→  ctx.scale(dpr, dpr);
   150→  ctx.clearRect(0, 0, w, height);
   151→  const totalFrames = samples.length / channels;
   152→  const s = Math.max(0, Math.min(totalFrames, Math.floor(startFrame)));
   153→  const e = Math.max(s, Math.min(totalFrames, Math.floor(endFrame)));
   154→  const span = e - s;
   155→  if (span <= 0) return canvas.toDataURL();
   156→  const mid = height / 2;
   157→  ctx.fillStyle = color;
   158→  for (let x = 0; x < w; x++) {
   159→    const f0 = s + Math.floor((x / w) * span);
   160→    const f1 = Math.min(e, s + Math.floor(((x + 1) / w) * span) + 1);
   161→    let peak = 0;
   162→    for (let i = f0; i < f1; i++) { const v = Math.abs(samples[i * channels] ?? 0); if (v > peak) peak = v; }
   163→    const bh = Math.max(1, peak * mid * 0.85);
   164→    ctx.fillRect(x, mid - bh, 1, bh * 2);
   165→  }
   166→  return canvas.toDataURL();
   167→}

   169→/** Waveform image for the buffer window the clip actually plays. With speed s,
   170→ *  a clip of `duration` timeline seconds consumes `duration*s` buffer seconds.
   171→ *
   172→ *  Two robustness fixes:
   173→ *  1. If `loadSamples` returns an EMPTY Float32Array (length 0) — which happens
   174→ *     when a freshly-split/created clip's IndexedDB write hasn't landed yet —
   175→ *     we DON'T cache the resulting blank PNG. Instead we show the `···`
   176→ *     placeholder and retry a few times (500ms apart) so the real waveform
   177→ *     appears once the samples are available.
   178→ *  2. We listen for `slate:audio-clip-changed` (fired by Normalize/Reverse/split)
   179→ *     and invalidate our cache entry for the current clipId, then force a
   180→ *     recompute via a `bust` counter — otherwise the memoised component would
   181→ *     keep showing the stale PNG even after the cache was cleared. */
   182→const WaveformImg = memo(function WaveformImg({ clipId, sampleKey, channels, sampleRate, offset, duration, speed, width, color }: {
   183→  clipId: string; sampleKey: string; channels: number; sampleRate: number;
   184→  offset: number; duration: number; speed: number; width: number; color: string;
   185→}) {
   186→  const height = TRACK_H - 6;
   187→  const [imgUrl, setImgUrl] = useState<string>('');
   188→  const [bust, setBust] = useState(0);
   189→  const retryRef = useRef(0);
   190→
   191→  // Invalidate cached PNGs for this clip when its samples change (normalize /
   192→  // reverse / split). The parent AudioEditor also invalidates + bumps version,
   193→  // but the memoised WaveformImg wouldn't otherwise recompute (its primitive
   194→  // props are unchanged) — the `bust` counter forces the load effect to re-run.
   195→  useEffect(() => {
   196→    const onChanged = (e: Event) => {
   197→      const detail = (e as CustomEvent<string>).detail;
   198→      // Match by clipId (normalize/reverse/split events) OR by sampleKey
   199→      // (multiplayer sample-arrival events from registerSampleSyncMap).
   200→      if (detail !== clipId && detail !== sampleKey) return;
   201→      for (const key of waveformPNGCache.keys()) {
   202→        if (key.startsWith(`${clipId}:`)) waveformPNGCache.delete(key);
   203→      }
   204→      retryRef.current = 0;
   205→      setBust((n) => n + 1);
   206→    };
   207→    window.addEventListener('slate:audio-clip-changed', onChanged as EventListener);
   208→    return () => window.removeEventListener('slate:audio-clip-changed', onChanged as EventListener);
   209→  }, [clipId, sampleKey]);
   210→
   211→  useEffect(() => {
   212→    const startFrame = Math.round(offset * sampleRate);
   213→    const endFrame = Math.round((offset + duration * speed) * sampleRate);
   214→    const cacheKey = `${clipId}:${sampleKey}:${startFrame}:${endFrame}:${Math.floor(width)}`;
   215→    const cached = waveformPNGCache.get(cacheKey);
   216→    if (cached) { setImgUrl(cached); return; }
   217→    let cancelled = false;
   218→    let retryTimer: ReturnType<typeof setTimeout> | null = null;
   219→
   220→    const attempt = () => {
   221→      if (cancelled) return;
   222→      void loadSamples(sampleKey).then((samples) => {
   223→        if (cancelled) return;
   224→        // Empty samples = the IndexedDB write for this sampleKey hasn't landed
   225→        // yet (e.g. brand-new clip from a split, OR a remote peer's clip whose
   226→        // sample blob is still in flight via the Yjs sync map). DON'T cache a
   227→        // blank PNG — retry for up to 10s (20 × 500ms) so the real waveform
   228→        // appears once the samples arrive. The previous 5-retry / 2.5s budget
   229→        // was too short for cross-peer sample sync, which can take several
   230→        // seconds for a 5MB blob over a slow link. The `slate:audio-clip-changed`
   231→        // event (dispatched by onClipsAdded when a remote clip is added, and by
   232→        // registerSampleSyncMap when samples arrive) resets `retryRef` to 0 so
   233→        // the WaveformImg re-tries immediately on the signal — this retry
   234→        // budget is the safety net for the case where the event fires before
   235→        // the WaveformImg mounts, or doesn't fire at all.
   236→        if (samples.length === 0) {
   237→          if (retryRef.current < 20) {
   238→            retryRef.current += 1;
   239→            retryTimer = setTimeout(attempt, 500);
   240→          }
   241→          return;
   242→        }
   243→        retryRef.current = 0;
   244→        const url = computeWaveformPNG(samples, channels, width, color, height, startFrame, endFrame);
   245→        waveformPNGCache.set(cacheKey, url);
   246→        setImgUrl(url);
   247→      });
   248→    };
   249→    attempt();
   250→
   251→    return () => {
   252→      cancelled = true;
   253→      if (retryTimer) clearTimeout(retryTimer);
   254→    };
   255→  }, [clipId, sampleKey, channels, sampleRate, offset, duration, speed, width, color, height, bust]);
   256→
   257→  if (!imgUrl) return <div className="flex h-full items-center justify-center text-[7px] text-text-dim">···</div>;
   258→  return <img src={imgUrl} alt="" className="pointer-events-none h-full w-full" style={{ objectFit: 'fill' }} />;
   259→});
   260→
   261→/** Drop cached waveform PNGs for a clip (call when its samples change). */
   262→function invalidateWaveform(clipId: string): void {
   263→  for (const key of waveformPNGCache.keys()) if (key.startsWith(`${clipId}:`)) waveformPNGCache.delete(key);
   264→}

   442→    // peer) re-applies the Yjs values to the live audio graph, so the settings
   443→    // panel is audible mid-playback without a restart.

   445→    // REMOTE MUTE/SOLO FIX: this subscription is what makes "user A mutes a
   446→    // track → user B hears the mute" work. The mute state IS in Yjs (it's a
   447→    // field on the track Y.Map), so it syncs to every peer. But sync alone

   449→    // nodes still hold the old value until something re-applies them.

   471→    // a clip's volume, fighting the user). The restart is debounced 500ms so
   472→    // a burst of additions (e.g. a peer imports 5 files) coalesces into one.
   473→    //
   474→    // LIVE WAVEFORM REFRESH: when a remote peer adds a clip, its metadata
   475→    // arrives via Yjs immediately but the sample blob travels separately via
   476→    // the audioSampleSync Y.Map (see sampleStore.ts). If we don't ping the
   477→    // WaveformImg, it might have already exhausted its empty-samples retry
   478→    // budget (now 10s, but a slow link can still exceed that) and given up
   479→    // showing a blank "···" placeholder until the user refreshes. So for
   480→    // each newly-added clip we:
   481→    //   1. clearCache(clipId) on the engine so the next play re-loads samples
   482→    //      (the cache might hold a null/empty entry from a prior failed getBuffer)
   483→    //   2. dispatch `slate:audio-clip-changed` with the clipId — the
   484→    //      AudioEditor's listener (below) re-invalidates the waveform cache +
   485→    //      clears the engine cache + bumps version, AND the WaveformImg's own
   486→    //      listener resets its retryRef to 0 and bumps `bust` to immediately
   487→    //      re-attempt loadSamples (which will now find the synced blob once
   488→    //      it lands).
   489→    //   3. scheduleRestart() so if we're currently playing, the new clip is
   490→    //      picked up atomically (debounced 500ms to coalesce bursts).

   492→      const addedIds: string[] = [];

   494→        const change = event.changes.keys.get(key);
   495→        if (change?.action === 'add') addedIds.push(key);
   496→      }

   498→      for (const id of addedIds) {
   499→        engineRef.current?.clearCache(id);
   500→        window.dispatchEvent(new CustomEvent('slate:audio-clip-changed', { detail: id }));
   501→      }
   502→      scheduleRestart();
   503→    };

   506→    return () => { clearTimeout(lateRead); tracks.unobserveDeep(bump); tracks.unobserveDeep(applyTracks); clips.unobserveDeep(bump); clips.unobserveDeep(restartOnClipEdit); audioMap.unobserve(bump); clips.unobserve(onClipsAdded); };

   521→    return () => {

   523→      undoRef.current = null;

   571→  // Ctrl+scroll zoom — centred on the playhead so the screen position under
   572→  // the cursor/playhead stays put instead of zooming toward the left edge.
   573→  /** Zoom while keeping one timeline point at the same screen offset.
   574→   *  The anchor is the time under `anchorClientX` (mouse-wheel zoom: the spot
   575→   *  under the cursor stays put — how every DAW/map zoom feels), or the
   576→   *  playhead when no anchor is given (toolbar buttons, fit-to-window).
   577→   *  Keeping time t fixed on screen means newScroll = oldScroll + t·Δpps,
   578→   *  since screenX(t) = timelineLeft − scrollLeft + t·pps. The `scrollLeft`
   579→   *  write happens in the layout effect below (after React commits the new
   580→   *  `minWidth` on the timeline div — setting it earlier would be clamped by
   581→   *  the stale `scrollWidth`). */
   582→  const zoomAnchored = useCallback((newPxPerSec: number, anchorClientX?: number) => {
   583→    const el = scrollRef.current;
   584→    const oldPxPerSec = pxRef.current;
   585→    if (!el || oldPxPerSec === newPxPerSec) {
   586→      setPxPerSec(newPxPerSec);
   587→      return;
   588→    }

   592→      if (tl) t = Math.max(0, (anchorClientX - tl.getBoundingClientRect().left) / oldPxPerSec);
   593→    }
   594→    pendingScrollRef.current = Math.max(0, el.scrollLeft + t * (newPxPerSec - oldPxPerSec));
   595→    setPxPerSec(newPxPerSec);
   596→  }, []);

   602→      pendingScrollRef.current = null;
   603→    }
   604→  }, [pxPerSec]);

   606→  useEffect(() => {

   613→    };

   615→    return () => el.removeEventListener('wheel', onWheel);
   616→  }, [zoomAnchored]);
   617→
   618→  // Global pointermove/up — pure DOM, zero React state.
   619→  // rAF-throttled: pointermove fires at the hardware's report rate (often
   620→  // 60-120Hz), and the move handler does a neighbours-overlap scan on every
   621→  // event. Coalescing multiple moves into a single rAF callback caps the work
   622→  // at the display refresh rate (~60fps) so a fast pointer drag never stalls
   623→  // the main thread. A no-op skip (cursor hasn't moved since the last
   624→  // processed frame) avoids redundant DOM writes for stationary pointers.
   625→  useEffect(() => {
   626→    /** Last pointer position we actually applied to the DOM. Lets us skip
   627→     *  work when pointermove fires without real movement (e.g. a touchpad
   628→     *  reporting pressure changes, or sub-pixel jitter). */
   629→    let lastProcessedX: number | null = null;
   630→    let lastProcessedY: number | null = null;
   631→
   632→    /** Magnet: pull `t` (a clip START time) to the nearest beat line, other
   633→     *  clip edge, or the playhead — considering BOTH the clip's start and end
   634→     *  edges — if one is within SNAP_PX at the current zoom. `dur` null =
   635→     *  only the start edge matters (trims). Returns the adjusted start. */
   636→    const snapStart = (t: number, dur: number | null, snapTimes: number[]): number => {
   637→      const pps = pxRef.current;
   638→      const thr = SNAP_PX / pps;
   639→      const beat = 60 / bpmRef.current;
   640→      let best = t;
   641→      let bestDist = thr;
   642→      const consider = (target: number, edgeOffset: number) => {
   643→        const dist = Math.abs(t + edgeOffset - target);
   644→        if (dist < bestDist) { bestDist = dist; best = target - edgeOffset; }
   645→      };
   646→      consider(Math.round(t / beat) * beat, 0);
   647→      if (dur !== null) consider(Math.round((t + dur) / beat) * beat, dur);
   648→      for (const st of snapTimes) {
   649→        consider(st, 0);
   650→        if (dur !== null) consider(st, dur);
   651→      }
   652→      return best;
   653→    };
   654→
   655→    const applyMove = () => {
   656→      moveRafRef.current = 0;
   657→      const d = dragRef.current;
   658→      if (!d) return;
   659→      const clientX = moveXRef.current;
   660→      if (clientX === null) return;
   661→      const clientY = moveYRef.current;
   662→      // Skip no-op moves — cursor hasn't moved to a new pixel since the last
   663→      // frame we processed. Saves an overlap scan + 1-2 style writes.
   664→      if (lastProcessedX === clientX && lastProcessedY === clientY) return;
   665→      lastProcessedX = clientX;
   666→      lastProcessedY = clientY;
   667→
   668→      // Dead zone: a plain click (pointer never leaves a 3px box) must not
   669→      // nudge the clip. Once exceeded, the drag is live for good.
   670→      if (!d.moved) {
   671→        if (Math.abs(clientX - d.sx) < DRAG_DEADZONE_PX && Math.abs(clientY - d.sy) < DRAG_DEADZONE_PX) return;
   672→        d.moved = true;
   673→        if (d.mode === 'drag') {
   674→          d.el.style.zIndex = '40'; // ride above other rows while crossing tracks
   675→          // Elevate the whole group too so the user sees the multi-clip
   676→          // selection ride above non-selected neighbours while in transit.
   677→          if (d.origins) for (const o of d.origins.values()) if (o.el !== d.el) o.el.style.zIndex = '40';
   678→        }
   679→      }
   680→
   681→      const pps = pxRef.current;
   682→      const dt = (clientX - d.sx) / pps;
   683→      const oldEnd = d.os + d.od;
   684→      const wantSnap = snapRef.current && !moveAltRef.current;
   685→      if (d.mode === 'drag') {
   686→        // MULTI-SELECT DRAG: when `origins` is populated, every selected clip
   687→        // moves by the same horizontal delta. Vertical track-switching is
   688→        // disabled for the group (each clip stays on its origin track) since
   689→        // reshuffling every selected clip onto one target track would be
   690→        // ambiguous and likely overlap-prone. The dragged clip's snap result
   691→        // is applied to the whole group so the magnet still feels right.
   692→        if (d.origins && d.origins.size > 0) {
   693→          const dragged = d.origins.get(d.clipId);
   694→          let snappedDt = dt;
   695→          if (wantSnap && dragged) {
   696→            const snappedStart = snapStart(dragged.os + dt, dragged.od, d.snapTimes);
   697→            snappedDt = snappedStart - dragged.os;
   698→          }
   699→          for (const o of d.origins.values()) {
   700→            const newStart = Math.max(0, o.os + snappedDt);
   701→            o.el.style.left = `${newStart * pps}px`;
   702→          }
   703→          return;
   704→        }
   705→        // The clip follows the mouse 1:1 (snapped to the magnet grid). It's
   706→        // allowed to ride OVER other clips while the drag is live — it's
   707→        // elevated to z-40 so the overlap reads as "in transit", and the
   708→        // no-overlap invariant is enforced once, on drop (see onUp). The
   709→        // previous mid-drag resolution parked the clip flush against a
   710→        // blocker and then teleported it across when the cursor passed the
   711→        // blocker's midpoint, which made dragging feel sticky and jumpy.
   712→        const duration = d.od;
   713→        // Vertical: whole-row steps move the clip across tracks.
   714→        const rowDelta = Math.max(
   715→          -d.trackIndex,
   716→          Math.min(d.trackIds.length - 1 - d.trackIndex, Math.round((clientY - d.sy) / TRACK_H)),
   717→        );
   718→        d.rowDelta = rowDelta;
   719→        d.el.style.transform = rowDelta === 0 ? '' : `translateY(${rowDelta * TRACK_H}px)`;
   720→
   721→        const desired = wantSnap ? snapStart(d.os + dt, duration, d.snapTimes) : d.os + dt;
   722→        d.el.style.left = `${Math.max(0, desired) * pps}px`;
   723→      } else if (d.mode === 'trimL') {
   724→        // Cut from the left: never past the left neighbour, the source start
   725→        // (offset ≥ 0 → limited by the trimmed head in timeline seconds), or a
   726→        // minimum width.
   727→        const minStart = Math.max(d.leftLimit, d.os - d.oo / d.speed);
   728→        const desired = wantSnap ? snapStart(d.os + dt, null, d.snapTimes) : d.os + dt;
   729→        const start = Math.min(oldEnd - 0.1, Math.max(minStart, desired));
   730→        d.el.style.left = `${start * pps}px`;
   731→        d.el.style.width = `${(oldEnd - start) * pps}px`;
   732→        // Shift the (fixed-width) waveform the opposite way so the audio stays
   733→        // anchored — the newly-trimmed head is clipped, not squashed.
   734→        if (d.waveEl) d.waveEl.style.left = `${-(start - d.os) * pps}px`;
   735→      } else if (d.mode === 'trimR') {
   736→        // Cut from the right: never past the right neighbour or a min width.
   737→        const desired = wantSnap ? snapStart(oldEnd + dt, null, d.snapTimes) : oldEnd + dt;
   738→        const end = Math.min(d.rightLimit, Math.max(d.os + 0.1, desired));
   739→        d.el.style.width = `${(end - d.os) * pps}px`;
   740→      }
   741→    };
   742→
   743→    const onMove = (ev: PointerEvent) => {
   744→      const d = dragRef.current; if (!d) return;
   745→      // Stash the latest pointer state — the rAF callback reads it. Cheap
   746→      // (ref writes) so even 120Hz pointermove streams don't pile up work.
   747→      moveXRef.current = ev.clientX;
   748→      moveYRef.current = ev.clientY;
   749→      moveAltRef.current = ev.altKey;
   750→      // Schedule a frame if one isn't already pending. The guard is what
   751→      // coalesces multiple moves within the same frame into one DOM write.
   752→      if (!moveRafRef.current) {
   753→        moveRafRef.current = requestAnimationFrame(applyMove);
   754→      }
   755→    };
   756→    const onUp = () => {
   757→      const d = dragRef.current; if (!d) return;
   758→      // Cancel any pending rAF — we'll process the final position
   759→      // SYNCHRONOUSLY here so the committed Yjs value reflects the exact
   760→      // pointer-up location (otherwise the last pointermove before pointerup
   761→      // could be dropped, leaving the committed position ~1 frame behind the
   762→      // cursor). lastProcessed is reset to bypass the no-op skip in case the
   763→      // final pointermove matched the prior processed frame but we still need
   764→      // to read the just-written style to commit to Yjs.
   765→      if (moveRafRef.current) {
   766→        cancelAnimationFrame(moveRafRef.current);
   767→        moveRafRef.current = 0;
   768→      }
   769→      if (moveXRef.current !== null) {
   770→        lastProcessedX = null; // bypass the no-op skip on the final flush
   771→        lastProcessedY = null;
   772→        applyMove();
   773→      }
   774→      // Undo the drag-only visual state React doesn't manage. For a multi-clip
   775→      // drag, also clear the elevated z-index on every selected clip's element.
   776→      if (d.origins) for (const o of d.origins.values()) { o.el.style.zIndex = ''; o.el.style.transform = ''; }
   777→      d.el.style.zIndex = '';
   778→      d.el.style.transform = '';
   779→      // A click that never left the dead zone selects but must not write —
   780→      // committing a parseFloat round-trip of the style would drift the clip.
   781→      if (d.moved) {
   782→        const pps = pxRef.current;
   783→        const left = parseFloat(d.el.style.left) / pps;
   784→        const width = parseFloat(d.el.style.width) / pps;
   785→        if (d.mode === 'drag') {
   786→          // MULTI-SELECT COMMIT: every selected clip moves by the same delta,
   787→          // clamped so the group stays overlap-free against non-selected
   788→          // neighbours on each clip's origin track. The delta is recomputed
   789→          // from the dragged clip's pointer-up position (so the snapped /
   790→          // clamped value the user sees is what gets committed), then applied
   791→          // to every clip in the group via updateAudioClip inside a single
   792→          // Yjs transaction.
   793→          if (d.origins && d.origins.size > 0) {
   794→            const dragged = d.origins.get(d.clipId);
   795→            const dt = dragged ? left - dragged.os : 0;
   796→            const clampedDt = clampGroupDt(dt, d.origins, d.byTrack);
   797→            // Write the resolved positions to the DOM too so there's no
   798→            // flicker between pointer-up and the Yjs-driven re-render.
   799→            for (const o of d.origins.values()) {
   800→              const newStart = Math.max(0, o.os + clampedDt);
   801→              o.el.style.left = `${newStart * pps}px`;
   802→            }
   803→            // Single transaction so the whole group commits atomically to the
   804→            // Yjs doc — peers see one update, not N interleaved with their
   805→            // own edits.
   806→            slate.doc.transact(() => {
   807→              for (const [cid, o] of d.origins!) {
   808→                updateAudioClip(slate, cid, { start: Math.max(0, o.os + clampedDt) });
   809→              }
   810→            });
   811→          } else {
   812→            const targetTrackId = d.trackIds[d.trackIndex + d.rowDelta] ?? d.trackIds[d.trackIndex]!;
   813→            const originTrackId = d.trackIds[d.trackIndex]!;
   814→            // Enforce the no-overlap invariant HERE (the drag itself follows
   815→            // the mouse freely): place the clip in the free gap nearest to the
   816→            // dropped position.
   817→            const start = nearestFreeStart(left, d.od, d.byTrack.get(targetTrackId) ?? []);
   818→            // Write the resolved position to the DOM too so there's no flicker
   819→            // between pointer-up and the Yjs-driven re-render.
   820→            d.el.style.left = `${start * pps}px`;
   821→            updateAudioClip(slate, d.clipId, {
   822→              start,
   823→              ...(targetTrackId !== originTrackId ? { trackId: targetTrackId } : {}),
   824→            });
   825→          }
   826→        }
   827→        else if (d.mode === 'trimL') updateAudioClip(slate, d.clipId, { start: left, duration: width, offset: Math.max(0, d.oo + (left - d.os) * d.speed) });
   828→        else if (d.mode === 'trimR') updateAudioClip(slate, d.clipId, { duration: width });
   829→      }
   830→      moveXRef.current = null;
   831→      lastProcessedX = null;
   832→      lastProcessedY = null;
   833→      dragRef.current = null;
   834→    };
   835→    window.addEventListener('pointermove', onMove);
   836→    window.addEventListener('pointerup', onUp);
   837→    return () => {
   838→      window.removeEventListener('pointermove', onMove);
   839→      window.removeEventListener('pointerup', onUp);
   840→      // Don't leak a pending rAF if the component unmounts mid-drag.
   841→      if (moveRafRef.current) {
   842→        cancelAnimationFrame(moveRafRef.current);
   843→        moveRafRef.current = 0;
   844→      }
   845→    };

   860→      // on (D/F/G/… play notes, Z/X shift octave) — Space/arrows/Delete still
   861→      // reach the transport, like a real DAW. Modifier chords (Ctrl+Z undo,
   862→      // Ctrl+C copy) pass through: the instrument ignores them too.

   866→      else if ((k === 'z' && (e.ctrlKey || e.metaKey) && e.shiftKey) || (k === 'y' && (e.ctrlKey || e.metaKey))) { e.preventDefault(); undoRef.current?.redo(); }

   868→      else if (k === 'v' && (e.ctrlKey || e.metaKey)) { e.preventDefault(); void pasteClips(); }

   873→      else if (k === 'm') { e.preventDefault(); setMetronome((n) => { engineRef.current?.setMetronome(!n); return !n; }); }

   908→  // When a clip's samples change (normalize/reverse from the settings panel,
   909→  // OR a remote peer's sample blob just landed via the Yjs sync map),
   910→  // drop the cached waveform + decoded buffer so both refresh, AND — if we're
   911→  // currently playing — schedule a debounced restartPlayback so the clip is
   912→  // picked up. This covers the case where a clip's metadata arrived but its
   913→  // samples were still in flight: getBuffer returned null on the first pass,
   914→  // and this restart re-schedules once the samples are usable.
   915→  useEffect(() => {
   916→    const onChanged = (e: Event) => {
   917→      const detail = (e as CustomEvent<string>).detail;
   918→      // The event's detail is a CLIP id for local edits, but a SAMPLE KEY for
   919→      // sample-sync arrivals (registerSampleSyncMap only knows the key).
   920→      // The engine's buffer cache and the waveform PNG cache are keyed by
   921→      // clip id — clearing them with a sampleKey was a silent no-op, so a
   922→      // peer whose clip was normalized/reversed/split by someone else got a
   923→      // fresh waveform but kept PLAYING the stale audio. Resolve the detail
   924→      // to every matching clip id (usually exactly one).
   925→      const clipIds: string[] = [];
   926→      const clipsMap = slateRef.current.audioClips();
   927→      if (clipsMap.get(detail)) clipIds.push(detail);
   928→      else clipsMap.forEach((m, cid) => { if (m.get('sampleKey') === detail) clipIds.push(cid); });
   929→      if (clipIds.length === 0) clipIds.push(detail); // unknown — invalidate as-is
   930→      for (const id of clipIds) {
   931→        invalidateWaveform(id);
   932→        engineRef.current?.clearCache(id);
   933→        // Pre-warm the engine's AudioBuffer cache even when not playing —
   934→        // otherwise a remote peer who's paused when samples land goes through
   935→        // the full getBuffer retry loop on their next play(), adding audible
   936→        // latency to the first playback. scheduleRestart() is a no-op when
   937→        // paused, so without this the cache stays cold until play. Safe to
   938→        // call when the engine has no AudioContext yet (preloadBuffer no-ops).
   939→        void engineRef.current?.preloadBuffer(slateRef.current, id);
   940→      }
   941→      // Re-announce under the resolved clip id when the event carried a
   942→      // sampleKey — the engine's in-flight sample-retry loops key their
   943→      // budget reset on clip ids. Terminates: the re-dispatched event's
   944→      // detail IS a clip id, so this branch won't fire again.
   945→      for (const id of clipIds) {
   946→        if (id !== detail) window.dispatchEvent(new CustomEvent('slate:audio-clip-changed', { detail: id }));
   947→      }
   948→      setVersion((v) => v + 1);
   949→      scheduleRestart();

   955→      toast({
   956→        title: 'Clip won’t sync to collaborators',
   957→        description: 'It plays for you, but this clip is too long or the board’s shared audio space is full. Collaborators won’t hear it.',
   958→        variant: 'error',
   959→      });
   960→    };
   961→    window.addEventListener('slate:audio-clip-changed', onChanged as EventListener);
   962→    window.addEventListener('slate:audio-sync-skipped', onSkipped);
   963→    return () => {
   964→      window.removeEventListener('slate:audio-clip-changed', onChanged as EventListener);
   965→      window.removeEventListener('slate:audio-sync-skipped', onSkipped);
   966→    };

   968→  }, []);

  1268→      if (!list) { list = []; byTrack.set(o.trackId, list); }

  1540→              setMarquee({ x1: s.startX, y1: s.startY, x2: ex, y2: ey });
  1541→              // Hit-test clips against the marquee rect
  1542→              const x1 = Math.min(s.startX, ex);
  1543→              const x2 = Math.max(s.startX, ex);
  1544→              const y1 = Math.min(s.startY, ey);
  1545→              const y2 = Math.max(s.startY, ey);
  1546→              const t1 = x1 / pxRef.current, t2 = x2 / pxRef.current;
  1547→              const startRowIdx = Math.floor(y1 / TRACK_H);
  1548→              const endRowIdx = Math.floor(y2 / TRACK_H);
  1549→              const next = new Set(s.origin);
  1550→              clips.forEach((c) => {
  1551→                const ti = tracks.findIndex((t) => t.id === c.trackId);
  1552→                if (ti < startRowIdx || ti > endRowIdx) return;
  1553→                const cEnd = c.start + c.duration;
  1554→                if (cEnd >= t1 && c.start <= t2) next.add(c.id);
  1555→              });
  1556→              setSelectedClipIds(next);
  1557→            }}
  1558→            onPointerUp={() => {
  1559→              const s = marqueeRef.current;
  1560→              marqueeRef.current = null;
  1561→              setMarquee(null);
  1562→              if (!s) return;
  1563→              if (!s.moved) {
  1564→                // Plain click → seek to the pointerdown position. Non-additive
  1565→                // clicks also clear the selection (clicking empty space
  1566→                // dismisses a multi-select); additive clicks (shift/cmd on
  1567→                // empty space) leave the selection alone.
  1568→                seek(s.seekTime);
  1569→                if (!s.additive) setSelectedClipIds(new Set());
  1570→              }
  1571→              // If `s.moved` is true the marquee selection was already
  1572→              // finalised incrementally in pointermove — nothing to do here.
  1573→            }}
  1574→          />
  1575→          {/* Marquee rect overlay */}
  1576→          {marquee && (
  1577→            <div className="pointer-events-none absolute z-30 border border-accent/70 bg-accent/15" style={{
  1578→              left: Math.min(marquee.x1, marquee.x2),
  1579→              top: 28 + Math.min(marquee.y1, marquee.y2),
  1580→              width: Math.abs(marquee.x2 - marquee.x1),
  1581→              height: Math.abs(marquee.y2 - marquee.y1),
  1582→            }} />
  1583→          )}
  1584→          {/* Grid background */}
  1585→          <div className="pointer-events-none absolute inset-0 top-7" style={gridStyle}>
  1586→            {looping && <div onPointerDown={(e) => startLoopDrag('move', e)} className="pointer-events-auto absolute top-0 bottom-0 cursor-grab bg-accent/8 border-x-2 border-accent/40" style={{ left: loopStart * pxPerSec, width: (loopEnd - loopStart) * pxPerSec }} />}
  1587→          </div>
  1588→          {/* Playhead */}
  1589→          <div ref={playheadRef} className="absolute top-0 bottom-0 z-20 w-0.5 bg-warn pointer-events-none" style={{ transform: 'translateX(0px)' }}>
  1590→            <div className="absolute top-0 -left-1 h-2 w-2 rounded-full bg-warn" />
  1591→          </div>
  1592→          {/* Clips */}
  1593→          {tracks.map((t) => (
  1594→            <div key={t.id} data-track-id={t.id} className="pointer-events-none relative border-b border-border/15" style={{ height: TRACK_H }}>
  1595→              {clips.filter((c) => c.trackId === t.id).map((c) => (
  1596→                <ClipBlock key={c.id} clip={c} pxPerSec={pxPerSec} selected={selectedClipIds.has(c.id)}
  1597→                  onDragStart={startDrag} onTrimStart={startTrim} />
  1598→              ))}
  1599→            </div>
  1600→          ))}
  1601→          {/* Remote playheads — overlay on top of clips, mirroring each peer's
  1602→           *  audio transport position. Rendered after clips so z-20 wins. */}
  1603→          <RemotePlayheads room={room} peerStates={peerStates} pxRef={pxRef} selfId={room.identity.peerId} />
  1604→        </div>
  1605→      </div>
  1606→
  1607→      {/* Status */}
  1608→      <div className="flex shrink-0 items-center gap-2 border-t border-border px-2 py-0.5 text-[8px] font-mono uppercase text-text-dim">
  1609→        <span>{tracks.length}T · {clips.length}C</span>
  1610→        {recording && <span className="text-danger">● Rec</span>}
  1611→        {playing && <span className="text-accent">▶ Play</span>}
  1612→        <span className="ml-auto">Space · Ctrl+Z=Undo · C=Split · D=Dup · Ctrl+C/V=Copy/Paste · Del · R=Rec · L=Loop · M=Met · ←→=Seek · Ctrl+Scroll=Zoom</span>
  1613→      </div>
  1614→    </div>
  1615→  );
  1616→}

  1620→const TrackHeader = memo(function TrackHeader({ track, hasSolo, slate, engineRef }: {
  1621→  track: AudioTrack; hasSolo: boolean;
  1622→  slate: ReturnType<typeof useRoom>['slate'];
  1623→  engineRef: React.RefObject<AudioEngine | null>;
  1624→}) {
  1625→  const [vol, setVol] = useState(track.volume);
  1626→  const [pan, setPan] = useState(track.pan);
  1627→  // Separate drag flags per slider — the previous single shared `isDraggingRef`
  1628→  // meant starting a volume drag also blocked the pan prop-sync effect (and
  1629→  // vice versa), which could cause one slider's committed Yjs value to be
  1630→  // momentarily ignored by the other's gating. Per-slider flags keep each
  1631→  // slider's prop-sync independent.
  1632→  const isDraggingVolRef = useRef(false);
  1633→  const isDraggingPanRef = useRef(false);
  1634→  useEffect(() => { if (!isDraggingVolRef.current) setVol(track.volume); }, [track.volume]);
  1635→  useEffect(() => { if (!isDraggingPanRef.current) setPan(track.pan); }, [track.pan]);
  1636→
  1637→  // Audibility — mirrors engine.setupTrackNodes so the live gain we write
  1638→  // during a volume drag respects mute/solo (don't briefly un-mute a muted
  1639→  // track just because the user is dragging its volume slider).
  1640→  const audible = hasSolo ? track.solo : !track.muted;
  1641→
  1642→  // `update` is for non-slider track edits (name/mute/solo/arm) — these DO
  1643→  // need a full updateTracks so the audio graph reflects the new state
  1644→  // (e.g. toggling solo rebalances every track's gain).
  1645→  const update = (patch: Partial<AudioTrack>) => { updateAudioTrack(slate, track.id, patch); engineRef.current?.updateTracks(slate); };
  1646→
  1647→  // Volume slider: update local state + the engine's gain node DIRECTLY for
  1648→  // immediate audio feedback. Do NOT call updateTracks (which re-reads every
  1649→  // track from Yjs and rebuilds the audio graph — far too expensive per
  1650→  // onChange event, and the source of the reported lag). Commit to Yjs once
  1651→  // on pointerup.
  1652→  const onVolDown = () => { isDraggingVolRef.current = true; };
  1653→  const onVol = (v: number) => { setVol(v); engineRef.current?.setTrackVolume(track.id, v, audible); };
  1654→  const onVolEnd = () => { isDraggingVolRef.current = false; updateAudioTrack(slate, track.id, { volume: vol }); };
  1655→
  1656→  // Pan slider: same pattern — set the panner node directly, commit on
  1657→  // pointerup. The pan slider is smaller than the volume slider (secondary
  1658→  // control) but still usable, with L/R labels so its function is obvious
  1659→  // (the user complaint was "two sliders, only one works" — partly because
  1660→  // neither slider was labelled, so it wasn't clear what each did).
  1661→  const onPanDown = () => { isDraggingPanRef.current = true; };
  1662→  const onPan = (p: number) => { setPan(p); engineRef.current?.setTrackPan(track.id, p); };
  1663→  const onPanEnd = () => { isDraggingPanRef.current = false; updateAudioTrack(slate, track.id, { pan: pan }); };
  1664→
  1665→  // ── Track kind: Audio ↔ MIDI ─────────────────────────────────────────────
  1666→  // A single icon button toggles the kind. On a MIDI track, an instrument
  1667→  // picker (compact <select>) sits next to the name so the user can choose
  1668→  // which synth preset or soundfont the track's MIDI clips play through.
  1669→  // Toggling kind to 'midi' defaults the instrument to the soundfont piano
  1670→  // (a sensible starting point for a fresh MIDI track); toggling back to
  1671→  // 'audio' leaves the instrumentId in place (harmless — it's only consulted
  1672→  // for MIDI clips).
  1673→  const isMidi = track.kind === 'midi';
  1674→  const toggleKind = () => {
  1675→    const nextKind: AudioTrack['kind'] = isMidi ? 'audio' : 'midi';
  1676→    // Switching to MIDI defaults the instrument to the soundfont piano if the
  1677→    // track doesn't already have one set; switching to audio clears the input
  1678→    // back to 'none' (a MIDI-armed audio track makes no sense).
  1679→    const patch: Partial<AudioTrack> = { kind: nextKind };
  1680→    if (nextKind === 'midi') {
  1681→      if (!track.instrumentId) patch.instrumentId = SOUNDFONT_PIANO_ID;
  1682→      // If the track was armed for mic, drop the arm — MIDI tracks arm for
  1683→      // instrument-take recording, not mic input.
  1684→      if (track.armed) { patch.armed = false; patch.input = 'none'; }
  1685→    } else if (track.input === 'midi') {
  1686→      patch.input = 'none';
  1687→      patch.armed = false;
  1688→    }
  1689→    update(patch);
  1690→  };
  1691→
  1692→  // Arm button: on an audio track → 'mic' input; on a MIDI track → 'midi'
  1693→  // input. The arm state still toggles; only the input source differs.
  1694→  const toggleArm = () => {
  1695→    if (track.armed) update({ armed: false, input: 'none' });
  1696→    else update({ armed: true, input: isMidi ? 'midi' : 'mic' });
  1697→  };
  1698→
  1699→  return (
  1700→    <div className="border-b border-border/15 px-2 py-1" style={{ height: TRACK_H }}>
  1701→      <div className="flex items-center gap-0.5">
  1702→        <span className="h-2 w-2 shrink-0 rounded-full" style={{ backgroundColor: track.color }} />
  1703→        <input type="text" value={track.name} onChange={(e) => update({ name: e.target.value })} className="min-w-0 flex-1 bg-transparent text-[11px] font-medium text-text outline-none" />
  1704→        {/* Kind toggle: Audio (Volume2 icon) ↔ MIDI (Piano icon). Highlighted
  1705→            when MIDI so the kind is visually distinct at a glance. */}
  1706→        <button onClick={toggleKind} className={`flex h-4 w-4 items-center justify-center rounded ${isMidi ? 'bg-accent/30 text-accent' : 'text-text-mid hover:bg-bg-3'}`} title={isMidi ? 'MIDI track (click for Audio)' : 'Audio track (click for MIDI)'}>{isMidi ? <Piano size={9} /> : <Volume2 size={9} />}</button>
  1707→        <button onClick={() => update({ muted: !track.muted })} className={`flex h-4 w-4 items-center justify-center rounded ${track.muted && !hasSolo ? 'bg-warn/30 text-warn' : 'text-text-mid hover:bg-bg-3'}`} title="M">{track.muted ? <VolumeX size={9} /> : <Volume2 size={9} />}</button>
  1708→        <button onClick={() => update({ solo: !track.solo })} className={`flex h-4 w-4 items-center justify-center rounded ${track.solo ? 'bg-accent/30 text-accent' : 'text-text-mid hover:bg-bg-3'}`} title="S"><Headphones size={9} /></button>
  1709→        <button onClick={toggleArm} className={`flex h-4 w-4 items-center justify-center rounded ${track.armed ? 'bg-danger/30 text-danger' : 'text-text-mid hover:bg-bg-3'}`} title={isMidi ? 'Arm for MIDI take' : 'Arm'}><div className="h-1.5 w-1.5 rounded-full" style={{ backgroundColor: track.armed ? 'currentColor' : 'transparent', border: '1px solid currentColor' }} /></button>
  1710→        <button onClick={() => deleteAudioTrack(slate, track.id)} className="flex h-4 w-4 items-center justify-center rounded text-text-mid hover:bg-bg-3 hover:text-danger" title="Del"><Trash2 size={9} /></button>
  1711→      </div>
  1712→      {/* MIDI tracks: instrument picker row replaces the pan slider row.
  1713→          Audio tracks keep the original volume + pan slider row. */}
  1714→      {isMidi ? (
  1715→        <div className="mt-0.5 flex items-center gap-1">
  1716→          <Piano size={9} className="shrink-0 text-accent" aria-hidden />
  1717→          <select
  1718→            value={track.instrumentId ?? SOUNDFONT_PIANO_ID}
  1719→            onChange={(e) => update({ instrumentId: e.target.value })}
  1720→            className="min-w-0 flex-1 rounded-sm border border-border bg-bg-3 px-1 py-0.5 text-[9px] text-text outline-none focus:border-accent"
  1721→            aria-label="Instrument"
  1722→            title="Which instrument plays this track's MIDI clips"
  1723→          >
  1724→            <option value={SOUNDFONT_PIANO_ID}>Soundfont Piano</option>
  1725→            <optgroup label="Synth presets">

  1728→            {(() => {
  1729→              const customs = loadCustomInstruments();
  1730→              return customs.length > 0 ? (
  1731→                <optgroup label="My instruments">
  1732→                  {customs.map((p) => <option key={p.id} value={p.id}>{p.name}</option>)}

  1753→const ClipBlock = memo(function ClipBlock({ clip, pxPerSec, selected, onDragStart, onTrimStart }: {
  1754→  clip: AudioClip; pxPerSec: number; selected: boolean;
  1755→  onDragStart: (clip: AudioClip, e: React.PointerEvent, el: HTMLElement, waveEl: HTMLElement | null) => void;
  1756→  onTrimStart: (clip: AudioClip, side: 'left' | 'right', e: React.PointerEvent, el: HTMLElement, waveEl: HTMLElement | null) => void;
  1757→}) {
  1758→  const left = clip.start * pxPerSec;
  1759→  const width = Math.max(4, clip.duration * pxPerSec);
  1760→  const elRef = useRef<HTMLDivElement | null>(null);
  1761→  const waveRef = useRef<HTMLDivElement | null>(null);
  1762→  const w = Math.max(2, Math.floor(width));
  1763→  // Fade overlay widths (clamped to the clip box so a too-long fade doesn't
  1764→  // spill past the opposite edge).
  1765→  const fadeInW = clip.fadeIn > 0 ? Math.min(width, clip.fadeIn * pxPerSec) : 0;
  1766→  const fadeOutW = clip.fadeOut > 0 ? Math.min(width, clip.fadeOut * pxPerSec) : 0;
  1767→
  1768→  // onDragStart handles selection (incl. Shift/Cmd additive) + the
  1769→  // clip-select event — selecting here too would clobber multi-select.
  1770→  // `data-clip-id` is set so the multi-drag startDrag can look up every
  1771→  // selected clip's DOM element via document.querySelector when assembling
  1772→  // the `origins` map (see startDrag's multiDrag branch).
  1773→  return (
  1774→    <div ref={elRef} data-clip-id={clip.id} onPointerDown={(e) => { if (elRef.current) onDragStart(clip, e, elRef.current, waveRef.current); }}
  1775→      className={`group pointer-events-auto absolute top-0.5 bottom-0.5 cursor-grab overflow-hidden rounded border ${selected ? 'border-warn' : 'border-black/30'} active:cursor-grabbing`} style={{ left, width, backgroundColor: `${clip.color}20` }}>
  1776→      {/* Fixed-width waveform layer — clipped by the box so trimming cuts the
  1777→          audio rather than squashing the whole wave into a smaller space. */}
  1778→      <div ref={waveRef} className="pointer-events-none absolute top-0 bottom-0 left-0" style={{ width }}>
  1779→        {clip.sampleKey && <WaveformImg clipId={clip.id} sampleKey={clip.sampleKey} channels={clip.channels} sampleRate={clip.sampleRate} offset={clip.offset} duration={clip.duration} speed={clip.speed ?? 1} width={w} color={clip.color} />}
  1780→      </div>
  1781→      {/* Fade-in overlay: triangle from full height at the outer (left) edge

  1785→        <div className="pointer-events-none absolute top-0 bottom-0 left-0 bg-black/30" style={{ width: fadeInW, clipPath: 'polygon(0% 0%, 0% 100%, 100% 50%)' }} />
  1786→      )}
  1787→      {/* Fade-out overlay: mirrored on the right side. */}
  1788→      {fadeOutW > 0 && (
  1789→        <div className="pointer-events-none absolute top-0 bottom-0 right-0 bg-black/30" style={{ width: fadeOutW, clipPath: 'polygon(100% 0%, 100% 100%, 0% 50%)' }} />
  1790→      )}
  1791→      <span className="absolute left-1 top-0 truncate text-[7px] font-medium text-text-mid/70 pointer-events-none">{clip.name}{clip.mute ? ' (muted)' : ''}</span>
  1792→      <div onPointerDown={(e) => { if (elRef.current) onTrimStart(clip, 'left', e, elRef.current, waveRef.current); }} className="absolute left-0 top-0 bottom-0 w-1.5 cursor-ew-resize bg-white/40 opacity-0 group-hover:opacity-100" />
  1793→      <div onPointerDown={(e) => { if (elRef.current) onTrimStart(clip, 'right', e, elRef.current, waveRef.current); }} className="absolute right-0 top-0 bottom-0 w-1.5 cursor-ew-resize bg-white/40 opacity-0 group-hover:opacity-100" />
  1794→    </div>
