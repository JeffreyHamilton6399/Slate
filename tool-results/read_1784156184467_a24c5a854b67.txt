    39→// ── Waveform cache: pre-computed PNG data URLs ───────────────────────────────
    40→// Key: `${clipId}:${sampleCount}:${width}` → data URL
    41→const waveformPNGCache = new Map<string, string>();
    42→
    43→// Draw only the window [startFrame, endFrame) of the sample across `width`
    44→// pixels. Rendering the *window* (not the whole sample stretched) is what makes
    45→// trimming look like cutting the audio away rather than squashing it.
    46→function computeWaveformPNG(samples: Float32Array, channels: number, width: number, color: string, height: number, startFrame: number, endFrame: number): string {
    47→  const canvas = document.createElement('canvas');
    48→  const dpr = window.devicePixelRatio || 1;
    49→  const w = Math.max(2, Math.floor(width));
    50→  canvas.width = w * dpr;
    51→  canvas.height = height * dpr;
    52→  const ctx = canvas.getContext('2d')!;
    53→  ctx.scale(dpr, dpr);
    54→  ctx.clearRect(0, 0, w, height);
    55→  const totalFrames = samples.length / channels;
    56→  const s = Math.max(0, Math.min(totalFrames, Math.floor(startFrame)));
    57→  const e = Math.max(s, Math.min(totalFrames, Math.floor(endFrame)));
    58→  const span = e - s;
    59→  if (span <= 0) return canvas.toDataURL();
    60→  const mid = height / 2;
    61→  ctx.fillStyle = color;
    62→  for (let x = 0; x < w; x++) {
    63→    const f0 = s + Math.floor((x / w) * span);
    64→    const f1 = Math.min(e, s + Math.floor(((x + 1) / w) * span) + 1);
    65→    let peak = 0;
    66→    for (let i = f0; i < f1; i++) { const v = Math.abs(samples[i * channels] ?? 0); if (v > peak) peak = v; }
    67→    const bh = Math.max(1, peak * mid * 0.85);
    68→    ctx.fillRect(x, mid - bh, 1, bh * 2);
    69→  }
    70→  return canvas.toDataURL();
    71→}

    73→/** Waveform image for the buffer window the clip actually plays. With speed s,
    74→ *  a clip of `duration` timeline seconds consumes `duration*s` buffer seconds.
    75→ *
    76→ *  Two robustness fixes:
    77→ *  1. If `loadSamples` returns an EMPTY Float32Array (length 0) — which happens
    78→ *     when a freshly-split/created clip's IndexedDB write hasn't landed yet —
    79→ *     we DON'T cache the resulting blank PNG. Instead we show the `···`
    80→ *     placeholder and retry a few times (500ms apart) so the real waveform
    81→ *     appears once the samples are available.
    82→ *  2. We listen for `slate:audio-clip-changed` (fired by Normalize/Reverse/split)
    83→ *     and invalidate our cache entry for the current clipId, then force a
    84→ *     recompute via a `bust` counter — otherwise the memoised component would
    85→ *     keep showing the stale PNG even after the cache was cleared. */
    86→const WaveformImg = memo(function WaveformImg({ clipId, sampleKey, channels, sampleRate, offset, duration, speed, width, color }: {
    87→  clipId: string; sampleKey: string; channels: number; sampleRate: number;
    88→  offset: number; duration: number; speed: number; width: number; color: string;
    89→}) {
    90→  const height = TRACK_H - 6;
    91→  const [imgUrl, setImgUrl] = useState<string>('');
    92→  const [bust, setBust] = useState(0);
    93→  const retryRef = useRef(0);
    94→
    95→  // Invalidate cached PNGs for this clip when its samples change (normalize /
    96→  // reverse / split). The parent AudioEditor also invalidates + bumps version,
    97→  // but the memoised WaveformImg wouldn't otherwise recompute (its primitive
    98→  // props are unchanged) — the `bust` counter forces the load effect to re-run.
    99→  useEffect(() => {
   100→    const onChanged = (e: Event) => {
   101→      const detail = (e as CustomEvent<string>).detail;
   102→      // Match by clipId (normalize/reverse/split events) OR by sampleKey
   103→      // (multiplayer sample-arrival events from registerSampleSyncMap).
   104→      if (detail !== clipId && detail !== sampleKey) return;
   105→      for (const key of waveformPNGCache.keys()) {
   106→        if (key.startsWith(`${clipId}:`)) waveformPNGCache.delete(key);
   107→      }
   108→      retryRef.current = 0;
   109→      setBust((n) => n + 1);
   110→    };
   111→    window.addEventListener('slate:audio-clip-changed', onChanged as EventListener);
   112→    return () => window.removeEventListener('slate:audio-clip-changed', onChanged as EventListener);
   113→  }, [clipId, sampleKey]);
   114→
   115→  useEffect(() => {
   116→    const startFrame = Math.round(offset * sampleRate);
   117→    const endFrame = Math.round((offset + duration * speed) * sampleRate);
   118→    const cacheKey = `${clipId}:${sampleKey}:${startFrame}:${endFrame}:${Math.floor(width)}`;
   119→    const cached = waveformPNGCache.get(cacheKey);
   120→    if (cached) { setImgUrl(cached); return; }
   121→    let cancelled = false;
   122→    let retryTimer: ReturnType<typeof setTimeout> | null = null;
   123→
   124→    const attempt = () => {
   125→      if (cancelled) return;
   126→      void loadSamples(sampleKey).then((samples) => {
   127→        if (cancelled) return;
   128→        // Empty samples = the IndexedDB write for this sampleKey hasn't landed
   129→        // yet (e.g. brand-new clip from a split). DON'T cache a blank PNG —
   130→        // retry a few times so the real waveform appears.
   131→        if (samples.length === 0) {
   132→          if (retryRef.current < 5) {
   133→            retryRef.current += 1;
   134→            retryTimer = setTimeout(attempt, 500);
   135→          }
   136→          return;
   137→        }
   138→        retryRef.current = 0;
   139→        const url = computeWaveformPNG(samples, channels, width, color, height, startFrame, endFrame);
   140→        waveformPNGCache.set(cacheKey, url);
   141→        setImgUrl(url);
   142→      });
   143→    };
   144→    attempt();
   145→
   146→    return () => {
   147→      cancelled = true;
   148→      if (retryTimer) clearTimeout(retryTimer);
   149→    };
   150→  }, [clipId, sampleKey, channels, sampleRate, offset, duration, speed, width, color, height, bust]);
   151→
   152→  if (!imgUrl) return <div className="flex h-full items-center justify-center text-[7px] text-text-dim">···</div>;
   153→  return <img src={imgUrl} alt="" className="pointer-events-none h-full w-full" style={{ objectFit: 'fill' }} />;
   154→});
   155→
   156→/** Drop cached waveform PNGs for a clip (call when its samples change). */
   157→function invalidateWaveform(clipId: string): void {
   158→  for (const key of waveformPNGCache.keys()) if (key.startsWith(`${clipId}:`)) waveformPNGCache.delete(key);
   159→}

   163→export function AudioEditor() {
   164→  const room = useRoom();
   165→  const slate = room.slate;
   166→  const [version, setVersion] = useState(0);
   167→  const [playing, setPlaying] = useState(false);
   168→  const [bpm, setBpmState] = useState(slate.audioBpm());
   169→  const [metronome, setMetronome] = useState(false);
   170→  const [recording, setRecording] = useState(false);
   171→  const [selectedClipId, setSelectedClipId] = useState<string | null>(null);
   172→  const [masterVol, setMasterVol] = useState(0.85);
   173→  const [looping, setLooping] = useState(false);
   174→  const [loopStart, setLoopStart] = useState(0);
   175→  const [loopEnd, setLoopEnd] = useState(8);
   176→  const [pxPerSec, setPxPerSec] = useState(80);
   177→  const engineRef = useRef<AudioEngine | null>(null);
   178→  const stopRecRef = useRef<(() => Promise<{ samples: number[]; sampleRate: number; channels: number; duration: number }>) | null>(null);
   179→  const rafRef = useRef(0);
   180→  const positionRef = useRef(0);
   181→  // Refs mirror the latest `playing` and `slate` so the long-lived Yjs / event
   182→  // listeners (which close over the initial render) can read fresh values
   183→  // without re-subscribing on every state change.
   184→  const playingRef = useRef(false);
   185→  playingRef.current = playing;
   186→  const slateRef = useRef(slate);
   187→  slateRef.current = slate;
   188→  /** Debounce timer for restartPlayback — coalesces rapid clip/sample changes
   189→   *  so we don't tear down and rebuild the scheduler 5× in a row when a peer
   190→   *  adds several clips at once. */
   191→  const restartTimerRef = useRef<ReturnType<typeof setTimeout> | null>(null);
   192→  const playheadRef = useRef<HTMLDivElement | null>(null);
   193→  const posDisplayRef = useRef<HTMLSpanElement | null>(null);
   194→  const scrollRef = useRef<HTMLDivElement | null>(null);
   195→  const dragRef = useRef<{
   196→    clipId: string; el: HTMLElement; waveEl: HTMLElement | null;
   197→    os: number; od: number; oo: number; speed: number; sx: number;
   198→    leftLimit: number; rightLimit: number; mode: 'drag' | 'trimL' | 'trimR';
   199→    neighbours: { start: number; end: number }[];
   200→  } | null>(null);
   201→  const selectedRef = useRef<string | null>(null);
   202→  selectedRef.current = selectedClipId;
   203→  const clipsRef = useRef<AudioClip[]>([]);
   204→  const pxRef = useRef(pxPerSec);
   205→  pxRef.current = pxPerSec;
   206→  /** Desired scrollLeft to apply after the next pxPerSec commit (used by
   207→   *  zoomAtPlayhead so the playhead stays at the same screen position when
   208→   *  zooming). Cleared in the layout effect below once applied. */
   209→  const pendingScrollRef = useRef<number | null>(null);
   210→
   211→  /** Schedule a debounced restartPlayback (500ms). Called when the clip set
   212→   *  changes mid-playback (remote peer adds a clip) or when a clip's samples
   213→   *  arrive after play started (remote sample blob landed via the sync map).
   214→   *  The debounce coalesces rapid bursts so we don't tear down and rebuild
   215→   *  the scheduler repeatedly. No-op when not playing. Reads only refs, so the
   216→   *  callback identity is stable for the lifetime of the component. */
   217→  const scheduleRestart = useCallback(() => {
   218→    if (!playingRef.current) return;
   219→    if (restartTimerRef.current) clearTimeout(restartTimerRef.current);
   220→    restartTimerRef.current = setTimeout(() => {
   221→      restartTimerRef.current = null;
   222→      const eng = engineRef.current;
   223→      if (eng && playingRef.current) eng.restartPlayback(slateRef.current, positionRef.current);
   224→    }, 500);
   225→  }, []);
   226→
   227→  // Yjs subscription.
   228→  useEffect(() => {
   229→    // Register the multiplayer sample sync map so audio clips imported by
   230→    // other peers are automatically received and stored locally.
   231→    registerSampleSyncMap(room);
   232→    const tracks = slate.audioTracks();
   233→    const clips = slate.audioClips();
   234→    const audioMap = slate.doc.getMap('audio');
   235→    let pending = false;
   236→    const bump = () => { if (pending) return; pending = true; requestAnimationFrame(() => { pending = false; setVersion((v) => v + 1); }); };
   237→    tracks.observeDeep(bump); clips.observeDeep(bump); audioMap.observe(bump); bump();
   238→    // Detect NEW clips being added while playing and schedule a debounced
   239→    // restartPlayback so the new clip is picked up mid-playback without a
   240→    // manual stop/play. We watch the shallow clips map (not observeDeep) so
   241→    // we only react to key additions, not to every property edit on an
   242→    // existing clip (which would restart playback every time a peer nudged
   243→    // a clip's volume, fighting the user). The restart is debounced 500ms so
   244→    // a burst of additions (e.g. a peer imports 5 files) coalesces into one.
   245→    const onClipsAdded = (event: Y.YMapEvent<Y.Map<unknown>>) => {
   246→      let hasAddition = false;
   247→      for (const key of event.keysChanged) {
   248→        const change = event.changes.keys.get(key);
   249→        if (change?.action === 'add') { hasAddition = true; break; }
   250→      }
   251→      if (hasAddition) scheduleRestart();
   252→    };
   253→    clips.observe(onClipsAdded);
   254→    const lateRead = setTimeout(bump, 200);
   255→    return () => { clearTimeout(lateRead); tracks.unobserveDeep(bump); clips.unobserveDeep(bump); audioMap.unobserve(bump); clips.unobserve(onClipsAdded); };
   256→    // eslint-disable-next-line react-hooks/exhaustive-deps
   257→  }, [slate]);
   258→
   259→  useEffect(() => {
   260→    engineRef.current = new AudioEngine();
   261→    engineRef.current.setMasterVolume(masterVol);
   262→    return () => {
   263→      // Cancel any pending restart so it doesn't fire after dispose().
   264→      if (restartTimerRef.current) { clearTimeout(restartTimerRef.current); restartTimerRef.current = null; }
   265→      engineRef.current?.dispose();
   266→      engineRef.current = null;
   267→    };
   268→    // eslint-disable-next-line react-hooks/exhaustive-deps
   269→  }, []);
   270→
   271→  // Playhead — direct DOM.
   272→  useEffect(() => {
   273→    if (!playing) {
   274→      // Playback just stopped — cancel any pending mid-playback restart so
   275→      // it doesn't fire and call restartPlayback (which would re-start audio
   276→      // after the user pressed pause).
   277→      if (restartTimerRef.current) { clearTimeout(restartTimerRef.current); restartTimerRef.current = null; }
   278→      return;
   279→    }
   280→    const tick = () => {
   281→      rafRef.current = requestAnimationFrame(tick);
   282→      const eng = engineRef.current; if (!eng) return;
   283→      const pos = eng.getPosition();
   284→      positionRef.current = pos;
   285→      if (playheadRef.current) playheadRef.current.style.transform = `translateX(${pos * pxRef.current}px)`;
   286→      if (posDisplayRef.current) posDisplayRef.current.textContent = `${pos.toFixed(1)}s`;
   287→      if (pos > timelineDuration + 2) { eng.stop(); setPlaying(false); }
   288→    };
   289→    rafRef.current = requestAnimationFrame(tick);
   290→    return () => cancelAnimationFrame(rafRef.current);
   291→    // eslint-disable-next-line react-hooks/exhaustive-deps
   292→  }, [playing]);
   293→
   294→  // Ctrl+scroll zoom — centred on the playhead so the screen position under
   295→  // the cursor/playhead stays put instead of zooming toward the left edge.
   296→  /** Zoom while keeping the playhead at the same screen offset. Records the
   297→   *  playhead's offset from the left edge of the visible viewport (relative to
   298→   *  the timeline content), changes `pxPerSec`, then schedules a scroll
   299→   *  correction so the playhead lands at the same offset in the new zoom.
   300→   *  The actual `scrollLeft` write happens in the layout effect below (after
   301→   *  React commits the new `minWidth` on the timeline div — setting it earlier
   302→   *  would be clamped by the stale `scrollWidth`). */
   303→  const zoomAtPlayhead = useCallback((newPxPerSec: number) => {
   304→    const el = scrollRef.current;
   305→    const oldPxPerSec = pxRef.current;
   306→    if (!el || oldPxPerSec === newPxPerSec) {
   307→      setPxPerSec(newPxPerSec);
   308→      return;
   309→    }
   310→    const scrollLeft = el.scrollLeft;
   311→    const playheadX = positionRef.current * oldPxPerSec;
   312→    const playheadOffset = playheadX - scrollLeft;
   313→    const newPlayheadX = positionRef.current * newPxPerSec;
   314→    const newScrollLeft = newPlayheadX - playheadOffset;
   315→    pendingScrollRef.current = newScrollLeft;
   316→    setPxPerSec(newPxPerSec);
   317→  }, []);
   318→
   319→  // Apply any pending scroll correction after pxPerSec commits to the DOM.
   320→  useLayoutEffect(() => {
   321→    if (pendingScrollRef.current !== null && scrollRef.current) {
   322→      scrollRef.current.scrollLeft = pendingScrollRef.current;
   323→      pendingScrollRef.current = null;
   324→    }
   325→  }, [pxPerSec]);
   326→
   327→  useEffect(() => {
   328→    const el = scrollRef.current; if (!el) return;
   329→    const onWheel = (e: WheelEvent) => {
   330→      if (!e.ctrlKey && !e.metaKey) return;
   331→      e.preventDefault();
   332→      const next = Math.max(MIN_PX_PER_SEC, Math.min(MAX_PX_PER_SEC, pxRef.current * (e.deltaY < 0 ? 1.2 : 1 / 1.2)));
   333→      zoomAtPlayhead(next);
   334→    };
   335→    el.addEventListener('wheel', onWheel, { passive: false });
   336→    return () => el.removeEventListener('wheel', onWheel);
   337→  }, [zoomAtPlayhead]);
   338→
   339→  // Global pointermove/up — pure DOM, zero React state.
   340→  useEffect(() => {
   341→    const onMove = (ev: PointerEvent) => {
   342→      const d = dragRef.current; if (!d) return;
   343→      const pps = pxRef.current;
   344→      const dt = (ev.clientX - d.sx) / pps;
   345→      const oldEnd = d.os + d.od;
   346→      if (d.mode === 'drag') {
   347→        // Snap-to-neighbour drag: the clip follows the mouse, but if the new
   348→        // position would overlap another clip on the same track, instead of
   349→        // stopping dead at the neighbour's edge we JUMP to the free side of
   350→        // that neighbour (right of it when dragging right, left of it when
   351→        // dragging left). Iterates so a chain of back-to-back clips resolves
   352→        // to the next free slot. A small overlap threshold avoids accidental
   353→        // snaps when the cursor barely crosses the boundary.
   354→        const duration = d.od;
   355→        const rawStart = d.os + dt;
   356→        let start = rawStart;
   357→        const overlapThreshold = 0.05; // s — ignore near-boundary jitter
   358→        for (let iter = 0; iter < d.neighbours.length + 1; iter++) {
   359→          const blocker = d.neighbours.find((n) => {
   360→            const ovStart = Math.max(start, n.start);
   361→            const ovEnd = Math.min(start + duration, n.end);
   362→            return ovEnd - ovStart > overlapThreshold;
   363→          });
   364→          if (!blocker) break;
   365→          if (dt >= 0) start = blocker.end;          // snap to RIGHT of blocker
   366→          else start = blocker.start - duration;     // snap to LEFT of blocker
   367→        }
   368→        // Fallback clamp: never let the clip start before t=0.
   369→        start = Math.max(0, start);
   370→        d.el.style.left = `${start * pps}px`;
   371→      } else if (d.mode === 'trimL') {
   372→        // Cut from the left: never past the left neighbour, the source start
   373→        // (offset ≥ 0 → limited by the trimmed head in timeline seconds), or a
   374→        // minimum width.
   375→        const minStart = Math.max(d.leftLimit, d.os - d.oo / d.speed);
   376→        const start = Math.min(oldEnd - 0.1, Math.max(minStart, d.os + dt));
   377→        d.el.style.left = `${start * pps}px`;
   378→        d.el.style.width = `${(oldEnd - start) * pps}px`;
   379→        // Shift the (fixed-width) waveform the opposite way so the audio stays
   380→        // anchored — the newly-trimmed head is clipped, not squashed.
   381→        if (d.waveEl) d.waveEl.style.left = `${-(start - d.os) * pps}px`;
   382→      } else if (d.mode === 'trimR') {
   383→        // Cut from the right: never past the right neighbour or a min width.
   384→        const end = Math.min(d.rightLimit, Math.max(d.os + 0.1, oldEnd + dt));
   385→        d.el.style.width = `${(end - d.os) * pps}px`;
   386→      }
   387→    };
   388→    const onUp = () => {
   389→      const d = dragRef.current; if (!d) return;
   390→      const pps = pxRef.current;
   391→      const left = parseFloat(d.el.style.left) / pps;
   392→      const width = parseFloat(d.el.style.width) / pps;
   393→      if (d.mode === 'drag') updateAudioClip(slate, d.clipId, { start: Math.max(0, left) });
   394→      else if (d.mode === 'trimL') updateAudioClip(slate, d.clipId, { start: left, duration: width, offset: Math.max(0, d.oo + (left - d.os) * d.speed) });
   395→      else if (d.mode === 'trimR') updateAudioClip(slate, d.clipId, { duration: width });
   396→      dragRef.current = null;
   397→    };
   398→    window.addEventListener('pointermove', onMove);
   399→    window.addEventListener('pointerup', onUp);
   400→    return () => { window.removeEventListener('pointermove', onMove); window.removeEventListener('pointerup', onUp); };
   401→  }, [slate]);
   402→
   403→  // Hotkeys.
   404→  useEffect(() => {
   405→    const onKey = (e: KeyboardEvent) => {
   406→      const t = e.target as HTMLElement;
   407→      if (t && /^(INPUT|TEXTAREA|SELECT)$/.test(t.tagName)) return;
   408→      const k = e.key.toLowerCase();
   409→      if (k === ' ') { e.preventDefault(); togglePlay(); }
   410→      else if (k === 'c' && !e.ctrlKey && selectedRef.current) { e.preventDefault(); void splitAudioClip(slate, selectedRef.current, positionRef.current); }
   411→      else if ((k === 'delete' || k === 'backspace') && selectedRef.current) { e.preventDefault(); deleteAudioClip(slate, selectedRef.current); setSelectedClipId(null); }
   412→      else if (k === 'd' && !e.ctrlKey && selectedRef.current) { e.preventDefault(); dupClip(selectedRef.current); }
   413→      else if (k === 'l') { e.preventDefault(); setLooping((n) => !n); }
   414→      else if (k === 'm') { e.preventDefault(); setMetronome((n) => { engineRef.current?.setMetronome(!n); return !n; }); }
   415→      else if (k === 'r' && !e.ctrlKey) { e.preventDefault(); void toggleRecord(); }
   416→      else if (k === 'arrowleft') { e.preventDefault(); seek(positionRef.current - 2); }
   417→      else if (k === 'arrowright') { e.preventDefault(); seek(positionRef.current + 2); }
   418→      else if (k === 'home') { e.preventDefault(); seek(0); }
   419→    };
   420→    window.addEventListener('keydown', onKey);
   421→    return () => window.removeEventListener('keydown', onKey);
   422→    // eslint-disable-next-line react-hooks/exhaustive-deps
   423→  }, [playing, recording]);
   424→
   425→  // Read from Yjs — lightweight (no samples copied).
   426→  const tracks: AudioTrack[] = useMemo(() => {

   445→  // currently playing — schedule a debounced restartPlayback so the clip is
   446→  // picked up. This covers the case where a clip's metadata arrived but its
   447→  // samples were still in flight: getBuffer returned null on the first pass,
   448→  // and this restart re-schedules once the samples are usable.

   460→  }, []);

   500→        toast({ title: 'Recording added' });

   523→  // The free interval [leftLimit, rightLimit] the clip may occupy without

   527→  const neighbourBounds = useCallback((clip: AudioClip): { leftLimit: number; rightLimit: number; neighbours: { start: number; end: number }[] } => {
   528→    let leftLimit = 0;
   529→    let rightLimit = Infinity;

   531→    const clipEnd = clip.start + clip.duration;

   536→      else if (o.start >= clipEnd - 1e-4) rightLimit = Math.min(rightLimit, o.start);
   537→      neighbours.push({ start: o.start, end: oEnd });

   568→    window.addEventListener('pointermove', onMove); window.addEventListener('pointerup', onUp);
   569→  }, [loopStart, loopEnd, pxPerSec]);

   571→  // ── Render ────────────────────────────────────────────────────────────────
   572→
   573→  const beatDur = 60 / bpm; // seconds per beat
   574→  const gridStyle = {
   575→    backgroundImage: `repeating-linear-gradient(to right, rgba(128,128,128,0.08) 0 1px, transparent 1px ${beatDur * pxPerSec}px), repeating-linear-gradient(to right, rgba(128,128,128,0.2) 0 1px, transparent 1px ${beatDur * 4 * pxPerSec}px)`,
   576→  };
   577→
   578→  // Adaptive ruler tick interval — picks a "nice" step (in seconds) whose
   579→  // pixel spacing stays readable at the current zoom: tight when zoomed in
   580→  // (so you get millisecond-level ticks), sparse when zoomed out (so a long
   581→  // mix doesn't turn into a solid wall of labels).
   582→  const { tickInterval, formatTick } = useMemo(() => {
   583→    if (pxPerSec >= 400) return { tickInterval: 0.1, formatTick: (t: number) => `${t.toFixed(1)}s` };
   584→    if (pxPerSec >= 100) return { tickInterval: 1, formatTick: (t: number) => `${t}s` };
   585→    if (pxPerSec >= 40) return { tickInterval: 5, formatTick: (t: number) => `${t}s` };
   586→    if (pxPerSec >= 10) return { tickInterval: 10, formatTick: (t: number) => `${t}s` };
   587→    return {
   588→      tickInterval: 60,
   589→      formatTick: (t: number) => {
   590→        const m = Math.floor(t / 60);
   591→        const s = Math.round(t - m * 60);
   592→        return s === 0 ? `${m}m` : `${m}m ${s}s`;
   593→      },
   594→    };
   595→  }, [pxPerSec]);
   596→
   597→  return (
   598→    <div className="flex h-full flex-col bg-bg overflow-hidden" onDragOver={(e) => { if (e.dataTransfer?.types?.includes('Files')) e.preventDefault(); }} onDrop={(e) => { e.preventDefault(); for (const f of [...(e.dataTransfer?.files ?? [])].filter((f) => /\.(mp3|wav|ogg|m4a|flac|aac)$/i.test(f.name))) void handleFileImport(f); }}>
   599→      {/* Transport */}
   600→      <div className="flex shrink-0 items-center gap-1 border-b border-border bg-bg-2 px-2 py-1.5">
   601→        <button onClick={() => seek(0)} className="flex h-7 w-7 items-center justify-center rounded text-text-mid hover:bg-bg-3" title="Start"><SkipBack size={14} /></button>
   602→        <button onClick={togglePlay} className={`flex h-9 w-9 items-center justify-center rounded-full text-white ${playing ? 'bg-warn' : 'bg-accent'} hover:opacity-80`} title="Play (Space)">{playing ? <Pause size={18} /> : <Play size={18} />}</button>
   603→        <button onClick={() => void toggleRecord()} className={`flex h-8 w-8 items-center justify-center rounded-full border ${recording ? 'border-danger bg-danger/20 text-danger animate-pulse' : 'border-border text-text-mid hover:bg-bg-3'}`} title="Record (R)"><Mic size={15} /></button>
   604→        <span ref={posDisplayRef} className="ml-1 min-w-[2.5rem] font-mono text-xs text-text">0.0s</span>
   605→        <div className="mx-1 h-5 w-px bg-border" />
   606→        <button onClick={() => selectedRef.current && splitAudioClip(slate, selectedRef.current, positionRef.current)} disabled={!selectedClipId} className="flex h-7 w-7 items-center justify-center rounded text-text-mid hover:bg-bg-3 disabled:opacity-30" title="Split (C)"><Scissors size={14} /></button>
   607→        <button onClick={() => selectedRef.current && dupClip(selectedRef.current)} disabled={!selectedClipId} className="flex h-7 w-7 items-center justify-center rounded text-text-mid hover:bg-bg-3 disabled:opacity-30" title="Duplicate (D)"><Copy size={14} /></button>
   608→        <button onClick={() => { if (selectedRef.current) { deleteAudioClip(slate, selectedRef.current); setSelectedClipId(null); } }} disabled={!selectedClipId} className="flex h-7 w-7 items-center justify-center rounded text-text-mid hover:bg-bg-3 hover:text-danger disabled:opacity-30" title="Delete (Del)"><Trash2 size={14} /></button>
   609→        <div className="mx-1 h-5 w-px bg-border" />
   610→        <label className="flex items-center gap-1 text-[11px] text-text-dim">BPM<input type="number" min={20} max={300} value={bpm} onChange={(e) => { setBpmState(Number(e.target.value)); setAudioBpm(slate, Number(e.target.value)); engineRef.current?.setBpm(Number(e.target.value)); }} className="w-14 rounded border border-border bg-bg-3 px-1 py-0.5 text-center font-mono text-xs text-text outline-none focus:border-accent" /></label>
   611→        <button onClick={() => { const n = !metronome; setMetronome(n); engineRef.current?.setMetronome(n); }} className={`flex h-7 w-7 items-center justify-center rounded ${metronome ? 'bg-accent/20 text-accent' : 'text-text-mid hover:bg-bg-3'}`} title="Metronome (M)"><Music size={13} /></button>
   612→        <button onClick={() => setLooping((n) => !n)} className={`flex h-7 w-7 items-center justify-center rounded ${looping ? 'bg-accent/20 text-accent' : 'text-text-mid hover:bg-bg-3'}`} title="Loop (L)"><Repeat size={13} /></button>
   613→        <div className="mx-1 h-5 w-px bg-border" />
   614→        <div className="flex items-center gap-1"><Volume2 size={12} className="text-text-mid" /><input type="range" min={0} max={1} step={0.01} value={masterVol} onChange={(e) => { setMasterVol(Number(e.target.value)); engineRef.current?.setMasterVolume(Number(e.target.value)); }} className="w-14 accent-accent" /></div>
   615→        <button onClick={() => zoomAtPlayhead(Math.max(MIN_PX_PER_SEC, pxRef.current / 1.3))} className="flex h-6 w-6 items-center justify-center rounded text-text-mid hover:bg-bg-3" title="Zoom out"><ZoomOut size={12} /></button>
   616→        <button onClick={fitToWindow} className="flex h-6 w-6 items-center justify-center rounded text-text-mid hover:bg-bg-3 hover:text-accent" title="Fit to window"><Maximize2 size={12} /></button>
   617→        <button onClick={() => zoomAtPlayhead(Math.min(MAX_PX_PER_SEC, pxRef.current * 1.3))} className="flex h-6 w-6 items-center justify-center rounded text-text-mid hover:bg-bg-3" title="Zoom in"><ZoomIn size={12} /></button>
   618→        <div className="flex-1" />
   619→        <label className="flex cursor-pointer items-center gap-1 rounded border border-border px-2 py-1 text-[11px] text-text-mid hover:bg-bg-3"><Upload size={12} />Import<input type="file" accept="audio/*" className="hidden" onChange={(e) => { const f = e.target.files?.[0]; if (f) void handleFileImport(f); e.target.value = ''; }} /></label>
   620→        <button onClick={() => addAudioTrack(slate)} className="flex items-center gap-1 rounded border border-accent/40 bg-accent/10 px-2 py-1 text-[11px] text-accent hover:bg-accent/20"><Plus size={12} />Track</button>
   621→      </div>
   622→
   623→      {/* Track area */}
   624→      <div ref={scrollRef} className="flex flex-1 min-h-0 overflow-auto">
   625→        {/* Headers */}

   630→        </div>
   631→
   632→        {/* Timeline */}
   633→        <div className="relative flex-1" style={{ minWidth: timelineDuration * pxPerSec }}>
   634→          {/* Ruler + loop handles */}
   635→          <div className="sticky top-0 z-10 border-b border-border bg-bg-2/95" style={{ height: 28 }}>
   636→            {Array.from({ length: Math.ceil(timelineDuration / tickInterval) + 1 }, (_, i) => {
   637→              const t = i * tickInterval;
   638→              return (
   639→                <span key={i} className="absolute top-1 pl-1 text-[8px] font-mono text-text-dim" style={{ left: t * pxPerSec }}>{formatTick(t)}</span>
   640→              );
   641→            })}
   642→            {looping && (
   643→              <>
   644→                <div onPointerDown={(e) => startLoopDrag('start', e)} className="absolute top-0 z-20 flex h-7 cursor-ew-resize items-center justify-center bg-accent/40 hover:bg-accent/60" style={{ left: loopStart * pxPerSec - 8, width: 8 }} title="Drag loop start"><ChevronLeft size={10} className="text-white" /></div>

   646→              </>
   647→            )}
   648→          </div>
   649→          {/* Seek layer — background click catcher (behind clips) */}
   650→          <div className="absolute inset-0 top-7" onPointerDown={(e) => { const r = e.currentTarget.getBoundingClientRect(); seek((e.clientX - r.left) / pxRef.current); }} />
   651→          {/* Grid background */}
   652→          <div className="pointer-events-none absolute inset-0 top-7" style={gridStyle}>
   653→            {looping && <div onPointerDown={(e) => startLoopDrag('move', e)} className="pointer-events-auto absolute top-0 bottom-0 cursor-grab bg-accent/8 border-x-2 border-accent/40" style={{ left: loopStart * pxPerSec, width: (loopEnd - loopStart) * pxPerSec }} />}
   654→          </div>
   655→          {/* Playhead */}
   656→          <div ref={playheadRef} className="absolute top-0 bottom-0 z-20 w-0.5 bg-warn pointer-events-none" style={{ transform: 'translateX(0px)' }}>
   657→            <div className="absolute top-0 -left-1 h-2 w-2 rounded-full bg-warn" />
   658→          </div>
   659→          {/* Clips */}
   660→          {tracks.map((t) => (
   661→            <div key={t.id} className="pointer-events-none relative border-b border-border/15" style={{ height: TRACK_H }}>
   662→              {clips.filter((c) => c.trackId === t.id).map((c) => (
   663→                <ClipBlock key={c.id} clip={c} pxPerSec={pxPerSec} selected={selectedClipId === c.id}
   664→                  onSelect={setSelectedClipId} onDragStart={startDrag} onTrimStart={startTrim} />
   665→              ))}
   666→            </div>
   667→          ))}
   668→        </div>
   669→      </div>
   670→
   671→      {/* Status */}
   672→      <div className="flex shrink-0 items-center gap-2 border-t border-border px-2 py-0.5 text-[8px] font-mono uppercase text-text-dim">
   673→        <span>{tracks.length}T · {clips.length}C</span>
   674→        {recording && <span className="text-danger">● Rec</span>}
   675→        {playing && <span className="text-accent">▶ Play</span>}
   676→        <span className="ml-auto">Space · C=Split · D=Dup · Del · R=Rec · L=Loop · M=Met · ←→=Seek · Ctrl+Scroll=Zoom</span>
   677→      </div>
   678→    </div>
   679→  );
   680→}

   684→const TrackHeader = memo(function TrackHeader({ track, hasSolo, slate, engineRef }: {
   685→  track: AudioTrack; hasSolo: boolean;
   686→  slate: ReturnType<typeof useRoom>['slate'];
   687→  engineRef: React.RefObject<AudioEngine | null>;
   688→}) {
   689→  const [vol, setVol] = useState(track.volume);
   690→  const [pan, setPan] = useState(track.pan);
   691→  // While the user is dragging either slider, DON'T let the prop-sync effects
   692→  // overwrite the local state — otherwise the Yjs observe fired by the engine's
   693→  // live `updateTracks` would clobber `vol`/`pan` mid-drag, causing the slider
   694→  // to fight the user (the "big slider does nothing" symptom).
   695→  const isDraggingRef = useRef(false);
   696→  useEffect(() => { if (!isDraggingRef.current) setVol(track.volume); }, [track.volume]);
   697→  useEffect(() => { if (!isDraggingRef.current) setPan(track.pan); }, [track.pan]);
   698→
   699→  const update = (patch: Partial<AudioTrack>) => { updateAudioTrack(slate, track.id, patch); engineRef.current?.updateTracks(slate); };
   700→  const onVolDown = () => { isDraggingRef.current = true; };
   701→  const onVol = (v: number) => { setVol(v); engineRef.current?.updateTracks(slate); };
   702→  const onVolEnd = () => { isDraggingRef.current = false; updateAudioTrack(slate, track.id, { volume: vol }); };
   703→  const onPanDown = () => { isDraggingRef.current = true; };
   704→  const onPan = (p: number) => { setPan(p); engineRef.current?.updateTracks(slate); };
   705→  const onPanEnd = () => { isDraggingRef.current = false; updateAudioTrack(slate, track.id, { pan: pan }); };
   706→
   707→  return (
   708→    <div className="border-b border-border/15 px-2 py-1" style={{ height: TRACK_H }}>
   709→      <div className="flex items-center gap-0.5">
   710→        <span className="h-2 w-2 shrink-0 rounded-full" style={{ backgroundColor: track.color }} />
   711→        <input type="text" value={track.name} onChange={(e) => update({ name: e.target.value })} className="min-w-0 flex-1 bg-transparent text-[11px] font-medium text-text outline-none" />
   712→        <button onClick={() => update({ muted: !track.muted })} className={`flex h-4 w-4 items-center justify-center rounded ${track.muted && !hasSolo ? 'bg-warn/30 text-warn' : 'text-text-mid hover:bg-bg-3'}`} title="M">{track.muted ? <VolumeX size={9} /> : <Volume2 size={9} />}</button>
   713→        <button onClick={() => update({ solo: !track.solo })} className={`flex h-4 w-4 items-center justify-center rounded ${track.solo ? 'bg-accent/30 text-accent' : 'text-text-mid hover:bg-bg-3'}`} title="S"><Headphones size={9} /></button>
   714→        <button onClick={() => update({ armed: !track.armed, input: !track.armed ? 'mic' : 'none' })} className={`flex h-4 w-4 items-center justify-center rounded ${track.armed ? 'bg-danger/30 text-danger' : 'text-text-mid hover:bg-bg-3'}`} title="Arm"><div className="h-1.5 w-1.5 rounded-full" style={{ backgroundColor: track.armed ? 'currentColor' : 'transparent', border: '1px solid currentColor' }} /></button>
   715→        <button onClick={() => deleteAudioTrack(slate, track.id)} className="flex h-4 w-4 items-center justify-center rounded text-text-mid hover:bg-bg-3 hover:text-danger" title="Del"><Trash2 size={9} /></button>
   716→      </div>
   717→      <div className="mt-0.5 flex items-center gap-1">
   718→        <Volume2 size={8} className="shrink-0 text-text-dim" />

   726→// ── Clip block ──────────────────────────────────────────────────────────────

   728→const ClipBlock = memo(function ClipBlock({ clip, pxPerSec, selected, onSelect, onDragStart, onTrimStart }: {
   729→  clip: AudioClip; pxPerSec: number; selected: boolean;
   730→  onSelect: (id: string) => void;
   731→  onDragStart: (clip: AudioClip, e: React.PointerEvent, el: HTMLElement, waveEl: HTMLElement | null) => void;
   732→  onTrimStart: (clip: AudioClip, side: 'left' | 'right', e: React.PointerEvent, el: HTMLElement, waveEl: HTMLElement | null) => void;
   733→}) {
   734→  const left = clip.start * pxPerSec;
   735→  const width = Math.max(4, clip.duration * pxPerSec);
   736→  const elRef = useRef<HTMLDivElement | null>(null);
   737→  const waveRef = useRef<HTMLDivElement | null>(null);
   738→  const w = Math.max(2, Math.floor(width));
   739→  // Fade overlay widths (clamped to the clip box so a too-long fade doesn't
   740→  // spill past the opposite edge).
   741→  const fadeInW = clip.fadeIn > 0 ? Math.min(width, clip.fadeIn * pxPerSec) : 0;
   742→  const fadeOutW = clip.fadeOut > 0 ? Math.min(width, clip.fadeOut * pxPerSec) : 0;
   743→
   744→  return (
   745→    <div ref={elRef} onPointerDown={(e) => { if (elRef.current) { e.stopPropagation(); onSelect(clip.id); window.dispatchEvent(new CustomEvent('slate:audio-clip-select', { detail: clip.id })); onDragStart(clip, e, elRef.current, waveRef.current); } }}
   746→      className={`group pointer-events-auto absolute top-0.5 bottom-0.5 cursor-grab overflow-hidden rounded border ${selected ? 'border-warn' : 'border-black/30'} active:cursor-grabbing`} style={{ left, width, backgroundColor: `${clip.color}20` }}>
   747→      {/* Fixed-width waveform layer — clipped by the box so trimming cuts the
   748→          audio rather than squashing the whole wave into a smaller space. */}
   749→      <div ref={waveRef} className="pointer-events-none absolute top-0 bottom-0 left-0" style={{ width }}>
   750→        {clip.sampleKey && <WaveformImg clipId={clip.id} sampleKey={clip.sampleKey} channels={clip.channels} sampleRate={clip.sampleRate} offset={clip.offset} duration={clip.duration} speed={clip.speed ?? 1} width={w} color={clip.color} />}
   751→      </div>
   752→      {/* Fade-in overlay: triangle from full height at the outer (left) edge
   753→          tapering to 0 at the inner edge — the dark wedge represents the
   754→          portion of the clip still under the fade. */}
   755→      {fadeInW > 0 && (
   756→        <div className="pointer-events-none absolute top-0 bottom-0 left-0 bg-black/30" style={{ width: fadeInW, clipPath: 'polygon(0% 0%, 0% 100%, 100% 50%)' }} />
   757→      )}
   758→      {/* Fade-out overlay: mirrored on the right side. */}
   759→      {fadeOutW > 0 && (
   760→        <div className="pointer-events-none absolute top-0 bottom-0 right-0 bg-black/30" style={{ width: fadeOutW, clipPath: 'polygon(100% 0%, 100% 100%, 0% 50%)' }} />
   761→      )}
   762→      <span className="absolute left-1 top-0 truncate text-[7px] font-medium text-text-mid/70 pointer-events-none">{clip.name}{clip.mute ? ' (muted)' : ''}</span>
   763→      <div onPointerDown={(e) => { if (elRef.current) onTrimStart(clip, 'left', e, elRef.current, waveRef.current); }} className="absolute left-0 top-0 bottom-0 w-1.5 cursor-ew-resize bg-white/40 opacity-0 group-hover:opacity-100" />
   764→      <div onPointerDown={(e) => { if (elRef.current) onTrimStart(clip, 'right', e, elRef.current, waveRef.current); }} className="absolute right-0 top-0 bottom-0 w-1.5 cursor-ew-resize bg-white/40 opacity-0 group-hover:opacity-100" />
   765→    </div>
   766→  );
   767→});
