   314тЖТ   314тЖТ      const idx = pairs.findIndex(([a, b]) => a === bestA && b === bestB);
   315тЖТ   315тЖТ      let nextPairs: [number, number][];
   316тЖТ   316тЖТ      if (additive) {

   318тЖТ   318тЖТ      } else {

  1087тЖТ  1087тЖТfunction PivotProjector({
  1088тЖТ  1088тЖТ  snapshot,
  1089тЖТ  1089тЖТ  out,
  1090тЖТ  1090тЖТ}: {
  1091тЖТ  1091тЖТ  snapshot: SceneSnapshot;
  1092тЖТ  1092тЖТ  out: React.MutableRefObject<{ x: number; y: number } | null>;
  1093тЖТ  1093тЖТ}) {
  1094тЖТ  1094тЖТ  const { camera, size } = useThree();
  1095тЖТ  1095тЖТ  const snapRef = useRef(snapshot);
  1096тЖТ  1096тЖТ  snapRef.current = snapshot;
  1097тЖТ  1097тЖТ  useFrame(() => {
  1098тЖТ  1098тЖТ    const sel = useScene3DStore.getState().selection;
  1099тЖТ  1099тЖТ    if (sel.length === 0) {
  1100тЖТ  1100тЖТ      out.current = null;
  1101тЖТ  1101тЖТ      return;
  1102тЖТ  1102тЖТ    }
  1103тЖТ  1103тЖТ    let x = 0, y = 0, z = 0, n = 0;
  1104тЖТ  1104тЖТ    for (const id of sel) {
  1105тЖТ  1105тЖТ      const o = snapRef.current.objects.find((ob) => ob.id === id);
  1106тЖТ  1106тЖТ      if (o) {
  1107тЖТ  1107тЖТ        x += o.transform.position.x;
  1108тЖТ  1108тЖТ        y += o.transform.position.y;
  1109тЖТ  1109тЖТ        z += o.transform.position.z;
  1110тЖТ  1110тЖТ        n++;
  1111тЖТ  1111тЖТ      }
  1112тЖТ  1112тЖТ    }
  1113тЖТ  1113тЖТ    if (n === 0) {
  1114тЖТ  1114тЖТ      out.current = null;
  1115тЖТ  1115тЖТ      return;
  1116тЖТ  1116тЖТ    }
  1117тЖТ  1117тЖТ    const v = new THREE.Vector3(x / n, y / n, z / n).project(camera);
  1118тЖТ  1118тЖТ    out.current = { x: (v.x * 0.5 + 0.5) * size.width, y: (-v.y * 0.5 + 0.5) * size.height };
  1119тЖТ  1119тЖТ  });
  1120тЖТ  1120тЖТ  return null;
  1121тЖТ  1121тЖТ}

  1124тЖТ  1124тЖТ * Publishes the camera basis + world-units-per-pixel each frame so the modal
  1125тЖТ  1125тЖТ * tool driver (which lives outside the R3F canvas) can map pointer motion
  1126тЖТ  1126тЖТ * into the screen plane at the right sensitivity.

  1128тЖТ  1128тЖТfunction CameraTracking({
  1129тЖТ  1129тЖТ  out,
  1130тЖТ  1130тЖТ  room,
  1131тЖТ  1131тЖТ}: {
  1132тЖТ  1132тЖТ  out: React.MutableRefObject<CameraInfo | null>;
  1133тЖТ  1133тЖТ  room: SlateRoom;
  1134тЖТ  1134тЖТ}) {
  1135тЖТ  1135тЖТ  const { camera, controls, size } = useThree();
  1136тЖТ  1136тЖТ  const lastPublish = useRef({ t: 0, key: '' });
  1137тЖТ  1137тЖТ  useFrame(() => {
  1138тЖТ  1138тЖТ    const cam = camera as THREE.PerspectiveCamera;
  1139тЖТ  1139тЖТ    const right = new THREE.Vector3().setFromMatrixColumn(cam.matrixWorld, 0);
  1140тЖТ  1140тЖТ    const up = new THREE.Vector3().setFromMatrixColumn(cam.matrixWorld, 1);
  1141тЖТ  1141тЖТ    const forward = new THREE.Vector3().setFromMatrixColumn(cam.matrixWorld, 2).negate();
  1142тЖТ  1142тЖТ    const target = (controls as { target?: THREE.Vector3 } | null)?.target;
  1143тЖТ  1143тЖТ    const dist = target ? cam.position.distanceTo(target) : Math.max(cam.position.length(), 1);
  1144тЖТ  1144тЖТ    const pxScale =
  1145тЖТ  1145тЖТ      (2 * Math.max(dist, 0.5) * Math.tan(THREE.MathUtils.degToRad(cam.fov) / 2)) /
  1146тЖТ  1146тЖТ      Math.max(size.height, 1);
  1147тЖТ  1147тЖТ    out.current = {
  1148тЖТ  1148тЖТ      basis: {
  1149тЖТ  1149тЖТ        right: { x: right.x, y: right.y, z: right.z },
  1150тЖТ  1150тЖТ        up: { x: up.x, y: up.y, z: up.z },
  1151тЖТ  1151тЖТ        forward: { x: forward.x, y: forward.y, z: forward.z },
  1152тЖТ  1152тЖТ      },
  1153тЖТ  1153тЖТ      pxScale,
  1154тЖТ  1154тЖТ    };
  1155тЖТ  1155тЖТ    // Publish the camera pose so other peers can see where we're looking.
  1156тЖТ  1156тЖТ    // Throttled and change-gated тАФ awareness messages count against the
  1157тЖТ  1157тЖТ    // server's per-peer rate limit.
  1158тЖТ  1158тЖТ    const now = performance.now();
  1159тЖТ  1159тЖТ    if (now - lastPublish.current.t < 150) return;
  1160тЖТ  1160тЖТ    const t = target ?? new THREE.Vector3();
  1161тЖТ  1161тЖТ    const r = (n: number) => Math.round(n * 100) / 100;
  1162тЖТ  1162тЖТ    const p: [number, number, number] = [r(cam.position.x), r(cam.position.y), r(cam.position.z)];
  1163тЖТ  1163тЖТ    const tt: [number, number, number] = [r(t.x), r(t.y), r(t.z)];
  1164тЖТ  1164тЖТ    const key = p.join(',') + '|' + tt.join(',');
  1165тЖТ  1165тЖТ    if (key === lastPublish.current.key) return;
  1166тЖТ  1166тЖТ    lastPublish.current = { t: now, key };
  1167тЖТ  1167тЖТ    room.setLocalAwareness({ cam: { p, t: tt } });
  1168тЖТ  1168тЖТ  });
  1169тЖТ  1169тЖТ  return null;
  1170тЖТ  1170тЖТ}

  1185тЖТ  1185тЖТ/** Blender-style camera glyph: wire frustum + up-triangle, apex at the eye. */
  1186тЖТ  1186тЖТconst PEER_CAM_LINES = (() => {
  1187тЖТ  1187тЖТ  const w = 0.42, h = 0.28, d = 0.85; // frustum half-extents + depth
  1188тЖТ  1188тЖТ  const a = [0, 0, 0];
  1189тЖТ  1189тЖТ  const c = [
  1190тЖТ  1190тЖТ    [-w, h, d], [w, h, d], [w, -h, d], [-w, -h, d],
  1191тЖТ  1191тЖТ  ];
  1192тЖТ  1192тЖТ  const seg: number[] = [];
  1193тЖТ  1193тЖТ  for (const p of c) seg.push(...a, ...p); // apex тЖТ corners
  1194тЖТ  1194тЖТ  for (let i = 0; i < 4; i++) seg.push(...c[i]!, ...c[(i + 1) % 4]!); // rect
  1195тЖТ  1195тЖТ  // Up indicator above the top edge.
  1196тЖТ  1196тЖТ  seg.push(-w * 0.5, h, d, 0, h + 0.18, d);
  1197тЖТ  1197тЖТ  seg.push(0, h + 0.18, d, w * 0.5, h, d);
  1198тЖТ  1198тЖТ  seg.push(w * 0.5, h, d, -w * 0.5, h, d);
  1199тЖТ  1199тЖТ  const g = new THREE.BufferGeometry();
  1200тЖТ  1200тЖТ  g.setAttribute('position', new THREE.BufferAttribute(new Float32Array(seg), 3));
  1201тЖТ  1201тЖТ  return g;
  1202тЖТ  1202тЖТ})();

  1204тЖТ  1204тЖТfunction PeerCamera({ state }: { state: AwarenessState }) {
  1205тЖТ  1205тЖТ  const cam = state.cam!;
  1206тЖТ  1206тЖТ  const groupRef = useRef<THREE.Group | null>(null);
  1207тЖТ  1207тЖТ  // Poses arrive ~6/s; damp per frame so remote cameras glide, not hop.
  1208тЖТ  1208тЖТ  const targetPos = useRef(new THREE.Vector3(cam.p[0], cam.p[1], cam.p[2]));
  1209тЖТ  1209тЖТ  const lookCur = useRef(new THREE.Vector3(cam.t[0], cam.t[1], cam.t[2]));
  1210тЖТ  1210тЖТ  const lookTarget = useRef(new THREE.Vector3(cam.t[0], cam.t[1], cam.t[2]));
  1211тЖТ  1211тЖТ  targetPos.current.set(cam.p[0], cam.p[1], cam.p[2]);
  1212тЖТ  1212тЖТ  lookTarget.current.set(cam.t[0], cam.t[1], cam.t[2]);
  1213тЖТ  1213тЖТ  useFrame((_, delta) => {
  1214тЖТ  1214тЖТ    const g = groupRef.current;
  1215тЖТ  1215тЖТ    if (!g) return;
  1216тЖТ  1216тЖТ    // Frame-rate-independent smoothing (~90 ms time constant); teleport when
  1217тЖТ  1217тЖТ    // the peer jumps far so the glyph never streaks across the whole scene.
  1218тЖТ  1218тЖТ    if (g.position.distanceToSquared(targetPos.current) > 400) {
  1219тЖТ  1219тЖТ      g.position.copy(targetPos.current);
  1220тЖТ  1220тЖТ      lookCur.current.copy(lookTarget.current);
  1221тЖТ  1221тЖТ    } else {
  1222тЖТ  1222тЖТ      const k = 1 - Math.exp(-delta * 11);
  1223тЖТ  1223тЖТ      g.position.lerp(targetPos.current, k);
  1224тЖТ  1224тЖТ      lookCur.current.lerp(lookTarget.current, k);
  1225тЖТ  1225тЖТ    }
  1226тЖТ  1226тЖТ    g.lookAt(lookCur.current);
  1227тЖТ  1227тЖТ  });
  1228тЖТ  1228тЖТ  return (
  1229тЖТ  1229тЖТ    <group ref={groupRef} position={[cam.p[0], cam.p[1], cam.p[2]]}>
  1230тЖТ  1230тЖТ      <lineSegments geometry={PEER_CAM_LINES} raycast={() => null}>
  1231тЖТ  1231тЖТ        <lineBasicMaterial color={state.color} transparent opacity={0.9} />
  1232тЖТ  1232тЖТ      </lineSegments>
  1233тЖТ  1233тЖТ      {/* Small solid eye so the camera reads at a distance. */}
  1234тЖТ  1234тЖТ      <mesh raycast={() => null}>
  1235тЖТ  1235тЖТ        <sphereGeometry args={[0.06, 10, 10]} />
  1236тЖТ  1236тЖТ        <meshBasicMaterial color={state.color} />
  1237тЖТ  1237тЖТ      </mesh>
  1238тЖТ  1238тЖТ      <Html position={[0, 0.55, 0]} center distanceFactor={14} occlude={false}>
  1239тЖТ  1239тЖТ        <div
  1240тЖТ  1240тЖТ          className="pointer-events-none w-max max-w-[140px] truncate rounded-full px-1.5 py-px text-[10px] font-medium leading-4 text-black/85 shadow"
  1241тЖТ  1241тЖТ          style={{ backgroundColor: state.color }}
  1242тЖТ  1242тЖТ        >
  1243тЖТ  1243тЖТ          {state.name}
  1244тЖТ  1244тЖТ        </div>
  1245тЖТ  1245тЖТ      </Html>
  1246тЖТ  1246тЖТ    </group>
  1247тЖТ  1247тЖТ  );
  1248тЖТ  1248тЖТ}

  1250тЖТ  1250тЖТfunction editHudText(

  1256тЖТ  1256тЖТ    edge: sel.edges.length / 2,

  1258тЖТ  1258тЖТ  };
  1259тЖТ  1259тЖТ  const n = counts[mode] ?? 0;
  1260тЖТ  1260тЖТ  if (n === 0) {
  1261тЖТ  1261тЖТ    const noun = mode === 'vertex' ? 'vertex' : mode;
  1262тЖТ  1262тЖТ    return `Click a ${noun} to select it ┬╖ Shift+Click adds ┬╖ 1/2/3 switch vert/edge/face`;
  1263тЖТ  1263тЖТ  }
  1264тЖТ  1264тЖТ  const plural = mode === 'vertex' ? 'vertices' : `${mode}s`;
  1265тЖТ  1265тЖТ  const verb = mode === 'face' ? ' ┬╖ E extrudes ┬╖ G moves' : ' ┬╖ G moves';
  1266тЖТ  1266тЖТ  return `${n} ${n === 1 ? (mode === 'vertex' ? 'vertex' : mode) : plural} selected${verb}`;
  1267тЖТ  1267тЖТ}

  1269тЖТ  1269тЖТ/** Blender-style numpad view presets: 1 front, 3 right, 7 top, 0 camera. */
  1270тЖТ  1270тЖТfunction ViewPresets({
  1271тЖТ  1271тЖТ  snapshot,
  1272тЖТ  1272тЖТ  lookThroughRef,
  1273тЖТ  1273тЖТ  cameraFollowRef,
  1274тЖТ  1274тЖТ}: {
  1275тЖТ  1275тЖТ  snapshot: SceneSnapshot;
  1276тЖТ  1276тЖТ  lookThroughRef: React.MutableRefObject<() => boolean>;
  1277тЖТ  1277тЖТ  cameraFollowRef?: React.MutableRefObject<(t: number) => boolean>;
  1278тЖТ  1278тЖТ}) {
  1279тЖТ  1279тЖТ  const { camera, controls } = useThree();
  1280тЖТ  1280тЖТ  const snapRef = useRef(snapshot);
  1281тЖТ  1281тЖТ  snapRef.current = snapshot;
  1282тЖТ  1282тЖТ
  1283тЖТ  1283тЖТ  // Point the R3F camera at the scene camera's pose sampled at time `t`, so
  1284тЖТ  1284тЖТ  // an animation render follows keyframed camera moves. Returns false when
  1285тЖТ  1285тЖТ  // there's no scene camera.
  1286тЖТ  1286тЖТ  useEffect(() => {
  1287тЖТ  1287тЖТ    if (!cameraFollowRef) return;
  1288тЖТ  1288тЖТ    cameraFollowRef.current = (t: number) => {
  1289тЖТ  1289тЖТ      const sel = useScene3DStore.getState().selection;
  1290тЖТ  1290тЖТ      const cams = snapRef.current.objects.filter((o) => o.type === 'camera');
  1291тЖТ  1291тЖТ      const cam = cams.find((o) => sel.includes(o.id)) ?? cams[0];
  1292тЖТ  1292тЖТ      if (!cam) return false;
  1293тЖТ  1293тЖТ      const tr = (cam.anim?.length ? sampleAnim(cam.anim, t) : null) ?? cam.transform;
  1294тЖТ  1294тЖТ      const dir = new THREE.Vector3(0, -1, 0).applyEuler(
  1295тЖТ  1295тЖТ        new THREE.Euler(tr.rotation.x, tr.rotation.y, tr.rotation.z),
  1296тЖТ  1296тЖТ      );
  1297тЖТ  1297тЖТ      camera.position.set(tr.position.x, tr.position.y, tr.position.z);
  1298тЖТ  1298тЖТ      const persp = camera as THREE.PerspectiveCamera;
  1299тЖТ  1299тЖТ      persp.fov = cam.camera?.fov ?? 50;
  1300тЖТ  1300тЖТ      persp.updateProjectionMatrix();
  1301тЖТ  1301тЖТ      camera.lookAt(new THREE.Vector3().copy(camera.position).addScaledVector(dir, 8));
  1302тЖТ  1302тЖТ      return true;
  1303тЖТ  1303тЖТ    };
  1304тЖТ  1304тЖТ  }, [camera, cameraFollowRef]);
  1305тЖТ  1305тЖТ
  1306тЖТ  1306тЖТ  // Jump the viewport into the selected (or first) scene camera. Shared by
  1307тЖТ  1307тЖТ  // Numpad 0 and the render button. Returns false when the scene has none.
  1308тЖТ  1308тЖТ  useEffect(() => {
  1309тЖТ  1309тЖТ    lookThroughRef.current = () => {
  1310тЖТ  1310тЖТ      const sel = useScene3DStore.getState().selection;
  1311тЖТ  1311тЖТ      const cams = snapRef.current.objects.filter((o) => o.type === 'camera');
  1312тЖТ  1312тЖТ      const cam = cams.find((o) => sel.includes(o.id)) ?? cams[0];
  1313тЖТ  1313тЖТ      if (!cam) return false;
  1314тЖТ  1314тЖТ      const t = cam.transform;
  1315тЖТ  1315тЖТ      const dir = new THREE.Vector3(0, -1, 0).applyEuler(
  1316тЖТ  1316тЖТ        new THREE.Euler(t.rotation.x, t.rotation.y, t.rotation.z),
  1317тЖТ  1317тЖТ      );
  1318тЖТ  1318тЖТ      camera.position.set(t.position.x, t.position.y, t.position.z);
  1319тЖТ  1319тЖТ      const persp = camera as THREE.PerspectiveCamera;
  1320тЖТ  1320тЖТ      persp.fov = cam.camera?.fov ?? 50;
  1321тЖТ  1321тЖТ      persp.updateProjectionMatrix();
  1322тЖТ  1322тЖТ      (controls as { target?: THREE.Vector3 } | null)?.target?.copy(
  1323тЖТ  1323тЖТ        new THREE.Vector3().copy(camera.position).addScaledVector(dir, 8),
  1324тЖТ  1324тЖТ      );
  1325тЖТ  1325тЖТ      (controls as { update?: () => void } | null)?.update?.();
  1326тЖТ  1326тЖТ      // Enter "camera view": the glyph hides and a frame HUD shows until the
  1327тЖТ  1327тЖТ      // user orbits/flies away (Blender's Numpad 0 behavior).
  1328тЖТ  1328тЖТ      useScene3DStore.getState().setViewingCamera(cam.id);
  1329тЖТ  1329тЖТ      return true;
  1330тЖТ  1330тЖТ    };
  1331тЖТ  1331тЖТ  }, [camera, controls, lookThroughRef]);
  1332тЖТ  1332тЖТ
  1333тЖТ  1333тЖТ  useEffect(() => {
  1334тЖТ  1334тЖТ    const onKey = (e: KeyboardEvent) => {
  1335тЖТ  1335тЖТ      const target = e.target as HTMLElement;
  1336тЖТ  1336тЖТ      if (target && /^(INPUT|TEXTAREA|SELECT)$/.test(target.tagName)) return;
  1337тЖТ  1337тЖТ      if (useScene3DStore.getState().flying) return;
  1338тЖТ  1338тЖТ      if (e.code === 'Numpad0') {
  1339тЖТ  1339тЖТ        if (lookThroughRef.current()) e.preventDefault();
  1340тЖТ  1340тЖТ        return;
  1341тЖТ  1341тЖТ      }
  1342тЖТ  1342тЖТ      let dir: THREE.Vector3 | null = null;
  1343тЖТ  1343тЖТ      if (e.code === 'Numpad1') dir = new THREE.Vector3(0, 0, 1);
  1344тЖТ  1344тЖТ      else if (e.code === 'Numpad3') dir = new THREE.Vector3(1, 0, 0);
  1345тЖТ  1345тЖТ      else if (e.code === 'Numpad7') dir = new THREE.Vector3(0.0001, 1, 0.0001);
  1346тЖТ  1346тЖТ      if (!dir) return;
  1347тЖТ  1347тЖТ      // Ctrl+Numpad flips to the opposite view (back/left/bottom), as in Blender.
  1348тЖТ  1348тЖТ      if (e.ctrlKey) dir.negate();
  1349тЖТ  1349тЖТ      e.preventDefault();
  1350тЖТ  1350тЖТ      const t = (controls as { target?: THREE.Vector3 } | null)?.target ?? new THREE.Vector3();
  1351тЖТ  1351тЖТ      const dist = camera.position.distanceTo(t) || 8;
  1352тЖТ  1352тЖТ      camera.position.copy(t.clone().add(dir.normalize().multiplyScalar(dist)));
  1353тЖТ  1353тЖТ      (controls as { update?: () => void } | null)?.update?.();
  1354тЖТ  1354тЖТ    };
  1355тЖТ  1355тЖТ    window.addEventListener('keydown', onKey);
  1356тЖТ  1356тЖТ    return () => window.removeEventListener('keydown', onKey);
  1357тЖТ  1357тЖТ  }, [camera, controls]);
  1358тЖТ  1358тЖТ  return null;
  1359тЖТ  1359тЖТ}

  1361тЖТ  1361тЖТ/**
  1362тЖТ  1362тЖТ * Tiny in-canvas component that captures `useThree` and exposes a callback
  1363тЖТ  1363тЖТ * to frame the current selection. Kept inside <Canvas> so it has access to
  1364тЖТ  1364тЖТ * the camera and controls.
  1365тЖТ  1365тЖТ */
  1366тЖТ  1366тЖТfunction FrameSelectedBinding({
  1367тЖТ  1367тЖТ  frameRef,
  1368тЖТ  1368тЖТ  snapshot,
  1369тЖТ  1369тЖТ  selection,
  1370тЖТ  1370тЖТ}: {
  1371тЖТ  1371тЖТ  frameRef: React.MutableRefObject<(all: boolean) => void>;
  1372тЖТ  1372тЖТ  snapshot: SceneSnapshot;
  1373тЖТ  1373тЖТ  selection: string[];
  1374тЖТ  1374тЖТ}) {
  1375тЖТ  1375тЖТ  const { camera, controls } = useThree();
  1376тЖТ  1376тЖТ  useFrame(() => {});
  1377тЖТ  1377тЖТ  useEffect(() => {
  1378тЖТ  1378тЖТ    frameRef.current = (all: boolean) => {
  1379тЖТ  1379тЖТ      // Frame the selection; with `all` (or nothing selected), frame the scene.
  1380тЖТ  1380тЖТ      const ids = !all && selection.length > 0 ? selection : snapshot.objects.map((o) => o.id);
  1381тЖТ  1381тЖТ      const positions: THREE.Vector3[] = [];
  1382тЖТ  1382тЖТ      for (const id of ids) {
  1383тЖТ  1383тЖТ        const obj = snapshot.objects.find((o) => o.id === id);
  1384тЖТ  1384тЖТ        if (!obj) continue;
  1385тЖТ  1385тЖТ        positions.push(
  1386тЖТ  1386тЖТ          new THREE.Vector3(obj.transform.position.x, obj.transform.position.y, obj.transform.position.z),
  1387тЖТ  1387тЖТ        );
  1388тЖТ  1388тЖТ      }
  1389тЖТ  1389тЖТ      if (positions.length === 0) return;
  1390тЖТ  1390тЖТ      const box = new THREE.Box3().setFromPoints(positions);
  1391тЖТ  1391тЖТ      const size = new THREE.Vector3();
  1392тЖТ  1392тЖТ      box.getSize(size);
  1393тЖТ  1393тЖТ      const center = new THREE.Vector3();
  1394тЖТ  1394тЖТ      box.getCenter(center);
  1395тЖТ  1395тЖТ      const radius = Math.max(size.length() / 2, 1) * 1.5;
  1396тЖТ  1396тЖТ      const dir = new THREE.Vector3()
  1397тЖТ  1397тЖТ        .subVectors(camera.position, (controls as { target?: THREE.Vector3 })?.target ?? new THREE.Vector3())
  1398тЖТ  1398тЖТ        .normalize();
  1399тЖТ  1399тЖТ      camera.position.copy(center.clone().add(dir.multiplyScalar(radius)));
  1400тЖТ  1400тЖТ      (controls as { target?: THREE.Vector3 } | null)?.target?.copy(center);
  1401тЖТ  1401тЖТ      (controls as { update?: () => void } | null)?.update?.();
  1402тЖТ  1402тЖТ    };
  1403тЖТ  1403тЖТ  }, [camera, controls, snapshot, selection, frameRef]);
  1404тЖТ  1404тЖТ  return null;
  1405тЖТ  1405тЖТ}
