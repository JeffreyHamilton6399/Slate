    60→/** Nearest start position at which a clip of `duration` fits without
    61→ *  overlapping any blocker. Merges blockers into occupied intervals, then
    62→ *  picks the gap position closest to `desired`. This replaces the previous
    63→ *  "slide to the nearest side of the current blocker" iteration, which could
    64→ *  ping-pong between two adjacent blockers (each resolution landing on the
    65→ *  other) and exhaust its iteration budget while STILL overlapping —
    66→ *  committing a clip on top of another one. Scanning gaps directly is
    67→ *  deterministic: the open-ended gap after the last blocker always fits, so a
    68→ *  valid position always exists. */
    69→function nearestFreeStart(desired: number, duration: number, blockersIn: { start: number; end: number }[]): number {
    70→  const eps = 1e-4;
    71→  desired = Math.max(0, desired);
    72→  if (blockersIn.length === 0) return desired;
    73→  const sorted = [...blockersIn].sort((a, b) => a.start - b.start);
    74→  const occupied: { start: number; end: number }[] = [];
    75→  for (const b of sorted) {
    76→    const last = occupied[occupied.length - 1];
    77→    if (last && b.start <= last.end + eps) last.end = Math.max(last.end, b.end);
    78→    else occupied.push({ start: b.start, end: b.end });
    79→  }
    80→  let best = desired;
    81→  let bestDist = Infinity;
    82→  const consider = (lo: number, hi: number) => {
    83→    if (hi - lo < duration - eps) return; // gap too small
    84→    const cand = Math.min(Math.max(desired, lo), hi - duration);
    85→    const dist = Math.abs(cand - desired);
    86→    if (dist < bestDist) { bestDist = dist; best = cand; }
    87→  };
    88→  let prevEnd = 0;
    89→  for (const o of occupied) {
    90→    consider(prevEnd, o.start);
    91→    prevEnd = Math.max(prevEnd, o.end);
    92→  }
    93→  consider(prevEnd, Infinity); // open-ended tail — always fits
    94→  return Math.max(0, best);
    95→}

    97→// ── Waveform cache: pre-computed PNG data URLs ───────────────────────────────
    98→// Key: `${clipId}:${sampleCount}:${width}` → data URL
    99→const waveformPNGCache = new Map<string, string>();
   100→
   101→// Draw only the window [startFrame, endFrame) of the sample across `width`
   102→// pixels. Rendering the *window* (not the whole sample stretched) is what makes
   103→// trimming look like cutting the audio away rather than squashing it.
   104→function computeWaveformPNG(samples: Float32Array, channels: number, width: number, color: string, height: number, startFrame: number, endFrame: number): string {
   105→  const canvas = document.createElement('canvas');
   106→  const dpr = window.devicePixelRatio || 1;
   107→  const w = Math.max(2, Math.floor(width));
   108→  canvas.width = w * dpr;
   109→  canvas.height = height * dpr;
   110→  const ctx = canvas.getContext('2d')!;
   111→  ctx.scale(dpr, dpr);
   112→  ctx.clearRect(0, 0, w, height);
   113→  const totalFrames = samples.length / channels;
   114→  const s = Math.max(0, Math.min(totalFrames, Math.floor(startFrame)));
   115→  const e = Math.max(s, Math.min(totalFrames, Math.floor(endFrame)));
   116→  const span = e - s;
   117→  if (span <= 0) return canvas.toDataURL();
   118→  const mid = height / 2;
   119→  ctx.fillStyle = color;
   120→  for (let x = 0; x < w; x++) {
   121→    const f0 = s + Math.floor((x / w) * span);
   122→    const f1 = Math.min(e, s + Math.floor(((x + 1) / w) * span) + 1);
   123→    let peak = 0;
   124→    for (let i = f0; i < f1; i++) { const v = Math.abs(samples[i * channels] ?? 0); if (v > peak) peak = v; }
   125→    const bh = Math.max(1, peak * mid * 0.85);
   126→    ctx.fillRect(x, mid - bh, 1, bh * 2);
   127→  }
   128→  return canvas.toDataURL();
   129→}

   131→/** Waveform image for the buffer window the clip actually plays. With speed s,
   132→ *  a clip of `duration` timeline seconds consumes `duration*s` buffer seconds.
   133→ *
   134→ *  Two robustness fixes:
   135→ *  1. If `loadSamples` returns an EMPTY Float32Array (length 0) — which happens
   136→ *     when a freshly-split/created clip's IndexedDB write hasn't landed yet —
   137→ *     we DON'T cache the resulting blank PNG. Instead we show the `···`
   138→ *     placeholder and retry a few times (500ms apart) so the real waveform
   139→ *     appears once the samples are available.
   140→ *  2. We listen for `slate:audio-clip-changed` (fired by Normalize/Reverse/split)
   141→ *     and invalidate our cache entry for the current clipId, then force a
   142→ *     recompute via a `bust` counter — otherwise the memoised component would
   143→ *     keep showing the stale PNG even after the cache was cleared. */
   144→const WaveformImg = memo(function WaveformImg({ clipId, sampleKey, channels, sampleRate, offset, duration, speed, width, color }: {
   145→  clipId: string; sampleKey: string; channels: number; sampleRate: number;
   146→  offset: number; duration: number; speed: number; width: number; color: string;
   147→}) {
   148→  const height = TRACK_H - 6;
   149→  const [imgUrl, setImgUrl] = useState<string>('');
   150→  const [bust, setBust] = useState(0);
   151→  const retryRef = useRef(0);
   152→
   153→  // Invalidate cached PNGs for this clip when its samples change (normalize /
   154→  // reverse / split). The parent AudioEditor also invalidates + bumps version,
   155→  // but the memoised WaveformImg wouldn't otherwise recompute (its primitive
   156→  // props are unchanged) — the `bust` counter forces the load effect to re-run.
   157→  useEffect(() => {
   158→    const onChanged = (e: Event) => {
   159→      const detail = (e as CustomEvent<string>).detail;
   160→      // Match by clipId (normalize/reverse/split events) OR by sampleKey
   161→      // (multiplayer sample-arrival events from registerSampleSyncMap).
   162→      if (detail !== clipId && detail !== sampleKey) return;
   163→      for (const key of waveformPNGCache.keys()) {
   164→        if (key.startsWith(`${clipId}:`)) waveformPNGCache.delete(key);
   165→      }
   166→      retryRef.current = 0;
   167→      setBust((n) => n + 1);
   168→    };
   169→    window.addEventListener('slate:audio-clip-changed', onChanged as EventListener);
   170→    return () => window.removeEventListener('slate:audio-clip-changed', onChanged as EventListener);
   171→  }, [clipId, sampleKey]);
   172→
   173→  useEffect(() => {
   174→    const startFrame = Math.round(offset * sampleRate);
   175→    const endFrame = Math.round((offset + duration * speed) * sampleRate);
   176→    const cacheKey = `${clipId}:${sampleKey}:${startFrame}:${endFrame}:${Math.floor(width)}`;
   177→    const cached = waveformPNGCache.get(cacheKey);
   178→    if (cached) { setImgUrl(cached); return; }
   179→    let cancelled = false;
   180→    let retryTimer: ReturnType<typeof setTimeout> | null = null;
   181→
   182→    const attempt = () => {
   183→      if (cancelled) return;
   184→      void loadSamples(sampleKey).then((samples) => {
   185→        if (cancelled) return;
   186→        // Empty samples = the IndexedDB write for this sampleKey hasn't landed
   187→        // yet (e.g. brand-new clip from a split, OR a remote peer's clip whose
   188→        // sample blob is still in flight via the Yjs sync map). DON'T cache a
   189→        // blank PNG — retry for up to 10s (20 × 500ms) so the real waveform
   190→        // appears once the samples arrive. The previous 5-retry / 2.5s budget
   191→        // was too short for cross-peer sample sync, which can take several
   192→        // seconds for a 5MB blob over a slow link. The `slate:audio-clip-changed`
   193→        // event (dispatched by onClipsAdded when a remote clip is added, and by
   194→        // registerSampleSyncMap when samples arrive) resets `retryRef` to 0 so
   195→        // the WaveformImg re-tries immediately on the signal — this retry
   196→        // budget is the safety net for the case where the event fires before
   197→        // the WaveformImg mounts, or doesn't fire at all.
   198→        if (samples.length === 0) {
   199→          if (retryRef.current < 20) {
   200→            retryRef.current += 1;
   201→            retryTimer = setTimeout(attempt, 500);
   202→          }
   203→          return;
   204→        }
   205→        retryRef.current = 0;
   206→        const url = computeWaveformPNG(samples, channels, width, color, height, startFrame, endFrame);
   207→        waveformPNGCache.set(cacheKey, url);
   208→        setImgUrl(url);
   209→      });
   210→    };
   211→    attempt();
   212→
   213→    return () => {
   214→      cancelled = true;
   215→      if (retryTimer) clearTimeout(retryTimer);
   216→    };
   217→  }, [clipId, sampleKey, channels, sampleRate, offset, duration, speed, width, color, height, bust]);
   218→
   219→  if (!imgUrl) return <div className="flex h-full items-center justify-center text-[7px] text-text-dim">···</div>;
   220→  return <img src={imgUrl} alt="" className="pointer-events-none h-full w-full" style={{ objectFit: 'fill' }} />;
   221→});
   222→
   223→/** Drop cached waveform PNGs for a clip (call when its samples change). */
   224→function invalidateWaveform(clipId: string): void {
   225→  for (const key of waveformPNGCache.keys()) if (key.startsWith(`${clipId}:`)) waveformPNGCache.delete(key);
   226→}

   328→  /** Schedule a debounced restartPlayback (500ms). Called when the clip set
   329→   *  changes mid-playback (remote peer adds a clip) or when a clip's samples
   330→   *  arrive after play started (remote sample blob landed via the sync map).
   331→   *  The debounce coalesces rapid bursts so we don't tear down and rebuild

   334→  const scheduleRestart = useCallback(() => {
   335→    if (!playingRef.current) return;
   336→    if (restartTimerRef.current) clearTimeout(restartTimerRef.current);

   354→          // excluded so they don't trigger a re-render.

   358→          const nextKey = states.map(key).sort().join(';');

   360→        });
   361→      }),

   363→  );

   373→
   374→  // Clear our audio playhead on unmount so other peers stop seeing us in the
   375→  // audio editor. (audio: null = "not in the audio editor".)

   381→
   382→  // Yjs subscription.
   383→  useEffect(() => {
   384→    // NOTE: registerSampleSyncMap(room) is now called from useSlateRoom's
   385→    // attach() — registering here only on AudioEditor mount meant a peer on
   386→    // a 2D/3D board (audio panel closed) never registered the sync map and
   387→    // never received remote sample blobs. See useSlateRoom.ts.
   388→    const tracks = slate.audioTracks();
   389→    const clips = slate.audioClips();

   392→    const bump = () => { if (pending) return; pending = true; requestAnimationFrame(() => { pending = false; setVersion((v) => v + 1); }); };
   393→    // Any track edit (volume, pan, EQ, sends, mute/solo — local knob or remote
   394→    // peer) re-applies the Yjs values to the live audio graph, so the settings
   395→    // panel is audible mid-playback without a restart.
   396→    const applyTracks = () => { engineRef.current?.updateTracks(slateRef.current); };
   397→    // Clip edits (position, trim, gain, pan, filters, speed/pitch — ours or a
   398→    // peer's) reschedule playback so they're audible mid-play. Debounced 500ms
   399→    // in scheduleRestart and a no-op while paused, so knob drags coalesce into
   400→    // one restart instead of stuttering. Track edits deliberately DON'T
   401→    // restart — applyTracks adjusts their nodes live.
   402→    const restartOnClipEdit = () => { scheduleRestart(); };
   403→    tracks.observeDeep(bump); clips.observeDeep(bump); audioMap.observe(bump); bump();
   404→    tracks.observeDeep(applyTracks);
   405→    clips.observeDeep(restartOnClipEdit);
   406→    // Detect NEW clips being added while playing and schedule a debounced
   407→    // restartPlayback so the new clip is picked up mid-playback without a
   408→    // manual stop/play. We watch the shallow clips map (not observeDeep) so
   409→    // we only react to key additions, not to every property edit on an
   410→    // existing clip (which would restart playback every time a peer nudged
   411→    // a clip's volume, fighting the user). The restart is debounced 500ms so
   412→    // a burst of additions (e.g. a peer imports 5 files) coalesces into one.
   413→    //
   414→    // LIVE WAVEFORM REFRESH: when a remote peer adds a clip, its metadata
   415→    // arrives via Yjs immediately but the sample blob travels separately via
   416→    // the audioSampleSync Y.Map (see sampleStore.ts). If we don't ping the
   417→    // WaveformImg, it might have already exhausted its empty-samples retry
   418→    // budget (now 10s, but a slow link can still exceed that) and given up
   419→    // showing a blank "···" placeholder until the user refreshes. So for
   420→    // each newly-added clip we:
   421→    //   1. clearCache(clipId) on the engine so the next play re-loads samples
   422→    //      (the cache might hold a null/empty entry from a prior failed getBuffer)
   423→    //   2. dispatch `slate:audio-clip-changed` with the clipId — the
   424→    //      AudioEditor's listener (below) re-invalidates the waveform cache +

   441→      }
   442→      scheduleRestart();

   445→    const lateRead = setTimeout(bump, 200);

   494→  // Ctrl+scroll zoom — centred on the playhead so the screen position under
   495→  // the cursor/playhead stays put instead of zooming toward the left edge.
   496→  /** Zoom while keeping one timeline point at the same screen offset.
   497→   *  The anchor is the time under `anchorClientX` (mouse-wheel zoom: the spot
   498→   *  under the cursor stays put — how every DAW/map zoom feels), or the
   499→   *  playhead when no anchor is given (toolbar buttons, fit-to-window).
   500→   *  Keeping time t fixed on screen means newScroll = oldScroll + t·Δpps,
   501→   *  since screenX(t) = timelineLeft − scrollLeft + t·pps. The `scrollLeft`
   502→   *  write happens in the layout effect below (after React commits the new

   519→  }, []);

   541→  // Global pointermove/up — pure DOM, zero React state.
   542→  // rAF-throttled: pointermove fires at the hardware's report rate (often
   543→  // 60-120Hz), and the move handler does a neighbours-overlap scan on every
   544→  // event. Coalescing multiple moves into a single rAF callback caps the work
   545→  // at the display refresh rate (~60fps) so a fast pointer drag never stalls
   546→  // the main thread. A no-op skip (cursor hasn't moved since the last
   547→  // processed frame) avoids redundant DOM writes for stationary pointers.
   548→  useEffect(() => {
   549→    /** Last pointer position we actually applied to the DOM. Lets us skip
   550→     *  work when pointermove fires without real movement (e.g. a touchpad
   551→     *  reporting pressure changes, or sub-pixel jitter). */
   552→    let lastProcessedX: number | null = null;
   553→    let lastProcessedY: number | null = null;
   554→
   555→    /** Magnet: pull `t` (a clip START time) to the nearest beat line, other
   556→     *  clip edge, or the playhead — considering BOTH the clip's start and end
   557→     *  edges — if one is within SNAP_PX at the current zoom. `dur` null =
   558→     *  only the start edge matters (trims). Returns the adjusted start. */
   559→    const snapStart = (t: number, dur: number | null, snapTimes: number[]): number => {
   560→      const pps = pxRef.current;
   561→      const thr = SNAP_PX / pps;
   562→      const beat = 60 / bpmRef.current;
   563→      let best = t;
   564→      let bestDist = thr;
   565→      const consider = (target: number, edgeOffset: number) => {
   566→        const dist = Math.abs(t + edgeOffset - target);
   567→        if (dist < bestDist) { bestDist = dist; best = target - edgeOffset; }
   568→      };
   569→      consider(Math.round(t / beat) * beat, 0);
   570→      if (dur !== null) consider(Math.round((t + dur) / beat) * beat, dur);
   571→      for (const st of snapTimes) {
   572→        consider(st, 0);
   573→        if (dur !== null) consider(st, dur);
   574→      }
   575→      return best;
   576→    };
   577→
   578→    const applyMove = () => {
   579→      moveRafRef.current = 0;
   580→      const d = dragRef.current;
   581→      if (!d) return;
   582→      const clientX = moveXRef.current;
   583→      if (clientX === null) return;
   584→      const clientY = moveYRef.current;
   585→      // Skip no-op moves — cursor hasn't moved to a new pixel since the last
   586→      // frame we processed. Saves an overlap scan + 1-2 style writes.
   587→      if (lastProcessedX === clientX && lastProcessedY === clientY) return;
   588→      lastProcessedX = clientX;
   589→      lastProcessedY = clientY;
   590→
   591→      // Dead zone: a plain click (pointer never leaves a 3px box) must not
   592→      // nudge the clip. Once exceeded, the drag is live for good.
   593→      if (!d.moved) {
   594→        if (Math.abs(clientX - d.sx) < DRAG_DEADZONE_PX && Math.abs(clientY - d.sy) < DRAG_DEADZONE_PX) return;
   595→        d.moved = true;
   596→        if (d.mode === 'drag') d.el.style.zIndex = '40'; // ride above other rows while crossing tracks
   597→      }
   598→
   599→      const pps = pxRef.current;
   600→      const dt = (clientX - d.sx) / pps;
   601→      const oldEnd = d.os + d.od;
   602→      const wantSnap = snapRef.current && !moveAltRef.current;
   603→      if (d.mode === 'drag') {
   604→        // The clip follows the mouse 1:1 (snapped to the magnet grid). It's
   605→        // allowed to ride OVER other clips while the drag is live — it's
   606→        // elevated to z-40 so the overlap reads as "in transit", and the
   607→        // no-overlap invariant is enforced once, on drop (see onUp). The
   608→        // previous mid-drag resolution parked the clip flush against a
   609→        // blocker and then teleported it across when the cursor passed the
   610→        // blocker's midpoint, which made dragging feel sticky and jumpy.
   611→        const duration = d.od;
   612→        // Vertical: whole-row steps move the clip across tracks.
   613→        const rowDelta = Math.max(
   614→          -d.trackIndex,
   615→          Math.min(d.trackIds.length - 1 - d.trackIndex, Math.round((clientY - d.sy) / TRACK_H)),
   616→        );
   617→        d.rowDelta = rowDelta;
   618→        d.el.style.transform = rowDelta === 0 ? '' : `translateY(${rowDelta * TRACK_H}px)`;
   619→
   620→        const desired = wantSnap ? snapStart(d.os + dt, duration, d.snapTimes) : d.os + dt;
   621→        d.el.style.left = `${Math.max(0, desired) * pps}px`;
   622→      } else if (d.mode === 'trimL') {
   623→        // Cut from the left: never past the left neighbour, the source start
   624→        // (offset ≥ 0 → limited by the trimmed head in timeline seconds), or a
   625→        // minimum width.
   626→        const minStart = Math.max(d.leftLimit, d.os - d.oo / d.speed);
   627→        const desired = wantSnap ? snapStart(d.os + dt, null, d.snapTimes) : d.os + dt;
   628→        const start = Math.min(oldEnd - 0.1, Math.max(minStart, desired));
   629→        d.el.style.left = `${start * pps}px`;
   630→        d.el.style.width = `${(oldEnd - start) * pps}px`;
   631→        // Shift the (fixed-width) waveform the opposite way so the audio stays
   632→        // anchored — the newly-trimmed head is clipped, not squashed.
   633→        if (d.waveEl) d.waveEl.style.left = `${-(start - d.os) * pps}px`;
   634→      } else if (d.mode === 'trimR') {
   635→        // Cut from the right: never past the right neighbour or a min width.
   636→        const desired = wantSnap ? snapStart(oldEnd + dt, null, d.snapTimes) : oldEnd + dt;
   637→        const end = Math.min(d.rightLimit, Math.max(d.os + 0.1, desired));
   638→        d.el.style.width = `${(end - d.os) * pps}px`;
   639→      }
   640→    };
   641→
   642→    const onMove = (ev: PointerEvent) => {
   643→      const d = dragRef.current; if (!d) return;
   644→      // Stash the latest pointer state — the rAF callback reads it. Cheap
   645→      // (ref writes) so even 120Hz pointermove streams don't pile up work.
   646→      moveXRef.current = ev.clientX;
   647→      moveYRef.current = ev.clientY;
   648→      moveAltRef.current = ev.altKey;
   649→      // Schedule a frame if one isn't already pending. The guard is what
   650→      // coalesces multiple moves within the same frame into one DOM write.
   651→      if (!moveRafRef.current) {
   652→        moveRafRef.current = requestAnimationFrame(applyMove);
   653→      }
   654→    };
   655→    const onUp = () => {
   656→      const d = dragRef.current; if (!d) return;
   657→      // Cancel any pending rAF — we'll process the final position
   658→      // SYNCHRONOUSLY here so the committed Yjs value reflects the exact
   659→      // pointer-up location (otherwise the last pointermove before pointerup
   660→      // could be dropped, leaving the committed position ~1 frame behind the
   661→      // cursor). lastProcessed is reset to bypass the no-op skip in case the
   662→      // final pointermove matched the prior processed frame but we still need
   663→      // to read the just-written style to commit to Yjs.
   664→      if (moveRafRef.current) {
   665→        cancelAnimationFrame(moveRafRef.current);
   666→        moveRafRef.current = 0;
   667→      }
   668→      if (moveXRef.current !== null) {
   669→        lastProcessedX = null; // bypass the no-op skip on the final flush
   670→        lastProcessedY = null;
   671→        applyMove();
   672→      }
   673→      // Undo the drag-only visual state React doesn't manage.
   674→      d.el.style.zIndex = '';
   675→      d.el.style.transform = '';
   676→      // A click that never left the dead zone selects but must not write —
   677→      // committing a parseFloat round-trip of the style would drift the clip.
   678→      if (d.moved) {
   679→        const pps = pxRef.current;
   680→        const left = parseFloat(d.el.style.left) / pps;
   681→        const width = parseFloat(d.el.style.width) / pps;
   682→        if (d.mode === 'drag') {
   683→          const targetTrackId = d.trackIds[d.trackIndex + d.rowDelta] ?? d.trackIds[d.trackIndex]!;
   684→          const originTrackId = d.trackIds[d.trackIndex]!;
   685→          // Enforce the no-overlap invariant HERE (the drag itself follows
   686→          // the mouse freely): place the clip in the free gap nearest to the
   687→          // dropped position.
   688→          const start = nearestFreeStart(left, d.od, d.byTrack.get(targetTrackId) ?? []);
   689→          // Write the resolved position to the DOM too so there's no flicker
   690→          // between pointer-up and the Yjs-driven re-render.
   691→          d.el.style.left = `${start * pps}px`;
   692→          updateAudioClip(slate, d.clipId, {
   693→            start,
   694→            ...(targetTrackId !== originTrackId ? { trackId: targetTrackId } : {}),
   695→          });
   696→        }
   697→        else if (d.mode === 'trimL') updateAudioClip(slate, d.clipId, { start: left, duration: width, offset: Math.max(0, d.oo + (left - d.os) * d.speed) });
   698→        else if (d.mode === 'trimR') updateAudioClip(slate, d.clipId, { duration: width });
   699→      }
   700→      moveXRef.current = null;
   701→      lastProcessedX = null;
   702→      lastProcessedY = null;
   703→      dragRef.current = null;
   704→    };
   705→    window.addEventListener('pointermove', onMove);
   706→    window.addEventListener('pointerup', onUp);
   707→    return () => {
   708→      window.removeEventListener('pointermove', onMove);
   709→      window.removeEventListener('pointerup', onUp);
   710→      // Don't leak a pending rAF if the component unmounts mid-drag.
   711→      if (moveRafRef.current) {
   712→        cancelAnimationFrame(moveRafRef.current);
   713→        moveRafRef.current = 0;
   714→      }
   715→    };
   716→  }, [slate]);

   718→  // Hotkeys. Gated: on an audio board the editor always owns them; on a 2D/3D
   719→  // board (where the editor is one docked panel among many) they only fire

   721→  // animation timeline (both played), and C/D/R fired while drawing.
   722→  useEffect(() => {
   723→    const onKey = (e: KeyboardEvent) => {
   724→      const t = e.target as HTMLElement;
   725→      if (t && /^(INPUT|TEXTAREA|SELECT)$/.test(t.tagName)) return;

   738→      else if (k === 'd' && !e.ctrlKey && selectedRef.current.size > 0) { e.preventDefault(); selectedRef.current.forEach(id => void dupClip(id)); }

   771→  // drop the cached waveform + decoded buffer so both refresh, AND — if we're
   772→  // currently playing — schedule a debounced restartPlayback so the clip is

   802→    window.addEventListener('slate:audio-sync-skipped', onSkipped);

   805→      window.removeEventListener('slate:audio-sync-skipped', onSkipped);

   827→  const togglePlay = useCallback(() => {

   830→    else {
   831→      if (looping) eng.setLoopRegion(loopStart, loopEnd); else eng.setLoopRegion(null, null);
   832→      // Best-effort: if the AudioContext is suspended (autoplay policy /

  1294→const TrackHeader = memo(function TrackHeader({ track, hasSolo, slate, engineRef }: {
  1295→  track: AudioTrack; hasSolo: boolean;
  1296→  slate: ReturnType<typeof useRoom>['slate'];
  1297→  engineRef: React.RefObject<AudioEngine | null>;
  1298→}) {
  1299→  const [vol, setVol] = useState(track.volume);
  1300→  const [pan, setPan] = useState(track.pan);
  1301→  // Separate drag flags per slider — the previous single shared `isDraggingRef`
  1302→  // meant starting a volume drag also blocked the pan prop-sync effect (and
  1303→  // vice versa), which could cause one slider's committed Yjs value to be
  1304→  // momentarily ignored by the other's gating. Per-slider flags keep each
  1305→  // slider's prop-sync independent.
  1306→  const isDraggingVolRef = useRef(false);
  1307→  const isDraggingPanRef = useRef(false);
  1308→  useEffect(() => { if (!isDraggingVolRef.current) setVol(track.volume); }, [track.volume]);
  1309→  useEffect(() => { if (!isDraggingPanRef.current) setPan(track.pan); }, [track.pan]);
  1310→
  1311→  // Audibility — mirrors engine.setupTrackNodes so the live gain we write
  1312→  // during a volume drag respects mute/solo (don't briefly un-mute a muted
  1313→  // track just because the user is dragging its volume slider).
  1314→  const audible = hasSolo ? track.solo : !track.muted;
  1315→
  1316→  // `update` is for non-slider track edits (name/mute/solo/arm) — these DO
  1317→  // need a full updateTracks so the audio graph reflects the new state
  1318→  // (e.g. toggling solo rebalances every track's gain).
  1319→  const update = (patch: Partial<AudioTrack>) => { updateAudioTrack(slate, track.id, patch); engineRef.current?.updateTracks(slate); };
  1320→
  1321→  // Volume slider: update local state + the engine's gain node DIRECTLY for
  1322→  // immediate audio feedback. Do NOT call updateTracks (which re-reads every
  1323→  // track from Yjs and rebuilds the audio graph — far too expensive per
  1324→  // onChange event, and the source of the reported lag). Commit to Yjs once
  1325→  // on pointerup.
  1326→  const onVolDown = () => { isDraggingVolRef.current = true; };
  1327→  const onVol = (v: number) => { setVol(v); engineRef.current?.setTrackVolume(track.id, v, audible); };
  1328→  const onVolEnd = () => { isDraggingVolRef.current = false; updateAudioTrack(slate, track.id, { volume: vol }); };
  1329→
  1330→  // Pan slider: same pattern — set the panner node directly, commit on
  1331→  // pointerup. The pan slider is smaller than the volume slider (secondary
  1332→  // control) but still usable, with L/R labels so its function is obvious
  1333→  // (the user complaint was "two sliders, only one works" — partly because
  1334→  // neither slider was labelled, so it wasn't clear what each did).
  1335→  const onPanDown = () => { isDraggingPanRef.current = true; };
  1336→  const onPan = (p: number) => { setPan(p); engineRef.current?.setTrackPan(track.id, p); };
  1337→  const onPanEnd = () => { isDraggingPanRef.current = false; updateAudioTrack(slate, track.id, { pan: pan }); };
  1338→
  1339→  return (
  1340→    <div className="border-b border-border/15 px-2 py-1" style={{ height: TRACK_H }}>
  1341→      <div className="flex items-center gap-0.5">
  1342→        <span className="h-2 w-2 shrink-0 rounded-full" style={{ backgroundColor: track.color }} />
  1343→        <input type="text" value={track.name} onChange={(e) => update({ name: e.target.value })} className="min-w-0 flex-1 bg-transparent text-[11px] font-medium text-text outline-none" />
  1344→        <button onClick={() => update({ muted: !track.muted })} className={`flex h-4 w-4 items-center justify-center rounded ${track.muted && !hasSolo ? 'bg-warn/30 text-warn' : 'text-text-mid hover:bg-bg-3'}`} title="M">{track.muted ? <VolumeX size={9} /> : <Volume2 size={9} />}</button>
  1345→        <button onClick={() => update({ solo: !track.solo })} className={`flex h-4 w-4 items-center justify-center rounded ${track.solo ? 'bg-accent/30 text-accent' : 'text-text-mid hover:bg-bg-3'}`} title="S"><Headphones size={9} /></button>
  1346→        <button onClick={() => update({ armed: !track.armed, input: !track.armed ? 'mic' : 'none' })} className={`flex h-4 w-4 items-center justify-center rounded ${track.armed ? 'bg-danger/30 text-danger' : 'text-text-mid hover:bg-bg-3'}`} title="Arm"><div className="h-1.5 w-1.5 rounded-full" style={{ backgroundColor: track.armed ? 'currentColor' : 'transparent', border: '1px solid currentColor' }} /></button>
  1347→        <button onClick={() => deleteAudioTrack(slate, track.id)} className="flex h-4 w-4 items-center justify-center rounded text-text-mid hover:bg-bg-3 hover:text-danger" title="Del"><Trash2 size={9} /></button>
  1348→      </div>
  1349→      <div className="mt-0.5 flex items-center gap-1">
  1350→        <Volume2 size={9} className="shrink-0 text-text-dim" aria-hidden />
  1351→        <input type="range" min={0} max={1} step={0.01} value={vol} aria-label="Volume" title="Volume" onPointerDown={onVolDown} onChange={(e) => onVol(Number(e.target.value))} onPointerUp={onVolEnd} className="h-1 min-w-0 flex-1 accent-accent" />
  1352→        <span className="shrink-0 text-[8px] font-medium leading-none text-text-dim" aria-hidden>L</span>

  1362→const ClipBlock = memo(function ClipBlock({ clip, pxPerSec, selected, onDragStart, onTrimStart }: {
  1363→  clip: AudioClip; pxPerSec: number; selected: boolean;

  1368→  const width = Math.max(4, clip.duration * pxPerSec);
