     1→/**
     2→ * CodeEditor — collaborative multi-file code surface for 'code' boards.
     3→ *
     4→ * CodeMirror 6 bound to one shared Y.Text per file via y-codemirror.next
     5→ * (yCollab): concurrent typing merges through Yjs, remote peers' carets and
     6→ * selections render inline with their presence color/name, and undo is a
     7→ * per-file Y.UndoManager scoped to YOUR edits (yUndoManagerKeymap — CM's own
     8→ * history is deliberately not installed, two histories would fight Ctrl+Z).
     9→ *
    10→ * Files live in the `code:files` Y.Map (id → { name }); each file's content
    11→ * is a top-level Y.Text (`code:text:<id>`) so every client resolves the same
    12→ * shared type. The creator seeds `main.js` on an empty board. Language is
    13→ * auto-detected from the filename via @codemirror/language-data, which
    14→ * dynamically imports only the grammars actually used.
    15→ *
    16→ * Polish (ROUND14-B):
    17→ *   - Fold gutter + indent-on-input + indent unit (2 spaces) so nested code
    18→ *     shows structure at a glance.
    19→ *   - Light/dark theme toggle. oneDark by default; the light theme is a CM
    20→ *     `EditorView.theme({ dark: false })` that lets the page tokens through.
    21→ *   - Tab strip above the editor: any file you click becomes a tab, middle-
    22→ *     click or the × on the tab closes it. The editor body always reflects
    23→ *     the active tab — clicking the same file in the rail is idempotent.
    24→ *   - A Find button in the toolbar that dispatches a synthetic Ctrl+F so
    25→ *     CM's built-in search panel (already bound via searchKeymap) opens.
    26→ *
    27→ * Split-view preview (ROUND18-A): an Eye toggle in the toolbar splits the
    28→ * editor 50/50 with a sandboxed <iframe srcdoc> that renders the project's
    29→ * HTML (via the shared buildPreview helper — local <link>/<script src> are
    30→ * inlined into the entry HTML, and a console bridge surfaces runtime logs
    31→ * inside the frame). The divider is drag-resizable (clamp 20%–80%); the
    32→ * preview auto-refreshes 400ms after any file change. The iframe runs with
    33→ * `allow-scripts` only (no `allow-same-origin`), so previewed code executes
    34→ * in a null origin and can't touch the Slate app or its sync state.
    35→ */

    71→/**
    72→ * Light theme — an empty rule set marked `{ dark: false }`. CM treats this as
    73→ * "light mode": it skips oneDark and lets the page tokens (via the
    74→ * `.slate-code-host .cm-editor { background: var(--bg) }` rules in
    75→ * codeEditor.css) carry the surface. Syntax tokens fall back to

    81→/**
    82→ * Indent guides — a tiny ViewPlugin that adds a `cm-indent-guide` line
    83→ * decoration to every visible line whose leading whitespace crosses one or
    84→ * more 2-space boundaries, and stashes the count in a `--cm-indent` CSS
    85→ * variable. codeEditor.css then paints vertical guides via a clipped
    86→ * `repeating-linear-gradient` background sized to that count.
    87→ *
    88→ * `@codemirror/view` shipped an official `indentationMarkers()` helper in a
    89→ * later 6.4x release; we're on 6.43.6 (latest published), so this is the
    90→ * local equivalent. Empty lines render with no guides — the trade-off is
    91→ * a few pixels of visual gap, but the implementation stays in one file.
    92→ */
    93→const indentGuides = ViewPlugin.fromClass(
    94→  class {
    95→    decos: DecorationSet;
    96→    constructor(view: EditorView) {
    97→      this.decos = this.build(view);
    98→    }
    99→    update(u: ViewUpdate) {
   100→      if (u.docChanged || u.viewportChanged || u.selectionSet) {
   101→        this.decos = this.build(u.view);
   102→      }
   103→    }
   104→    build(view: EditorView) {
   105→      const items: { from: number; value: ReturnType<typeof Decoration.line> }[] = [];
   106→      for (const { from, to } of view.visibleRanges) {
   107→        for (let pos = from; pos <= to; ) {
   108→          const line = view.state.doc.lineAt(pos);
   109→          // Count leading whitespace in 2-space units — a leading tab counts
   110→          // as one guide so mixed-indent files still get something useful.
   111→          const m = /^([ \t]+)/.exec(line.text);
   112→          const spaces = m?.[1] ? m[1].replace(/\t/g, '  ').length : 0;
   113→          const levels = Math.floor(spaces / 2);
   114→          if (levels > 0) {
   115→            items.push({
   116→              from: line.from,
   117→              value: Decoration.line({
   118→                class: 'cm-indent-guide',
   119→                attributes: { style: `--cm-indent: ${levels}` },
   120→              }),
   121→            });
   122→          }
   123→          pos = line.to + 1;
   124→        }
   125→      }
   126→      return Decoration.set(
   127→        items.map((i) => i.value.range(i.from)),
   128→        true,
   129→      );
   130→    }
   131→  },
   132→  { decorations: (v) => v.decos },
   133→);

   136→ *  rule; the +/- buttons step in 1px increments and clamp to [10, 24]). */
   137→const DEFAULT_FONT_SIZE = 13;
   138→const MIN_FONT_SIZE = 10;
   139→const MAX_FONT_SIZE = 24;

   145→    '.cm-gutters': { fontSize: `${px}px` },
   146→    '&': { fontSize: `${px}px` },
   147→  });
   148→}

   945→            >
   946→              <div className="flex items-center gap-2 border-b border-border px-3 py-2">
   947→                <CommandIcon size={14} className="shrink-0 text-text-dim" />
   948→                <input
   949→                  ref={paletteInputRef}
   950→                  type="text"
   951→                  value={paletteQuery}
   952→                  onChange={(e) => setPaletteQuery(e.target.value)}
   953→                  onKeyDown={(e) => {
   954→                    if (e.key === 'Escape') {
   955→                      e.preventDefault();
   956→                      closePalette();
   957→                    } else if (e.key === 'Enter') {
   958→                      e.preventDefault();
   959→                      const cmd = filteredCommands[paletteIndex];
   960→                      if (cmd) cmd.run();
   961→                    } else if (e.key === 'ArrowDown') {
   962→                      e.preventDefault();
   963→                      setPaletteIndex((i) => Math.min(filteredCommands.length - 1, i + 1));
   964→                    } else if (e.key === 'ArrowUp') {
   965→                      e.preventDefault();
   966→                      setPaletteIndex((i) => Math.max(0, i - 1));
   967→                    }
   968→                  }}
   969→                  placeholder="Type a command…"
   970→                  className="w-full bg-transparent text-sm text-text outline-none placeholder:text-text-dim"
   971→                />
   972→                <span className="shrink-0 rounded border border-border px-1.5 py-0.5 font-mono text-[10px] text-text-dim">Esc</span>
   973→              </div>
   974→              <ul className="max-h-72 overflow-y-auto py-1">
   975→                {filteredCommands.length === 0 && (
   976→                  <li className="px-3 py-3 text-center text-xs text-text-dim">No matching commands</li>
   977→                )}
   978→                {filteredCommands.map((cmd, i) => (
   979→                  <li key={cmd.id}>
   980→                    <button
   981→                      type="button"
   982→                      onMouseEnter={() => setPaletteIndex(i)}
   983→                      onClick={() => cmd.run()}
   984→                      className={`flex w-full items-center justify-between gap-2 px-3 py-1.5 text-left text-xs ${
   985→                        i === paletteIndex ? 'bg-accent/15 text-accent' : 'text-text-mid hover:bg-bg-3'
   986→                      }`}
   987→                    >
   988→                      <span>{cmd.label}</span>
   989→                      {cmd.hint && <span className="font-mono text-[10px] text-text-dim">{cmd.hint}</span>}
   990→                      {i === paletteIndex && <CornerDownLeft size={10} className="text-text-dim" />}
   991→                    </button>
   992→                  </li>
   993→                ))}
   994→              </ul>
   995→              <button
   996→                type="button"
   997→                aria-label="Close command palette"
   998→                onClick={closePalette}
   999→                className="block w-full border-t border-border bg-bg-3/40 py-1 text-center font-mono text-[10px] text-text-dim hover:bg-bg-3"
  1000→              >
  1001→                Ctrl+Shift+P to open · ↑↓ to navigate · Enter to run
  1002→              </button>
  1003→            </div>
  1004→          )}
  1005→        </div>
  1006→        {/* Terminal strip — full-width along the bottom of the editor, above
  1007→            the status bar. Drag the top edge to resize. Reuses the same panel
  1008→            component as the dockable version; it shows the live Preview's
  1009→            console output. */}
  1010→        {showTerminal && (
  1011→          <div className="flex shrink-0 flex-col border-t border-border" style={{ height: terminalHeight }}>
  1012→            <div
  1013→              role="separator"
  1014→              aria-orientation="horizontal"
  1015→              aria-label="Resize terminal"
  1016→              onMouseDown={startTerminalDrag}
  1017→              className="h-1 shrink-0 cursor-row-resize bg-border hover:bg-accent/50"
  1018→            />
  1019→            <div className="min-h-0 flex-1">
  1020→              <CodeTerminalPanel />
  1021→            </div>
  1022→          </div>
  1023→        )}
  1024→        <div className="flex items-center gap-3 border-t border-border bg-bg-2 px-3 py-1 font-mono text-[10px] text-text-dim">
  1025→          <span>{langName}</span>
  1026→          <span>{lineCount} {lineCount === 1 ? 'line' : 'lines'}</span>
  1027→          <div className="flex-1" />
  1028→          <span>shared · every keystroke syncs</span>
  1029→        </div>
  1030→      </div>
  1031→    </div>
  1032→  );
  1033→}

  1035→export default CodeEditor;
