    50тЖТ    50тЖТ// Draw only the window [startFrame, endFrame) of the sample across `width`
    51тЖТ    51тЖТ// pixels. Rendering the *window* (not the whole sample stretched) is what makes
    52тЖТ    52тЖТ// trimming look like cutting the audio away rather than squashing it.
    53тЖТ    53тЖТfunction computeWaveformPNG(samples: Float32Array, channels: number, width: number, color: string, height: number, startFrame: number, endFrame: number): string {
    54тЖТ    54тЖТ  const canvas = document.createElement('canvas');
    55тЖТ    55тЖТ  const dpr = window.devicePixelRatio || 1;
    56тЖТ    56тЖТ  const w = Math.max(2, Math.floor(width));
    57тЖТ    57тЖТ  canvas.width = w * dpr;
    58тЖТ    58тЖТ  canvas.height = height * dpr;
    59тЖТ    59тЖТ  const ctx = canvas.getContext('2d')!;
    60тЖТ    60тЖТ  ctx.scale(dpr, dpr);
    61тЖТ    61тЖТ  ctx.clearRect(0, 0, w, height);
    62тЖТ    62тЖТ  const totalFrames = samples.length / channels;
    63тЖТ    63тЖТ  const s = Math.max(0, Math.min(totalFrames, Math.floor(startFrame)));
    64тЖТ    64тЖТ  const e = Math.max(s, Math.min(totalFrames, Math.floor(endFrame)));
    65тЖТ    65тЖТ  const span = e - s;
    66тЖТ    66тЖТ  if (span <= 0) return canvas.toDataURL();
    67тЖТ    67тЖТ  const mid = height / 2;
    68тЖТ    68тЖТ  ctx.fillStyle = color;
    69тЖТ    69тЖТ  for (let x = 0; x < w; x++) {
    70тЖТ    70тЖТ    const f0 = s + Math.floor((x / w) * span);
    71тЖТ    71тЖТ    const f1 = Math.min(e, s + Math.floor(((x + 1) / w) * span) + 1);
    72тЖТ    72тЖТ    let peak = 0;
    73тЖТ    73тЖТ    for (let i = f0; i < f1; i++) { const v = Math.abs(samples[i * channels] ?? 0); if (v > peak) peak = v; }
    74тЖТ    74тЖТ    const bh = Math.max(1, peak * mid * 0.85);
    75тЖТ    75тЖТ    ctx.fillRect(x, mid - bh, 1, bh * 2);
    76тЖТ    76тЖТ  }
    77тЖТ    77тЖТ  return canvas.toDataURL();
    78тЖТ    78тЖТ}

    80тЖТ    80тЖТ/** Waveform image for the buffer window the clip actually plays. With speed s,
    81тЖТ    81тЖТ *  a clip of `duration` timeline seconds consumes `duration*s` buffer seconds.
    82тЖТ    82тЖТ *
    83тЖТ    83тЖТ *  Two robustness fixes:
    84тЖТ    84тЖТ *  1. If `loadSamples` returns an EMPTY Float32Array (length 0) тАФ which happens
    85тЖТ    85тЖТ *     when a freshly-split/created clip's IndexedDB write hasn't landed yet тАФ
    86тЖТ    86тЖТ *     we DON'T cache the resulting blank PNG. Instead we show the `┬╖┬╖┬╖`
    87тЖТ    87тЖТ *     placeholder and retry a few times (500ms apart) so the real waveform
    88тЖТ    88тЖТ *     appears once the samples are available.
    89тЖТ    89тЖТ *  2. We listen for `slate:audio-clip-changed` (fired by Normalize/Reverse/split)
    90тЖТ    90тЖТ *     and invalidate our cache entry for the current clipId, then force a
    91тЖТ    91тЖТ *     recompute via a `bust` counter тАФ otherwise the memoised component would
    92тЖТ    92тЖТ *     keep showing the stale PNG even after the cache was cleared. */
    93тЖТ    93тЖТconst WaveformImg = memo(function WaveformImg({ clipId, sampleKey, channels, sampleRate, offset, duration, speed, width, color }: {
    94тЖТ    94тЖТ  clipId: string; sampleKey: string; channels: number; sampleRate: number;
    95тЖТ    95тЖТ  offset: number; duration: number; speed: number; width: number; color: string;
    96тЖТ    96тЖТ}) {
    97тЖТ    97тЖТ  const height = TRACK_H - 6;
    98тЖТ    98тЖТ  const [imgUrl, setImgUrl] = useState<string>('');
    99тЖТ    99тЖТ  const [bust, setBust] = useState(0);
   100тЖТ   100тЖТ  const retryRef = useRef(0);
   101тЖТ   101тЖТ
   102тЖТ   102тЖТ  // Invalidate cached PNGs for this clip when its samples change (normalize /
   103тЖТ   103тЖТ  // reverse / split). The parent AudioEditor also invalidates + bumps version,
   104тЖТ   104тЖТ  // but the memoised WaveformImg wouldn't otherwise recompute (its primitive
   105тЖТ   105тЖТ  // props are unchanged) тАФ the `bust` counter forces the load effect to re-run.
   106тЖТ   106тЖТ  useEffect(() => {
   107тЖТ   107тЖТ    const onChanged = (e: Event) => {
   108тЖТ   108тЖТ      const detail = (e as CustomEvent<string>).detail;
   109тЖТ   109тЖТ      // Match by clipId (normalize/reverse/split events) OR by sampleKey
   110тЖТ   110тЖТ      // (multiplayer sample-arrival events from registerSampleSyncMap).
   111тЖТ   111тЖТ      if (detail !== clipId && detail !== sampleKey) return;
   112тЖТ   112тЖТ      for (const key of waveformPNGCache.keys()) {
   113тЖТ   113тЖТ        if (key.startsWith(`${clipId}:`)) waveformPNGCache.delete(key);
   114тЖТ   114тЖТ      }
   115тЖТ   115тЖТ      retryRef.current = 0;
   116тЖТ   116тЖТ      setBust((n) => n + 1);
   117тЖТ   117тЖТ    };
   118тЖТ   118тЖТ    window.addEventListener('slate:audio-clip-changed', onChanged as EventListener);
   119тЖТ   119тЖТ    return () => window.removeEventListener('slate:audio-clip-changed', onChanged as EventListener);
   120тЖТ   120тЖТ  }, [clipId, sampleKey]);
   121тЖТ   121тЖТ
   122тЖТ   122тЖТ  useEffect(() => {
   123тЖТ   123тЖТ    const startFrame = Math.round(offset * sampleRate);
   124тЖТ   124тЖТ    const endFrame = Math.round((offset + duration * speed) * sampleRate);
   125тЖТ   125тЖТ    const cacheKey = `${clipId}:${sampleKey}:${startFrame}:${endFrame}:${Math.floor(width)}`;
   126тЖТ   126тЖТ    const cached = waveformPNGCache.get(cacheKey);
   127тЖТ   127тЖТ    if (cached) { setImgUrl(cached); return; }
   128тЖТ   128тЖТ    let cancelled = false;
   129тЖТ   129тЖТ    let retryTimer: ReturnType<typeof setTimeout> | null = null;
   130тЖТ   130тЖТ
   131тЖТ   131тЖТ    const attempt = () => {
   132тЖТ   132тЖТ      if (cancelled) return;
   133тЖТ   133тЖТ      void loadSamples(sampleKey).then((samples) => {
   134тЖТ   134тЖТ        if (cancelled) return;
   135тЖТ   135тЖТ        // Empty samples = the IndexedDB write for this sampleKey hasn't landed
   136тЖТ   136тЖТ        // yet (e.g. brand-new clip from a split, OR a remote peer's clip whose
   137тЖТ   137тЖТ        // sample blob is still in flight via the Yjs sync map). DON'T cache a
   138тЖТ   138тЖТ        // blank PNG тАФ retry for up to 10s (20 ├Ч 500ms) so the real waveform
   139тЖТ   139тЖТ        // appears once the samples arrive. The previous 5-retry / 2.5s budget
   140тЖТ   140тЖТ        // was too short for cross-peer sample sync, which can take several
   141тЖТ   141тЖТ        // seconds for a 5MB blob over a slow link. The `slate:audio-clip-changed`
   142тЖТ   142тЖТ        // event (dispatched by onClipsAdded when a remote clip is added, and by
   143тЖТ   143тЖТ        // registerSampleSyncMap when samples arrive) resets `retryRef` to 0 so
   144тЖТ   144тЖТ        // the WaveformImg re-tries immediately on the signal тАФ this retry
   145тЖТ   145тЖТ        // budget is the safety net for the case where the event fires before
   146тЖТ   146тЖТ        // the WaveformImg mounts, or doesn't fire at all.
   147тЖТ   147тЖТ        if (samples.length === 0) {
   148тЖТ   148тЖТ          if (retryRef.current < 20) {
   149тЖТ   149тЖТ            retryRef.current += 1;
   150тЖТ   150тЖТ            retryTimer = setTimeout(attempt, 500);
   151тЖТ   151тЖТ          }
   152тЖТ   152тЖТ          return;
   153тЖТ   153тЖТ        }
   154тЖТ   154тЖТ        retryRef.current = 0;
   155тЖТ   155тЖТ        const url = computeWaveformPNG(samples, channels, width, color, height, startFrame, endFrame);
   156тЖТ   156тЖТ        waveformPNGCache.set(cacheKey, url);
   157тЖТ   157тЖТ        setImgUrl(url);
   158тЖТ   158тЖТ      });
   159тЖТ   159тЖТ    };
   160тЖТ   160тЖТ    attempt();
   161тЖТ   161тЖТ
   162тЖТ   162тЖТ    return () => {
   163тЖТ   163тЖТ      cancelled = true;
   164тЖТ   164тЖТ      if (retryTimer) clearTimeout(retryTimer);
   165тЖТ   165тЖТ    };
   166тЖТ   166тЖТ  }, [clipId, sampleKey, channels, sampleRate, offset, duration, speed, width, color, height, bust]);
   167тЖТ   167тЖТ
   168тЖТ   168тЖТ  if (!imgUrl) return <div className="flex h-full items-center justify-center text-[7px] text-text-dim">┬╖┬╖┬╖</div>;
   169тЖТ   169тЖТ  return <img src={imgUrl} alt="" className="pointer-events-none h-full w-full" style={{ objectFit: 'fill' }} />;
   170тЖТ   170тЖТ});

   447тЖТ   447тЖТ  // Global pointermove/up тАФ pure DOM, zero React state.
   448тЖТ   448тЖТ  // rAF-throttled: pointermove fires at the hardware's report rate (often
   449тЖТ   449тЖТ  // 60-120Hz), and the move handler does a neighbours-overlap scan on every
   450тЖТ   450тЖТ  // event. Coalescing multiple moves into a single rAF callback caps the work
   451тЖТ   451тЖТ  // at the display refresh rate (~60fps) so a fast pointer drag never stalls
   452тЖТ   452тЖТ  // the main thread. A no-op skip (cursor hasn't moved since the last
   453тЖТ   453тЖТ  // processed frame) avoids redundant DOM writes for stationary pointers.
   454тЖТ   454тЖТ  useEffect(() => {
   455тЖТ   455тЖТ    /** Last clientX we actually applied to the DOM. Lets us skip work when
   456тЖТ   456тЖТ     *  pointermove fires with the same x (e.g. a touchpad reporting pressure
   457тЖТ   457тЖТ     *  changes without movement, or sub-pixel jitter that rounds to the same
   458тЖТ   458тЖТ     *  px). */

   460тЖТ   460тЖТ
   461тЖТ   461тЖТ    const applyMove = () => {
   462тЖТ   462тЖТ      moveRafRef.current = 0;

   464тЖТ   464тЖТ      if (!d) return;

   468тЖТ   468тЖТ      // frame we processed. Saves a neighbours scan + 1-2 style writes.

   472тЖТ   472тЖТ      const pps = pxRef.current;
   473тЖТ   473тЖТ      const dt = (clientX - d.sx) / pps;
   474тЖТ   474тЖТ      const oldEnd = d.os + d.od;
   475тЖТ   475тЖТ      if (d.mode === 'drag') {
   476тЖТ   476тЖТ        // Snap-to-neighbour drag: the clip follows the mouse, but if the new
   477тЖТ   477тЖТ        // position would overlap another clip on the same track, instead of
   478тЖТ   478тЖТ        // stopping dead at the neighbour's edge we JUMP to the free side of
   479тЖТ   479тЖТ        // that neighbour (right of it when dragging right, left of it when
   480тЖТ   480тЖТ        // dragging left). Iterates so a chain of back-to-back clips resolves
   481тЖТ   481тЖТ        // to the next free slot. A small overlap threshold avoids accidental
   482тЖТ   482тЖТ        // snaps when the cursor barely crosses the boundary.

   486тЖТ   486тЖТ        const overlapThreshold = 0.05; // s тАФ ignore near-boundary jitter

   488тЖТ   488тЖТ          const blocker = d.neighbours.find((n) => {
   489тЖТ   489тЖТ            const ovStart = Math.max(start, n.start);
   490тЖТ   490тЖТ            const ovEnd = Math.min(start + duration, n.end);
   491тЖТ   491тЖТ            return ovEnd - ovStart > overlapThreshold;
   492тЖТ   492тЖТ          });

   496тЖТ   496тЖТ        }

   522тЖТ   522тЖТ      moveXRef.current = ev.clientX;

   524тЖТ   524тЖТ      // coalesces multiple moves within the same frame into one DOM write.

   544тЖТ   544тЖТ        applyMove();

   546тЖТ   546тЖТ      const pps = pxRef.current;
   547тЖТ   547тЖТ      const left = parseFloat(d.el.style.left) / pps;

   553тЖТ   553тЖТ      lastProcessedX = null;
   554тЖТ   554тЖТ      dragRef.current = null;

   557тЖТ   557тЖТ    window.addEventListener('pointerup', onUp);
