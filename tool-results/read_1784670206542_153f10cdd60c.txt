    91тЖТ    91тЖТfunction readAccentColor(): string {
    92тЖТ    92тЖТ  if (typeof window === 'undefined') return '#7c6aff';
    93тЖТ    93тЖТ  const v = getComputedStyle(document.documentElement).getPropertyValue('--accent').trim();

    97тЖТ    97тЖТ/** Darken a hex color by mixing it with black (factor 0..1). */
    98тЖТ    98тЖТfunction darken(hex: string, factor: number): string {
    99тЖТ    99тЖТ  const m = /^#?([0-9a-fA-F]{6})$/.exec(hex);
   100тЖТ   100тЖТ  if (!m) return hex;
   101тЖТ   101тЖТ  const r = parseInt(m[1]!.slice(0, 2), 16);
   102тЖТ   102тЖТ  const g = parseInt(m[1]!.slice(2, 4), 16);
   103тЖТ   103тЖТ  const b = parseInt(m[1]!.slice(4, 6), 16);
   104тЖТ   104тЖТ  const dr = Math.round(r * (1 - factor));
   105тЖТ   105тЖТ  const dg = Math.round(g * (1 - factor));
   106тЖТ   106тЖТ  const db = Math.round(b * (1 - factor));
   107тЖТ   107тЖТ  return `#${dr.toString(16).padStart(2, '0')}${dg.toString(16).padStart(2, '0')}${db.toString(16).padStart(2, '0')}`;
   108тЖТ   108тЖТ}

   313тЖТ   313тЖТ      const vpos = (vi: number) => ({
   314тЖТ   314тЖТ        x: mesh.vertices[vi * 3] ?? 0,
   315тЖТ   315тЖТ        y: mesh.vertices[vi * 3 + 1] ?? 0,
   316тЖТ   316тЖТ        z: mesh.vertices[vi * 3 + 2] ?? 0,
   317тЖТ   317тЖТ      });
   318тЖТ   318тЖТ      if (store.selectMode === 'vertex') {
   319тЖТ   319тЖТ        let best = poly[0]!;
   320тЖТ   320тЖТ        let bestD = Infinity;
   321тЖТ   321тЖТ        for (const vi of poly) {
   322тЖТ   322тЖТ          const p = vpos(vi);
   323тЖТ   323тЖТ          const d = (p.x - hit.local.x) ** 2 + (p.y - hit.local.y) ** 2 + (p.z - hit.local.z) ** 2;
   324тЖТ   324тЖТ          if (d < bestD) {
   325тЖТ   325тЖТ            bestD = d;
   326тЖТ   326тЖТ            best = vi;
   327тЖТ   327тЖТ          }
   328тЖТ   328тЖТ        }
   329тЖТ   329тЖТ        const cur = store.editSelection.verts;
   330тЖТ   330тЖТ        const next = additive
   331тЖТ   331тЖТ          ? cur.includes(best)
   332тЖТ   332тЖТ            ? cur.filter((v) => v !== best)
   333тЖТ   333тЖТ            : [...cur, best]
   334тЖТ   334тЖТ          : cur.length === 1 && cur[0] === best
   335тЖТ   335тЖТ            ? []
   336тЖТ   336тЖТ            : [best];
   337тЖТ   337тЖТ        setEditSelection({ verts: next, faces: [], edges: [] });
   338тЖТ   338тЖТ        return;
   339тЖТ   339тЖТ      }
   340тЖТ   340тЖТ
   341тЖТ   341тЖТ      // Edge mode: nearest polygon edge to the hit point.
   342тЖТ   342тЖТ      let bestA = poly[0]!;
   343тЖТ   343тЖТ      let bestB = poly[1] ?? poly[0]!;
   344тЖТ   344тЖТ      let bestD = Infinity;
   345тЖТ   345тЖТ      for (let i = 0; i < poly.length; i++) {
   346тЖТ   346тЖТ        const a = poly[i]!;
   347тЖТ   347тЖТ        const b = poly[(i + 1) % poly.length]!;
   348тЖТ   348тЖТ        const pa = vpos(a);
   349тЖТ   349тЖТ        const pb = vpos(b);
   350тЖТ   350тЖТ        const abx = pb.x - pa.x,
   351тЖТ   351тЖТ          aby = pb.y - pa.y,
   352тЖТ   352тЖТ          abz = pb.z - pa.z;
   353тЖТ   353тЖТ        const len2 = abx * abx + aby * aby + abz * abz || 1;
   354тЖТ   354тЖТ        let t =
   355тЖТ   355тЖТ          ((hit.local.x - pa.x) * abx + (hit.local.y - pa.y) * aby + (hit.local.z - pa.z) * abz) /
   356тЖТ   356тЖТ          len2;
   357тЖТ   357тЖТ        t = Math.max(0, Math.min(1, t));
   358тЖТ   358тЖТ        const dx = hit.local.x - (pa.x + t * abx);
   359тЖТ   359тЖТ        const dy = hit.local.y - (pa.y + t * aby);
   360тЖТ   360тЖТ        const dz = hit.local.z - (pa.z + t * abz);
   361тЖТ   361тЖТ        const d = dx * dx + dy * dy + dz * dz;
   362тЖТ   362тЖТ        if (d < bestD) {
   363тЖТ   363тЖТ          bestD = d;
   364тЖТ   364тЖТ          bestA = Math.min(a, b);
   365тЖТ   365тЖТ          bestB = Math.max(a, b);
   366тЖТ   366тЖТ        }
   367тЖТ   367тЖТ      }
   368тЖТ   368тЖТ      const cur = store.editSelection.edges;
   369тЖТ   369тЖТ      const pairs: [number, number][] = [];
   370тЖТ   370тЖТ      for (let i = 0; i + 1 < cur.length; i += 2) pairs.push([cur[i]!, cur[i + 1]!]);
   371тЖТ   371тЖТ      const idx = pairs.findIndex(([a, b]) => a === bestA && b === bestB);
   372тЖТ   372тЖТ      let nextPairs: [number, number][];
   373тЖТ   373тЖТ      if (additive) {
   374тЖТ   374тЖТ        nextPairs = idx >= 0 ? pairs.filter((_, i) => i !== idx) : [...pairs, [bestA, bestB]];
   375тЖТ   375тЖТ      } else {
   376тЖТ   376тЖТ        nextPairs = pairs.length === 1 && idx === 0 ? [] : [[bestA, bestB]];
   377тЖТ   377тЖТ      }
   378тЖТ   378тЖТ      setEditSelection({ edges: nextPairs.flat(), faces: [], verts: [] });
   379тЖТ   379тЖТ    },
   380тЖТ   380тЖТ    [setEditSelection],
   381тЖТ   381тЖТ  );

  1261тЖТ  1261тЖТfunction PivotProjector({
  1262тЖТ  1262тЖТ  snapshot,
  1263тЖТ  1263тЖТ  out,
  1264тЖТ  1264тЖТ}: {
  1265тЖТ  1265тЖТ  snapshot: SceneSnapshot;
  1266тЖТ  1266тЖТ  out: React.MutableRefObject<{ x: number; y: number } | null>;
  1267тЖТ  1267тЖТ}) {
  1268тЖТ  1268тЖТ  const { camera, size } = useThree();
  1269тЖТ  1269тЖТ  const snapRef = useRef(snapshot);
  1270тЖТ  1270тЖТ  snapRef.current = snapshot;
  1271тЖТ  1271тЖТ  useFrame(() => {
  1272тЖТ  1272тЖТ    const sel = useScene3DStore.getState().selection;
  1273тЖТ  1273тЖТ    if (sel.length === 0) {
  1274тЖТ  1274тЖТ      out.current = null;
  1275тЖТ  1275тЖТ      return;
  1276тЖТ  1276тЖТ    }
  1277тЖТ  1277тЖТ    let x = 0, y = 0, z = 0, n = 0;
  1278тЖТ  1278тЖТ    for (const id of sel) {
  1279тЖТ  1279тЖТ      const o = snapRef.current.objects.find((ob) => ob.id === id);
  1280тЖТ  1280тЖТ      if (o) {
  1281тЖТ  1281тЖТ        x += o.transform.position.x;
  1282тЖТ  1282тЖТ        y += o.transform.position.y;
  1283тЖТ  1283тЖТ        z += o.transform.position.z;
  1284тЖТ  1284тЖТ        n++;
  1285тЖТ  1285тЖТ      }
  1286тЖТ  1286тЖТ    }
  1287тЖТ  1287тЖТ    if (n === 0) {
  1288тЖТ  1288тЖТ      out.current = null;
  1289тЖТ  1289тЖТ      return;
  1290тЖТ  1290тЖТ    }
  1291тЖТ  1291тЖТ    const v = new THREE.Vector3(x / n, y / n, z / n).project(camera);
  1292тЖТ  1292тЖТ    out.current = { x: (v.x * 0.5 + 0.5) * size.width, y: (-v.y * 0.5 + 0.5) * size.height };
  1293тЖТ  1293тЖТ  });
  1294тЖТ  1294тЖТ  return null;
  1295тЖТ  1295тЖТ}

  1302тЖТ  1302тЖТfunction CameraTracking({
  1303тЖТ  1303тЖТ  out,
  1304тЖТ  1304тЖТ  room,
  1305тЖТ  1305тЖТ}: {
  1306тЖТ  1306тЖТ  out: React.MutableRefObject<CameraInfo | null>;
  1307тЖТ  1307тЖТ  room: SlateRoom;
  1308тЖТ  1308тЖТ}) {
  1309тЖТ  1309тЖТ  const { camera, controls, size } = useThree();
  1310тЖТ  1310тЖТ  const lastPublish = useRef({ t: 0, key: '' });
  1311тЖТ  1311тЖТ  useFrame(() => {
  1312тЖТ  1312тЖТ    const cam = camera as THREE.PerspectiveCamera;
  1313тЖТ  1313тЖТ    const right = new THREE.Vector3().setFromMatrixColumn(cam.matrixWorld, 0);
  1314тЖТ  1314тЖТ    const up = new THREE.Vector3().setFromMatrixColumn(cam.matrixWorld, 1);
  1315тЖТ  1315тЖТ    const forward = new THREE.Vector3().setFromMatrixColumn(cam.matrixWorld, 2).negate();
  1316тЖТ  1316тЖТ    const target = (controls as { target?: THREE.Vector3 } | null)?.target;
  1317тЖТ  1317тЖТ    const dist = target ? cam.position.distanceTo(target) : Math.max(cam.position.length(), 1);
  1318тЖТ  1318тЖТ    const pxScale =
  1319тЖТ  1319тЖТ      (2 * Math.max(dist, 0.5) * Math.tan(THREE.MathUtils.degToRad(cam.fov) / 2)) /
  1320тЖТ  1320тЖТ      Math.max(size.height, 1);
  1321тЖТ  1321тЖТ    out.current = {
  1322тЖТ  1322тЖТ      basis: {
  1323тЖТ  1323тЖТ        right: { x: right.x, y: right.y, z: right.z },
  1324тЖТ  1324тЖТ        up: { x: up.x, y: up.y, z: up.z },
  1325тЖТ  1325тЖТ        forward: { x: forward.x, y: forward.y, z: forward.z },
  1326тЖТ  1326тЖТ      },
  1327тЖТ  1327тЖТ      pxScale,
  1328тЖТ  1328тЖТ    };
  1329тЖТ  1329тЖТ    // Publish "where we're looking" so a model import lands at the orbit target
  1330тЖТ  1330тЖТ    // sized to the current view, not at the world origin at a fixed size.
  1331тЖТ  1331тЖТ    const focusT = target ?? new THREE.Vector3();
  1332тЖТ  1332тЖТ    setCameraFocus({
  1333тЖТ  1333тЖТ      center: { x: focusT.x, y: focusT.y, z: focusT.z },
  1334тЖТ  1334тЖТ      viewSize: pxScale * size.height,
  1335тЖТ  1335тЖТ      ts: performance.now(),
  1336тЖТ  1336тЖТ    });
  1337тЖТ  1337тЖТ    // Publish the camera pose so other peers can see where we're looking.
  1338тЖТ  1338тЖТ    // Throttled and change-gated тАФ awareness messages count against the
  1339тЖТ  1339тЖТ    // server's per-peer rate limit. During fly mode (walk navigation) we
  1340тЖТ  1340тЖТ    // publish more often so remote peers see the camera move smoothly in
  1341тЖТ  1341тЖТ    // real time instead of snapping once on release.
  1342тЖТ  1342тЖТ    const now = performance.now();
  1343тЖТ  1343тЖТ    const flying = useScene3DStore.getState().flying;
  1344тЖТ  1344тЖТ    const interval = flying ? 50 : 150;
  1345тЖТ  1345тЖТ    if (now - lastPublish.current.t < interval) return;
  1346тЖТ  1346тЖТ    const t = target ?? new THREE.Vector3();
  1347тЖТ  1347тЖТ    const r = (n: number) => Math.round(n * 100) / 100;
  1348тЖТ  1348тЖТ    const p: [number, number, number] = [r(cam.position.x), r(cam.position.y), r(cam.position.z)];
  1349тЖТ  1349тЖТ    const tt: [number, number, number] = [r(t.x), r(t.y), r(t.z)];
  1350тЖТ  1350тЖТ    const key = p.join(',') + '|' + tt.join(',');
  1351тЖТ  1351тЖТ    if (key === lastPublish.current.key) return;
  1352тЖТ  1352тЖТ    lastPublish.current = { t: now, key };
  1353тЖТ  1353тЖТ    room.setLocalAwareness({ cam: { p, t: tt } });
  1354тЖТ  1354тЖТ  });
  1355тЖТ  1355тЖТ  return null;
  1356тЖТ  1356тЖТ}

  1390тЖТ  1390тЖТfunction PeerCamera({ state }: { state: AwarenessState }) {
  1391тЖТ  1391тЖТ  const cam = state.cam!;
  1392тЖТ  1392тЖТ  const groupRef = useRef<THREE.Group | null>(null);
  1393тЖТ  1393тЖТ  // Poses arrive ~6/s; damp per frame so remote cameras glide, not hop.
  1394тЖТ  1394тЖТ  const targetPos = useRef(new THREE.Vector3(cam.p[0], cam.p[1], cam.p[2]));
  1395тЖТ  1395тЖТ  const lookCur = useRef(new THREE.Vector3(cam.t[0], cam.t[1], cam.t[2]));
  1396тЖТ  1396тЖТ  const lookTarget = useRef(new THREE.Vector3(cam.t[0], cam.t[1], cam.t[2]));
  1397тЖТ  1397тЖТ  targetPos.current.set(cam.p[0], cam.p[1], cam.p[2]);
  1398тЖТ  1398тЖТ  lookTarget.current.set(cam.t[0], cam.t[1], cam.t[2]);
  1399тЖТ  1399тЖТ  useFrame((_, delta) => {
  1400тЖТ  1400тЖТ    const g = groupRef.current;
  1401тЖТ  1401тЖТ    if (!g) return;
  1402тЖТ  1402тЖТ    // Frame-rate-independent smoothing (~90 ms time constant); teleport when
  1403тЖТ  1403тЖТ    // the peer jumps far so the glyph never streaks across the whole scene.
  1404тЖТ  1404тЖТ    if (g.position.distanceToSquared(targetPos.current) > 400) {
  1405тЖТ  1405тЖТ      g.position.copy(targetPos.current);
  1406тЖТ  1406тЖТ      lookCur.current.copy(lookTarget.current);
  1407тЖТ  1407тЖТ    } else {
  1408тЖТ  1408тЖТ      const k = 1 - Math.exp(-delta * 11);
  1409тЖТ  1409тЖТ      g.position.lerp(targetPos.current, k);
  1410тЖТ  1410тЖТ      lookCur.current.lerp(lookTarget.current, k);
  1411тЖТ  1411тЖТ    }
  1412тЖТ  1412тЖТ    g.lookAt(lookCur.current);
  1413тЖТ  1413тЖТ  });
  1414тЖТ  1414тЖТ  return (
  1415тЖТ  1415тЖТ    <group ref={groupRef} position={[cam.p[0], cam.p[1], cam.p[2]]}>
  1416тЖТ  1416тЖТ      <lineSegments geometry={PEER_CAM_LINES} raycast={() => null}>
  1417тЖТ  1417тЖТ        <lineBasicMaterial color={state.color} transparent opacity={0.9} />
  1418тЖТ  1418тЖТ      </lineSegments>
  1419тЖТ  1419тЖТ      {/* Small solid eye so the camera reads at a distance. */}
  1420тЖТ  1420тЖТ      <mesh raycast={() => null}>
  1421тЖТ  1421тЖТ        <sphereGeometry args={[0.06, 10, 10]} />
  1422тЖТ  1422тЖТ        <meshBasicMaterial color={state.color} />
  1423тЖТ  1423тЖТ      </mesh>
  1424тЖТ  1424тЖТ      <Html position={[0, 0.55, 0]} center distanceFactor={14} occlude={false}>
  1425тЖТ  1425тЖТ        <div
  1426тЖТ  1426тЖТ          className="pointer-events-none w-max max-w-[140px] truncate rounded-full px-1.5 py-px text-[10px] font-medium leading-4 text-black/85 shadow"
  1427тЖТ  1427тЖТ          style={{ backgroundColor: state.color }}
  1428тЖТ  1428тЖТ        >
  1429тЖТ  1429тЖТ          {state.name}
  1430тЖТ  1430тЖТ        </div>
  1431тЖТ  1431тЖТ      </Html>
  1432тЖТ  1432тЖТ    </group>
  1433тЖТ  1433тЖТ  );
  1434тЖТ  1434тЖТ}

  1436тЖТ  1436тЖТfunction editHudText(
  1437тЖТ  1437тЖТ  mode: 'vertex' | 'edge' | 'face',
  1438тЖТ  1438тЖТ  sel: { verts: number[]; edges: number[]; faces: number[] },
  1439тЖТ  1439тЖТ  vertices?: number[],
  1440тЖТ  1440тЖТ  transform?: { scale: { x: number; y: number; z: number } },
  1441тЖТ  1441тЖТ): string {
  1442тЖТ  1442тЖТ  const counts: Record<string, number> = {
  1443тЖТ  1443тЖТ    vertex: sel.verts.length,
  1444тЖТ  1444тЖТ    edge: sel.edges.length / 2,
  1445тЖТ  1445тЖТ    face: sel.faces.length,
  1446тЖТ  1446тЖТ  };
  1447тЖТ  1447тЖТ  const n = counts[mode] ?? 0;
  1448тЖТ  1448тЖТ  if (n === 0) {
  1449тЖТ  1449тЖТ    const noun = mode === 'vertex' ? 'vertex' : mode;
  1450тЖТ  1450тЖТ    return `Click a ${noun} to select it ┬╖ Shift+Click adds ┬╖ 1/2/3 switch vert/edge/face`;
  1451тЖТ  1451тЖТ  }
  1452тЖТ  1452тЖТ  const plural = mode === 'vertex' ? 'vertices' : `${mode}s`;
  1453тЖТ  1453тЖТ  // CAD-style measurements (Blender's N-panel): edge lengths + face area.
  1454тЖТ  1454тЖТ  if (mode === 'edge' && vertices && vertices.length >= 3 && transform) {
  1455тЖТ  1455тЖТ    let total = 0;
  1456тЖТ  1456тЖТ    for (let i = 0; i + 1 < sel.edges.length; i += 2) {
  1457тЖТ  1457тЖТ      const a = sel.edges[i]!;
  1458тЖТ  1458тЖТ      const b = sel.edges[i + 1]!;
  1459тЖТ  1459тЖТ      const ax = (vertices[a * 3] ?? 0) * transform.scale.x;
  1460тЖТ  1460тЖТ      const ay = (vertices[a * 3 + 1] ?? 0) * transform.scale.y;
  1461тЖТ  1461тЖТ      const az = (vertices[a * 3 + 2] ?? 0) * transform.scale.z;
  1462тЖТ  1462тЖТ      const bx = (vertices[b * 3] ?? 0) * transform.scale.x;
  1463тЖТ  1463тЖТ      const by = (vertices[b * 3 + 1] ?? 0) * transform.scale.y;
  1464тЖТ  1464тЖТ      const bz = (vertices[b * 3 + 2] ?? 0) * transform.scale.z;
  1465тЖТ  1465тЖТ      total += Math.hypot(bx - ax, by - ay, bz - az);
  1466тЖТ  1466тЖТ    }
  1467тЖТ  1467тЖТ    return `${n} ${n === 1 ? 'edge' : plural} selected тАФ length ${total.toFixed(3)} m ┬╖ G moves`;
  1468тЖТ  1468тЖТ  }
  1469тЖТ  1469тЖТ  if (mode === 'face') {
  1470тЖТ  1470тЖТ    return `${n} ${n === 1 ? 'face' : plural} selected ┬╖ E extrudes ┬╖ G moves`;
  1471тЖТ  1471тЖТ  }
  1472тЖТ  1472тЖТ  const verb = ' ┬╖ G moves';
  1473тЖТ  1473тЖТ  return `${n} ${n === 1 ? (mode === 'vertex' ? 'vertex' : mode) : plural} selected${verb}`;
  1474тЖТ  1474тЖТ}

  1477тЖТ  1477тЖТfunction ViewPresets({
  1478тЖТ  1478тЖТ  snapshot,
  1479тЖТ  1479тЖТ  lookThroughRef,
  1480тЖТ  1480тЖТ  cameraFollowRef,
  1481тЖТ  1481тЖТ}: {
  1482тЖТ  1482тЖТ  snapshot: SceneSnapshot;
  1483тЖТ  1483тЖТ  lookThroughRef: React.MutableRefObject<() => boolean>;
  1484тЖТ  1484тЖТ  cameraFollowRef?: React.MutableRefObject<(t: number) => boolean>;
  1485тЖТ  1485тЖТ}) {
  1486тЖТ  1486тЖТ  const { camera, controls } = useThree();
  1487тЖТ  1487тЖТ  const snapRef = useRef(snapshot);
  1488тЖТ  1488тЖТ  snapRef.current = snapshot;
  1489тЖТ  1489тЖТ
  1490тЖТ  1490тЖТ  // Point the R3F camera at the scene camera's pose sampled at time `t`, so
  1491тЖТ  1491тЖТ  // an animation render follows keyframed camera moves. Returns false when
  1492тЖТ  1492тЖТ  // there's no scene camera.
  1493тЖТ  1493тЖТ  useEffect(() => {
  1494тЖТ  1494тЖТ    if (!cameraFollowRef) return;
  1495тЖТ  1495тЖТ    cameraFollowRef.current = (t: number) => {
  1496тЖТ  1496тЖТ      const sel = useScene3DStore.getState().selection;
  1497тЖТ  1497тЖТ      const cams = snapRef.current.objects.filter((o) => o.type === 'camera');
  1498тЖТ  1498тЖТ      const cam = cams.find((o) => sel.includes(o.id)) ?? cams[0];
  1499тЖТ  1499тЖТ      if (!cam) return false;
  1500тЖТ  1500тЖТ      const tr = (cam.anim?.length ? sampleAnim(cam.anim, t) : null) ?? cam.transform;
  1501тЖТ  1501тЖТ      const dir = new THREE.Vector3(0, -1, 0).applyEuler(
  1502тЖТ  1502тЖТ        new THREE.Euler(tr.rotation.x, tr.rotation.y, tr.rotation.z),
  1503тЖТ  1503тЖТ      );
  1504тЖТ  1504тЖТ      camera.position.set(tr.position.x, tr.position.y, tr.position.z);
  1505тЖТ  1505тЖТ      const persp = camera as THREE.PerspectiveCamera;
  1506тЖТ  1506тЖТ      persp.fov = cam.camera?.fov ?? 50;
  1507тЖТ  1507тЖТ      persp.updateProjectionMatrix();
  1508тЖТ  1508тЖТ      camera.lookAt(new THREE.Vector3().copy(camera.position).addScaledVector(dir, 8));
  1509тЖТ  1509тЖТ      return true;
  1510тЖТ  1510тЖТ    };
  1511тЖТ  1511тЖТ  }, [camera, cameraFollowRef]);
  1512тЖТ  1512тЖТ
  1513тЖТ  1513тЖТ  // Jump the viewport into the selected (or first) scene camera. Shared by
  1514тЖТ  1514тЖТ  // Numpad 0 and the render button. Returns false when the scene has none.
  1515тЖТ  1515тЖТ  useEffect(() => {
  1516тЖТ  1516тЖТ    lookThroughRef.current = () => {
  1517тЖТ  1517тЖТ      const sel = useScene3DStore.getState().selection;
  1518тЖТ  1518тЖТ      const cams = snapRef.current.objects.filter((o) => o.type === 'camera');
  1519тЖТ  1519тЖТ      const cam = cams.find((o) => sel.includes(o.id)) ?? cams[0];
  1520тЖТ  1520тЖТ      if (!cam) return false;
  1521тЖТ  1521тЖТ      const t = cam.transform;
  1522тЖТ  1522тЖТ      const dir = new THREE.Vector3(0, -1, 0).applyEuler(
  1523тЖТ  1523тЖТ        new THREE.Euler(t.rotation.x, t.rotation.y, t.rotation.z),
  1524тЖТ  1524тЖТ      );
  1525тЖТ  1525тЖТ      camera.position.set(t.position.x, t.position.y, t.position.z);
  1526тЖТ  1526тЖТ      const persp = camera as THREE.PerspectiveCamera;
  1527тЖТ  1527тЖТ      persp.fov = cam.camera?.fov ?? 50;
  1528тЖТ  1528тЖТ      persp.updateProjectionMatrix();
  1529тЖТ  1529тЖТ      (controls as { target?: THREE.Vector3 } | null)?.target?.copy(
  1530тЖТ  1530тЖТ        new THREE.Vector3().copy(camera.position).addScaledVector(dir, 8),
  1531тЖТ  1531тЖТ      );
  1532тЖТ  1532тЖТ      (controls as { update?: () => void } | null)?.update?.();
  1533тЖТ  1533тЖТ      // Enter "camera view": the glyph hides and a frame HUD shows until the
  1534тЖТ  1534тЖТ      // user orbits/flies away (Blender's Numpad 0 behavior).
  1535тЖТ  1535тЖТ      useScene3DStore.getState().setViewingCamera(cam.id);
  1536тЖТ  1536тЖТ      return true;
  1537тЖТ  1537тЖТ    };
  1538тЖТ  1538тЖТ  }, [camera, controls, lookThroughRef]);
  1539тЖТ  1539тЖТ
  1540тЖТ  1540тЖТ  useEffect(() => {
  1541тЖТ  1541тЖТ    const onKey = (e: KeyboardEvent) => {
  1542тЖТ  1542тЖТ      const target = e.target as HTMLElement;
  1543тЖТ  1543тЖТ      if (target && /^(INPUT|TEXTAREA|SELECT)$/.test(target.tagName)) return;
  1544тЖТ  1544тЖТ      if (useScene3DStore.getState().flying) return;
  1545тЖТ  1545тЖТ      if (e.code === 'Numpad0') {
  1546тЖТ  1546тЖТ        if (lookThroughRef.current()) e.preventDefault();
  1547тЖТ  1547тЖТ        return;
  1548тЖТ  1548тЖТ      }
  1549тЖТ  1549тЖТ      let dir: THREE.Vector3 | null = null;
  1550тЖТ  1550тЖТ      if (e.code === 'Numpad1') dir = new THREE.Vector3(0, 0, 1);
  1551тЖТ  1551тЖТ      else if (e.code === 'Numpad3') dir = new THREE.Vector3(1, 0, 0);
  1552тЖТ  1552тЖТ      else if (e.code === 'Numpad7') dir = new THREE.Vector3(0.0001, 1, 0.0001);
  1553тЖТ  1553тЖТ      if (!dir) return;
  1554тЖТ  1554тЖТ      // Ctrl+Numpad flips to the opposite view (back/left/bottom), as in Blender.
  1555тЖТ  1555тЖТ      if (e.ctrlKey) dir.negate();
  1556тЖТ  1556тЖТ      e.preventDefault();
  1557тЖТ  1557тЖТ      const t = (controls as { target?: THREE.Vector3 } | null)?.target ?? new THREE.Vector3();
  1558тЖТ  1558тЖТ      const dist = camera.position.distanceTo(t) || 8;
  1559тЖТ  1559тЖТ      camera.position.copy(t.clone().add(dir.normalize().multiplyScalar(dist)));
  1560тЖТ  1560тЖТ      (controls as { update?: () => void } | null)?.update?.();
  1561тЖТ  1561тЖТ    };
  1562тЖТ  1562тЖТ    window.addEventListener('keydown', onKey);
  1563тЖТ  1563тЖТ    return () => window.removeEventListener('keydown', onKey);
  1564тЖТ  1564тЖТ  }, [camera, controls]);
  1565тЖТ  1565тЖТ  return null;
  1566тЖТ  1566тЖТ}

  1568тЖТ  1568тЖТ/**
  1569тЖТ  1569тЖТ * Tiny in-canvas component that captures `useThree` and exposes a callback
  1570тЖТ  1570тЖТ * to frame the current selection. Kept inside <Canvas> so it has access to
  1571тЖТ  1571тЖТ * the camera and controls.
  1572тЖТ  1572тЖТ */
  1573тЖТ  1573тЖТfunction FrameSelectedBinding({
  1574тЖТ  1574тЖТ  frameRef,
  1575тЖТ  1575тЖТ  snapshot,
  1576тЖТ  1576тЖТ  selection,
  1577тЖТ  1577тЖТ}: {
  1578тЖТ  1578тЖТ  frameRef: React.MutableRefObject<(all: boolean) => void>;
  1579тЖТ  1579тЖТ  snapshot: SceneSnapshot;
  1580тЖТ  1580тЖТ  selection: string[];
  1581тЖТ  1581тЖТ}) {
  1582тЖТ  1582тЖТ  const { camera, controls } = useThree();
  1583тЖТ  1583тЖТ  useFrame(() => {});
  1584тЖТ  1584тЖТ  useEffect(() => {
  1585тЖТ  1585тЖТ    frameRef.current = (all: boolean) => {
  1586тЖТ  1586тЖТ      // Frame the selection; with `all` (or nothing selected), frame the scene.
  1587тЖТ  1587тЖТ      const ids = !all && selection.length > 0 ? selection : snapshot.objects.map((o) => o.id);
  1588тЖТ  1588тЖТ      // Build a world-space bounding box from each object's mesh vertices
  1589тЖТ  1589тЖТ      // (transformed by position/rotation/scale) тАФ not just its origin point.
  1590тЖТ  1590тЖТ      // A model imported at the origin would otherwise frame as a zero-size
  1591тЖТ  1591тЖТ      // point and the camera would dive inside it.
  1592тЖТ  1592тЖТ      const box = new THREE.Box3();
  1593тЖТ  1593тЖТ      let any = false;
  1594тЖТ  1594тЖТ      for (const id of ids) {
  1595тЖТ  1595тЖТ        const obj = snapshot.objects.find((o) => o.id === id);
  1596тЖТ  1596тЖТ        if (!obj) continue;
  1597тЖТ  1597тЖТ        const mesh = obj.meshId ? snapshot.meshes.get(obj.meshId) : undefined;
  1598тЖТ  1598тЖТ        if (mesh && mesh.vertices.length > 0) {
  1599тЖТ  1599тЖТ          // Local AABB of the raw vertices.
  1600тЖТ  1600тЖТ          const min = new THREE.Vector3(Infinity, Infinity, Infinity);
  1601тЖТ  1601тЖТ          const max = new THREE.Vector3(-Infinity, -Infinity, -Infinity);
  1602тЖТ  1602тЖТ          for (let i = 0; i < mesh.vertices.length; i += 3) {
  1603тЖТ  1603тЖТ            min.min(new THREE.Vector3(mesh.vertices[i]!, mesh.vertices[i + 1]!, mesh.vertices[i + 2]!));
  1604тЖТ  1604тЖТ            max.max(new THREE.Vector3(mesh.vertices[i]!, mesh.vertices[i + 1]!, mesh.vertices[i + 2]!));
  1605тЖТ  1605тЖТ          }
  1606тЖТ  1606тЖТ          // Transform the 8 corners into world space and expand the box.
  1607тЖТ  1607тЖТ          const m = new THREE.Matrix4().compose(
  1608тЖТ  1608тЖТ            new THREE.Vector3(obj.transform.position.x, obj.transform.position.y, obj.transform.position.z),
  1609тЖТ  1609тЖТ            new THREE.Quaternion().setFromEuler(
  1610тЖТ  1610тЖТ              new THREE.Euler(obj.transform.rotation.x, obj.transform.rotation.y, obj.transform.rotation.z),
  1611тЖТ  1611тЖТ            ),
  1612тЖТ  1612тЖТ            new THREE.Vector3(obj.transform.scale.x, obj.transform.scale.y, obj.transform.scale.z),
  1613тЖТ  1613тЖТ          );
  1614тЖТ  1614тЖТ          const corners = [
  1615тЖТ  1615тЖТ            new THREE.Vector3(min.x, min.y, min.z),
  1616тЖТ  1616тЖТ            new THREE.Vector3(max.x, min.y, min.z),
  1617тЖТ  1617тЖТ            new THREE.Vector3(min.x, max.y, min.z),
  1618тЖТ  1618тЖТ            new THREE.Vector3(max.x, max.y, min.z),
  1619тЖТ  1619тЖТ            new THREE.Vector3(min.x, min.y, max.z),
  1620тЖТ  1620тЖТ            new THREE.Vector3(max.x, min.y, max.z),
  1621тЖТ  1621тЖТ            new THREE.Vector3(min.x, max.y, max.z),
  1622тЖТ  1622тЖТ            new THREE.Vector3(max.x, max.y, max.z),
  1623тЖТ  1623тЖТ          ].map((c) => c.applyMatrix4(m));
  1624тЖТ  1624тЖТ          for (const c of corners) box.expandByPoint(c);
  1625тЖТ  1625тЖТ          any = true;
  1626тЖТ  1626тЖТ        } else {
  1627тЖТ  1627тЖТ          // Lights / empties: just their position.
  1628тЖТ  1628тЖТ          box.expandByPoint(
  1629тЖТ  1629тЖТ            new THREE.Vector3(obj.transform.position.x, obj.transform.position.y, obj.transform.position.z),
  1630тЖТ  1630тЖТ          );
  1631тЖТ  1631тЖТ          any = true;
  1632тЖТ  1632тЖТ        }
  1633тЖТ  1633тЖТ      }
  1634тЖТ  1634тЖТ      if (!any) return;
  1635тЖТ  1635тЖТ      const size = new THREE.Vector3();
  1636тЖТ  1636тЖТ      const center = new THREE.Vector3();
  1637тЖТ  1637тЖТ      box.getSize(size);
  1638тЖТ  1638тЖТ      box.getCenter(center);
  1639тЖТ  1639тЖТ      // Pad so the object isn't flush against the frame edge, and never let a
  1640тЖТ  1640тЖТ      // degenerate (point) box dive the camera inside the model.
  1641тЖТ  1641тЖТ      const padded = size.clone().max(new THREE.Vector3(0.5, 0.5, 0.5)).multiplyScalar(1.25);
  1642тЖТ  1642тЖТ      const radius = Math.max(padded.length() / 2, 1.5);
  1643тЖТ  1643тЖТ      const dir = new THREE.Vector3()
  1644тЖТ  1644тЖТ        .subVectors(camera.position, (controls as { target?: THREE.Vector3 })?.target ?? new THREE.Vector3())
  1645тЖТ  1645тЖТ        .normalize();
  1646тЖТ  1646тЖТ      if (!Number.isFinite(dir.x) || dir.lengthSq() < 1e-9) dir.set(1, 0.8, 1).normalize();
  1647тЖТ  1647тЖТ      camera.position.copy(center.clone().add(dir.multiplyScalar(radius)));
  1648тЖТ  1648тЖТ      (controls as { target?: THREE.Vector3 } | null)?.target?.copy(center);
  1649тЖТ  1649тЖТ      (controls as { update?: () => void } | null)?.update?.();
  1650тЖТ  1650тЖТ    };
  1651тЖТ  1651тЖТ  }, [camera, controls, snapshot, selection, frameRef]);
  1652тЖТ  1652тЖТ  return null;
  1653тЖТ  1653тЖТ}
