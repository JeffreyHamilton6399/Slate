   102‚Üí   102‚Üí  constructor() {

   104‚Üí   104‚Üí    // clip arrive (or when the clip is otherwise mutated). Without this, a
   105‚Üí   105‚Üí    // remote peer whose sample blob arrives JUST after the 20√ó500ms budget
   106‚Üí   106‚Üí    // expired would never hear the clip until the next play() ‚Äî the cache

   108‚Üí   108‚Üí    // re-requested. Resetting mid-loop extends the budget on the signal
   109‚Üí   109‚Üí    // that the samples are now available, so the in-flight loop wins on
   110‚Üí   110‚Üí    // its next iteration instead of giving up.

   114‚Üí   114‚Üí        this.retryAttempts.set(id, 0);
   115‚Üí   115‚Üí      }
   116‚Üí   116‚Üí    };
   117‚Üí   117‚Üí    window.addEventListener('slate:audio-clip-changed', this.clipChangedHandler);
   118‚Üí   118‚Üí  }

   120‚Üí   120‚Üí  /** Ensure the AudioContext is created. Browsers' autoplay policies suspend
   121‚Üí   121‚Üí   *  AudioContexts created before a user gesture ‚Äî `ctx.resume()` from within
   122‚Üí   122‚Üí   *  a non-gesture call stack is silently ignored, so the previous
   123‚Üí   123‚Üí   *  `void this.ctx.resume()` was a no-op for a remote peer who just joined
   124‚Üí   124‚Üí   *  and hadn't clicked anything. Instead of silently failing, we register a
   125‚Üí   125‚Üí   *  one-time gesture listener (pointerdown/keydown/touchstart) that resumes
   126‚Üí   126‚Üí   *  the context on the first interaction. Callers that KNOW they're in a
   127‚Üí   127‚Üí   *  gesture handler (e.g. a click on the play button) should also call
   128‚Üí   128‚Üí   *  `resumeOnGesture()` directly for immediate effect. */

   130‚Üí   130‚Üí    if (!this.ctx) {
   131‚Üí   131‚Üí      const AudioCtx = window.AudioContext || (window as unknown as { webkitAudioContext: typeof AudioContext }).webkitAudioContext;

   134‚Üí   134‚Üí      this.masterGain.gain.value = this.masterVolume;
   135‚Üí   135‚Üí      this.masterGain.connect(this.ctx.destination);
   136‚Üí   136‚Üí    }
   137‚Üí   137‚Üí    if (this.ctx.state === 'suspended') {
   138‚Üí   138‚Üí       
   139‚Üí   139‚Üí      console.warn('[slate-audio] AudioContext is suspended ‚Äî will resume on the first user gesture (browser autoplay policy). Click anywhere to enable audio.');
   140‚Üí   140‚Üí      this.attachGestureListener();

   142‚Üí   142‚Üí    return this.ctx;
   143‚Üí   143‚Üí  }
   144‚Üí   144‚Üí
   145‚Üí   145‚Üí  /** Register a one-time pointerdown/keydown/touchstart listener that resumes
   146‚Üí   146‚Üí   *  the suspended AudioContext on the first user interaction. Idempotent ‚Äî
   147‚Üí   147‚Üí   *  multiple calls while suspended coalesce into one listener. The listener
   148‚Üí   148‚Üí   *  removes itself from all three event targets when it fires. */
   149‚Üí   149‚Üí  private attachGestureListener(): void {
   150‚Üí   150‚Üí    if (this.gestureHandler) return;
   151‚Üí   151‚Üí    const handler = () => {
   152‚Üí   152‚Üí      if (this.ctx && this.ctx.state === 'suspended') void this.ctx.resume();
   153‚Üí   153‚Üí      window.removeEventListener('pointerdown', handler);
   154‚Üí   154‚Üí      window.removeEventListener('keydown', handler);
   155‚Üí   155‚Üí      window.removeEventListener('touchstart', handler);
   156‚Üí   156‚Üí      this.gestureHandler = null;
   157‚Üí   157‚Üí    };
   158‚Üí   158‚Üí    this.gestureHandler = handler;
   159‚Üí   159‚Üí    window.addEventListener('pointerdown', handler);
   160‚Üí   160‚Üí    window.addEventListener('keydown', handler);
   161‚Üí   161‚Üí    window.addEventListener('touchstart', handler);
   162‚Üí   162‚Üí  }
   163‚Üí   163‚Üí
   164‚Üí   164‚Üí  /** Resume the AudioContext if it's suspended. Safe to call from any user
   165‚Üí   165‚Üí   *  gesture handler (click, pointerdown, keydown, touchstart) ‚Äî browsers
   166‚Üí   166‚Üí   *  only allow `resume()` to take effect from within a gesture call stack.

   172‚Üí   172‚Üí      // supersedes it.
   173‚Üí   173‚Üí      if (this.gestureHandler) {
   174‚Üí   174‚Üí        window.removeEventListener('pointerdown', this.gestureHandler);
   175‚Üí   175‚Üí        window.removeEventListener('keydown', this.gestureHandler);
   176‚Üí   176‚Üí        window.removeEventListener('touchstart', this.gestureHandler);
   177‚Üí   177‚Üí        this.gestureHandler = null;
   178‚Üí   178‚Üí      }
   179‚Üí   179‚Üí    }
   180‚Üí   180‚Üí  }
   181‚Üí   181‚Üí
   182‚Üí   182‚Üí  /** Get or create an AudioBuffer for a clip ‚Äî loads samples from IndexedDB.
   183‚Üí   183‚Üí   *  If the samples haven't arrived yet (empty Float32Array, which happens when
   184‚Üí   184‚Üí   *  a remote peer's sample-blob is still in flight via the Yjs sync map),
   185‚Üí   185‚Üí   *  retries up to 20 times at 500ms intervals (10s total) before giving up
   186‚Üí   186‚Üí   *  and returning null. The clip is then skipped on this play() pass, but
   187‚Üí   187‚Üí   *  the AudioEditor's restart-on-sample-arrival will re-schedule it once the
   188‚Üí   188‚Üí   *  samples land and the cache is populated. The 20√ó500ms budget matches
   189‚Üí   189‚Üí   *  the WaveformImg retry budget ‚Äî large multi-MB samples arriving as
   190‚Üí   190‚Üí   *  ~512KB Yjs chunks over a slow link can take several seconds, and the
   191‚Üí   191‚Üí   *  previous 10√ó300ms (3s) budget was too short. The `slate:audio-clip-changed`
   192‚Üí   192‚Üí   *  event (fired when samples arrive) resets `retryAttempts` to 0 for any
   193‚Üí   193‚Üí   *  clip currently in the loop, so the budget extends on the very signal
   194‚Üí   194‚Üí   *  that the samples are now available. */
   195‚Üí   195‚Üí  private async getBuffer(clip: AudioClip): Promise<AudioBuffer | null> {
   196‚Üí   196‚Üí    const ctx = this.ctx!;
   197‚Üí   197‚Üí    const cached = this.bufferCache.get(clip.id);
   198‚Üí   198‚Üí    if (cached) return cached;
   199‚Üí   199‚Üí
   200‚Üí   200‚Üí    // If a retry loop is already running for this clip, bail ‚Äî another
   201‚Üí   201‚Üí    // play()/restartPlayback() call will pick up the cached buffer once it
   202‚Üí   202‚Üí    // lands (the AudioEditor restarts on `slate:audio-clip-changed`).
   203‚Üí   203‚Üí    if (this.retryingClips.has(clip.id)) return null;
   204‚Üí   204‚Üí
   205‚Üí   205‚Üí    const buildBuffer = async (): Promise<AudioBuffer | null> => {
   206‚Üí   206‚Üí      const samples = await loadSamples(clip.sampleKey);
   207‚Üí   207‚Üí      const channels = clip.channels;
   208‚Üí   208‚Üí      const length = samples.length / channels;
   209‚Üí   209‚Üí      if (length === 0) return null;
   210‚Üí   210‚Üí      const buf = ctx.createBuffer(channels, length, clip.sampleRate);
   211‚Üí   211‚Üí      for (let ch = 0; ch < channels; ch++) {
   212‚Üí   212‚Üí        const data = buf.getChannelData(ch);
   213‚Üí   213‚Üí        for (let i = 0; i < length; i++) {
   214‚Üí   214‚Üí          data[i] = samples[i * channels + ch] ?? 0;
   215‚Üí   215‚Üí        }
   216‚Üí   216‚Üí      }
   217‚Üí   217‚Üí      return buf;
   218‚Üí   218‚Üí    };
   219‚Üí   219‚Üí
   220‚Üí   220‚Üí    // Fast path ‚Äî samples already present (the overwhelmingly common case).
   221‚Üí   221‚Üí    let buf = await buildBuffer();
   222‚Üí   222‚Üí    if (buf) {
   223‚Üí   223‚Üí      this.bufferCache.set(clip.id, buf);
   224‚Üí   224‚Üí      return buf;
   225‚Üí   225‚Üí    }
   226‚Üí   226‚Üí
   227‚Üí   227‚Üí    // Samples not yet arrived ‚Äî retry for up to 10s (20 √ó 500ms). This
   228‚Üí   228‚Üí    // covers the gap between a remote clip's metadata landing in Yjs and
   229‚Üí   229‚Üí    // its sample blob landing in the sync map + being written to local
   230‚Üí   230‚Üí    // IndexedDB. The `retryAttempts` counter is read fresh on every
   231‚Üí   231‚Üí    // iteration so a `slate:audio-clip-changed` event (samples arrived)
   232‚Üí   232‚Üí    // can reset it to 0 mid-loop, extending the budget on the signal that
   233‚Üí   233‚Üí    // the wait is over.
   234‚Üí   234‚Üí    this.retryingClips.add(clip.id);
   235‚Üí   235‚Üí    this.retryAttempts.set(clip.id, 0);
   236‚Üí   236‚Üí    try {
   237‚Üí   237‚Üí      while ((this.retryAttempts.get(clip.id) ?? 0) < 20) {
   238‚Üí   238‚Üí        this.retryAttempts.set(clip.id, (this.retryAttempts.get(clip.id) ?? 0) + 1);
   239‚Üí   239‚Üí        await new Promise<void>((resolve) => setTimeout(resolve, 500));
   240‚Üí   240‚Üí        buf = await buildBuffer();
   241‚Üí   241‚Üí        if (buf) {
   242‚Üí   242‚Üí          this.bufferCache.set(clip.id, buf);
   243‚Üí   243‚Üí          return buf;
   244‚Üí   244‚Üí        }
   245‚Üí   245‚Üí      }
   246‚Üí   246‚Üí    } finally {
   247‚Üí   247‚Üí      this.retryingClips.delete(clip.id);
   248‚Üí   248‚Üí      this.retryAttempts.delete(clip.id);
   249‚Üí   249‚Üí    }
   250‚Üí   250‚Üí    return null;
   251‚Üí   251‚Üí  }
   252‚Üí   252‚Üí
   253‚Üí   253‚Üí  /** Pre-load (decode + cache) the AudioBuffer for a clip WITHOUT scheduling
   254‚Üí   254‚Üí   *  playback. Used to warm the cache when samples arrive for a clip while
   255‚Üí   255‚Üí   *  the user is paused ‚Äî without this, the first play() goes through the
   256‚Üí   256‚Üí   *  full getBuffer retry loop, adding audible latency to the first
   257‚Üí   257‚Üí   *  playback after a remote sample lands. No-op if the clip doesn't exist
   258‚Üí   258‚Üí   *  or the AudioContext hasn't been created yet (no gesture ‚Üí no decoding). */
   259‚Üí   259‚Üí  async preloadBuffer(slate: SlateDoc, clipId: string): Promise<void> {
   260‚Üí   260‚Üí    if (!this.ctx) return;
   261‚Üí   261‚Üí    const yo = slate.audioClips().get(clipId);
   262‚Üí   262‚Üí    if (!yo) return;
   263‚Üí   263‚Üí    const clip = readAudioClip(yo, clipId);
   264‚Üí   264‚Üí    if (!clip) return;
   265‚Üí   265‚Üí    await this.getBuffer(clip); // pre-warm the cache
   266‚Üí   266‚Üí  }
   267‚Üí   267‚Üí
   268‚Üí   268‚Üí  /** Lazily build the shared FX bus (one reverb + one feedback echo the whole
   269‚Üí   269‚Üí   *  mix shares; per-track send gains feed them). */
   270‚Üí   270‚Üí  private ensureFxBus(): void {
   271‚Üí   271‚Üí    if (this.fxReverb || !this.ctx || !this.masterGain) return;
   272‚Üí   272‚Üí    const ctx = this.ctx;
   273‚Üí   273‚Üí    this.fxReverb = ctx.createConvolver();
   274‚Üí   274‚Üí    this.fxReverb.buffer = this.makeReverbIR(ctx, 2.2, 3);
   275‚Üí   275‚Üí    this.fxReverb.connect(this.masterGain);
   276‚Üí   276‚Üí    this.fxDelay = ctx.createDelay(1);
   277‚Üí   277‚Üí    this.fxDelay.delayTime.value = this.delayTimeForBpm();
   278‚Üí   278‚Üí    this.fxDelayFeedback = ctx.createGain();
   279‚Üí   279‚Üí    this.fxDelayFeedback.gain.value = 0.35;
   280‚Üí   280‚Üí    this.fxDelay.connect(this.fxDelayFeedback);
   281‚Üí   281‚Üí    this.fxDelayFeedback.connect(this.fxDelay);
   282‚Üí   282‚Üí    this.fxDelay.connect(this.masterGain);
   283‚Üí   283‚Üí  }

   285‚Üí   285‚Üí  /** Dotted-eighth echo synced to the project tempo (the classic slap). */
   286‚Üí   286‚Üí  private delayTimeForBpm(): number {
   287‚Üí   287‚Üí    return Math.min(1, 0.375 * (60 / this.bpm));
   288‚Üí   288‚Üí  }
   289‚Üí   289‚Üí
   290‚Üí   290‚Üí  /** Build the per-track chain for the current set of tracks:
   291‚Üí   291‚Üí   *  gain ‚Üí EQ low/mid/high ‚Üí panner ‚Üí master, with post-EQ sends into the
   292‚Üí   292‚Üí   *  shared reverb/delay bus. Values (volume, pan, EQ dB, send levels) are
   293‚Üí   293‚Üí   *  re-applied on every call, so this doubles as the "apply Yjs ‚Üí graph"
   294‚Üí   294‚Üí   *  step for live edits from the settings panel. */
   295‚Üí   295‚Üí  private setupTrackNodes(tracks: AudioTrack[]): void {
   296‚Üí   296‚Üí    const ctx = this.ctx!;
   297‚Üí   297‚Üí    this.ensureFxBus();
   298‚Üí   298‚Üí    // Remove stale nodes.
   299‚Üí   299‚Üí    for (const [id] of this.trackGains) {
   300‚Üí   300‚Üí      if (!tracks.find((t) => t.id === id)) {
   301‚Üí   301‚Üí        this.trackGains.get(id)?.disconnect();
   302‚Üí   302‚Üí        this.trackPanners.get(id)?.disconnect();
   303‚Üí   303‚Üí        const eq = this.trackEqs.get(id);
   304‚Üí   304‚Üí        if (eq) { eq.low.disconnect(); eq.mid.disconnect(); eq.high.disconnect(); }
   305‚Üí   305‚Üí        const sends = this.trackSends.get(id);
   306‚Üí   306‚Üí        if (sends) { sends.reverb.disconnect(); sends.delay.disconnect(); }
   307‚Üí   307‚Üí        this.trackGains.delete(id);
   308‚Üí   308‚Üí        this.trackPanners.delete(id);
   309‚Üí   309‚Üí        this.trackEqs.delete(id);
   310‚Üí   310‚Üí        this.trackSends.delete(id);
   311‚Üí   311‚Üí      }
   312‚Üí   312‚Üí    }
   313‚Üí   313‚Üí    // Create missing nodes.
   314‚Üí   314‚Üí    for (const track of tracks) {
   315‚Üí   315‚Üí      if (!this.trackGains.has(track.id)) {
   316‚Üí   316‚Üí        const gain = ctx.createGain();
   317‚Üí   317‚Üí        const low = ctx.createBiquadFilter();
   318‚Üí   318‚Üí        low.type = 'lowshelf';
   319‚Üí   319‚Üí        low.frequency.value = 200;
   320‚Üí   320‚Üí        const mid = ctx.createBiquadFilter();
   321‚Üí   321‚Üí        mid.type = 'peaking';
   322‚Üí   322‚Üí        mid.frequency.value = 1000;
   323‚Üí   323‚Üí        mid.Q.value = 0.9;
   324‚Üí   324‚Üí        const high = ctx.createBiquadFilter();
   325‚Üí   325‚Üí        high.type = 'highshelf';
   326‚Üí   326‚Üí        high.frequency.value = 4000;
   327‚Üí   327‚Üí        const panner = ctx.createStereoPanner();
   328‚Üí   328‚Üí        gain.connect(low);
   329‚Üí   329‚Üí        low.connect(mid);
   330‚Üí   330‚Üí        mid.connect(high);
   331‚Üí   331‚Üí        high.connect(panner);
   332‚Üí   332‚Üí        panner.connect(this.masterGain!);
   333‚Üí   333‚Üí        const reverbSend = ctx.createGain();
   334‚Üí   334‚Üí        reverbSend.gain.value = 0;
   335‚Üí   335‚Üí        const delaySend = ctx.createGain();
   336‚Üí   336‚Üí        delaySend.gain.value = 0;
   337‚Üí   337‚Üí        high.connect(reverbSend);
   338‚Üí   338‚Üí        high.connect(delaySend);
   339‚Üí   339‚Üí        if (this.fxReverb) reverbSend.connect(this.fxReverb);
   340‚Üí   340‚Üí        if (this.fxDelay) delaySend.connect(this.fxDelay);
   341‚Üí   341‚Üí        this.trackGains.set(track.id, gain);
   342‚Üí   342‚Üí        this.trackPanners.set(track.id, panner);
   343‚Üí   343‚Üí        this.trackEqs.set(track.id, { low, mid, high });
   344‚Üí   344‚Üí        this.trackSends.set(track.id, { reverb: reverbSend, delay: delaySend });
   345‚Üí   345‚Üí      }
   346‚Üí   346‚Üí      // Update values.
   347‚Üí   347‚Üí      const gain = this.trackGains.get(track.id)!;
   348‚Üí   348‚Üí      const panner = this.trackPanners.get(track.id)!;
   349‚Üí   349‚Üí      const eq = this.trackEqs.get(track.id)!;
   350‚Üí   350‚Üí      const sends = this.trackSends.get(track.id)!;
   351‚Üí   351‚Üí      const anySolo = tracks.some((t) => t.solo);
   352‚Üí   352‚Üí      const audible = anySolo ? track.solo : !track.muted;
   353‚Üí   353‚Üí      gain.gain.value = audible ? track.volume : 0;
   354‚Üí   354‚Üí      panner.pan.value = track.pan;
   355‚Üí   355‚Üí      eq.low.gain.value = track.eqLow ?? 0;
   356‚Üí   356‚Üí      eq.mid.gain.value = track.eqMid ?? 0;
   357‚Üí   357‚Üí      eq.high.gain.value = track.eqHigh ?? 0;
   358‚Üí   358‚Üí      sends.reverb.gain.value = track.reverbSend ?? 0;
   359‚Üí   359‚Üí      sends.delay.gain.value = track.delaySend ?? 0;
   360‚Üí   360‚Üí    }
   361‚Üí   361‚Üí  }

   363‚Üí   363‚Üí  /** Play all clips that intersect the current playhead. */
   364‚Üí   364‚Üí  async play(slate: SlateDoc, offset: number): Promise<void> {
   365‚Üí   365‚Üí    const ctx = this.ensureContext();
   366‚Üí   366‚Üí    // Browser autoplay policy: if the AudioContext is still suspended (no
   367‚Üí   367‚Üí    // user gesture yet, or the gesture listener hasn't fired), scheduling
   368‚Üí   368‚Üí    // sources is a silent no-op ‚Äî the user would see the playhead move but
   369‚Üí   369‚Üí    // hear nothing. Bail with an actionable toast instead. The gesture
   370‚Üí   370‚Üí    // listener registered in ensureContext() will resume on the first click,
   371‚Üí   371‚Üí    // and the user can press play again.
   372‚Üí   372‚Üí    if (ctx.state === 'suspended') {
   373‚Üí   373‚Üí      toast({
   374‚Üí   374‚Üí        title: 'Click anywhere to enable audio',
   375‚Üí   375‚Üí        description: 'Your browser blocked audio until you interact with the page.',
   376‚Üí   376‚Üí      });
   377‚Üí   377‚Üí      return;
   378‚Üí   378‚Üí    }
   379‚Üí   379‚Üí    if (this.loopTimer !== null) { clearTimeout(this.loopTimer); this.loopTimer = null; }
   380‚Üí   380‚Üí    this.playing = true;
   381‚Üí   381‚Üí    this.startOffset = offset;
   382‚Üí   382‚Üí    this.startTime = ctx.currentTime;
   383‚Üí   383‚Üí
   384‚Üí   384‚Üí    // Read tracks + clips from the doc.
   385‚Üí   385‚Üí    const tracks: AudioTrack[] = [];
   386‚Üí   386‚Üí    slate.audioTracks().forEach((m, id) => {
   387‚Üí   387‚Üí      const t = readAudioTrack(m, id);
   388‚Üí   388‚Üí      if (t) tracks.push(t);
   389‚Üí   389‚Üí    });
   390‚Üí   390‚Üí    this.setupTrackNodes(tracks);
   391‚Üí   391‚Üí
   392‚Üí   392‚Üí    const clips: AudioClip[] = [];
   393‚Üí   393‚Üí    slate.audioClips().forEach((m, id) => {
   394‚Üí   394‚Üí      const c = readAudioClip(m, id);
   395‚Üí   395‚Üí      if (c) clips.push(c);
   396‚Üí   396‚Üí    });
   397‚Üí   397‚Üí
   398‚Üí   398‚Üí    // Preload EVERY buffer before scheduling anything. Awaiting each buffer
   399‚Üí   399‚Üí    // inside the scheduling loop anchored each clip to a drifting
   400‚Üí   400‚Üí    // ctx.currentTime ‚Äî a clip whose decode (or sample-arrival retry) took
   401‚Üí   401‚Üí    // 500ms started 500ms late relative to the playhead AND to clips scheduled
   402‚Üí   402‚Üí    // before it, so multi-track projects audibly fell out of sync.
   403‚Üí   403‚Üí    //
   404‚Üí   404‚Üí    // MIDI clips have no IndexedDB sample blob (their notes are inline in
   405‚Üí   405‚Üí    // Yjs), so they're skipped in the audio-buffer preload pass ‚Äî they get
   406‚Üí   406‚Üí    // their own preload pass below (warm the soundfont sample cache for
   407‚Üí   407‚Üí    // soundfont-piano clips so the first note doesn't drop).
   408‚Üí   408‚Üí    const buffers = new Map<string, AudioBuffer>();
   409‚Üí   409‚Üí    await Promise.all(
   410‚Üí   410‚Üí      clips.map(async (clip) => {
   411‚Üí   411‚Üí        if (clip.mute) return;
   412‚Üí   412‚Üí        if (clip.kind === 'midi') return; // no PCM samples
   413‚Üí   413‚Üí        const buffer = await this.getBuffer(clip);
   414‚Üí   414‚Üí        if (buffer) buffers.set(clip.id, buffer);
   415‚Üí   415‚Üí      }),
   416‚Üí   416‚Üí    );
   417‚Üí   417‚Üí    if (!this.playing) return; // stopped while buffers were loading
   418‚Üí   418‚Üí
   419‚Üí   419‚Üí    // Register the pitch-shift worklet if any clip needs time-stretch or
   420‚Üí   420‚Üí    // pitch-shift (speed ‚âÝ 1 or pitch ‚âÝ 0). One addModule per context ‚Äî
   421‚Üí   421‚Üí    // memoized, so this is a cheap await after the first play(). Resolves
   422‚Üí   422‚Üí    // false where AudioWorklet is unavailable; scheduling then falls back to
   423‚Üí   423‚Üí    // the old coupled detune behaviour.
   424‚Üí   424‚Üí    const needsShift = clips.some(
   425‚Üí   425‚Üí      (c) => !c.mute && c.kind !== 'midi' && ((c.pitch ?? 0) !== 0 || (c.speed != null && c.speed > 0 && c.speed !== 1)),
   426‚Üí   426‚Üí    );
   427‚Üí   427‚Üí    const shiftReady = needsShift ? await ensurePitchWorklet(ctx) : false;
   428‚Üí   428‚Üí    if (!this.playing) return; // stopped while the worklet was loading
   429‚Üí   429‚Üí
   430‚Üí   430‚Üí    // Pre-warm the soundfont for MIDI clips that use it. Without this the
   431‚Üí   431‚Üí    // first note of each clip would silently fetch its sample and miss its
   432‚Üí   432‚Üí    // scheduled start time (noteOn returns null while loading). We collect
   433‚Üí   433‚Üí    // every distinct midi number across all soundfont-piano clips and ask
   434‚Üí   434‚Üí    // the SoundfontInstrument to preload them in parallel ‚Äî by the time the
   435‚Üí   435‚Üí    // scheduling loop runs, every note is cached and noteOn fires instantly.
   436‚Üí   436‚Üí    const sfMidis = new Set<number>();
   437‚Üí   437‚Üí    for (const clip of clips) {
   438‚Üí   438‚Üí      if (clip.mute || clip.kind !== 'midi') continue;
   439‚Üí   439‚Üí      const instId = this.resolveInstrumentId(clip, tracks);
   440‚Üí   440‚Üí      if (instId === SOUNDFONT_PIANO_ID) {
   441‚Üí   441‚Üí        for (const n of clip.notes ?? []) sfMidis.add(n.midi);
   442‚Üí   442‚Üí      }
   443‚Üí   443‚Üí    }
   444‚Üí   444‚Üí    if (sfMidis.size > 0) {
   445‚Üí   445‚Üí      const sf = this.ensureSoundfont();
   446‚Üí   446‚Üí      if (sf) await sf.preloadNotes(Array.from(sfMidis));
   447‚Üí   447‚Üí    }
   448‚Üí   448‚Üí    if (!this.playing) return; // stopped during soundfont preload
   449‚Üí   449‚Üí
   450‚Üí   450‚Üí    // Re-anchor NOW that everything is ready: all clips schedule against one
   451‚Üí   451‚Üí    // shared clock read, so their relative timing is sample-accurate.
   452‚Üí   452‚Üí    this.startTime = ctx.currentTime;
   453‚Üí   453‚Üí
   454‚Üí   454‚Üí    // Schedule each clip that starts after the playhead or is currently playing.
   455‚Üí   455‚Üí    for (const clip of clips) {
   456‚Üí   456‚Üí      const track = tracks.find((t) => t.id === clip.trackId);
   457‚Üí   457‚Üí      if (!track) continue;
   458‚Üí   458‚Üí      if (clip.mute) continue; // clip muted individually
   459‚Üí   459‚Üí
   460‚Üí   460‚Üí      const clipEnd = clip.start + clip.duration;
   461‚Üí   461‚Üí      if (clipEnd <= offset) continue; // clip already finished
   462‚Üí   462‚Üí
   463‚Üí   463‚Üí      // ‚îÄ‚îÄ MIDI branch ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ
   464‚Üí   464‚Üí      // For MIDI clips, schedule each note as an instrument voice (soundfont
   465‚Üí   465‚Üí      // sample or oscillator synth preset) at startTime + (clip.start +
   466‚Üí   466‚Üí      // note.start - offset). Notes whose start has already passed (the
   467‚Üí   467‚Üí      // playhead is partway through the clip) are scheduled with a clamped
   468‚Üí   468‚Üí      // start time so they fire immediately rather than in the past (Web Audio
   469‚Üí   469‚Üí      // throws if source.start(when) is called with when < currentTime).
   470‚Üí   470‚Üí      if (clip.kind === 'midi') {
   471‚Üí   471‚Üí        const trackGain = this.trackGains.get(track.id) ?? this.masterGain!;
   472‚Üí   472‚Üí        this.scheduleMidiClip(ctx, clip, track, offset, trackGain);
   473‚Üí   473‚Üí        continue;
   474‚Üí   474‚Üí      }
   475‚Üí   475‚Üí
   476‚Üí   476‚Üí      const buffer = buffers.get(clip.id);
   477‚Üí   477‚Üí      if (!buffer) continue;
   478‚Üí   478‚Üí
   479‚Üí   479‚Üí      const speed = clip.speed && clip.speed > 0 ? clip.speed : 1;
   480‚Üí   480‚Üí      const pitchCents = clip.pitch ?? 0;
   481‚Üí   481‚Üí      const clipOffset = clip.offset ?? 0; // trim from the source (buffer seconds)
   482‚Üí   482‚Üí      const clipVol = clip.gain ?? 1;
   483‚Üí   483‚Üí
   484‚Üí   484‚Üí      const source = ctx.createBufferSource();
   485‚Üí   485‚Üí      source.buffer = buffer;
   486‚Üí   486‚Üí      // Speed and pitch, DECOUPLED like a real editor (Audition's Stretch &
   487‚Üí   487‚Üí      // Pitch / Premiere's "maintain pitch"):
   488‚Üí   488‚Üí      //   - playbackRate = speed sets the timeline duration (and would shift
   489‚Üí   489‚Üí      //     pitch by √óspeed as a Web Audio side effect);
   490‚Üí   490‚Üí      //   - a granular pitch-shift worklet after the source scales frequency
   491‚Üí   491‚Üí      //     by 2^(cents/1200) / speed ‚Äî cancelling the varispeed side effect
   492‚Üí   492‚Üí      //     and applying the user's pitch knob on top.
   493‚Üí   493‚Üí      // Net: Speed changes duration but keeps pitch; Pitch shifts semitones
   494‚Üí   494‚Üí      // but keeps duration. Unity ratio bypasses the shifter entirely (the
   495‚Üí   495‚Üí      // common path stays two nodes). If the worklet isn't available, fall
   496‚Üí   496‚Üí      // back to the old coupled detune.
   497‚Üí   497‚Üí      source.playbackRate.value = speed;
   498‚Üí   498‚Üí      const shiftRatio = Math.pow(2, pitchCents / 1200) / speed;
   499‚Üí   499‚Üí      let shifter: AudioWorkletNode | undefined;
   500‚Üí   500‚Üí      let clipHead: AudioNode = source;
   501‚Üí   501‚Üí      if (shiftReady && Math.abs(shiftRatio - 1) > 0.0005) {
   502‚Üí   502‚Üí        shifter = createPitchShifter(ctx, shiftRatio);
   503‚Üí   503‚Üí        source.connect(shifter);
   504‚Üí   504‚Üí        clipHead = shifter;
   505‚Üí   505‚Üí        // Let the shifter ring out its buffered tail and self-terminate when
   506‚Üí   506‚Üí        // the source ends (naturally or via stop()) ‚Äî worklet nodes whose
   507‚Üí   507‚Üí        // process() keeps returning true are never GC'd.
   508‚Üí   508‚Üí        const s = shifter;
   509‚Üí   509‚Üí        source.onended = () => releasePitchShifter(s);
   510‚Üí   510‚Üí      } else if (!shiftReady && pitchCents !== 0) {
   511‚Üí   511‚Üí        source.detune.value = pitchCents; // legacy coupled fallback
   512‚Üí   512‚Üí      }
   513‚Üí   513‚Üí
   514‚Üí   514‚Üí      // Per-clip gain (fades + clip volume) ‚Üí per-clip panner ‚Üí optional
   515‚Üí   515‚Üí      // HP/LP filters ‚Üí track chain. Filters are only inserted when engaged
   516‚Üí   516‚Üí      // (cutoff moved off the ends of the audible range) so the default path
   517‚Üí   517‚Üí      // stays two nodes.
   518‚Üí   518‚Üí      const clipGain = ctx.createGain();
   519‚Üí   519‚Üí      const clipPan = ctx.createStereoPanner();
   520‚Üí   520‚Üí      clipPan.pan.value = Math.max(-1, Math.min(1, clip.pan ?? 0));
   521‚Üí   521‚Üí      clipHead.connect(clipGain);
   522‚Üí   522‚Üí      clipGain.connect(clipPan);
   523‚Üí   523‚Üí      let clipTail: AudioNode = clipPan;
   524‚Üí   524‚Üí      const hp = clip.hpCutoff ?? 20;
   525‚Üí   525‚Üí      if (hp > 22) {
   526‚Üí   526‚Üí        const hpf = ctx.createBiquadFilter();
   527‚Üí   527‚Üí        hpf.type = 'highpass';
   528‚Üí   528‚Üí        hpf.frequency.value = hp;
   529‚Üí   529‚Üí        clipTail.connect(hpf);
   530‚Üí   530‚Üí        clipTail = hpf;
   531‚Üí   531‚Üí      }
   532‚Üí   532‚Üí      const lp = clip.lpCutoff ?? 20000;
   533‚Üí   533‚Üí      if (lp < 19500) {
   534‚Üí   534‚Üí        const lpf = ctx.createBiquadFilter();
   535‚Üí   535‚Üí        lpf.type = 'lowpass';
   536‚Üí   536‚Üí        lpf.frequency.value = lp;
   537‚Üí   537‚Üí        clipTail.connect(lpf);
   538‚Üí   538‚Üí        clipTail = lpf;
   539‚Üí   539‚Üí      }
   540‚Üí   540‚Üí      const trackGain = this.trackGains.get(track.id);
   541‚Üí   541‚Üí      if (trackGain) clipTail.connect(trackGain);
   542‚Üí   542‚Üí      else clipTail.connect(this.masterGain!);
   543‚Üí   543‚Üí
   544‚Üí   544‚Üí      // How far the playhead is already into the clip (real seconds), and where
   545‚Üí   545‚Üí      // that lands in the source buffer (buffer seconds ‚Üí scaled by speed).
   546‚Üí   546‚Üí      const skipTo = Math.max(0, offset - clip.start);
   547‚Üí   547‚Üí      const whenToStart = this.startTime + Math.max(0, clip.start - offset);
   548‚Üí   548‚Üí      const playDuration = clip.duration - skipTo; // real seconds left on the timeline
   549‚Üí   549‚Üí      const bufStart = clipOffset + skipTo * speed;
   550‚Üí   550‚Üí      const bufDuration = playDuration * speed;
   551‚Üí   551‚Üí
   552‚Üí   552‚Üí      // Latency compensation: the shifter's granular window delays audio by
   553‚Üí   553‚Üí      // ~45ms on average, so shifted clips would land late against unshifted
   554‚Üí   554‚Üí      // ones. Start the SOURCE early by that amount when there's headroom ‚Äî
   555‚Üí   555‚Üí      // the audio emerges from the shifter on time. (Fades below stay
   556‚Üí   556‚Üí      // anchored to whenToStart: the gain node sits after the shifter, in
   557‚Üí   557‚Üí      // timeline time.)
   558‚Üí   558‚Üí      const srcStart = shifter ? Math.max(ctx.currentTime, whenToStart - PITCH_SHIFT_LATENCY) : whenToStart;
   559‚Üí   559‚Üí      source.start(srcStart, bufStart, bufDuration);
   560‚Üí   560‚Üí
   561‚Üí   561‚Üí      // Fades (real-time). Base level is the clip gain.
   562‚Üí   562‚Üí      clipGain.gain.setValueAtTime(clip.fadeIn > 0 ? 0 : clipVol, whenToStart);
   563‚Üí   563‚Üí      if (clip.fadeIn > 0) {
   564‚Üí   564‚Üí        clipGain.gain.linearRampToValueAtTime(clipVol, whenToStart + clip.fadeIn);
   565‚Üí   565‚Üí      }
   566‚Üí   566‚Üí      if (clip.fadeOut > 0) {
   567‚Üí   567‚Üí        clipGain.gain.setValueAtTime(clipVol, whenToStart + playDuration - clip.fadeOut);
   568‚Üí   568‚Üí        clipGain.gain.linearRampToValueAtTime(0, whenToStart + playDuration);
   569‚Üí   569‚Üí      }
   570‚Üí   570‚Üí
   571‚Üí   571‚Üí      this.playingClips.push({ source, gain: clipGain, clipId: clip.id, shifter });
   572‚Üí   572‚Üí    }
   573‚Üí   573‚Üí
   574‚Üí   574‚Üí    // Loop region: getPosition() wraps the PLAYHEAD at the loop end, but the
   575‚Üí   575‚Üí    // scheduled sources would happily play straight through it ‚Äî the display
   576‚Üí   576‚Üí    // looped while the audio didn't. Schedule a restart at the boundary so the
   577‚Üí   577‚Üí    // audio wraps with the playhead.
   578‚Üí   578‚Üí    if (this.loopRegion && this.loopRegion.end > this.loopRegion.start && offset < this.loopRegion.end) {
   579‚Üí   579‚Üí      const untilWrap = this.loopRegion.end - offset;
   580‚Üí   580‚Üí      this.loopTimer = window.setTimeout(() => {
   581‚Üí   581‚Üí        this.loopTimer = null;
   582‚Üí   582‚Üí        if (this.playing && this.loopRegion) this.restartPlayback(slate, this.loopRegion.start);
   583‚Üí   583‚Üí      }, untilWrap * 1000);
   584‚Üí   584‚Üí    }
   585‚Üí   585‚Üí
   586‚Üí   586‚Üí    // Start metronome if enabled.
   587‚Üí   587‚Üí    if (this.metronomeOn) this.startMetronome();
   588‚Üí   588‚Üí  }

   590‚Üí   590‚Üí  /** Stop all playback. */
   591‚Üí   591‚Üí  stop(): void {
   592‚Üí   592‚Üí    this.playing = false;
   593‚Üí   593‚Üí    if (this.loopTimer !== null) { clearTimeout(this.loopTimer); this.loopTimer = null; }
   594‚Üí   594‚Üí    for (const { source, shifter } of this.playingClips) {
   595‚Üí   595‚Üí      try { source.stop(); } catch { /* already stopped */ }
   596‚Üí   596‚Üí      if (shifter) releasePitchShifter(shifter); // idempotent (onended also fires)
   597‚Üí   597‚Üí    }
   598‚Üí   598‚Üí    this.playingClips = [];
   599‚Üí   599‚Üí    // Release every live MIDI voice too ‚Äî without this, oscillator-based
   600‚Üí   600‚Üí    // presets would ring past stop() until their natural decay finished, and
   601‚Üí   601‚Üí    // soundfont samples would play out their full release tail.
   602‚Üí   602‚Üí    if (this.ctx) {
   603‚Üí   603‚Üí      const now = this.ctx.currentTime;
   604‚Üí   604‚Üí      for (const v of this.playingMidiVoices) {
   605‚Üí   605‚Üí        try { v.stop(now); } catch { /* already stopped */ }
   606‚Üí   606‚Üí      }
   607‚Üí   607‚Üí    }
   608‚Üí   608‚Üí    this.playingMidiVoices = [];
   609‚Üí   609‚Üí    this.stopMetronome();
   610‚Üí   610‚Üí  }
   611‚Üí   611‚Üí
   612‚Üí   612‚Üí  /** Stop all current sources and immediately re-schedule all clips from
   613‚Üí   613‚Üí   *  `offset`, atomically ‚Äî `playing` stays true throughout so getPosition()
   614‚Üí   614‚Üí   *  keeps tracking from the new startTime (no jump back to 0, no UI glitch).
   615‚Üí   615‚Üí   *  Used when the clip set changes mid-playback (a remote peer adds a clip,
   616‚Üí   616‚Üí   *  or a clip's samples arrive after play started) so the new clip is picked
   617‚Üí   617‚Üí   *  up without a full stop+play round-trip from the UI. There's a brief audio
   618‚Üí   618‚Üí   *  gap while buffers reload, but the playhead and `playing` state never
   619‚Üí   619‚Üí   *  flicker. If the AudioContext hasn't been created yet (no user gesture),
   620‚Üí   620‚Üí   *  falls back to a plain play() which creates it. */
   621‚Üí   621‚Üí  restartPlayback(slate: SlateDoc, offset: number): void {
   622‚Üí   622‚Üí    if (!this.ctx) {
   623‚Üí   623‚Üí      void this.play(slate, offset);
   624‚Üí   624‚Üí      return;
   625‚Üí   625‚Üí    }
   626‚Üí   626‚Üí    // Stop + forget current sources, but keep `playing` true so getPosition()
   627‚Üí   627‚Üí    // continues to track. play() will reset startTime/startOffset.
   628‚Üí   628‚Üí    for (const { source, shifter } of this.playingClips) {
   629‚Üí   629‚Üí      try { source.stop(); } catch { /* already stopped */ }
   630‚Üí   630‚Üí      if (shifter) releasePitchShifter(shifter); // idempotent (onended also fires)
   631‚Üí   631‚Üí    }
   632‚Üí   632‚Üí    this.playingClips = [];
   633‚Üí   633‚Üí    // Release live MIDI voices too (see stop()).
   634‚Üí   634‚Üí    {
   635‚Üí   635‚Üí      const now = this.ctx.currentTime;
   636‚Üí   636‚Üí      for (const v of this.playingMidiVoices) {
   637‚Üí   637‚Üí        try { v.stop(now); } catch { /* already stopped */ }
   638‚Üí   638‚Üí      }
   639‚Üí   639‚Üí    }
   640‚Üí   640‚Üí    this.playingMidiVoices = [];
   641‚Üí   641‚Üí    this.stopMetronome();
   642‚Üí   642‚Üí    // Re-schedule. play() re-reads the doc (so new clips are included),
   643‚Üí   643‚Üí    // resets startTime/startOffset, and restarts the metronome if enabled.
   644‚Üí   644‚Üí    void this.play(slate, offset);
   645‚Üí   645‚Üí  }

   647‚Üí   647‚Üí  /** Get the current playhead position (seconds). Loops if a loop region is set. */
   648‚Üí   648‚Üí  getPosition(): number {
   649‚Üí   649‚Üí    if (!this.playing || !this.ctx) return this.startOffset;
   650‚Üí   650‚Üí    let pos = this.startOffset + (this.ctx.currentTime - this.startTime);
   651‚Üí   651‚Üí    if (this.loopRegion) {
   652‚Üí   652‚Üí      const loopLen = this.loopRegion.end - this.loopRegion.start;
   653‚Üí   653‚Üí      if (loopLen > 0 && pos >= this.loopRegion.end) {
   654‚Üí   654‚Üí        // Wrap around the loop region.
   655‚Üí   655‚Üí        pos = this.loopRegion.start + ((pos - this.loopRegion.start) % loopLen);
   656‚Üí   656‚Üí      }
   657‚Üí   657‚Üí    }
   658‚Üí   658‚Üí    return pos;
   659‚Üí   659‚Üí  }

   693‚Üí   693‚Üí  /** Create a simple reverb impulse response for the ConvolverNode. */
   694‚Üí   694‚Üí  private makeReverbIR(ctx: AudioContext, seconds: number, decay: number): AudioBuffer {
   695‚Üí   695‚Üí    const rate = ctx.sampleRate;
   696‚Üí   696‚Üí    const length = Math.max(1, Math.floor(rate * seconds));
   697‚Üí   697‚Üí    const ir = ctx.createBuffer(2, length, rate);
   698‚Üí   698‚Üí    for (let ch = 0; ch < 2; ch++) {
   699‚Üí   699‚Üí      const data = ir.getChannelData(ch);
   700‚Üí   700‚Üí      for (let i = 0; i < length; i++) {
   701‚Üí   701‚Üí        data[i] = (Math.random() * 2 - 1) * Math.pow(1 - i / length, decay);
   702‚Üí   702‚Üí      }
   703‚Üí   703‚Üí    }
   704‚Üí   704‚Üí    return ir;
   705‚Üí   705‚Üí  }

   746‚Üí   746‚Üí  private click(time: number): void {
   747‚Üí   747‚Üí    if (!this.ctx || !this.masterGain) return;
   748‚Üí   748‚Üí    const osc = this.ctx.createOscillator();
   749‚Üí   749‚Üí    const gain = this.ctx.createGain();
   750‚Üí   750‚Üí    osc.frequency.value = 1000; // Click pitch
   751‚Üí   751‚Üí    gain.gain.setValueAtTime(0.3, time);
   752‚Üí   752‚Üí    gain.gain.exponentialRampToValueAtTime(0.001, time + 0.05);
   753‚Üí   753‚Üí    osc.connect(gain);
   754‚Üí   754‚Üí    gain.connect(this.masterGain);
   755‚Üí   755‚Üí    osc.start(time);
   756‚Üí   756‚Üí    osc.stop(time + 0.05);
   757‚Üí   757‚Üí  }

   759‚Üí   759‚Üí  /** Start recording from a microphone input. Returns a stop callback that
   760‚Üí   760‚Üí   *  resolves to the recorded PCM samples. */
   761‚Üí   761‚Üí  async startRecording(): Promise<() => Promise<{ samples: number[]; sampleRate: number; channels: number; duration: number }>> {
   762‚Üí   762‚Üí    const ctx = this.ensureContext();
   763‚Üí   763‚Üí    const stream = await navigator.mediaDevices.getUserMedia({ audio: true });
   764‚Üí   764‚Üí    const source = ctx.createMediaStreamSource(stream);
   765‚Üí   765‚Üí    const processor = ctx.createScriptProcessor(4096, 1, 1);

   774‚Üí   774‚Üí

   776‚Üí   776‚Üí      processor.disconnect();

   778‚Üí   778‚Üí      stream.getTracks().forEach((t) => t.stop());
   779‚Üí   779‚Üí      this.recording = false;
   780‚Üí   780‚Üí      const duration = chunks.length / ctx.sampleRate;
   781‚Üí   781‚Üí      return { samples: chunks, sampleRate: ctx.sampleRate, channels: 1, duration };
   782‚Üí   782‚Üí    };

   840‚Üí   840‚Üí   *  audio clips ‚Äî but per-track routing for MIDI clips is handled in
   841‚Üí   841‚Üí   *  scheduleMidiClip(), which passes the track's gain-chain tail as `dest`
   842‚Üí   842‚Üí   *  to a per-call noteOn (the SoundfontInstrument constructor only needs a
   843‚Üí   843‚Üí   *  fallback destination; the per-clip routing overrides it).
   844‚Üí   844‚Üí   *
   845‚Üí   845‚Üí   *  Returns null if the AudioContext doesn't exist yet (no gesture). The
   846‚Üí   846‚Üí   *  caller (play()) bails on suspended ctx anyway, so this is mostly a
   847‚Üí   847‚Üí   *  belt-and-braces guard. */
   848‚Üí   848‚Üí  private ensureSoundfont(): SoundfontInstrument | null {
   849‚Üí   849‚Üí    if (!this.ctx || !this.masterGain) return null;
   850‚Üí   850‚Üí    if (!this.soundfont) {
   851‚Üí   851‚Üí      // The dest here is a fallback only ‚Äî scheduleMidiClip routes each note
   852‚Üí   852‚Üí      // through the track's gain chain. The SoundfontInstrument connects its
   853‚Üí   853‚Üí      // per-note gain nodes to whatever `dest` is at noteOn time, so the
   854‚Üí   854‚Üí      // constructor's dest is only used if noteOn is called without a dest
   855‚Üí   855‚Üí      // override (it isn't, in our usage).
   856‚Üí   856‚Üí      this.soundfont = new SoundfontInstrument(this.ctx, this.masterGain);
   857‚Üí   857‚Üí    }
   858‚Üí   858‚Üí    return this.soundfont;
   859‚Üí   859‚Üí  }
   860‚Üí   860‚Üí
   861‚Üí   861‚Üí  /** Resolve which instrument a MIDI clip should play through. Clip-level
   862‚Üí   862‚Üí   *  `instrumentId` wins; otherwise the track's `instrumentId` is used; if
   863‚Üí   863‚Üí   *  neither is set, fall back to the grand-piano preset (a sensible default
   864‚Üí   864‚Üí   *  for any MIDI clip ‚Äî it's what a user expects when they create a MIDI
   865‚Üí   865‚Üí   *  track and drop notes on it without picking an instrument). */
   866‚Üí   866‚Üí  private resolveInstrumentId(clip: AudioClip, tracks: AudioTrack[]): string {
   867‚Üí   867‚Üí    if (clip.instrumentId) return clip.instrumentId;
   868‚Üí   868‚Üí    const track = tracks.find((t) => t.id === clip.trackId);
   869‚Üí   869‚Üí    if (track?.instrumentId) return track.instrumentId;
   870‚Üí   870‚Üí    return INSTRUMENT_PRESETS[0]!.id; // 'inst-grand-piano'
   871‚Üí   871‚Üí  }
   872‚Üí   872‚Üí
   873‚Üí   873‚Üí  /** Look up a synth preset by id (factory presets first, then custom
   874‚Üí   874‚Üí   *  instruments from localStorage). Falls back to the grand-piano preset
   875‚Üí   875‚Üí   *  if the id is unknown ‚Äî better to play *something* than silently drop
   876‚Üí   876‚Üí   *  the clip because the user deleted a custom instrument mid-session. */
   877‚Üí   877‚Üí  private resolveSynthParams(id: string): InstrumentParams {
   878‚Üí   878‚Üí    const factory = INSTRUMENT_PRESETS.find((p) => p.id === id);
   879‚Üí   879‚Üí    if (factory) return factory;
   880‚Üí   880‚Üí    // loadCustomInstruments reads localStorage ‚Äî cheap, but only worth doing
   881‚Üí   881‚Üí    // when the id isn't a factory preset. We don't memoize because the set of
   882‚Üí   882‚Üí    // custom instruments can change mid-session (user saves/deletes one).
   883‚Üí   883‚Üí    const custom = loadCustomInstruments().find((p) => p.id === id);
   884‚Üí   884‚Üí    if (custom) return custom;
   885‚Üí   885‚Üí    return INSTRUMENT_PRESETS[0]!;
   886‚Üí   886‚Üí  }
   887‚Üí   887‚Üí
   888‚Üí   888‚Üí  /** Schedule every note in a MIDI clip. Each note fires at
   889‚Üí   889‚Üí   *  `startTime + (clip.start + note.start - offset)` and releases at the
   890‚Üí   890‚Üí   *  note's start + duration. Notes whose start time is already in the past
   891‚Üí   891‚Üí   *  (the playhead is partway through the clip) are clamped to fire
   892‚Üí   892‚Üí   *  immediately ‚Äî Web Audio throws if source.start(when) is called with a
   893‚Üí   893‚Üí   *  `when` < ctx.currentTime.
   894‚Üí   894‚Üí   *
   895‚Üí   895‚Üí   *  The instrument is selected by `resolveInstrumentId`:
   896‚Üí   896‚Üí   *  - 'soundfont-piano' ‚Üí SoundfontInstrument (sample-based, freepats)
   897‚Üí   897‚Üí   *  - any other id ‚Üí oscillator-based LiveInstrument preset (startVoice)
   898‚Üí   898‚Üí   *
   899‚Üí   899‚Üí   *  The voice's output is routed through the track's gain chain (the same
   900‚Üí   900‚Üí   *  one audio clips use) so MIDI clips get the same volume/pan/EQ/sends/
   901‚Üí   901‚Üí   *  mute/solo treatment as audio clips. */
   902‚Üí   902‚Üí  private scheduleMidiClip(
   903‚Üí   903‚Üí    ctx: AudioContext,
   904‚Üí   904‚Üí    clip: AudioClip,
   905‚Üí   905‚Üí    _track: AudioTrack,
   906‚Üí   906‚Üí    offset: number,
   907‚Üí   907‚Üí    dest: AudioNode,
   908‚Üí   908‚Üí  ): void {
   909‚Üí   909‚Üí    const notes = clip.notes ?? [];
   910‚Üí   910‚Üí    if (notes.length === 0) return;
   911‚Üí   911‚Üí    const instId = this.resolveInstrumentId(clip, [_track]);
   912‚Üí   912‚Üí    const clipVol = clip.gain ?? 1;
   913‚Üí   913‚Üí
   914‚Üí   914‚Üí    if (instId === SOUNDFONT_PIANO_ID) {
   915‚Üí   915‚Üí      const sf = this.ensureSoundfont();
   916‚Üí   916‚Üí      if (!sf) return;
   917‚Üí   917‚Üí      for (const n of notes) {
   918‚Üí   918‚Üí        const noteAbsStart = clip.start + n.start;
   919‚Üí   919‚Üí        if (noteAbsStart + n.duration <= offset) continue; // note already ended
   920‚Üí   920‚Üí        // Clamp to currentTime so a note whose scheduled start is in the past
   921‚Üí   921‚Üí        // (playhead partway through the clip) fires NOW instead of throwing.
   922‚Üí   922‚Üí        const when = Math.max(ctx.currentTime, this.startTime + Math.max(0, noteAbsStart - offset));
   923‚Üí   923‚Üí        const voice = sf.noteOn(n.midi, n.velocity * clipVol, when);
   924‚Üí   924‚Üí        if (!voice) continue; // sample not loaded (will be on next play)
   925‚Üí   925‚Üí        const stopWhen = Math.max(when + 0.01, this.startTime + Math.max(0, noteAbsStart + n.duration - offset));
   926‚Üí   926‚Üí        voice.stop(stopWhen);
   927‚Üí   927‚Üí        this.playingMidiVoices.push({ clipId: clip.id, stop: voice.stop });
   928‚Üí   928‚Üí      }
   929‚Üí   929‚Üí      return;
   930‚Üí   930‚Üí    }
   931‚Üí   931‚Üí
   932‚Üí   932‚Üí    // Oscillator-based synth preset.
   933‚Üí   933‚Üí    const params = this.resolveSynthParams(instId);
   934‚Üí   934‚Üí    for (const n of notes) {
   935‚Üí   935‚Üí      const noteAbsStart = clip.start + n.start;
   936‚Üí   936‚Üí      if (noteAbsStart + n.duration <= offset) continue; // note already ended
   937‚Üí   937‚Üí      const when = Math.max(ctx.currentTime, this.startTime + Math.max(0, noteAbsStart - offset));
   938‚Üí   938‚Üí      const voice = startVoice(ctx, dest, params, n.midi, n.velocity * clipVol, when);
   939‚Üí   939‚Üí      const stopWhen = Math.max(when + 0.02, this.startTime + Math.max(0, noteAbsStart + n.duration - offset));
   940‚Üí   940‚Üí      voice.stop(stopWhen);
   941‚Üí   941‚Üí      this.playingMidiVoices.push({ clipId: clip.id, stop: voice.stop });
   942‚Üí   942‚Üí    }
   943‚Üí   943‚Üí  }
   944‚Üí   944‚Üí
   945‚Üí   945‚Üí  dispose(): void {
   946‚Üí   946‚Üí    this.stop();
   947‚Üí   947‚Üí    if (this.gestureHandler) {
   948‚Üí   948‚Üí      window.removeEventListener('pointerdown', this.gestureHandler);
   949‚Üí   949‚Üí      window.removeEventListener('keydown', this.gestureHandler);
   950‚Üí   950‚Üí      window.removeEventListener('touchstart', this.gestureHandler);
   951‚Üí   951‚Üí      this.gestureHandler = null;
   952‚Üí   952‚Üí    }
   953‚Üí   953‚Üí    if (this.clipChangedHandler) {
   954‚Üí   954‚Üí      window.removeEventListener('slate:audio-clip-changed', this.clipChangedHandler);
   955‚Üí   955‚Üí      this.clipChangedHandler = null;
   956‚Üí   956‚Üí    }
   957‚Üí   957‚Üí    this.trackGains.forEach((g) => g.disconnect());
   958‚Üí   958‚Üí    this.trackPanners.forEach((p) => p.disconnect());
   959‚Üí   959‚Üí    this.trackEqs.forEach((eq) => { eq.low.disconnect(); eq.mid.disconnect(); eq.high.disconnect(); });
   960‚Üí   960‚Üí    this.trackSends.forEach((s) => { s.reverb.disconnect(); s.delay.disconnect(); });
   961‚Üí   961‚Üí    this.trackGains.clear();
   962‚Üí   962‚Üí    this.trackPanners.clear();
   963‚Üí   963‚Üí    this.trackEqs.clear();
   964‚Üí   964‚Üí    this.trackSends.clear();
   965‚Üí   965‚Üí    this.fxReverb?.disconnect();
   966‚Üí   966‚Üí    this.fxDelay?.disconnect();
   967‚Üí   967‚Üí    this.fxDelayFeedback?.disconnect();
   968‚Üí   968‚Üí    this.fxReverb = null;
   969‚Üí   969‚Üí    this.fxDelay = null;
   970‚Üí   970‚Üí    this.fxDelayFeedback = null;
   971‚Üí   971‚Üí    this.bufferCache.clear();
   972‚Üí   972‚Üí    this.retryingClips.clear();
   973‚Üí   973‚Üí    this.retryAttempts.clear();
   974‚Üí   974‚Üí    this.soundfont?.dispose();
   975‚Üí   975‚Üí    this.soundfont = null;
   976‚Üí   976‚Üí    if (this.ctx) {
   977‚Üí   977‚Üí      void this.ctx.close();
   978‚Üí   978‚Üí      this.ctx = null;
   979‚Üí   979‚Üí      this.masterGain = null;
   980‚Üí   980‚Üí    }
   981‚Üí   981‚Üí  }
   982‚Üí   982‚Üí}
