    62тЖТ    62тЖТ/** Nearest start position at which a clip of `duration` fits without
    63тЖТ    63тЖТ *  overlapping any blocker. Merges blockers into occupied intervals, then
    64тЖТ    64тЖТ *  picks the gap position closest to `desired`. This replaces the previous
    65тЖТ    65тЖТ *  "slide to the nearest side of the current blocker" iteration, which could
    66тЖТ    66тЖТ *  ping-pong between two adjacent blockers (each resolution landing on the
    67тЖТ    67тЖТ *  other) and exhaust its iteration budget while STILL overlapping тАФ
    68тЖТ    68тЖТ *  committing a clip on top of another one. Scanning gaps directly is
    69тЖТ    69тЖТ *  deterministic: the open-ended gap after the last blocker always fits, so a
    70тЖТ    70тЖТ *  valid position always exists. */
    71тЖТ    71тЖТexport function nearestFreeStart(desired: number, duration: number, blockersIn: { start: number; end: number }[]): number {
    72тЖТ    72тЖТ  const eps = 1e-4;
    73тЖТ    73тЖТ  desired = Math.max(0, desired);
    74тЖТ    74тЖТ  if (blockersIn.length === 0) return desired;
    75тЖТ    75тЖТ  const sorted = [...blockersIn].sort((a, b) => a.start - b.start);
    76тЖТ    76тЖТ  const occupied: { start: number; end: number }[] = [];
    77тЖТ    77тЖТ  for (const b of sorted) {
    78тЖТ    78тЖТ    const last = occupied[occupied.length - 1];
    79тЖТ    79тЖТ    if (last && b.start <= last.end + eps) last.end = Math.max(last.end, b.end);
    80тЖТ    80тЖТ    else occupied.push({ start: b.start, end: b.end });
    81тЖТ    81тЖТ  }
    82тЖТ    82тЖТ  let best = desired;
    83тЖТ    83тЖТ  let bestDist = Infinity;
    84тЖТ    84тЖТ  const consider = (lo: number, hi: number) => {
    85тЖТ    85тЖТ    if (hi - lo < duration - eps) return; // gap too small
    86тЖТ    86тЖТ    const cand = Math.min(Math.max(desired, lo), hi - duration);
    87тЖТ    87тЖТ    const dist = Math.abs(cand - desired);
    88тЖТ    88тЖТ    if (dist < bestDist) { bestDist = dist; best = cand; }
    89тЖТ    89тЖТ  };
    90тЖТ    90тЖТ  let prevEnd = 0;
    91тЖТ    91тЖТ  for (const o of occupied) {
    92тЖТ    92тЖТ    consider(prevEnd, o.start);
    93тЖТ    93тЖТ    prevEnd = Math.max(prevEnd, o.end);
    94тЖТ    94тЖТ  }
    95тЖТ    95тЖТ  consider(prevEnd, Infinity); // open-ended tail тАФ always fits
    96тЖТ    96тЖТ  return Math.max(0, best);
    97тЖТ    97тЖТ}
    98тЖТ    98тЖТ
    99тЖТ    99тЖТ/** Clamp a multi-clip drag delta so the WHOLE group stays overlap-free.
   100тЖТ   100тЖТ *  Each selected clip moves by the same `dt`; for each clip we collect the
   101тЖТ   101тЖТ *  range of `dt` values that keep it off its non-selected neighbours on the
   102тЖТ   102тЖТ *  same track, then intersect those ranges across all clips. If the desired
   103тЖТ   103тЖТ *  `dt` falls outside the intersection, we clamp to the nearest bound тАФ so a
   104тЖТ   104тЖТ *  group dragged into a wall stops flush against it instead of teleporting
   105тЖТ   105тЖТ *  past. The origin track per clip is used (multi-drag is horizontal-only тАФ
   106тЖТ   106тЖТ *  no cross-track reshuffle, which would be ambiguous for a group). */
   107тЖТ   107тЖТfunction clampGroupDt(
   108тЖТ   108тЖТ  dt: number,
   109тЖТ   109тЖТ  origins: Map<string, { os: number; od: number; trackId: string }>,
   110тЖТ   110тЖТ  byTrack: Map<string, { start: number; end: number }[]>,
   111тЖТ   111тЖТ): number {
   112тЖТ   112тЖТ  let minRight = Infinity; // upper bound on dt (blocker to the right of a clip)
   113тЖТ   113тЖТ  let maxLeft = -Infinity; // lower bound on dt (blocker to the left of a clip)
   114тЖТ   114тЖТ  for (const o of origins.values()) {
   115тЖТ   115тЖТ    const blockers = byTrack.get(o.trackId) ?? [];
   116тЖТ   116тЖТ    for (const b of blockers) {
   117тЖТ   117тЖТ      // Blocker entirely to the LEFT of the clip's origin тЖТ moving LEFT past
   118тЖТ   118тЖТ      // it would overlap. Constraint: dt >= b.end - o.os.
   119тЖТ   119тЖТ      if (b.end <= o.os + 1e-4) {
   120тЖТ   120тЖТ        maxLeft = Math.max(maxLeft, b.end - o.os);
   121тЖТ   121тЖТ      }
   122тЖТ   122тЖТ      // Blocker entirely to the RIGHT of the clip's origin тЖТ moving RIGHT past
   123тЖТ   123тЖТ      // it would overlap. Constraint: dt <= b.start - o.os - o.od.
   124тЖТ   124тЖТ      else if (b.start >= o.os + o.od - 1e-4) {
   125тЖТ   125тЖТ        minRight = Math.min(minRight, b.start - o.os - o.od);
   126тЖТ   126тЖТ      }
   127тЖТ   127тЖТ      // A blocker that straddles the clip's original interval would mean the
   128тЖТ   128тЖТ      // clip was ALREADY overlapping before the drag тАФ ignore (shouldn't happen).
   129тЖТ   129тЖТ    }
   130тЖТ   130тЖТ  }
   131тЖТ   131тЖТ  if (maxLeft > minRight) return 0; // conflicting constraints тАФ no movement
   132тЖТ   132тЖТ  return Math.max(maxLeft, Math.min(minRight, dt));
   133тЖТ   133тЖТ}

   135тЖТ   135тЖТ// тФАтФА Waveform cache: pre-computed PNG data URLs тФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФА
   136тЖТ   136тЖТ// Key: `${clipId}:${sampleCount}:${width}` тЖТ data URL
   137тЖТ   137тЖТconst waveformPNGCache = new Map<string, string>();
   138тЖТ   138тЖТ
   139тЖТ   139тЖТ// Draw only the window [startFrame, endFrame) of the sample across `width`
   140тЖТ   140тЖТ// pixels. Rendering the *window* (not the whole sample stretched) is what makes
   141тЖТ   141тЖТ// trimming look like cutting the audio away rather than squashing it.
   142тЖТ   142тЖТfunction computeWaveformPNG(samples: Float32Array, channels: number, width: number, color: string, height: number, startFrame: number, endFrame: number): string {
   143тЖТ   143тЖТ  const canvas = document.createElement('canvas');
   144тЖТ   144тЖТ  const dpr = window.devicePixelRatio || 1;
   145тЖТ   145тЖТ  const w = Math.max(2, Math.floor(width));
   146тЖТ   146тЖТ  canvas.width = w * dpr;
   147тЖТ   147тЖТ  canvas.height = height * dpr;
   148тЖТ   148тЖТ  const ctx = canvas.getContext('2d')!;
   149тЖТ   149тЖТ  ctx.scale(dpr, dpr);
   150тЖТ   150тЖТ  ctx.clearRect(0, 0, w, height);
   151тЖТ   151тЖТ  const totalFrames = samples.length / channels;
   152тЖТ   152тЖТ  const s = Math.max(0, Math.min(totalFrames, Math.floor(startFrame)));
   153тЖТ   153тЖТ  const e = Math.max(s, Math.min(totalFrames, Math.floor(endFrame)));
   154тЖТ   154тЖТ  const span = e - s;
   155тЖТ   155тЖТ  if (span <= 0) return canvas.toDataURL();
   156тЖТ   156тЖТ  const mid = height / 2;
   157тЖТ   157тЖТ  ctx.fillStyle = color;
   158тЖТ   158тЖТ  for (let x = 0; x < w; x++) {
   159тЖТ   159тЖТ    const f0 = s + Math.floor((x / w) * span);
   160тЖТ   160тЖТ    const f1 = Math.min(e, s + Math.floor(((x + 1) / w) * span) + 1);
   161тЖТ   161тЖТ    let peak = 0;
   162тЖТ   162тЖТ    for (let i = f0; i < f1; i++) { const v = Math.abs(samples[i * channels] ?? 0); if (v > peak) peak = v; }
   163тЖТ   163тЖТ    const bh = Math.max(1, peak * mid * 0.85);
   164тЖТ   164тЖТ    ctx.fillRect(x, mid - bh, 1, bh * 2);
   165тЖТ   165тЖТ  }
   166тЖТ   166тЖТ  return canvas.toDataURL();
   167тЖТ   167тЖТ}
   168тЖТ   168тЖТ
   169тЖТ   169тЖТ/** Waveform image for the buffer window the clip actually plays. With speed s,
   170тЖТ   170тЖТ *  a clip of `duration` timeline seconds consumes `duration*s` buffer seconds.
   171тЖТ   171тЖТ *
   172тЖТ   172тЖТ *  Two robustness fixes:
   173тЖТ   173тЖТ *  1. If `loadSamples` returns an EMPTY Float32Array (length 0) тАФ which happens
   174тЖТ   174тЖТ *     when a freshly-split/created clip's IndexedDB write hasn't landed yet тАФ
   175тЖТ   175тЖТ *     we DON'T cache the resulting blank PNG. Instead we show the `┬╖┬╖┬╖`
   176тЖТ   176тЖТ *     placeholder and retry a few times (500ms apart) so the real waveform
   177тЖТ   177тЖТ *     appears once the samples are available.
   178тЖТ   178тЖТ *  2. We listen for `slate:audio-clip-changed` (fired by Normalize/Reverse/split)
   179тЖТ   179тЖТ *     and invalidate our cache entry for the current clipId, then force a
   180тЖТ   180тЖТ *     recompute via a `bust` counter тАФ otherwise the memoised component would
   181тЖТ   181тЖТ *     keep showing the stale PNG even after the cache was cleared. */
   182тЖТ   182тЖТconst WaveformImg = memo(function WaveformImg({ clipId, sampleKey, channels, sampleRate, offset, duration, speed, width, color }: {
   183тЖТ   183тЖТ  clipId: string; sampleKey: string; channels: number; sampleRate: number;
   184тЖТ   184тЖТ  offset: number; duration: number; speed: number; width: number; color: string;
   185тЖТ   185тЖТ}) {
   186тЖТ   186тЖТ  const height = TRACK_H - 6;
   187тЖТ   187тЖТ  const [imgUrl, setImgUrl] = useState<string>('');
   188тЖТ   188тЖТ  const [bust, setBust] = useState(0);
   189тЖТ   189тЖТ  const retryRef = useRef(0);
   190тЖТ   190тЖТ
   191тЖТ   191тЖТ  // Invalidate cached PNGs for this clip when its samples change (normalize /
   192тЖТ   192тЖТ  // reverse / split). The parent AudioEditor also invalidates + bumps version,
   193тЖТ   193тЖТ  // but the memoised WaveformImg wouldn't otherwise recompute (its primitive
   194тЖТ   194тЖТ  // props are unchanged) тАФ the `bust` counter forces the load effect to re-run.
   195тЖТ   195тЖТ  useEffect(() => {
   196тЖТ   196тЖТ    const onChanged = (e: Event) => {
   197тЖТ   197тЖТ      const detail = (e as CustomEvent<string>).detail;
   198тЖТ   198тЖТ      // Match by clipId (normalize/reverse/split events) OR by sampleKey
   199тЖТ   199тЖТ      // (multiplayer sample-arrival events from registerSampleSyncMap).
   200тЖТ   200тЖТ      if (detail !== clipId && detail !== sampleKey) return;
   201тЖТ   201тЖТ      for (const key of waveformPNGCache.keys()) {
   202тЖТ   202тЖТ        if (key.startsWith(`${clipId}:`)) waveformPNGCache.delete(key);
   203тЖТ   203тЖТ      }
   204тЖТ   204тЖТ      retryRef.current = 0;
   205тЖТ   205тЖТ      setBust((n) => n + 1);
   206тЖТ   206тЖТ    };
   207тЖТ   207тЖТ    window.addEventListener('slate:audio-clip-changed', onChanged as EventListener);
   208тЖТ   208тЖТ    return () => window.removeEventListener('slate:audio-clip-changed', onChanged as EventListener);
   209тЖТ   209тЖТ  }, [clipId, sampleKey]);
   210тЖТ   210тЖТ
   211тЖТ   211тЖТ  useEffect(() => {
   212тЖТ   212тЖТ    const startFrame = Math.round(offset * sampleRate);
   213тЖТ   213тЖТ    const endFrame = Math.round((offset + duration * speed) * sampleRate);
   214тЖТ   214тЖТ    const cacheKey = `${clipId}:${sampleKey}:${startFrame}:${endFrame}:${Math.floor(width)}`;
   215тЖТ   215тЖТ    const cached = waveformPNGCache.get(cacheKey);
   216тЖТ   216тЖТ    if (cached) { setImgUrl(cached); return; }
   217тЖТ   217тЖТ    let cancelled = false;
   218тЖТ   218тЖТ    let retryTimer: ReturnType<typeof setTimeout> | null = null;
   219тЖТ   219тЖТ
   220тЖТ   220тЖТ    const attempt = () => {
   221тЖТ   221тЖТ      if (cancelled) return;
   222тЖТ   222тЖТ      void loadSamples(sampleKey).then((samples) => {
   223тЖТ   223тЖТ        if (cancelled) return;
   224тЖТ   224тЖТ        // Empty samples = the IndexedDB write for this sampleKey hasn't landed
   225тЖТ   225тЖТ        // yet (e.g. brand-new clip from a split, OR a remote peer's clip whose
   226тЖТ   226тЖТ        // sample blob is still in flight via the Yjs sync map). DON'T cache a
   227тЖТ   227тЖТ        // blank PNG тАФ retry for up to 10s (20 ├Ч 500ms) so the real waveform
   228тЖТ   228тЖТ        // appears once the samples arrive. The previous 5-retry / 2.5s budget
   229тЖТ   229тЖТ        // was too short for cross-peer sample sync, which can take several
   230тЖТ   230тЖТ        // seconds for a 5MB blob over a slow link. The `slate:audio-clip-changed`
   231тЖТ   231тЖТ        // event (dispatched by onClipsAdded when a remote clip is added, and by
   232тЖТ   232тЖТ        // registerSampleSyncMap when samples arrive) resets `retryRef` to 0 so
   233тЖТ   233тЖТ        // the WaveformImg re-tries immediately on the signal тАФ this retry
   234тЖТ   234тЖТ        // budget is the safety net for the case where the event fires before
   235тЖТ   235тЖТ        // the WaveformImg mounts, or doesn't fire at all.
   236тЖТ   236тЖТ        if (samples.length === 0) {
   237тЖТ   237тЖТ          if (retryRef.current < 20) {
   238тЖТ   238тЖТ            retryRef.current += 1;
   239тЖТ   239тЖТ            retryTimer = setTimeout(attempt, 500);
   240тЖТ   240тЖТ          }
   241тЖТ   241тЖТ          return;
   242тЖТ   242тЖТ        }
   243тЖТ   243тЖТ        retryRef.current = 0;
   244тЖТ   244тЖТ        const url = computeWaveformPNG(samples, channels, width, color, height, startFrame, endFrame);
   245тЖТ   245тЖТ        waveformPNGCache.set(cacheKey, url);
   246тЖТ   246тЖТ        setImgUrl(url);
   247тЖТ   247тЖТ      });
   248тЖТ   248тЖТ    };
   249тЖТ   249тЖТ    attempt();
   250тЖТ   250тЖТ
   251тЖТ   251тЖТ    return () => {
   252тЖТ   252тЖТ      cancelled = true;
   253тЖТ   253тЖТ      if (retryTimer) clearTimeout(retryTimer);
   254тЖТ   254тЖТ    };
   255тЖТ   255тЖТ  }, [clipId, sampleKey, channels, sampleRate, offset, duration, speed, width, color, height, bust]);
   256тЖТ   256тЖТ
   257тЖТ   257тЖТ  if (!imgUrl) return <div className="flex h-full items-center justify-center text-[7px] text-text-dim">┬╖┬╖┬╖</div>;
   258тЖТ   258тЖТ  return <img src={imgUrl} alt="" className="pointer-events-none h-full w-full" style={{ objectFit: 'fill' }} />;
   259тЖТ   259тЖТ});

   261тЖТ   261тЖТ/** Drop cached waveform PNGs for a clip (call when its samples change). */

   263тЖТ   263тЖТ  for (const key of waveformPNGCache.keys()) if (key.startsWith(`${clipId}:`)) waveformPNGCache.delete(key);
   264тЖТ   264тЖТ}

   567тЖТ   567тЖТ    return () => cancelAnimationFrame(rafRef.current);

   569тЖТ   569тЖТ  }, [playing]);
   570тЖТ   570тЖТ
   571тЖТ   571тЖТ  // Ctrl+scroll zoom тАФ centred on the playhead so the screen position under
   572тЖТ   572тЖТ  // the cursor/playhead stays put instead of zooming toward the left edge.
   573тЖТ   573тЖТ  /** Zoom while keeping one timeline point at the same screen offset.
   574тЖТ   574тЖТ   *  The anchor is the time under `anchorClientX` (mouse-wheel zoom: the spot
   575тЖТ   575тЖТ   *  under the cursor stays put тАФ how every DAW/map zoom feels), or the
   576тЖТ   576тЖТ   *  playhead when no anchor is given (toolbar buttons, fit-to-window).
   577тЖТ   577тЖТ   *  Keeping time t fixed on screen means newScroll = oldScroll + t┬╖╬Фpps,
   578тЖТ   578тЖТ   *  since screenX(t) = timelineLeft тИТ scrollLeft + t┬╖pps. The `scrollLeft`
   579тЖТ   579тЖТ   *  write happens in the layout effect below (after React commits the new
   580тЖТ   580тЖТ   *  `minWidth` on the timeline div тАФ setting it earlier would be clamped by
   581тЖТ   581тЖТ   *  the stale `scrollWidth`). */
   582тЖТ   582тЖТ  const zoomAnchored = useCallback((newPxPerSec: number, anchorClientX?: number) => {
   583тЖТ   583тЖТ    const el = scrollRef.current;
   584тЖТ   584тЖТ    const oldPxPerSec = pxRef.current;
   585тЖТ   585тЖТ    if (!el || oldPxPerSec === newPxPerSec) {
   586тЖТ   586тЖТ      setPxPerSec(newPxPerSec);
   587тЖТ   587тЖТ      return;
   588тЖТ   588тЖТ    }
   589тЖТ   589тЖТ    let t = positionRef.current; // default anchor: the playhead
   590тЖТ   590тЖТ    if (anchorClientX !== undefined) {
   591тЖТ   591тЖТ      const tl = el.querySelector('[data-timeline]') as HTMLElement | null;
   592тЖТ   592тЖТ      if (tl) t = Math.max(0, (anchorClientX - tl.getBoundingClientRect().left) / oldPxPerSec);
   593тЖТ   593тЖТ    }
   594тЖТ   594тЖТ    pendingScrollRef.current = Math.max(0, el.scrollLeft + t * (newPxPerSec - oldPxPerSec));
   595тЖТ   595тЖТ    setPxPerSec(newPxPerSec);
   596тЖТ   596тЖТ  }, []);
   597тЖТ   597тЖТ
   598тЖТ   598тЖТ  // Apply any pending scroll correction after pxPerSec commits to the DOM.
   599тЖТ   599тЖТ  useLayoutEffect(() => {
   600тЖТ   600тЖТ    if (pendingScrollRef.current !== null && scrollRef.current) {
   601тЖТ   601тЖТ      scrollRef.current.scrollLeft = pendingScrollRef.current;
   602тЖТ   602тЖТ      pendingScrollRef.current = null;
   603тЖТ   603тЖТ    }
   604тЖТ   604тЖТ  }, [pxPerSec]);
   605тЖТ   605тЖТ
   606тЖТ   606тЖТ  useEffect(() => {
   607тЖТ   607тЖТ    const el = scrollRef.current; if (!el) return;
   608тЖТ   608тЖТ    const onWheel = (e: WheelEvent) => {
   609тЖТ   609тЖТ      if (!e.ctrlKey && !e.metaKey) return;
   610тЖТ   610тЖТ      e.preventDefault();

   612тЖТ   612тЖТ      zoomAnchored(next, e.clientX); // zoom toward the cursor
   613тЖТ   613тЖТ    };

   615тЖТ   615тЖТ    return () => el.removeEventListener('wheel', onWheel);
   616тЖТ   616тЖТ  }, [zoomAnchored]);
   617тЖТ   617тЖТ
   618тЖТ   618тЖТ  // Global pointermove/up тАФ pure DOM, zero React state.
   619тЖТ   619тЖТ  // rAF-throttled: pointermove fires at the hardware's report rate (often
   620тЖТ   620тЖТ  // 60-120Hz), and the move handler does a neighbours-overlap scan on every
   621тЖТ   621тЖТ  // event. Coalescing multiple moves into a single rAF callback caps the work
   622тЖТ   622тЖТ  // at the display refresh rate (~60fps) so a fast pointer drag never stalls
   623тЖТ   623тЖТ  // the main thread. A no-op skip (cursor hasn't moved since the last
   624тЖТ   624тЖТ  // processed frame) avoids redundant DOM writes for stationary pointers.
   625тЖТ   625тЖТ  useEffect(() => {
   626тЖТ   626тЖТ    /** Last pointer position we actually applied to the DOM. Lets us skip
   627тЖТ   627тЖТ     *  work when pointermove fires without real movement (e.g. a touchpad
   628тЖТ   628тЖТ     *  reporting pressure changes, or sub-pixel jitter). */
   629тЖТ   629тЖТ    let lastProcessedX: number | null = null;
   630тЖТ   630тЖТ    let lastProcessedY: number | null = null;
   631тЖТ   631тЖТ
   632тЖТ   632тЖТ    /** Magnet: pull `t` (a clip START time) to the nearest beat line, other
   633тЖТ   633тЖТ     *  clip edge, or the playhead тАФ considering BOTH the clip's start and end
   634тЖТ   634тЖТ     *  edges тАФ if one is within SNAP_PX at the current zoom. `dur` null =
   635тЖТ   635тЖТ     *  only the start edge matters (trims). Returns the adjusted start. */
   636тЖТ   636тЖТ    const snapStart = (t: number, dur: number | null, snapTimes: number[]): number => {
   637тЖТ   637тЖТ      const pps = pxRef.current;
   638тЖТ   638тЖТ      const thr = SNAP_PX / pps;
   639тЖТ   639тЖТ      const beat = 60 / bpmRef.current;
   640тЖТ   640тЖТ      let best = t;
   641тЖТ   641тЖТ      let bestDist = thr;
   642тЖТ   642тЖТ      const consider = (target: number, edgeOffset: number) => {
   643тЖТ   643тЖТ        const dist = Math.abs(t + edgeOffset - target);
   644тЖТ   644тЖТ        if (dist < bestDist) { bestDist = dist; best = target - edgeOffset; }
   645тЖТ   645тЖТ      };
   646тЖТ   646тЖТ      consider(Math.round(t / beat) * beat, 0);
   647тЖТ   647тЖТ      if (dur !== null) consider(Math.round((t + dur) / beat) * beat, dur);
   648тЖТ   648тЖТ      for (const st of snapTimes) {
   649тЖТ   649тЖТ        consider(st, 0);
   650тЖТ   650тЖТ        if (dur !== null) consider(st, dur);
   651тЖТ   651тЖТ      }
   652тЖТ   652тЖТ      return best;
   653тЖТ   653тЖТ    };
   654тЖТ   654тЖТ
   655тЖТ   655тЖТ    const applyMove = () => {
   656тЖТ   656тЖТ      moveRafRef.current = 0;
   657тЖТ   657тЖТ      const d = dragRef.current;
   658тЖТ   658тЖТ      if (!d) return;
   659тЖТ   659тЖТ      const clientX = moveXRef.current;
   660тЖТ   660тЖТ      if (clientX === null) return;
   661тЖТ   661тЖТ      const clientY = moveYRef.current;
   662тЖТ   662тЖТ      // Skip no-op moves тАФ cursor hasn't moved to a new pixel since the last
   663тЖТ   663тЖТ      // frame we processed. Saves an overlap scan + 1-2 style writes.
   664тЖТ   664тЖТ      if (lastProcessedX === clientX && lastProcessedY === clientY) return;
   665тЖТ   665тЖТ      lastProcessedX = clientX;
   666тЖТ   666тЖТ      lastProcessedY = clientY;
   667тЖТ   667тЖТ
   668тЖТ   668тЖТ      // Dead zone: a plain click (pointer never leaves a 3px box) must not
   669тЖТ   669тЖТ      // nudge the clip. Once exceeded, the drag is live for good.
   670тЖТ   670тЖТ      if (!d.moved) {
   671тЖТ   671тЖТ        if (Math.abs(clientX - d.sx) < DRAG_DEADZONE_PX && Math.abs(clientY - d.sy) < DRAG_DEADZONE_PX) return;
   672тЖТ   672тЖТ        d.moved = true;
   673тЖТ   673тЖТ        if (d.mode === 'drag') {
   674тЖТ   674тЖТ          d.el.style.zIndex = '40'; // ride above other rows while crossing tracks
   675тЖТ   675тЖТ          // Elevate the whole group too so the user sees the multi-clip
   676тЖТ   676тЖТ          // selection ride above non-selected neighbours while in transit.
   677тЖТ   677тЖТ          if (d.origins) for (const o of d.origins.values()) if (o.el !== d.el) o.el.style.zIndex = '40';
   678тЖТ   678тЖТ        }
   679тЖТ   679тЖТ      }
   680тЖТ   680тЖТ
   681тЖТ   681тЖТ      const pps = pxRef.current;
   682тЖТ   682тЖТ      const dt = (clientX - d.sx) / pps;
   683тЖТ   683тЖТ      const oldEnd = d.os + d.od;
   684тЖТ   684тЖТ      const wantSnap = snapRef.current && !moveAltRef.current;
   685тЖТ   685тЖТ      if (d.mode === 'drag') {
   686тЖТ   686тЖТ        // MULTI-SELECT DRAG: when `origins` is populated, every selected clip
   687тЖТ   687тЖТ        // moves by the same horizontal delta. Vertical track-switching is
   688тЖТ   688тЖТ        // disabled for the group (each clip stays on its origin track) since
   689тЖТ   689тЖТ        // reshuffling every selected clip onto one target track would be
   690тЖТ   690тЖТ        // ambiguous and likely overlap-prone. The dragged clip's snap result
   691тЖТ   691тЖТ        // is applied to the whole group so the magnet still feels right.
   692тЖТ   692тЖТ        if (d.origins && d.origins.size > 0) {
   693тЖТ   693тЖТ          const dragged = d.origins.get(d.clipId);
   694тЖТ   694тЖТ          let snappedDt = dt;
   695тЖТ   695тЖТ          if (wantSnap && dragged) {
   696тЖТ   696тЖТ            const snappedStart = snapStart(dragged.os + dt, dragged.od, d.snapTimes);
   697тЖТ   697тЖТ            snappedDt = snappedStart - dragged.os;
   698тЖТ   698тЖТ          }
   699тЖТ   699тЖТ          for (const o of d.origins.values()) {
   700тЖТ   700тЖТ            const newStart = Math.max(0, o.os + snappedDt);
   701тЖТ   701тЖТ            o.el.style.left = `${newStart * pps}px`;
   702тЖТ   702тЖТ          }
   703тЖТ   703тЖТ          return;
   704тЖТ   704тЖТ        }
   705тЖТ   705тЖТ        // The clip follows the mouse 1:1 (snapped to the magnet grid). It's
   706тЖТ   706тЖТ        // allowed to ride OVER other clips while the drag is live тАФ it's
   707тЖТ   707тЖТ        // elevated to z-40 so the overlap reads as "in transit", and the
   708тЖТ   708тЖТ        // no-overlap invariant is enforced once, on drop (see onUp). The
   709тЖТ   709тЖТ        // previous mid-drag resolution parked the clip flush against a
   710тЖТ   710тЖТ        // blocker and then teleported it across when the cursor passed the
   711тЖТ   711тЖТ        // blocker's midpoint, which made dragging feel sticky and jumpy.
   712тЖТ   712тЖТ        const duration = d.od;
   713тЖТ   713тЖТ        // Vertical: whole-row steps move the clip across tracks.
   714тЖТ   714тЖТ        const rowDelta = Math.max(
   715тЖТ   715тЖТ          -d.trackIndex,
   716тЖТ   716тЖТ          Math.min(d.trackIds.length - 1 - d.trackIndex, Math.round((clientY - d.sy) / TRACK_H)),
   717тЖТ   717тЖТ        );
   718тЖТ   718тЖТ        d.rowDelta = rowDelta;
   719тЖТ   719тЖТ        d.el.style.transform = rowDelta === 0 ? '' : `translateY(${rowDelta * TRACK_H}px)`;
   720тЖТ   720тЖТ
   721тЖТ   721тЖТ        const desired = wantSnap ? snapStart(d.os + dt, duration, d.snapTimes) : d.os + dt;
   722тЖТ   722тЖТ        d.el.style.left = `${Math.max(0, desired) * pps}px`;
   723тЖТ   723тЖТ      } else if (d.mode === 'trimL') {
   724тЖТ   724тЖТ        // Cut from the left: never past the left neighbour, the source start
   725тЖТ   725тЖТ        // (offset тЙе 0 тЖТ limited by the trimmed head in timeline seconds), or a
   726тЖТ   726тЖТ        // minimum width.
   727тЖТ   727тЖТ        const minStart = Math.max(d.leftLimit, d.os - d.oo / d.speed);
   728тЖТ   728тЖТ        const desired = wantSnap ? snapStart(d.os + dt, null, d.snapTimes) : d.os + dt;
   729тЖТ   729тЖТ        const start = Math.min(oldEnd - 0.1, Math.max(minStart, desired));
   730тЖТ   730тЖТ        d.el.style.left = `${start * pps}px`;
   731тЖТ   731тЖТ        d.el.style.width = `${(oldEnd - start) * pps}px`;
   732тЖТ   732тЖТ        // Shift the (fixed-width) waveform the opposite way so the audio stays
   733тЖТ   733тЖТ        // anchored тАФ the newly-trimmed head is clipped, not squashed.
   734тЖТ   734тЖТ        if (d.waveEl) d.waveEl.style.left = `${-(start - d.os) * pps}px`;
   735тЖТ   735тЖТ      } else if (d.mode === 'trimR') {
   736тЖТ   736тЖТ        // Cut from the right: never past the right neighbour or a min width.
   737тЖТ   737тЖТ        const desired = wantSnap ? snapStart(oldEnd + dt, null, d.snapTimes) : oldEnd + dt;
   738тЖТ   738тЖТ        const end = Math.min(d.rightLimit, Math.max(d.os + 0.1, desired));
   739тЖТ   739тЖТ        d.el.style.width = `${(end - d.os) * pps}px`;
   740тЖТ   740тЖТ      }
   741тЖТ   741тЖТ    };
   742тЖТ   742тЖТ
   743тЖТ   743тЖТ    const onMove = (ev: PointerEvent) => {
   744тЖТ   744тЖТ      const d = dragRef.current; if (!d) return;
   745тЖТ   745тЖТ      // Stash the latest pointer state тАФ the rAF callback reads it. Cheap
   746тЖТ   746тЖТ      // (ref writes) so even 120Hz pointermove streams don't pile up work.
   747тЖТ   747тЖТ      moveXRef.current = ev.clientX;
   748тЖТ   748тЖТ      moveYRef.current = ev.clientY;
   749тЖТ   749тЖТ      moveAltRef.current = ev.altKey;
   750тЖТ   750тЖТ      // Schedule a frame if one isn't already pending. The guard is what
   751тЖТ   751тЖТ      // coalesces multiple moves within the same frame into one DOM write.
   752тЖТ   752тЖТ      if (!moveRafRef.current) {
   753тЖТ   753тЖТ        moveRafRef.current = requestAnimationFrame(applyMove);
   754тЖТ   754тЖТ      }
   755тЖТ   755тЖТ    };
   756тЖТ   756тЖТ    const onUp = () => {
   757тЖТ   757тЖТ      const d = dragRef.current; if (!d) return;
   758тЖТ   758тЖТ      // Cancel any pending rAF тАФ we'll process the final position
   759тЖТ   759тЖТ      // SYNCHRONOUSLY here so the committed Yjs value reflects the exact
   760тЖТ   760тЖТ      // pointer-up location (otherwise the last pointermove before pointerup
   761тЖТ   761тЖТ      // could be dropped, leaving the committed position ~1 frame behind the
   762тЖТ   762тЖТ      // cursor). lastProcessed is reset to bypass the no-op skip in case the
   763тЖТ   763тЖТ      // final pointermove matched the prior processed frame but we still need
   764тЖТ   764тЖТ      // to read the just-written style to commit to Yjs.
   765тЖТ   765тЖТ      if (moveRafRef.current) {
   766тЖТ   766тЖТ        cancelAnimationFrame(moveRafRef.current);
   767тЖТ   767тЖТ        moveRafRef.current = 0;
   768тЖТ   768тЖТ      }
   769тЖТ   769тЖТ      if (moveXRef.current !== null) {
   770тЖТ   770тЖТ        lastProcessedX = null; // bypass the no-op skip on the final flush
   771тЖТ   771тЖТ        lastProcessedY = null;
   772тЖТ   772тЖТ        applyMove();
   773тЖТ   773тЖТ      }
   774тЖТ   774тЖТ      // Undo the drag-only visual state React doesn't manage. For a multi-clip
   775тЖТ   775тЖТ      // drag, also clear the elevated z-index on every selected clip's element.
   776тЖТ   776тЖТ      if (d.origins) for (const o of d.origins.values()) { o.el.style.zIndex = ''; o.el.style.transform = ''; }
   777тЖТ   777тЖТ      d.el.style.zIndex = '';
   778тЖТ   778тЖТ      d.el.style.transform = '';
   779тЖТ   779тЖТ      // A click that never left the dead zone selects but must not write тАФ
   780тЖТ   780тЖТ      // committing a parseFloat round-trip of the style would drift the clip.
   781тЖТ   781тЖТ      if (d.moved) {
   782тЖТ   782тЖТ        const pps = pxRef.current;
   783тЖТ   783тЖТ        const left = parseFloat(d.el.style.left) / pps;
   784тЖТ   784тЖТ        const width = parseFloat(d.el.style.width) / pps;
   785тЖТ   785тЖТ        if (d.mode === 'drag') {
   786тЖТ   786тЖТ          // MULTI-SELECT COMMIT: every selected clip moves by the same delta,
   787тЖТ   787тЖТ          // clamped so the group stays overlap-free against non-selected
   788тЖТ   788тЖТ          // neighbours on each clip's origin track. The delta is recomputed
   789тЖТ   789тЖТ          // from the dragged clip's pointer-up position (so the snapped /
   790тЖТ   790тЖТ          // clamped value the user sees is what gets committed), then applied
   791тЖТ   791тЖТ          // to every clip in the group via updateAudioClip inside a single
   792тЖТ   792тЖТ          // Yjs transaction.
   793тЖТ   793тЖТ          if (d.origins && d.origins.size > 0) {
   794тЖТ   794тЖТ            const dragged = d.origins.get(d.clipId);
   795тЖТ   795тЖТ            const dt = dragged ? left - dragged.os : 0;
   796тЖТ   796тЖТ            const clampedDt = clampGroupDt(dt, d.origins, d.byTrack);
   797тЖТ   797тЖТ            // Write the resolved positions to the DOM too so there's no
   798тЖТ   798тЖТ            // flicker between pointer-up and the Yjs-driven re-render.
   799тЖТ   799тЖТ            for (const o of d.origins.values()) {
   800тЖТ   800тЖТ              const newStart = Math.max(0, o.os + clampedDt);
   801тЖТ   801тЖТ              o.el.style.left = `${newStart * pps}px`;
   802тЖТ   802тЖТ            }
   803тЖТ   803тЖТ            // Single transaction so the whole group commits atomically to the
   804тЖТ   804тЖТ            // Yjs doc тАФ peers see one update, not N interleaved with their
   805тЖТ   805тЖТ            // own edits.
   806тЖТ   806тЖТ            slate.doc.transact(() => {
   807тЖТ   807тЖТ              for (const [cid, o] of d.origins!) {
   808тЖТ   808тЖТ                updateAudioClip(slate, cid, { start: Math.max(0, o.os + clampedDt) });
   809тЖТ   809тЖТ              }
   810тЖТ   810тЖТ            });
   811тЖТ   811тЖТ          } else {
   812тЖТ   812тЖТ            const targetTrackId = d.trackIds[d.trackIndex + d.rowDelta] ?? d.trackIds[d.trackIndex]!;
   813тЖТ   813тЖТ            const originTrackId = d.trackIds[d.trackIndex]!;
   814тЖТ   814тЖТ            // Enforce the no-overlap invariant HERE (the drag itself follows
   815тЖТ   815тЖТ            // the mouse freely): place the clip in the free gap nearest to the
   816тЖТ   816тЖТ            // dropped position.
   817тЖТ   817тЖТ            const start = nearestFreeStart(left, d.od, d.byTrack.get(targetTrackId) ?? []);
   818тЖТ   818тЖТ            // Write the resolved position to the DOM too so there's no flicker
   819тЖТ   819тЖТ            // between pointer-up and the Yjs-driven re-render.
   820тЖТ   820тЖТ            d.el.style.left = `${start * pps}px`;
   821тЖТ   821тЖТ            updateAudioClip(slate, d.clipId, {
   822тЖТ   822тЖТ              start,
   823тЖТ   823тЖТ              ...(targetTrackId !== originTrackId ? { trackId: targetTrackId } : {}),
   824тЖТ   824тЖТ            });
   825тЖТ   825тЖТ          }
   826тЖТ   826тЖТ        }
   827тЖТ   827тЖТ        else if (d.mode === 'trimL') updateAudioClip(slate, d.clipId, { start: left, duration: width, offset: Math.max(0, d.oo + (left - d.os) * d.speed) });
   828тЖТ   828тЖТ        else if (d.mode === 'trimR') updateAudioClip(slate, d.clipId, { duration: width });
   829тЖТ   829тЖТ      }
   830тЖТ   830тЖТ      moveXRef.current = null;
   831тЖТ   831тЖТ      lastProcessedX = null;
   832тЖТ   832тЖТ      lastProcessedY = null;
   833тЖТ   833тЖТ      dragRef.current = null;
   834тЖТ   834тЖТ    };
