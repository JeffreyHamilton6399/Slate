    25→const TRACK_H = 60;

    27→// ── Waveform cache: pre-computed PNG data URLs ───────────────────────────────
    28→// Key: `${clipId}:${sampleCount}:${width}` → data URL
    29→const waveformPNGCache = new Map<string, string>();
    30→
    31→// Draw only the window [startFrame, endFrame) of the sample across `width`
    32→// pixels. Rendering the *window* (not the whole sample stretched) is what makes
    33→// trimming look like cutting the audio away rather than squashing it.
    34→function computeWaveformPNG(samples: Float32Array, channels: number, width: number, color: string, height: number, startFrame: number, endFrame: number): string {
    35→  const canvas = document.createElement('canvas');
    36→  const dpr = window.devicePixelRatio || 1;
    37→  const w = Math.max(2, Math.floor(width));
    38→  canvas.width = w * dpr;
    39→  canvas.height = height * dpr;
    40→  const ctx = canvas.getContext('2d')!;
    41→  ctx.scale(dpr, dpr);
    42→  ctx.clearRect(0, 0, w, height);
    43→  const totalFrames = samples.length / channels;
    44→  const s = Math.max(0, Math.min(totalFrames, Math.floor(startFrame)));
    45→  const e = Math.max(s, Math.min(totalFrames, Math.floor(endFrame)));
    46→  const span = e - s;
    47→  if (span <= 0) return canvas.toDataURL();
    48→  const mid = height / 2;
    49→  ctx.fillStyle = color;
    50→  for (let x = 0; x < w; x++) {
    51→    const f0 = s + Math.floor((x / w) * span);
    52→    const f1 = Math.min(e, s + Math.floor(((x + 1) / w) * span) + 1);
    53→    let peak = 0;
    54→    for (let i = f0; i < f1; i++) { const v = Math.abs(samples[i * channels] ?? 0); if (v > peak) peak = v; }
    55→    const bh = Math.max(1, peak * mid * 0.85);
    56→    ctx.fillRect(x, mid - bh, 1, bh * 2);
    57→  }
    58→  return canvas.toDataURL();
    59→}

    61→/** Waveform image for the buffer window the clip actually plays. With speed s,
    62→ *  a clip of `duration` timeline seconds consumes `duration*s` buffer seconds.
    63→ *
    64→ *  Two robustness fixes:
    65→ *  1. If `loadSamples` returns an EMPTY Float32Array (length 0) — which happens
    66→ *     when a freshly-split/created clip's IndexedDB write hasn't landed yet —
    67→ *     we DON'T cache the resulting blank PNG. Instead we show the `···`
    68→ *     placeholder and retry a few times (500ms apart) so the real waveform
    69→ *     appears once the samples are available.
    70→ *  2. We listen for `slate:audio-clip-changed` (fired by Normalize/Reverse/split)
    71→ *     and invalidate our cache entry for the current clipId, then force a
    72→ *     recompute via a `bust` counter — otherwise the memoised component would
    73→ *     keep showing the stale PNG even after the cache was cleared. */
    74→const WaveformImg = memo(function WaveformImg({ clipId, sampleKey, channels, sampleRate, offset, duration, speed, width, color }: {
    75→  clipId: string; sampleKey: string; channels: number; sampleRate: number;
    76→  offset: number; duration: number; speed: number; width: number; color: string;
    77→}) {
    78→  const height = TRACK_H - 6;
    79→  const [imgUrl, setImgUrl] = useState<string>('');
    80→  const [bust, setBust] = useState(0);
    81→  const retryRef = useRef(0);
    82→
    83→  // Invalidate cached PNGs for this clip when its samples change (normalize /
    84→  // reverse / split). The parent AudioEditor also invalidates + bumps version,
    85→  // but the memoised WaveformImg wouldn't otherwise recompute (its primitive
    86→  // props are unchanged) — the `bust` counter forces the load effect to re-run.
    87→  useEffect(() => {
    88→    const onChanged = (e: Event) => {
    89→      const id = (e as CustomEvent<string>).detail;
    90→      if (id !== clipId) return;
    91→      for (const key of waveformPNGCache.keys()) {
    92→        if (key.startsWith(`${clipId}:`)) waveformPNGCache.delete(key);
    93→      }
    94→      retryRef.current = 0;
    95→      setBust((n) => n + 1);
    96→    };
    97→    window.addEventListener('slate:audio-clip-changed', onChanged as EventListener);
    98→    return () => window.removeEventListener('slate:audio-clip-changed', onChanged as EventListener);
    99→  }, [clipId]);
   100→
   101→  useEffect(() => {
   102→    const startFrame = Math.round(offset * sampleRate);
   103→    const endFrame = Math.round((offset + duration * speed) * sampleRate);
   104→    const cacheKey = `${clipId}:${sampleKey}:${startFrame}:${endFrame}:${Math.floor(width)}`;
   105→    const cached = waveformPNGCache.get(cacheKey);
   106→    if (cached) { setImgUrl(cached); return; }
   107→    let cancelled = false;
   108→    let retryTimer: ReturnType<typeof setTimeout> | null = null;
   109→
   110→    const attempt = () => {
   111→      if (cancelled) return;
   112→      void loadSamples(sampleKey).then((samples) => {
   113→        if (cancelled) return;
   114→        // Empty samples = the IndexedDB write for this sampleKey hasn't landed
   115→        // yet (e.g. brand-new clip from a split). DON'T cache a blank PNG —
   116→        // retry a few times so the real waveform appears.
   117→        if (samples.length === 0) {
   118→          if (retryRef.current < 5) {
   119→            retryRef.current += 1;
   120→            retryTimer = setTimeout(attempt, 500);
   121→          }
   122→          return;
   123→        }
   124→        retryRef.current = 0;
   125→        const url = computeWaveformPNG(samples, channels, width, color, height, startFrame, endFrame);
   126→        waveformPNGCache.set(cacheKey, url);
   127→        setImgUrl(url);
   128→      });
   129→    };
   130→    attempt();
   131→
   132→    return () => {
   133→      cancelled = true;
   134→      if (retryTimer) clearTimeout(retryTimer);
   135→    };
   136→  }, [clipId, sampleKey, channels, sampleRate, offset, duration, speed, width, color, height, bust]);
   137→
   138→  if (!imgUrl) return <div className="flex h-full items-center justify-center text-[7px] text-text-dim">···</div>;
   139→  return <img src={imgUrl} alt="" className="pointer-events-none h-full w-full" style={{ objectFit: 'fill' }} />;
   140→});
   141→
   142→/** Drop cached waveform PNGs for a clip (call when its samples change). */
   143→function invalidateWaveform(clipId: string): void {
   144→  for (const key of waveformPNGCache.keys()) if (key.startsWith(`${clipId}:`)) waveformPNGCache.delete(key);
   145→}

   148→
   149→export function AudioEditor() {
   150→  const room = useRoom();
   151→  const slate = room.slate;
   152→  const [version, setVersion] = useState(0);
   153→  const [playing, setPlaying] = useState(false);
   154→  const [bpm, setBpmState] = useState(slate.audioBpm());
   155→  const [metronome, setMetronome] = useState(false);
   156→  const [recording, setRecording] = useState(false);
   157→  const [selectedClipId, setSelectedClipId] = useState<string | null>(null);
   158→  const [masterVol, setMasterVol] = useState(0.85);
   159→  const [looping, setLooping] = useState(false);
   160→  const [loopStart, setLoopStart] = useState(0);
   161→  const [loopEnd, setLoopEnd] = useState(8);
   162→  const [pxPerSec, setPxPerSec] = useState(80);
   163→  const engineRef = useRef<AudioEngine | null>(null);
   164→  const stopRecRef = useRef<(() => Promise<{ samples: number[]; sampleRate: number; channels: number; duration: number }>) | null>(null);
   165→  const rafRef = useRef(0);
   166→  const positionRef = useRef(0);
   167→  const playheadRef = useRef<HTMLDivElement | null>(null);
   168→  const posDisplayRef = useRef<HTMLSpanElement | null>(null);
   169→  const scrollRef = useRef<HTMLDivElement | null>(null);
   170→  const dragRef = useRef<{
   171→    clipId: string; el: HTMLElement; waveEl: HTMLElement | null;
   172→    os: number; od: number; oo: number; speed: number; sx: number;
   173→    leftLimit: number; rightLimit: number; mode: 'drag' | 'trimL' | 'trimR';
   174→    neighbours: { start: number; end: number }[];
   175→  } | null>(null);
   176→  const selectedRef = useRef<string | null>(null);
   177→  selectedRef.current = selectedClipId;
   178→  const clipsRef = useRef<AudioClip[]>([]);
   179→  const pxRef = useRef(pxPerSec);
   180→  pxRef.current = pxPerSec;
   181→
   182→  // Yjs subscription.
   183→  useEffect(() => {
   184→    const tracks = slate.audioTracks();
   185→    const clips = slate.audioClips();
   186→    const audioMap = slate.doc.getMap('audio');
   187→    let pending = false;
   188→    const bump = () => { if (pending) return; pending = true; requestAnimationFrame(() => { pending = false; setVersion((v) => v + 1); }); };
   189→    tracks.observeDeep(bump); clips.observeDeep(bump); audioMap.observe(bump); bump();
   190→    const lateRead = setTimeout(bump, 200);
   191→    return () => { clearTimeout(lateRead); tracks.unobserveDeep(bump); clips.unobserveDeep(bump); audioMap.unobserve(bump); };
   192→  }, [slate]);
   193→
   194→  useEffect(() => {
   195→    engineRef.current = new AudioEngine();
   196→    engineRef.current.setMasterVolume(masterVol);
   197→    return () => { engineRef.current?.dispose(); engineRef.current = null; };
   198→    // eslint-disable-next-line react-hooks/exhaustive-deps
   199→  }, []);
   200→
   201→  // Playhead — direct DOM.
   202→  useEffect(() => {
   203→    if (!playing) return;
   204→    const tick = () => {
   205→      rafRef.current = requestAnimationFrame(tick);
   206→      const eng = engineRef.current; if (!eng) return;
   207→      const pos = eng.getPosition();
   208→      positionRef.current = pos;
   209→      if (playheadRef.current) playheadRef.current.style.transform = `translateX(${pos * pxRef.current}px)`;
   210→      if (posDisplayRef.current) posDisplayRef.current.textContent = `${pos.toFixed(1)}s`;
   211→      if (pos > timelineDuration + 2) { eng.stop(); setPlaying(false); }
   212→    };
   213→    rafRef.current = requestAnimationFrame(tick);
   214→    return () => cancelAnimationFrame(rafRef.current);
   215→    // eslint-disable-next-line react-hooks/exhaustive-deps
   216→  }, [playing]);
   217→
   218→  // Ctrl+scroll zoom.
   219→  useEffect(() => {
   220→    const el = scrollRef.current; if (!el) return;
   221→    const onWheel = (e: WheelEvent) => {
   222→      if (!e.ctrlKey && !e.metaKey) return;
   223→      e.preventDefault();
   224→      setPxPerSec((c) => Math.max(10, Math.min(800, c * (e.deltaY < 0 ? 1.2 : 1 / 1.2))));
   225→    };
   226→    el.addEventListener('wheel', onWheel, { passive: false });
   227→    return () => el.removeEventListener('wheel', onWheel);
   228→  }, []);
   229→
   230→  // Global pointermove/up — pure DOM, zero React state.
   231→  useEffect(() => {
   232→    const onMove = (ev: PointerEvent) => {
   233→      const d = dragRef.current; if (!d) return;
   234→      const pps = pxRef.current;
   235→      const dt = (ev.clientX - d.sx) / pps;
   236→      const oldEnd = d.os + d.od;
   237→      if (d.mode === 'drag') {
   238→        // Snap-to-neighbour drag: the clip follows the mouse, but if the new
   239→        // position would overlap another clip on the same track, instead of
   240→        // stopping dead at the neighbour's edge we JUMP to the free side of
   241→        // that neighbour (right of it when dragging right, left of it when
   242→        // dragging left). Iterates so a chain of back-to-back clips resolves
   243→        // to the next free slot. A small overlap threshold avoids accidental
   244→        // snaps when the cursor barely crosses the boundary.
   245→        const duration = d.od;
   246→        const rawStart = d.os + dt;
   247→        let start = rawStart;
   248→        const overlapThreshold = 0.05; // s — ignore near-boundary jitter
   249→        for (let iter = 0; iter < d.neighbours.length + 1; iter++) {
   250→          const blocker = d.neighbours.find((n) => {
   251→            const ovStart = Math.max(start, n.start);
   252→            const ovEnd = Math.min(start + duration, n.end);
   253→            return ovEnd - ovStart > overlapThreshold;
   254→          });
   255→          if (!blocker) break;
   256→          if (dt >= 0) start = blocker.end;          // snap to RIGHT of blocker
   257→          else start = blocker.start - duration;     // snap to LEFT of blocker
   258→        }
   259→        // Fallback clamp: never let the clip start before t=0.
   260→        start = Math.max(0, start);
   261→        d.el.style.left = `${start * pps}px`;
   262→      } else if (d.mode === 'trimL') {
   263→        // Cut from the left: never past the left neighbour, the source start
   264→        // (offset ≥ 0 → limited by the trimmed head in timeline seconds), or a
   265→        // minimum width.
   266→        const minStart = Math.max(d.leftLimit, d.os - d.oo / d.speed);
   267→        const start = Math.min(oldEnd - 0.1, Math.max(minStart, d.os + dt));
   268→        d.el.style.left = `${start * pps}px`;
   269→        d.el.style.width = `${(oldEnd - start) * pps}px`;
   270→        // Shift the (fixed-width) waveform the opposite way so the audio stays
   271→        // anchored — the newly-trimmed head is clipped, not squashed.
   272→        if (d.waveEl) d.waveEl.style.left = `${-(start - d.os) * pps}px`;
   273→      } else if (d.mode === 'trimR') {
   274→        // Cut from the right: never past the right neighbour or a min width.
   275→        const end = Math.min(d.rightLimit, Math.max(d.os + 0.1, oldEnd + dt));
   276→        d.el.style.width = `${(end - d.os) * pps}px`;
   277→      }
   278→    };
   279→    const onUp = () => {
   280→      const d = dragRef.current; if (!d) return;
   281→      const pps = pxRef.current;
   282→      const left = parseFloat(d.el.style.left) / pps;
   283→      const width = parseFloat(d.el.style.width) / pps;
   284→      if (d.mode === 'drag') updateAudioClip(slate, d.clipId, { start: Math.max(0, left) });
   285→      else if (d.mode === 'trimL') updateAudioClip(slate, d.clipId, { start: left, duration: width, offset: Math.max(0, d.oo + (left - d.os) * d.speed) });
   286→      else if (d.mode === 'trimR') updateAudioClip(slate, d.clipId, { duration: width });
   287→      dragRef.current = null;
   288→    };
   289→    window.addEventListener('pointermove', onMove);
   290→    window.addEventListener('pointerup', onUp);
   291→    return () => { window.removeEventListener('pointermove', onMove); window.removeEventListener('pointerup', onUp); };
   292→  }, [slate]);
   293→
   294→  // Hotkeys.
   295→  useEffect(() => {
   296→    const onKey = (e: KeyboardEvent) => {
   297→      const t = e.target as HTMLElement;
   298→      if (t && /^(INPUT|TEXTAREA|SELECT)$/.test(t.tagName)) return;
   299→      const k = e.key.toLowerCase();
   300→      if (k === ' ') { e.preventDefault(); togglePlay(); }
   301→      else if (k === 'c' && !e.ctrlKey && selectedRef.current) { e.preventDefault(); void splitAudioClip(slate, selectedRef.current, positionRef.current); }
   302→      else if ((k === 'delete' || k === 'backspace') && selectedRef.current) { e.preventDefault(); deleteAudioClip(slate, selectedRef.current); setSelectedClipId(null); }
   303→      else if (k === 'd' && !e.ctrlKey && selectedRef.current) { e.preventDefault(); dupClip(selectedRef.current); }
   304→      else if (k === 'l') { e.preventDefault(); setLooping((n) => !n); }
   305→      else if (k === 'm') { e.preventDefault(); setMetronome((n) => { engineRef.current?.setMetronome(!n); return !n; }); }
   306→      else if (k === 'r' && !e.ctrlKey) { e.preventDefault(); void toggleRecord(); }
   307→      else if (k === 'arrowleft') { e.preventDefault(); seek(positionRef.current - 2); }
   308→      else if (k === 'arrowright') { e.preventDefault(); seek(positionRef.current + 2); }
   309→      else if (k === 'home') { e.preventDefault(); seek(0); }
   310→    };
   311→    window.addEventListener('keydown', onKey);
   312→    return () => window.removeEventListener('keydown', onKey);
   313→    // eslint-disable-next-line react-hooks/exhaustive-deps
   314→  }, [playing, recording]);
   315→
   316→  // Read from Yjs — lightweight (no samples copied).
   317→  const tracks: AudioTrack[] = useMemo(() => {
   318→    const list: AudioTrack[] = [];
   319→    slate.audioTracks().forEach((m, id) => { const t = readAudioTrack(m, id); if (t) list.push(t); });
   320→    list.sort((a, b) => a.order - b.order);
   321→    return list;
   322→    // eslint-disable-next-line react-hooks/exhaustive-deps
   323→  }, [slate, version]);
   324→
   325→  const clips: AudioClip[] = useMemo(() => {
   326→    const list: AudioClip[] = [];
   327→    slate.audioClips().forEach((m, id) => { const c = readAudioClip(m, id); if (c) list.push(c); });
   328→    return list;
   329→    // eslint-disable-next-line react-hooks/exhaustive-deps
   330→  }, [slate, version]);
   331→  clipsRef.current = clips;
   332→
   333→  // When a clip's samples change (normalize/reverse from the settings panel),
   334→  // drop the cached waveform + decoded buffer so both refresh.
   335→  useEffect(() => {
   336→    const onChanged = (e: Event) => {
   337→      const id = (e as CustomEvent<string>).detail;
   338→      invalidateWaveform(id);
   339→      engineRef.current?.clearCache(id);
   340→      setVersion((v) => v + 1);
   341→    };
   342→    window.addEventListener('slate:audio-clip-changed', onChanged as EventListener);
   343→    return () => window.removeEventListener('slate:audio-clip-changed', onChanged as EventListener);
   344→  }, []);
   345→
   346→  const timelineDuration = Math.max(30, ...clips.map((c) => c.start + c.duration), positionRef.current + 10);
   347→
   348→  // ── Transport ─────────────────────────────────────────────────────────────
   349→
   350→  const togglePlay = useCallback(() => {
   351→    const eng = engineRef.current; if (!eng) return;
   352→    if (playing) { eng.stop(); setPlaying(false); }
   353→    else { if (looping) eng.setLoopRegion(loopStart, loopEnd); else eng.setLoopRegion(null, null); void eng.play(slate, positionRef.current); setPlaying(true); }
   354→  }, [playing, slate, looping, loopStart, loopEnd]);
   355→
   356→  const seek = useCallback((t: number) => {
   357→    const pos = Math.max(0, t);
   358→    positionRef.current = pos;
   359→    if (playheadRef.current) playheadRef.current.style.transform = `translateX(${pos * pxRef.current}px)`;
   360→    if (posDisplayRef.current) posDisplayRef.current.textContent = `${pos.toFixed(1)}s`;
   361→  }, []);
   362→
   363→  const toggleRecord = useCallback(async () => {
   364→    if (recording) {
   365→      const stopFn = stopRecRef.current;
   366→      if (stopFn) {
   367→        const r = await stopFn(); stopRecRef.current = null; setRecording(false);
   368→        let tid = tracks.find((t) => t.armed)?.id;
   369→        if (!tid) tid = addAudioTrack(slate, { name: 'Recording' });
   370→        addAudioClip(slate, tid, { start: positionRef.current, samples: r.samples, sampleRate: r.sampleRate, channels: r.channels, duration: r.duration, name: `Rec ${new Date().toLocaleTimeString()}` });
   371→        toast({ title: 'Recording added' });
   372→      }
   373→    } else {
   374→      try { const sf = await engineRef.current?.startRecording(); stopRecRef.current = sf ?? null; setRecording(true); }
   375→      catch { toast({ title: 'Mic denied', variant: 'error' }); }
   376→    }
   377→  }, [recording, tracks, slate]);
   378→
   379→  const dupClip = useCallback(async (id: string) => {
   380→    await duplicateAudioClip(slate, id);
   381→  }, [slate]);
   382→
   383→  const handleFileImport = useCallback(async (file: File) => {
   384→    try {
   385→      const d = await decodeAudioFile(file);
   386→      const tid = addAudioTrack(slate, { name: file.name.replace(/\.[^.]+$/, '') });
   387→      addAudioClip(slate, tid, { start: positionRef.current, samples: d.samples, sampleRate: d.sampleRate, channels: d.channels, duration: d.duration, name: file.name });
   388→      toast({ title: 'Imported', description: file.name });
   389→    } catch (err) { toast({ title: 'Import failed', description: (err as Error).message, variant: 'error' }); }
   390→  }, [slate]);
   391→
   392→  // ── Drag start ────────────────────────────────────────────────────────────
   393→
   394→  // The free interval [leftLimit, rightLimit] the clip may occupy without
   395→  // overlapping any other clip on the same track. `neighbours` is the full
   396→  // list of same-track clip bounds (used by drag-snap to jump past a clip to
   397→  // the free slot on its other side).
   398→  const neighbourBounds = useCallback((clip: AudioClip): { leftLimit: number; rightLimit: number; neighbours: { start: number; end: number }[] } => {
   399→    let leftLimit = 0;
   400→    let rightLimit = Infinity;
   401→    const neighbours: { start: number; end: number }[] = [];
   402→    const clipEnd = clip.start + clip.duration;
   403→    for (const o of clipsRef.current) {
   404→      if (o.id === clip.id || o.trackId !== clip.trackId) continue;
   405→      const oEnd = o.start + o.duration;
   406→      if (oEnd <= clip.start + 1e-4) leftLimit = Math.max(leftLimit, oEnd);
   407→      else if (o.start >= clipEnd - 1e-4) rightLimit = Math.min(rightLimit, o.start);
   408→      neighbours.push({ start: o.start, end: oEnd });
   409→    }
   410→    return { leftLimit, rightLimit, neighbours };
   411→  }, []);
   412→
   413→  const startDrag = useCallback((clip: AudioClip, e: React.PointerEvent, el: HTMLElement, waveEl: HTMLElement | null) => {
   414→    e.stopPropagation(); setSelectedClipId(clip.id);
   415→    const { leftLimit, rightLimit, neighbours } = neighbourBounds(clip);
   416→    dragRef.current = { clipId: clip.id, el, waveEl, os: clip.start, od: clip.duration, oo: clip.offset, speed: clip.speed ?? 1, sx: e.clientX, leftLimit, rightLimit, neighbours, mode: 'drag' };
   417→  }, [neighbourBounds]);
   418→
   419→  const startTrim = useCallback((clip: AudioClip, side: 'left' | 'right', e: React.PointerEvent, el: HTMLElement, waveEl: HTMLElement | null) => {
   420→    e.stopPropagation(); setSelectedClipId(clip.id);
   421→    const { leftLimit, rightLimit, neighbours } = neighbourBounds(clip);

   437→    };
   438→    const onUp = () => { loopDragRef.current = null; window.removeEventListener('pointermove', onMove); window.removeEventListener('pointerup', onUp); };
   439→    window.addEventListener('pointermove', onMove); window.addEventListener('pointerup', onUp);
   440→  }, [loopStart, loopEnd, pxPerSec]);

   461→        <div className="mx-1 h-5 w-px bg-border" />

'bg-accent/20 text-accent' : 'text-text-mid hover:bg-bg-3'}`} title="Metronome (M)"><Music size={13} /></button>

   468→        <button onClick={() => setPxPerSec((c) => Math.min(800, c * 1.3))} className="flex h-6 w-6 items-center justify-center rounded text-text-mid hover:bg-bg-3" title="Zoom in"><ZoomIn size={12} /></button>
   469→        <div className="flex-1" />

   471→        <button onClick={() => addAudioTrack(slate)} className="flex items-center gap-1 rounded border border-accent/40 bg-accent/10 px-2 py-1 text-[11px] text-accent hover:bg-accent/20"><Plus size={12} />Track</button>
   472→      </div>

   484→        <div className="relative flex-1" style={{ minWidth: timelineDuration * pxPerSec }}>

   492→                <div onPointerDown={(e) => startLoopDrag('start', e)} className="absolute top-0 z-20 flex h-7 cursor-ew-resize items-center justify-center bg-accent/40 hover:bg-accent/60" style={{ left: loopStart * pxPerSec - 8, width: 8 }} title="Drag loop start"><ChevronLeft size={10} className="text-white" /></div>
   493→                <div onPointerDown={(e) => startLoopDrag('end', e)} className="absolute top-0 z-20 flex h-7 cursor-ew-resize items-center justify-center bg-accent/40 hover:bg-accent/60" style={{ left: loopEnd * pxPerSec, width: 8 }} title="Drag loop end"><ChevronRight size={10} className="text-white" /></div>
   494→              </>
   495→            )}
   496→          </div>
   497→          {/* Seek layer — background click catcher (behind clips) */}
   498→          <div className="absolute inset-0 top-7" onPointerDown={(e) => { const r = e.currentTarget.getBoundingClientRect(); seek((e.clientX - r.left) / pxRef.current); }} />
   499→          {/* Grid background */}
   500→          <div className="pointer-events-none absolute inset-0 top-7" style={gridStyle}>
   501→            {looping && <div onPointerDown={(e) => startLoopDrag('move', e)} className="pointer-events-auto absolute top-0 bottom-0 cursor-grab bg-accent/8 border-x-2 border-accent/40" style={{ left: loopStart * pxPerSec, width: (loopEnd - loopStart) * pxPerSec }} />}
   502→          </div>
   503→          {/* Playhead */}
   504→          <div ref={playheadRef} className="absolute top-0 bottom-0 z-20 w-0.5 bg-warn pointer-events-none" style={{ transform: 'translateX(0px)' }}>
   505→            <div className="absolute top-0 -left-1 h-2 w-2 rounded-full bg-warn" />
   506→          </div>
   507→          {/* Clips */}
   508→          {tracks.map((t) => (
   509→            <div key={t.id} className="pointer-events-none relative border-b border-border/15" style={{ height: TRACK_H }}>
   510→              {clips.filter((c) => c.trackId === t.id).map((c) => (
   511→                <ClipBlock key={c.id} clip={c} pxPerSec={pxPerSec} selected={selectedClipId === c.id}
   512→                  onSelect={setSelectedClipId} onDragStart={startDrag} onTrimStart={startTrim} />
   513→              ))}
   514→            </div>
   515→          ))}
   516→        </div>
   517→      </div>
   518→
   519→      {/* Status */}
   520→      <div className="flex shrink-0 items-center gap-2 border-t border-border px-2 py-0.5 text-[8px] font-mono uppercase text-text-dim">
   521→        <span>{tracks.length}T · {clips.length}C</span>
   522→        {recording && <span className="text-danger">● Rec</span>}
   523→        {playing && <span className="text-accent">▶ Play</span>}
   524→        <span className="ml-auto">Space · C=Split · D=Dup · Del · R=Rec · L=Loop · M=Met · ←→=Seek · Ctrl+Scroll=Zoom</span>
   525→      </div>
   526→    </div>
   527→  );
   528→}

   532→const TrackHeader = memo(function TrackHeader({ track, hasSolo, slate, engineRef }: {

   541→
   542→  const update = (patch: Partial<AudioTrack>) => { updateAudioTrack(slate, track.id, patch); engineRef.current?.updateTracks(slate); };

'currentColor' : 'transparent', border: '1px solid currentColor' }} /></button>

   569→const ClipBlock = memo(function ClipBlock({ clip, pxPerSec, selected, onSelect, onDragStart, onTrimStart }: {
   570→  clip: AudioClip; pxPerSec: number; selected: boolean;
   571→  onSelect: (id: string) => void;
   572→  onDragStart: (clip: AudioClip, e: React.PointerEvent, el: HTMLElement, waveEl: HTMLElement | null) => void;
   573→  onTrimStart: (clip: AudioClip, side: 'left' | 'right', e: React.PointerEvent, el: HTMLElement, waveEl: HTMLElement | null) => void;
   574→}) {
   575→  const left = clip.start * pxPerSec;
   576→  const width = Math.max(4, clip.duration * pxPerSec);
   577→  const elRef = useRef<HTMLDivElement | null>(null);
   578→  const waveRef = useRef<HTMLDivElement | null>(null);
   579→  const w = Math.max(2, Math.floor(width));
   580→  // Fade overlay widths (clamped to the clip box so a too-long fade doesn't
   581→  // spill past the opposite edge).
   582→  const fadeInW = clip.fadeIn > 0 ? Math.min(width, clip.fadeIn * pxPerSec) : 0;
   583→  const fadeOutW = clip.fadeOut > 0 ? Math.min(width, clip.fadeOut * pxPerSec) : 0;
   584→
   585→  return (
   586→    <div ref={elRef} onPointerDown={(e) => { if (elRef.current) { e.stopPropagation(); onSelect(clip.id); window.dispatchEvent(new CustomEvent('slate:audio-clip-select', { detail: clip.id })); onDragStart(clip, e, elRef.current, waveRef.current); } }}
   587→      className={`group pointer-events-auto absolute top-0.5 bottom-0.5 cursor-grab overflow-hidden rounded border ${selected ? 'border-warn' : 'border-black/30'} active:cursor-grabbing`} style={{ left, width, backgroundColor: `${clip.color}20` }}>
   588→      {/* Fixed-width waveform layer — clipped by the box so trimming cuts the
   589→          audio rather than squashing the whole wave into a smaller space. */}
   590→      <div ref={waveRef} className="pointer-events-none absolute top-0 bottom-0 left-0" style={{ width }}>
   591→        {clip.sampleKey && <WaveformImg clipId={clip.id} sampleKey={clip.sampleKey} channels={clip.channels} sampleRate={clip.sampleRate} offset={clip.offset} duration={clip.duration} speed={clip.speed ?? 1} width={w} color={clip.color} />}
   592→      </div>
   593→      {/* Fade-in overlay: triangle from full height at the outer (left) edge
   594→          tapering to 0 at the inner edge — the dark wedge represents the
   595→          portion of the clip still under the fade. */}
   596→      {fadeInW > 0 && (
   597→        <div className="pointer-events-none absolute top-0 bottom-0 left-0 bg-black/30" style={{ width: fadeInW, clipPath: 'polygon(0% 0%, 0% 100%, 100% 50%)' }} />
   598→      )}
   599→      {/* Fade-out overlay: mirrored on the right side. */}
   600→      {fadeOutW > 0 && (
   601→        <div className="pointer-events-none absolute top-0 bottom-0 right-0 bg-black/30" style={{ width: fadeOutW, clipPath: 'polygon(100% 0%, 100% 100%, 0% 50%)' }} />
   602→      )}
   603→      <span className="absolute left-1 top-0 truncate text-[7px] font-medium text-text-mid/70 pointer-events-none">{clip.name}{clip.mute ? ' (muted)' : ''}</span>
   604→      <div onPointerDown={(e) => { if (elRef.current) onTrimStart(clip, 'left', e, elRef.current, waveRef.current); }} className="absolute left-0 top-0 bottom-0 w-1.5 cursor-ew-resize bg-white/40 opacity-0 group-hover:opacity-100" />
   605→      <div onPointerDown={(e) => { if (elRef.current) onTrimStart(clip, 'right', e, elRef.current, waveRef.current); }} className="absolute right-0 top-0 bottom-0 w-1.5 cursor-ew-resize bg-white/40 opacity-0 group-hover:opacity-100" />
   606→    </div>
   607→  );
   608→});
