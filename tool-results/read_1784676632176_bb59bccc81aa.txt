    94тЖТ    94тЖТ  class {

    96тЖТ    96тЖТ    constructor(view: EditorView) {
    97тЖТ    97тЖТ      this.decos = this.build(view);
    98тЖТ    98тЖТ    }
    99тЖТ    99тЖТ    update(u: ViewUpdate) {
   100тЖТ   100тЖТ      if (u.docChanged || u.viewportChanged || u.selectionSet) {
   101тЖТ   101тЖТ        this.decos = this.build(u.view);
   102тЖТ   102тЖТ      }
   103тЖТ   103тЖТ    }
   104тЖТ   104тЖТ    build(view: EditorView) {
   105тЖТ   105тЖТ      const items: { from: number; value: ReturnType<typeof Decoration.line> }[] = [];
   106тЖТ   106тЖТ      for (const { from, to } of view.visibleRanges) {
   107тЖТ   107тЖТ        for (let pos = from; pos <= to; ) {
   108тЖТ   108тЖТ          const line = view.state.doc.lineAt(pos);
   109тЖТ   109тЖТ          // Count leading whitespace in 2-space units тАФ a leading tab counts
   110тЖТ   110тЖТ          // as one guide so mixed-indent files still get something useful.
   111тЖТ   111тЖТ          const m = /^([ \t]+)/.exec(line.text);
   112тЖТ   112тЖТ          const spaces = m?.[1] ? m[1].replace(/\t/g, '  ').length : 0;
   113тЖТ   113тЖТ          const levels = Math.floor(spaces / 2);
   114тЖТ   114тЖТ          if (levels > 0) {
   115тЖТ   115тЖТ            items.push({
   116тЖТ   116тЖТ              from: line.from,
   117тЖТ   117тЖТ              value: Decoration.line({
   118тЖТ   118тЖТ                class: 'cm-indent-guide',
   119тЖТ   119тЖТ                attributes: { style: `--cm-indent: ${levels}` },
   120тЖТ   120тЖТ              }),
   121тЖТ   121тЖТ            });
   122тЖТ   122тЖТ          }
   123тЖТ   123тЖТ          pos = line.to + 1;
   124тЖТ   124тЖТ        }
   125тЖТ   125тЖТ      }
   126тЖТ   126тЖТ      return Decoration.set(
   127тЖТ   127тЖТ        items.map((i) => i.value.range(i.from)),
   128тЖТ   128тЖТ        true,
   129тЖТ   129тЖТ      );
   130тЖТ   130тЖТ    }
