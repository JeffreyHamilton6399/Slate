    61→/** Nearest start position at which a clip of `duration` fits without
    62→ *  overlapping any blocker. Merges blockers into occupied intervals, then
    63→ *  picks the gap position closest to `desired`. This replaces the previous
    64→ *  "slide to the nearest side of the current blocker" iteration, which could
    65→ *  ping-pong between two adjacent blockers (each resolution landing on the
    66→ *  other) and exhaust its iteration budget while STILL overlapping —
    67→ *  committing a clip on top of another one. Scanning gaps directly is
    68→ *  deterministic: the open-ended gap after the last blocker always fits, so a
    69→ *  valid position always exists. */
    70→function nearestFreeStart(desired: number, duration: number, blockersIn: { start: number; end: number }[]): number {
    71→  const eps = 1e-4;
    72→  desired = Math.max(0, desired);
    73→  if (blockersIn.length === 0) return desired;
    74→  const sorted = [...blockersIn].sort((a, b) => a.start - b.start);
    75→  const occupied: { start: number; end: number }[] = [];
    76→  for (const b of sorted) {
    77→    const last = occupied[occupied.length - 1];
    78→    if (last && b.start <= last.end + eps) last.end = Math.max(last.end, b.end);
    79→    else occupied.push({ start: b.start, end: b.end });
    80→  }
    81→  let best = desired;
    82→  let bestDist = Infinity;
    83→  const consider = (lo: number, hi: number) => {
    84→    if (hi - lo < duration - eps) return; // gap too small
    85→    const cand = Math.min(Math.max(desired, lo), hi - duration);
    86→    const dist = Math.abs(cand - desired);
    87→    if (dist < bestDist) { bestDist = dist; best = cand; }
    88→  };
    89→  let prevEnd = 0;
    90→  for (const o of occupied) {
    91→    consider(prevEnd, o.start);
    92→    prevEnd = Math.max(prevEnd, o.end);
    93→  }
    94→  consider(prevEnd, Infinity); // open-ended tail — always fits
    95→  return Math.max(0, best);
    96→}

    98→// ── Waveform cache: pre-computed PNG data URLs ───────────────────────────────
    99→// Key: `${clipId}:${sampleCount}:${width}` → data URL
   100→const waveformPNGCache = new Map<string, string>();
   101→
   102→// Draw only the window [startFrame, endFrame) of the sample across `width`
   103→// pixels. Rendering the *window* (not the whole sample stretched) is what makes
   104→// trimming look like cutting the audio away rather than squashing it.
   105→function computeWaveformPNG(samples: Float32Array, channels: number, width: number, color: string, height: number, startFrame: number, endFrame: number): string {
   106→  const canvas = document.createElement('canvas');
   107→  const dpr = window.devicePixelRatio || 1;
   108→  const w = Math.max(2, Math.floor(width));
   109→  canvas.width = w * dpr;
   110→  canvas.height = height * dpr;
   111→  const ctx = canvas.getContext('2d')!;
   112→  ctx.scale(dpr, dpr);
   113→  ctx.clearRect(0, 0, w, height);
   114→  const totalFrames = samples.length / channels;
   115→  const s = Math.max(0, Math.min(totalFrames, Math.floor(startFrame)));
   116→  const e = Math.max(s, Math.min(totalFrames, Math.floor(endFrame)));
   117→  const span = e - s;
   118→  if (span <= 0) return canvas.toDataURL();
   119→  const mid = height / 2;
   120→  ctx.fillStyle = color;
   121→  for (let x = 0; x < w; x++) {
   122→    const f0 = s + Math.floor((x / w) * span);
   123→    const f1 = Math.min(e, s + Math.floor(((x + 1) / w) * span) + 1);
   124→    let peak = 0;
   125→    for (let i = f0; i < f1; i++) { const v = Math.abs(samples[i * channels] ?? 0); if (v > peak) peak = v; }
   126→    const bh = Math.max(1, peak * mid * 0.85);
   127→    ctx.fillRect(x, mid - bh, 1, bh * 2);
   128→  }
   129→  return canvas.toDataURL();
   130→}

   132→/** Waveform image for the buffer window the clip actually plays. With speed s,
   133→ *  a clip of `duration` timeline seconds consumes `duration*s` buffer seconds.
   134→ *
   135→ *  Two robustness fixes:
   136→ *  1. If `loadSamples` returns an EMPTY Float32Array (length 0) — which happens
   137→ *     when a freshly-split/created clip's IndexedDB write hasn't landed yet —
   138→ *     we DON'T cache the resulting blank PNG. Instead we show the `···`
   139→ *     placeholder and retry a few times (500ms apart) so the real waveform
   140→ *     appears once the samples are available.
   141→ *  2. We listen for `slate:audio-clip-changed` (fired by Normalize/Reverse/split)
   142→ *     and invalidate our cache entry for the current clipId, then force a
   143→ *     recompute via a `bust` counter — otherwise the memoised component would
   144→ *     keep showing the stale PNG even after the cache was cleared. */
   145→const WaveformImg = memo(function WaveformImg({ clipId, sampleKey, channels, sampleRate, offset, duration, speed, width, color }: {
   146→  clipId: string; sampleKey: string; channels: number; sampleRate: number;
   147→  offset: number; duration: number; speed: number; width: number; color: string;
   148→}) {
   149→  const height = TRACK_H - 6;
   150→  const [imgUrl, setImgUrl] = useState<string>('');
   151→  const [bust, setBust] = useState(0);
   152→  const retryRef = useRef(0);
   153→
   154→  // Invalidate cached PNGs for this clip when its samples change (normalize /
   155→  // reverse / split). The parent AudioEditor also invalidates + bumps version,
   156→  // but the memoised WaveformImg wouldn't otherwise recompute (its primitive
   157→  // props are unchanged) — the `bust` counter forces the load effect to re-run.
   158→  useEffect(() => {
   159→    const onChanged = (e: Event) => {
   160→      const detail = (e as CustomEvent<string>).detail;
   161→      // Match by clipId (normalize/reverse/split events) OR by sampleKey
   162→      // (multiplayer sample-arrival events from registerSampleSyncMap).
   163→      if (detail !== clipId && detail !== sampleKey) return;
   164→      for (const key of waveformPNGCache.keys()) {
   165→        if (key.startsWith(`${clipId}:`)) waveformPNGCache.delete(key);
   166→      }
   167→      retryRef.current = 0;
   168→      setBust((n) => n + 1);
   169→    };
   170→    window.addEventListener('slate:audio-clip-changed', onChanged as EventListener);
   171→    return () => window.removeEventListener('slate:audio-clip-changed', onChanged as EventListener);
   172→  }, [clipId, sampleKey]);
   173→
   174→  useEffect(() => {
   175→    const startFrame = Math.round(offset * sampleRate);
   176→    const endFrame = Math.round((offset + duration * speed) * sampleRate);
   177→    const cacheKey = `${clipId}:${sampleKey}:${startFrame}:${endFrame}:${Math.floor(width)}`;
   178→    const cached = waveformPNGCache.get(cacheKey);
   179→    if (cached) { setImgUrl(cached); return; }
   180→    let cancelled = false;
   181→    let retryTimer: ReturnType<typeof setTimeout> | null = null;
   182→
   183→    const attempt = () => {
   184→      if (cancelled) return;
   185→      void loadSamples(sampleKey).then((samples) => {
   186→        if (cancelled) return;
   187→        // Empty samples = the IndexedDB write for this sampleKey hasn't landed
   188→        // yet (e.g. brand-new clip from a split, OR a remote peer's clip whose
   189→        // sample blob is still in flight via the Yjs sync map). DON'T cache a
   190→        // blank PNG — retry for up to 10s (20 × 500ms) so the real waveform
   191→        // appears once the samples arrive. The previous 5-retry / 2.5s budget
   192→        // was too short for cross-peer sample sync, which can take several
   193→        // seconds for a 5MB blob over a slow link. The `slate:audio-clip-changed`
   194→        // event (dispatched by onClipsAdded when a remote clip is added, and by
   195→        // registerSampleSyncMap when samples arrive) resets `retryRef` to 0 so
   196→        // the WaveformImg re-tries immediately on the signal — this retry
   197→        // budget is the safety net for the case where the event fires before
   198→        // the WaveformImg mounts, or doesn't fire at all.
   199→        if (samples.length === 0) {
   200→          if (retryRef.current < 20) {
   201→            retryRef.current += 1;
   202→            retryTimer = setTimeout(attempt, 500);
   203→          }
   204→          return;
   205→        }
   206→        retryRef.current = 0;
   207→        const url = computeWaveformPNG(samples, channels, width, color, height, startFrame, endFrame);
   208→        waveformPNGCache.set(cacheKey, url);
   209→        setImgUrl(url);
   210→      });
   211→    };
   212→    attempt();
   213→
   214→    return () => {
   215→      cancelled = true;
   216→      if (retryTimer) clearTimeout(retryTimer);
   217→    };
   218→  }, [clipId, sampleKey, channels, sampleRate, offset, duration, speed, width, color, height, bust]);
   219→
   220→  if (!imgUrl) return <div className="flex h-full items-center justify-center text-[7px] text-text-dim">···</div>;
   221→  return <img src={imgUrl} alt="" className="pointer-events-none h-full w-full" style={{ objectFit: 'fill' }} />;
   222→});
   223→
   224→/** Drop cached waveform PNGs for a clip (call when its samples change). */
   225→function invalidateWaveform(clipId: string): void {
   226→  for (const key of waveformPNGCache.keys()) if (key.startsWith(`${clipId}:`)) waveformPNGCache.delete(key);
   227→}

   331→   *  arrive after play started (remote sample blob landed via the sync map).
   332→   *  The debounce coalesces rapid bursts so we don't tear down and rebuild

   361→        });

   375→  // Clear our audio playhead on unmount so other peers stop seeing us in the

   382→
   383→  // Yjs subscription.
   384→  useEffect(() => {
   385→    // NOTE: registerSampleSyncMap(room) is now called from useSlateRoom's
   386→    // attach() — registering here only on AudioEditor mount meant a peer on
   387→    // a 2D/3D board (audio panel closed) never registered the sync map and

   390→    const clips = slate.audioClips();

   393→    const bump = () => { if (pending) return; pending = true; requestAnimationFrame(() => { pending = false; setVersion((v) => v + 1); }); };
   394→    // Any track edit (volume, pan, EQ, sends, mute/solo — local knob or remote
   395→    // peer) re-applies the Yjs values to the live audio graph, so the settings
   396→    // panel is audible mid-playback without a restart.
   397→    //
   398→    // REMOTE MUTE/SOLO FIX: this subscription is what makes "user A mutes a
   399→    // track → user B hears the mute" work. The mute state IS in Yjs (it's a
   400→    // field on the track Y.Map), so it syncs to every peer. But sync alone
   401→    // doesn't affect the audio graph — the local engine's per-track gain
   402→    // nodes still hold the old value until something re-applies them.
   403→    // `applyTracks` is that "something": it reads the fresh Yjs tracks and
   404→    // calls `setupTrackNodes`, which writes `gain.gain.value = audible ?
   405→    // volume : 0` per track (audible = solo ? track.solo : !track.muted).
   406→    // Without this subscription, remote mute/solo edits would land in Yjs
   407→    // (visible in the TrackHeader) but never reach the audio graph until the
   408→    // user pressed stop/play (which re-runs setupTrackNodes via play()).
   409→    const applyTracks = () => { engineRef.current?.updateTracks(slateRef.current); };
   410→    // Clip edits (position, trim, gain, pan, filters, speed/pitch — ours or a
   411→    // peer's) reschedule playback so they're audible mid-play. Debounced 500ms
   412→    // in scheduleRestart and a no-op while paused, so knob drags coalesce into
   413→    // one restart instead of stuttering. Track edits deliberately DON'T
   414→    // restart — applyTracks adjusts their nodes live.
   415→    const restartOnClipEdit = () => { scheduleRestart(); };
   416→    tracks.observeDeep(bump); clips.observeDeep(bump); audioMap.observe(bump); bump();
   417→    tracks.observeDeep(applyTracks);
   418→    clips.observeDeep(restartOnClipEdit);
   419→    // Detect NEW clips being added while playing and schedule a debounced
   420→    // restartPlayback so the new clip is picked up mid-playback without a
   421→    // manual stop/play. We watch the shallow clips map (not observeDeep) so
   422→    // we only react to key additions, not to every property edit on an
   423→    // existing clip (which would restart playback every time a peer nudged
   424→    // a clip's volume, fighting the user). The restart is debounced 500ms so
   425→    // a burst of additions (e.g. a peer imports 5 files) coalesces into one.
   426→    //
   427→    // LIVE WAVEFORM REFRESH: when a remote peer adds a clip, its metadata
   428→    // arrives via Yjs immediately but the sample blob travels separately via
   429→    // the audioSampleSync Y.Map (see sampleStore.ts). If we don't ping the
   430→    // WaveformImg, it might have already exhausted its empty-samples retry
   431→    // budget (now 10s, but a slow link can still exceed that) and given up
   432→    // showing a blank "···" placeholder until the user refreshes. So for
   433→    // each newly-added clip we:
   434→    //   1. clearCache(clipId) on the engine so the next play re-loads samples
   435→    //      (the cache might hold a null/empty entry from a prior failed getBuffer)
   436→    //   2. dispatch `slate:audio-clip-changed` with the clipId — the
   437→    //      AudioEditor's listener (below) re-invalidates the waveform cache +
   438→    //      clears the engine cache + bumps version, AND the WaveformImg's own
   439→    //      listener resets its retryRef to 0 and bumps `bust` to immediately
   440→    //      re-attempt loadSamples (which will now find the synced blob once
   441→    //      it lands).
   442→    //   3. scheduleRestart() so if we're currently playing, the new clip is
   443→    //      picked up atomically (debounced 500ms to coalesce bursts).
   444→    const onClipsAdded = (event: Y.YMapEvent<Y.Map<unknown>>) => {
   445→      const addedIds: string[] = [];
   446→      for (const key of event.keysChanged) {
   447→        const change = event.changes.keys.get(key);
   448→        if (change?.action === 'add') addedIds.push(key);
   449→      }
   450→      if (addedIds.length === 0) return;
   451→      for (const id of addedIds) {
   452→        engineRef.current?.clearCache(id);
   453→        window.dispatchEvent(new CustomEvent('slate:audio-clip-changed', { detail: id }));
   454→      }
   455→      scheduleRestart();
   456→    };

   458→    const lateRead = setTimeout(bump, 200);
   459→    return () => { clearTimeout(lateRead); tracks.unobserveDeep(bump); tracks.unobserveDeep(applyTracks); clips.unobserveDeep(bump); clips.unobserveDeep(restartOnClipEdit); audioMap.unobserve(bump); clips.unobserve(onClipsAdded); };

   509→  /** Zoom while keeping one timeline point at the same screen offset.
   510→   *  The anchor is the time under `anchorClientX` (mouse-wheel zoom: the spot
   511→   *  under the cursor stays put — how every DAW/map zoom feels), or the
   512→   *  playhead when no anchor is given (toolbar buttons, fit-to-window).

   554→  // Global pointermove/up — pure DOM, zero React state.
   555→  // rAF-throttled: pointermove fires at the hardware's report rate (often
   556→  // 60-120Hz), and the move handler does a neighbours-overlap scan on every
   557→  // event. Coalescing multiple moves into a single rAF callback caps the work
   558→  // at the display refresh rate (~60fps) so a fast pointer drag never stalls
   559→  // the main thread. A no-op skip (cursor hasn't moved since the last
   560→  // processed frame) avoids redundant DOM writes for stationary pointers.
   561→  useEffect(() => {
   562→    /** Last pointer position we actually applied to the DOM. Lets us skip
   563→     *  work when pointermove fires without real movement (e.g. a touchpad
   564→     *  reporting pressure changes, or sub-pixel jitter). */
   565→    let lastProcessedX: number | null = null;
   566→    let lastProcessedY: number | null = null;
   567→
   568→    /** Magnet: pull `t` (a clip START time) to the nearest beat line, other
   569→     *  clip edge, or the playhead — considering BOTH the clip's start and end
   570→     *  edges — if one is within SNAP_PX at the current zoom. `dur` null =
   571→     *  only the start edge matters (trims). Returns the adjusted start. */
   572→    const snapStart = (t: number, dur: number | null, snapTimes: number[]): number => {
   573→      const pps = pxRef.current;
   574→      const thr = SNAP_PX / pps;
   575→      const beat = 60 / bpmRef.current;
   576→      let best = t;
   577→      let bestDist = thr;
   578→      const consider = (target: number, edgeOffset: number) => {
   579→        const dist = Math.abs(t + edgeOffset - target);
   580→        if (dist < bestDist) { bestDist = dist; best = target - edgeOffset; }
   581→      };
   582→      consider(Math.round(t / beat) * beat, 0);
   583→      if (dur !== null) consider(Math.round((t + dur) / beat) * beat, dur);
   584→      for (const st of snapTimes) {
   585→        consider(st, 0);
   586→        if (dur !== null) consider(st, dur);
   587→      }
   588→      return best;
   589→    };

   591→    const applyMove = () => {
   592→      moveRafRef.current = 0;
   593→      const d = dragRef.current;
   594→      if (!d) return;
   595→      const clientX = moveXRef.current;
   596→      if (clientX === null) return;
   597→      const clientY = moveYRef.current;
   598→      // Skip no-op moves — cursor hasn't moved to a new pixel since the last
   599→      // frame we processed. Saves an overlap scan + 1-2 style writes.
   600→      if (lastProcessedX === clientX && lastProcessedY === clientY) return;
   601→      lastProcessedX = clientX;
   602→      lastProcessedY = clientY;
   603→
   604→      // Dead zone: a plain click (pointer never leaves a 3px box) must not
   605→      // nudge the clip. Once exceeded, the drag is live for good.
   606→      if (!d.moved) {
   607→        if (Math.abs(clientX - d.sx) < DRAG_DEADZONE_PX && Math.abs(clientY - d.sy) < DRAG_DEADZONE_PX) return;
   608→        d.moved = true;
   609→        if (d.mode === 'drag') d.el.style.zIndex = '40'; // ride above other rows while crossing tracks
   610→      }
   611→
   612→      const pps = pxRef.current;
   613→      const dt = (clientX - d.sx) / pps;
   614→      const oldEnd = d.os + d.od;
   615→      const wantSnap = snapRef.current && !moveAltRef.current;
   616→      if (d.mode === 'drag') {
   617→        // The clip follows the mouse 1:1 (snapped to the magnet grid). It's
   618→        // allowed to ride OVER other clips while the drag is live — it's
   619→        // elevated to z-40 so the overlap reads as "in transit", and the
   620→        // no-overlap invariant is enforced once, on drop (see onUp). The
   621→        // previous mid-drag resolution parked the clip flush against a
   622→        // blocker and then teleported it across when the cursor passed the
   623→        // blocker's midpoint, which made dragging feel sticky and jumpy.
   624→        const duration = d.od;
   625→        // Vertical: whole-row steps move the clip across tracks.
   626→        const rowDelta = Math.max(
   627→          -d.trackIndex,
   628→          Math.min(d.trackIds.length - 1 - d.trackIndex, Math.round((clientY - d.sy) / TRACK_H)),
   629→        );
   630→        d.rowDelta = rowDelta;
   631→        d.el.style.transform = rowDelta === 0 ? '' : `translateY(${rowDelta * TRACK_H}px)`;
   632→
   633→        const desired = wantSnap ? snapStart(d.os + dt, duration, d.snapTimes) : d.os + dt;
   634→        d.el.style.left = `${Math.max(0, desired) * pps}px`;
   635→      } else if (d.mode === 'trimL') {
   636→        // Cut from the left: never past the left neighbour, the source start
   637→        // (offset ≥ 0 → limited by the trimmed head in timeline seconds), or a
   638→        // minimum width.
   639→        const minStart = Math.max(d.leftLimit, d.os - d.oo / d.speed);
   640→        const desired = wantSnap ? snapStart(d.os + dt, null, d.snapTimes) : d.os + dt;
   641→        const start = Math.min(oldEnd - 0.1, Math.max(minStart, desired));
   642→        d.el.style.left = `${start * pps}px`;
   643→        d.el.style.width = `${(oldEnd - start) * pps}px`;
   644→        // Shift the (fixed-width) waveform the opposite way so the audio stays
   645→        // anchored — the newly-trimmed head is clipped, not squashed.
   646→        if (d.waveEl) d.waveEl.style.left = `${-(start - d.os) * pps}px`;
   647→      } else if (d.mode === 'trimR') {
   648→        // Cut from the right: never past the right neighbour or a min width.
   649→        const desired = wantSnap ? snapStart(oldEnd + dt, null, d.snapTimes) : oldEnd + dt;
   650→        const end = Math.min(d.rightLimit, Math.max(d.os + 0.1, desired));
   651→        d.el.style.width = `${(end - d.os) * pps}px`;
   652→      }
   653→    };

   655→    const onMove = (ev: PointerEvent) => {
   656→      const d = dragRef.current; if (!d) return;
   657→      // Stash the latest pointer state — the rAF callback reads it. Cheap
   658→      // (ref writes) so even 120Hz pointermove streams don't pile up work.
   659→      moveXRef.current = ev.clientX;
   660→      moveYRef.current = ev.clientY;
   661→      moveAltRef.current = ev.altKey;

   663→      // coalesces multiple moves within the same frame into one DOM write.
   664→      if (!moveRafRef.current) {

   667→    };
   668→    const onUp = () => {
   669→      const d = dragRef.current; if (!d) return;
   670→      // Cancel any pending rAF — we'll process the final position
   671→      // SYNCHRONOUSLY here so the committed Yjs value reflects the exact
   672→      // pointer-up location (otherwise the last pointermove before pointerup
   673→      // could be dropped, leaving the committed position ~1 frame behind the
   674→      // cursor). lastProcessed is reset to bypass the no-op skip in case the
   675→      // final pointermove matched the prior processed frame but we still need
   676→      // to read the just-written style to commit to Yjs.
   677→      if (moveRafRef.current) {
   678→        cancelAnimationFrame(moveRafRef.current);
   679→        moveRafRef.current = 0;
   680→      }
   681→      if (moveXRef.current !== null) {
   682→        lastProcessedX = null; // bypass the no-op skip on the final flush
   683→        lastProcessedY = null;
   684→        applyMove();
   685→      }
   686→      // Undo the drag-only visual state React doesn't manage.
   687→      d.el.style.zIndex = '';
   688→      d.el.style.transform = '';
   689→      // A click that never left the dead zone selects but must not write —
   690→      // committing a parseFloat round-trip of the style would drift the clip.
   691→      if (d.moved) {
   692→        const pps = pxRef.current;
   693→        const left = parseFloat(d.el.style.left) / pps;
   694→        const width = parseFloat(d.el.style.width) / pps;
   695→        if (d.mode === 'drag') {
   696→          const targetTrackId = d.trackIds[d.trackIndex + d.rowDelta] ?? d.trackIds[d.trackIndex]!;
   697→          const originTrackId = d.trackIds[d.trackIndex]!;
   698→          // Enforce the no-overlap invariant HERE (the drag itself follows
   699→          // the mouse freely): place the clip in the free gap nearest to the
   700→          // dropped position.
   701→          const start = nearestFreeStart(left, d.od, d.byTrack.get(targetTrackId) ?? []);
   702→          // Write the resolved position to the DOM too so there's no flicker
   703→          // between pointer-up and the Yjs-driven re-render.
   704→          d.el.style.left = `${start * pps}px`;

   706→            start,
   707→            ...(targetTrackId !== originTrackId ? { trackId: targetTrackId } : {}),
   708→          });
   709→        }
   710→        else if (d.mode === 'trimL') updateAudioClip(slate, d.clipId, { start: left, duration: width, offset: Math.max(0, d.oo + (left - d.os) * d.speed) });
   711→        else if (d.mode === 'trimR') updateAudioClip(slate, d.clipId, { duration: width });
   712→      }
   713→      moveXRef.current = null;
   714→      lastProcessedX = null;
   715→      lastProcessedY = null;
   716→      dragRef.current = null;
   717→    };

   719→    window.addEventListener('pointerup', onUp);

   721→      window.removeEventListener('pointermove', onMove);
   722→      window.removeEventListener('pointerup', onUp);

   725→        cancelAnimationFrame(moveRafRef.current);
   726→        moveRafRef.current = 0;
   727→      }
   728→    };

  1321→const TrackHeader = memo(function TrackHeader({ track, hasSolo, slate, engineRef }: {
  1322→  track: AudioTrack; hasSolo: boolean;
  1323→  slate: ReturnType<typeof useRoom>['slate'];
  1324→  engineRef: React.RefObject<AudioEngine | null>;
  1325→}) {
  1326→  const [vol, setVol] = useState(track.volume);
  1327→  const [pan, setPan] = useState(track.pan);
  1328→  // Separate drag flags per slider — the previous single shared `isDraggingRef`
  1329→  // meant starting a volume drag also blocked the pan prop-sync effect (and
  1330→  // vice versa), which could cause one slider's committed Yjs value to be
  1331→  // momentarily ignored by the other's gating. Per-slider flags keep each
  1332→  // slider's prop-sync independent.
  1333→  const isDraggingVolRef = useRef(false);
  1334→  const isDraggingPanRef = useRef(false);
  1335→  useEffect(() => { if (!isDraggingVolRef.current) setVol(track.volume); }, [track.volume]);
  1336→  useEffect(() => { if (!isDraggingPanRef.current) setPan(track.pan); }, [track.pan]);
  1337→
  1338→  // Audibility — mirrors engine.setupTrackNodes so the live gain we write
  1339→  // during a volume drag respects mute/solo (don't briefly un-mute a muted
  1340→  // track just because the user is dragging its volume slider).
  1341→  const audible = hasSolo ? track.solo : !track.muted;
  1342→
  1343→  // `update` is for non-slider track edits (name/mute/solo/arm) — these DO
  1344→  // need a full updateTracks so the audio graph reflects the new state
  1345→  // (e.g. toggling solo rebalances every track's gain).
  1346→  const update = (patch: Partial<AudioTrack>) => { updateAudioTrack(slate, track.id, patch); engineRef.current?.updateTracks(slate); };
  1347→
  1348→  // Volume slider: update local state + the engine's gain node DIRECTLY for
  1349→  // immediate audio feedback. Do NOT call updateTracks (which re-reads every
  1350→  // track from Yjs and rebuilds the audio graph — far too expensive per
  1351→  // onChange event, and the source of the reported lag). Commit to Yjs once
  1352→  // on pointerup.
  1353→  const onVolDown = () => { isDraggingVolRef.current = true; };
  1354→  const onVol = (v: number) => { setVol(v); engineRef.current?.setTrackVolume(track.id, v, audible); };
  1355→  const onVolEnd = () => { isDraggingVolRef.current = false; updateAudioTrack(slate, track.id, { volume: vol }); };
  1356→
  1357→  // Pan slider: same pattern — set the panner node directly, commit on
  1358→  // pointerup. The pan slider is smaller than the volume slider (secondary
  1359→  // control) but still usable, with L/R labels so its function is obvious
  1360→  // (the user complaint was "two sliders, only one works" — partly because
  1361→  // neither slider was labelled, so it wasn't clear what each did).
  1362→  const onPanDown = () => { isDraggingPanRef.current = true; };
  1363→  const onPan = (p: number) => { setPan(p); engineRef.current?.setTrackPan(track.id, p); };
  1364→  const onPanEnd = () => { isDraggingPanRef.current = false; updateAudioTrack(slate, track.id, { pan: pan }); };
  1365→
  1366→  // ── Track kind: Audio ↔ MIDI ─────────────────────────────────────────────
  1367→  // A single icon button toggles the kind. On a MIDI track, an instrument
  1368→  // picker (compact <select>) sits next to the name so the user can choose
  1369→  // which synth preset or soundfont the track's MIDI clips play through.
  1370→  // Toggling kind to 'midi' defaults the instrument to the soundfont piano
  1371→  // (a sensible starting point for a fresh MIDI track); toggling back to
  1372→  // 'audio' leaves the instrumentId in place (harmless — it's only consulted
  1373→  // for MIDI clips).
  1374→  const isMidi = track.kind === 'midi';
  1375→  const toggleKind = () => {
  1376→    const nextKind: AudioTrack['kind'] = isMidi ? 'audio' : 'midi';
  1377→    // Switching to MIDI defaults the instrument to the soundfont piano if the
  1378→    // track doesn't already have one set; switching to audio clears the input
  1379→    // back to 'none' (a MIDI-armed audio track makes no sense).
  1380→    const patch: Partial<AudioTrack> = { kind: nextKind };
  1381→    if (nextKind === 'midi') {
  1382→      if (!track.instrumentId) patch.instrumentId = SOUNDFONT_PIANO_ID;
  1383→      // If the track was armed for mic, drop the arm — MIDI tracks arm for
  1384→      // instrument-take recording, not mic input.
  1385→      if (track.armed) { patch.armed = false; patch.input = 'none'; }
  1386→    } else if (track.input === 'midi') {
  1387→      patch.input = 'none';
  1388→      patch.armed = false;
  1389→    }
  1390→    update(patch);
  1391→  };
  1392→
  1393→  // Arm button: on an audio track → 'mic' input; on a MIDI track → 'midi'
  1394→  // input. The arm state still toggles; only the input source differs.
  1395→  const toggleArm = () => {
  1396→    if (track.armed) update({ armed: false, input: 'none' });
  1397→    else update({ armed: true, input: isMidi ? 'midi' : 'mic' });
  1398→  };
  1399→
  1400→  return (
  1401→    <div className="border-b border-border/15 px-2 py-1" style={{ height: TRACK_H }}>
  1402→      <div className="flex items-center gap-0.5">
  1403→        <span className="h-2 w-2 shrink-0 rounded-full" style={{ backgroundColor: track.color }} />
  1404→        <input type="text" value={track.name} onChange={(e) => update({ name: e.target.value })} className="min-w-0 flex-1 bg-transparent text-[11px] font-medium text-text outline-none" />
  1405→        {/* Kind toggle: Audio (Volume2 icon) ↔ MIDI (Piano icon). Highlighted
  1406→            when MIDI so the kind is visually distinct at a glance. */}
  1407→        <button onClick={toggleKind} className={`flex h-4 w-4 items-center justify-center rounded ${isMidi ? 'bg-accent/30 text-accent' : 'text-text-mid hover:bg-bg-3'}`} title={isMidi ? 'MIDI track (click for Audio)' : 'Audio track (click for MIDI)'}>{isMidi ? <Piano size={9} /> : <Volume2 size={9} />}</button>
  1408→        <button onClick={() => update({ muted: !track.muted })} className={`flex h-4 w-4 items-center justify-center rounded ${track.muted && !hasSolo ? 'bg-warn/30 text-warn' : 'text-text-mid hover:bg-bg-3'}`} title="M">{track.muted ? <VolumeX size={9} /> : <Volume2 size={9} />}</button>
  1409→        <button onClick={() => update({ solo: !track.solo })} className={`flex h-4 w-4 items-center justify-center rounded ${track.solo ? 'bg-accent/30 text-accent' : 'text-text-mid hover:bg-bg-3'}`} title="S"><Headphones size={9} /></button>
  1410→        <button onClick={toggleArm} className={`flex h-4 w-4 items-center justify-center rounded ${track.armed ? 'bg-danger/30 text-danger' : 'text-text-mid hover:bg-bg-3'}`} title={isMidi ? 'Arm for MIDI take' : 'Arm'}><div className="h-1.5 w-1.5 rounded-full" style={{ backgroundColor: track.armed ? 'currentColor' : 'transparent', border: '1px solid currentColor' }} /></button>
  1411→        <button onClick={() => deleteAudioTrack(slate, track.id)} className="flex h-4 w-4 items-center justify-center rounded text-text-mid hover:bg-bg-3 hover:text-danger" title="Del"><Trash2 size={9} /></button>
  1412→      </div>
  1413→      {/* MIDI tracks: instrument picker row replaces the pan slider row.
  1414→          Audio tracks keep the original volume + pan slider row. */}
  1415→      {isMidi ? (
  1416→        <div className="mt-0.5 flex items-center gap-1">
  1417→          <Piano size={9} className="shrink-0 text-accent" aria-hidden />
  1418→          <select
  1419→            value={track.instrumentId ?? SOUNDFONT_PIANO_ID}
  1420→            onChange={(e) => update({ instrumentId: e.target.value })}
  1421→            className="min-w-0 flex-1 rounded-sm border border-border bg-bg-3 px-1 py-0.5 text-[9px] text-text outline-none focus:border-accent"
  1422→            aria-label="Instrument"
  1423→            title="Which instrument plays this track's MIDI clips"
  1424→          >
  1425→            <option value={SOUNDFONT_PIANO_ID}>Soundfont Piano</option>
  1426→            <optgroup label="Synth presets">
  1427→              {INSTRUMENT_PRESETS.map((p) => <option key={p.id} value={p.id}>{p.name}</option>)}
  1428→            </optgroup>
  1429→            {(() => {
  1430→              const customs = loadCustomInstruments();
  1431→              return customs.length > 0 ? (
  1432→                <optgroup label="My instruments">
  1433→                  {customs.map((p) => <option key={p.id} value={p.id}>{p.name}</option>)}
  1434→                </optgroup>
  1435→              ) : null;
  1436→            })()}
  1437→          </select>
  1438→        </div>
  1439→      ) : (
  1440→        <div className="mt-0.5 flex items-center gap-1">
  1441→          <Volume2 size={9} className="shrink-0 text-text-dim" aria-hidden />
  1442→          <input type="range" min={0} max={1} step={0.01} value={vol} aria-label="Volume" title="Volume" onPointerDown={onVolDown} onChange={(e) => onVol(Number(e.target.value))} onPointerUp={onVolEnd} className="h-1 min-w-0 flex-1 accent-accent" />
  1443→          <span className="shrink-0 text-[8px] font-medium leading-none text-text-dim" aria-hidden>L</span>

  1445→          <span className="shrink-0 text-[8px] font-medium leading-none text-text-dim" aria-hidden>R</span>

  1454→const ClipBlock = memo(function ClipBlock({ clip, pxPerSec, selected, onDragStart, onTrimStart }: {
  1455→  clip: AudioClip; pxPerSec: number; selected: boolean;
  1456→  onDragStart: (clip: AudioClip, e: React.PointerEvent, el: HTMLElement, waveEl: HTMLElement | null) => void;

  1458→}) {
  1459→  const left = clip.start * pxPerSec;
  1460→  const width = Math.max(4, clip.duration * pxPerSec);
  1461→  const elRef = useRef<HTMLDivElement | null>(null);
  1462→  const waveRef = useRef<HTMLDivElement | null>(null);
  1463→  const w = Math.max(2, Math.floor(width));

  1465→  // spill past the opposite edge).
  1466→  const fadeInW = clip.fadeIn > 0 ? Math.min(width, clip.fadeIn * pxPerSec) : 0;
  1467→  const fadeOutW = clip.fadeOut > 0 ? Math.min(width, clip.fadeOut * pxPerSec) : 0;
  1468→
  1469→  // onDragStart handles selection (incl. Shift/Cmd additive) + the
  1470→  // clip-select event — selecting here too would clobber multi-select.
  1471→  return (
  1472→    <div ref={elRef} onPointerDown={(e) => { if (elRef.current) onDragStart(clip, e, elRef.current, waveRef.current); }}
  1473→      className={`group pointer-events-auto absolute top-0.5 bottom-0.5 cursor-grab overflow-hidden rounded border ${selected ? 'border-warn' : 'border-black/30'} active:cursor-grabbing`} style={{ left, width, backgroundColor: `${clip.color}20` }}>
  1474→      {/* Fixed-width waveform layer — clipped by the box so trimming cuts the
  1475→          audio rather than squashing the whole wave into a smaller space. */}
  1476→      <div ref={waveRef} className="pointer-events-none absolute top-0 bottom-0 left-0" style={{ width }}>
  1477→        {clip.sampleKey && <WaveformImg clipId={clip.id} sampleKey={clip.sampleKey} channels={clip.channels} sampleRate={clip.sampleRate} offset={clip.offset} duration={clip.duration} speed={clip.speed ?? 1} width={w} color={clip.color} />}
  1478→      </div>
  1479→      {/* Fade-in overlay: triangle from full height at the outer (left) edge
  1480→          tapering to 0 at the inner edge — the dark wedge represents the
  1481→          portion of the clip still under the fade. */}
