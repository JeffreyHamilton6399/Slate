    25→const TRACK_H = 60;
    26→
    27→// ── Waveform cache: pre-computed PNG data URLs ───────────────────────────────
    28→// Key: `${clipId}:${sampleCount}:${width}` → data URL
    29→const waveformPNGCache = new Map<string, string>();
    30→
    31→// Draw only the window [startFrame, endFrame) of the sample across `width`
    32→// pixels. Rendering the *window* (not the whole sample stretched) is what makes
    33→// trimming look like cutting the audio away rather than squashing it.
    34→function computeWaveformPNG(samples: Float32Array, channels: number, width: number, color: string, height: number, startFrame: number, endFrame: number): string {
    35→  const canvas = document.createElement('canvas');
    36→  const dpr = window.devicePixelRatio || 1;
    37→  const w = Math.max(2, Math.floor(width));
    38→  canvas.width = w * dpr;
    39→  canvas.height = height * dpr;
    40→  const ctx = canvas.getContext('2d')!;
    41→  ctx.scale(dpr, dpr);
    42→  ctx.clearRect(0, 0, w, height);
    43→  const totalFrames = samples.length / channels;
    44→  const s = Math.max(0, Math.min(totalFrames, Math.floor(startFrame)));
    45→  const e = Math.max(s, Math.min(totalFrames, Math.floor(endFrame)));
    46→  const span = e - s;
    47→  if (span <= 0) return canvas.toDataURL();
    48→  const mid = height / 2;
    49→  ctx.fillStyle = color;
    50→  for (let x = 0; x < w; x++) {
    51→    const f0 = s + Math.floor((x / w) * span);
    52→    const f1 = Math.min(e, s + Math.floor(((x + 1) / w) * span) + 1);
    53→    let peak = 0;
    54→    for (let i = f0; i < f1; i++) { const v = Math.abs(samples[i * channels] ?? 0); if (v > peak) peak = v; }
    55→    const bh = Math.max(1, peak * mid * 0.85);
    56→    ctx.fillRect(x, mid - bh, 1, bh * 2);
    57→  }
    58→  return canvas.toDataURL();
    59→}

    61→/** Waveform image for the buffer window the clip actually plays. With speed s,
    62→ *  a clip of `duration` timeline seconds consumes `duration*s` buffer seconds. */
    63→const WaveformImg = memo(function WaveformImg({ clipId, sampleKey, channels, sampleRate, offset, duration, speed, width, color }: {
    64→  clipId: string; sampleKey: string; channels: number; sampleRate: number;
    65→  offset: number; duration: number; speed: number; width: number; color: string;
    66→}) {
    67→  const height = TRACK_H - 6;
    68→  const [imgUrl, setImgUrl] = useState<string>('');
    69→
    70→  useEffect(() => {
    71→    const startFrame = Math.round(offset * sampleRate);
    72→    const endFrame = Math.round((offset + duration * speed) * sampleRate);
    73→    const cacheKey = `${clipId}:${sampleKey}:${startFrame}:${endFrame}:${Math.floor(width)}`;
    74→    const cached = waveformPNGCache.get(cacheKey);
    75→    if (cached) { setImgUrl(cached); return; }
    76→    let cancelled = false;
    77→    void loadSamples(sampleKey).then((samples) => {
    78→      if (cancelled) return;
    79→      const url = computeWaveformPNG(samples, channels, width, color, height, startFrame, endFrame);
    80→      waveformPNGCache.set(cacheKey, url);
    81→      setImgUrl(url);
    82→    });
    83→    return () => { cancelled = true; };
    84→  }, [clipId, sampleKey, channels, sampleRate, offset, duration, speed, width, color, height]);
    85→
    86→  if (!imgUrl) return <div className="flex h-full items-center justify-center text-[7px] text-text-dim">···</div>;
    87→  return <img src={imgUrl} alt="" className="pointer-events-none h-full w-full" style={{ objectFit: 'fill' }} />;
    88→});
    89→
    90→/** Drop cached waveform PNGs for a clip (call when its samples change). */
    91→function invalidateWaveform(clipId: string): void {
    92→  for (const key of waveformPNGCache.keys()) if (key.startsWith(`${clipId}:`)) waveformPNGCache.delete(key);
    93→}

    97→export function AudioEditor() {
    98→  const room = useRoom();
    99→  const slate = room.slate;
   100→  const [version, setVersion] = useState(0);
   101→  const [playing, setPlaying] = useState(false);
   102→  const [bpm, setBpmState] = useState(slate.audioBpm());
   103→  const [metronome, setMetronome] = useState(false);
   104→  const [recording, setRecording] = useState(false);
   105→  const [selectedClipId, setSelectedClipId] = useState<string | null>(null);
   106→  const [masterVol, setMasterVol] = useState(0.85);
   107→  const [looping, setLooping] = useState(false);
   108→  const [loopStart, setLoopStart] = useState(0);
   109→  const [loopEnd, setLoopEnd] = useState(8);
   110→  const [pxPerSec, setPxPerSec] = useState(80);
   111→  const engineRef = useRef<AudioEngine | null>(null);
   112→  const stopRecRef = useRef<(() => Promise<{ samples: number[]; sampleRate: number; channels: number; duration: number }>) | null>(null);
   113→  const rafRef = useRef(0);
   114→  const positionRef = useRef(0);
   115→  const playheadRef = useRef<HTMLDivElement | null>(null);
   116→  const posDisplayRef = useRef<HTMLSpanElement | null>(null);
   117→  const scrollRef = useRef<HTMLDivElement | null>(null);
   118→  const dragRef = useRef<{
   119→    clipId: string; el: HTMLElement; waveEl: HTMLElement | null;
   120→    os: number; od: number; oo: number; speed: number; sx: number;
   121→    leftLimit: number; rightLimit: number; mode: 'drag' | 'trimL' | 'trimR';
   122→  } | null>(null);
   123→  const selectedRef = useRef<string | null>(null);
   124→  selectedRef.current = selectedClipId;
   125→  const clipsRef = useRef<AudioClip[]>([]);
   126→  const pxRef = useRef(pxPerSec);
   127→  pxRef.current = pxPerSec;
   128→
   129→  // Yjs subscription.
   130→  useEffect(() => {
   131→    const tracks = slate.audioTracks();
   132→    const clips = slate.audioClips();
   133→    const audioMap = slate.doc.getMap('audio');
   134→    let pending = false;
   135→    const bump = () => { if (pending) return; pending = true; requestAnimationFrame(() => { pending = false; setVersion((v) => v + 1); }); };
   136→    tracks.observeDeep(bump); clips.observeDeep(bump); audioMap.observe(bump); bump();
   137→    const lateRead = setTimeout(bump, 200);
   138→    return () => { clearTimeout(lateRead); tracks.unobserveDeep(bump); clips.unobserveDeep(bump); audioMap.unobserve(bump); };
   139→  }, [slate]);
   140→
   141→  useEffect(() => {
   142→    engineRef.current = new AudioEngine();
   143→    engineRef.current.setMasterVolume(masterVol);
   144→    return () => { engineRef.current?.dispose(); engineRef.current = null; };
   145→    // eslint-disable-next-line react-hooks/exhaustive-deps
   146→  }, []);
   147→
   148→  // Playhead — direct DOM.
   149→  useEffect(() => {
   150→    if (!playing) return;
   151→    const tick = () => {
   152→      rafRef.current = requestAnimationFrame(tick);
   153→      const eng = engineRef.current; if (!eng) return;
   154→      const pos = eng.getPosition();
   155→      positionRef.current = pos;
   156→      if (playheadRef.current) playheadRef.current.style.transform = `translateX(${pos * pxRef.current}px)`;
   157→      if (posDisplayRef.current) posDisplayRef.current.textContent = `${pos.toFixed(1)}s`;
   158→      if (pos > timelineDuration + 2) { eng.stop(); setPlaying(false); }
   159→    };
   160→    rafRef.current = requestAnimationFrame(tick);
   161→    return () => cancelAnimationFrame(rafRef.current);
   162→    // eslint-disable-next-line react-hooks/exhaustive-deps
   163→  }, [playing]);
   164→
   165→  // Ctrl+scroll zoom.
   166→  useEffect(() => {
   167→    const el = scrollRef.current; if (!el) return;
   168→    const onWheel = (e: WheelEvent) => {
   169→      if (!e.ctrlKey && !e.metaKey) return;
   170→      e.preventDefault();
   171→      setPxPerSec((c) => Math.max(10, Math.min(800, c * (e.deltaY < 0 ? 1.2 : 1 / 1.2))));
   172→    };
   173→    el.addEventListener('wheel', onWheel, { passive: false });
   174→    return () => el.removeEventListener('wheel', onWheel);
   175→  }, []);
   176→
   177→  // Global pointermove/up — pure DOM, zero React state.
   178→  useEffect(() => {
   179→    const onMove = (ev: PointerEvent) => {
   180→      const d = dragRef.current; if (!d) return;
   181→      const pps = pxRef.current;
   182→      const dt = (ev.clientX - d.sx) / pps;
   183→      const oldEnd = d.os + d.od;
   184→      if (d.mode === 'drag') {
   185→        // Keep the whole clip inside the free gap between its neighbours.
   186→        const maxStart = Math.max(d.leftLimit, d.rightLimit - d.od);
   187→        const start = Math.min(maxStart, Math.max(d.leftLimit, d.os + dt));
   188→        d.el.style.left = `${start * pps}px`;
   189→      } else if (d.mode === 'trimL') {
   190→        // Cut from the left: never past the left neighbour, the source start
   191→        // (offset ≥ 0 → limited by the trimmed head in timeline seconds), or a
   192→        // minimum width.
   193→        const minStart = Math.max(d.leftLimit, d.os - d.oo / d.speed);
   194→        const start = Math.min(oldEnd - 0.1, Math.max(minStart, d.os + dt));
   195→        d.el.style.left = `${start * pps}px`;
   196→        d.el.style.width = `${(oldEnd - start) * pps}px`;
   197→        // Shift the (fixed-width) waveform the opposite way so the audio stays
   198→        // anchored — the newly-trimmed head is clipped, not squashed.
   199→        if (d.waveEl) d.waveEl.style.left = `${-(start - d.os) * pps}px`;
   200→      } else if (d.mode === 'trimR') {
   201→        // Cut from the right: never past the right neighbour or a min width.
   202→        const end = Math.min(d.rightLimit, Math.max(d.os + 0.1, oldEnd + dt));
   203→        d.el.style.width = `${(end - d.os) * pps}px`;
   204→      }
   205→    };
   206→    const onUp = () => {
   207→      const d = dragRef.current; if (!d) return;
   208→      const pps = pxRef.current;
   209→      const left = parseFloat(d.el.style.left) / pps;
   210→      const width = parseFloat(d.el.style.width) / pps;
   211→      if (d.mode === 'drag') updateAudioClip(slate, d.clipId, { start: Math.max(0, left) });
   212→      else if (d.mode === 'trimL') updateAudioClip(slate, d.clipId, { start: left, duration: width, offset: Math.max(0, d.oo + (left - d.os) * d.speed) });
   213→      else if (d.mode === 'trimR') updateAudioClip(slate, d.clipId, { duration: width });
   214→      dragRef.current = null;
   215→    };
   216→    window.addEventListener('pointermove', onMove);
   217→    window.addEventListener('pointerup', onUp);
   218→    return () => { window.removeEventListener('pointermove', onMove); window.removeEventListener('pointerup', onUp); };
   219→  }, [slate]);
   220→
   221→  // Hotkeys.
   222→  useEffect(() => {
   223→    const onKey = (e: KeyboardEvent) => {
   224→      const t = e.target as HTMLElement;
   225→      if (t && /^(INPUT|TEXTAREA|SELECT)$/.test(t.tagName)) return;
   226→      const k = e.key.toLowerCase();
   227→      if (k === ' ') { e.preventDefault(); togglePlay(); }
   228→      else if (k === 'c' && !e.ctrlKey && selectedRef.current) { e.preventDefault(); void splitAudioClip(slate, selectedRef.current, positionRef.current); }
   229→      else if ((k === 'delete' || k === 'backspace') && selectedRef.current) { e.preventDefault(); deleteAudioClip(slate, selectedRef.current); setSelectedClipId(null); }
   230→      else if (k === 'd' && !e.ctrlKey && selectedRef.current) { e.preventDefault(); dupClip(selectedRef.current); }
   231→      else if (k === 'l') { e.preventDefault(); setLooping((n) => !n); }
   232→      else if (k === 'm') { e.preventDefault(); setMetronome((n) => { engineRef.current?.setMetronome(!n); return !n; }); }
   233→      else if (k === 'r' && !e.ctrlKey) { e.preventDefault(); void toggleRecord(); }
   234→      else if (k === 'arrowleft') { e.preventDefault(); seek(positionRef.current - 2); }
   235→      else if (k === 'arrowright') { e.preventDefault(); seek(positionRef.current + 2); }
   236→      else if (k === 'home') { e.preventDefault(); seek(0); }
   237→    };
   238→    window.addEventListener('keydown', onKey);
   239→    return () => window.removeEventListener('keydown', onKey);
   240→    // eslint-disable-next-line react-hooks/exhaustive-deps
   241→  }, [playing, recording]);
   242→
   243→  // Read from Yjs — lightweight (no samples copied).
   244→  const tracks: AudioTrack[] = useMemo(() => {
   245→    const list: AudioTrack[] = [];
   246→    slate.audioTracks().forEach((m, id) => { const t = readAudioTrack(m, id); if (t) list.push(t); });
   247→    list.sort((a, b) => a.order - b.order);
   248→    return list;
   249→    // eslint-disable-next-line react-hooks/exhaustive-deps
   250→  }, [slate, version]);
   251→
   252→  const clips: AudioClip[] = useMemo(() => {
   253→    const list: AudioClip[] = [];
   254→    slate.audioClips().forEach((m, id) => { const c = readAudioClip(m, id); if (c) list.push(c); });
   255→    return list;
   256→    // eslint-disable-next-line react-hooks/exhaustive-deps
   257→  }, [slate, version]);
   258→  clipsRef.current = clips;
   259→
   260→  // When a clip's samples change (normalize/reverse from the settings panel),
   261→  // drop the cached waveform + decoded buffer so both refresh.
   262→  useEffect(() => {
   263→    const onChanged = (e: Event) => {
   264→      const id = (e as CustomEvent<string>).detail;
   265→      invalidateWaveform(id);
   266→      engineRef.current?.clearCache(id);
   267→      setVersion((v) => v + 1);
   268→    };
   269→    window.addEventListener('slate:audio-clip-changed', onChanged as EventListener);
   270→    return () => window.removeEventListener('slate:audio-clip-changed', onChanged as EventListener);
   271→  }, []);
   272→
   273→  const timelineDuration = Math.max(30, ...clips.map((c) => c.start + c.duration), positionRef.current + 10);
   274→
   275→  // ── Transport ─────────────────────────────────────────────────────────────
   276→
   277→  const togglePlay = useCallback(() => {
   278→    const eng = engineRef.current; if (!eng) return;
   279→    if (playing) { eng.stop(); setPlaying(false); }
   280→    else { if (looping) eng.setLoopRegion(loopStart, loopEnd); else eng.setLoopRegion(null, null); void eng.play(slate, positionRef.current); setPlaying(true); }
   281→  }, [playing, slate, looping, loopStart, loopEnd]);
   282→
   283→  const seek = useCallback((t: number) => {
   284→    const pos = Math.max(0, t);
   285→    positionRef.current = pos;
   286→    if (playheadRef.current) playheadRef.current.style.transform = `translateX(${pos * pxRef.current}px)`;
   287→    if (posDisplayRef.current) posDisplayRef.current.textContent = `${pos.toFixed(1)}s`;
   288→  }, []);
   289→
   290→  const toggleRecord = useCallback(async () => {
   291→    if (recording) {
   292→      const stopFn = stopRecRef.current;
   293→      if (stopFn) {
   294→        const r = await stopFn(); stopRecRef.current = null; setRecording(false);
   295→        let tid = tracks.find((t) => t.armed)?.id;
   296→        if (!tid) tid = addAudioTrack(slate, { name: 'Recording' });
   297→        addAudioClip(slate, tid, { start: positionRef.current, samples: r.samples, sampleRate: r.sampleRate, channels: r.channels, duration: r.duration, name: `Rec ${new Date().toLocaleTimeString()}` });
   298→        toast({ title: 'Recording added' });
   299→      }
   300→    } else {
   301→      try { const sf = await engineRef.current?.startRecording(); stopRecRef.current = sf ?? null; setRecording(true); }
   302→      catch { toast({ title: 'Mic denied', variant: 'error' }); }
   303→    }
   304→  }, [recording, tracks, slate]);
   305→
   306→  const dupClip = useCallback(async (id: string) => {
   307→    const yo = slate.audioClips().get(id); if (!yo) return;
   308→    const c = readAudioClip(yo, id); if (!c) return;
   309→    const samples = await loadSamples(c.sampleKey);
   310→    await addAudioClip(slate, c.trackId, { start: c.start + c.duration, samples: float32ToNumberArray(samples), sampleRate: c.sampleRate, channels: c.channels, duration: c.duration, name: `${c.name} copy` });
   311→  }, [slate]);
   312→
   313→  const handleFileImport = useCallback(async (file: File) => {
   314→    try {
   315→      const d = await decodeAudioFile(file);
   316→      const tid = addAudioTrack(slate, { name: file.name.replace(/\.[^.]+$/, '') });
   317→      addAudioClip(slate, tid, { start: positionRef.current, samples: d.samples, sampleRate: d.sampleRate, channels: d.channels, duration: d.duration, name: file.name });
   318→      toast({ title: 'Imported', description: file.name });
   319→    } catch (err) { toast({ title: 'Import failed', description: (err as Error).message, variant: 'error' }); }
   320→  }, [slate]);
   321→
   322→  // ── Drag start ────────────────────────────────────────────────────────────
   323→
   324→  // The free interval [leftLimit, rightLimit] the clip may occupy without
   325→  // overlapping any other clip on the same track.
   326→  const neighbourBounds = useCallback((clip: AudioClip): { leftLimit: number; rightLimit: number } => {
   327→    let leftLimit = 0;
   328→    let rightLimit = Infinity;
   329→    const clipEnd = clip.start + clip.duration;
   330→    for (const o of clipsRef.current) {
   331→      if (o.id === clip.id || o.trackId !== clip.trackId) continue;
   332→      const oEnd = o.start + o.duration;
   333→      if (oEnd <= clip.start + 1e-4) leftLimit = Math.max(leftLimit, oEnd);
   334→      else if (o.start >= clipEnd - 1e-4) rightLimit = Math.min(rightLimit, o.start);
   335→    }
   336→    return { leftLimit, rightLimit };
   337→  }, []);
   338→
   339→  const startDrag = useCallback((clip: AudioClip, e: React.PointerEvent, el: HTMLElement, waveEl: HTMLElement | null) => {
   340→    e.stopPropagation(); setSelectedClipId(clip.id);
   341→    const { leftLimit, rightLimit } = neighbourBounds(clip);
   342→    dragRef.current = { clipId: clip.id, el, waveEl, os: clip.start, od: clip.duration, oo: clip.offset, speed: clip.speed ?? 1, sx: e.clientX, leftLimit, rightLimit, mode: 'drag' };
   343→  }, [neighbourBounds]);
   344→
   345→  const startTrim = useCallback((clip: AudioClip, side: 'left' | 'right', e: React.PointerEvent, el: HTMLElement, waveEl: HTMLElement | null) => {
   346→    e.stopPropagation(); setSelectedClipId(clip.id);
   347→    const { leftLimit, rightLimit } = neighbourBounds(clip);
   348→    dragRef.current = { clipId: clip.id, el, waveEl, os: clip.start, od: clip.duration, oo: clip.offset, speed: clip.speed ?? 1, sx: e.clientX, leftLimit, rightLimit, mode: side === 'left' ? 'trimL' : 'trimR' };
   349→  }, [neighbourBounds]);
   350→
   351→  // ── Loop region drag ──────────────────────────────────────────────────────
   352→
   353→  const loopDragRef = useRef<{ mode: 'start' | 'end' | 'move'; sx: number; os: number; oe: number } | null>(null);
   354→  const startLoopDrag = useCallback((mode: 'start' | 'end' | 'move', e: React.PointerEvent) => {
   355→    e.stopPropagation();
   356→    loopDragRef.current = { mode, sx: e.clientX, os: loopStart, oe: loopEnd };
   357→    const onMove = (ev: PointerEvent) => {
   358→      const d = loopDragRef.current; if (!d) return;
   359→      const dt = (ev.clientX - d.sx) / pxRef.current;
   360→      if (d.mode === 'start') { setLoopStart(Math.max(0, Math.min(d.oe - 0.5, d.os + dt))); }
   361→      else if (d.mode === 'end') { setLoopEnd(Math.max(d.os + 0.5, d.oe + dt)); }
   362→      else if (d.mode === 'move') { setLoopStart(Math.max(0, d.os + dt)); setLoopEnd(Math.max(0.5, d.oe + dt)); }
   363→    };
   364→    const onUp = () => { loopDragRef.current = null; window.removeEventListener('pointermove', onMove); window.removeEventListener('pointerup', onUp); };
   365→    window.addEventListener('pointermove', onMove); window.addEventListener('pointerup', onUp);
   366→  }, [loopStart, loopEnd, pxPerSec]);
   367→
   368→  // ── Render ────────────────────────────────────────────────────────────────
   369→
   370→  const beatDur = 60 / bpm; // seconds per beat
   371→  const gridStyle = {
   372→    backgroundImage: `repeating-linear-gradient(to right, rgba(128,128,128,0.08) 0 1px, transparent 1px ${beatDur * pxPerSec}px), repeating-linear-gradient(to right, rgba(128,128,128,0.2) 0 1px, transparent 1px ${beatDur * 4 * pxPerSec}px)`,
   373→  };
   374→
   375→  return (
   376→    <div className="flex h-full flex-col bg-bg overflow-hidden" onDragOver={(e) => { if (e.dataTransfer?.types?.includes('Files')) e.preventDefault(); }} onDrop={(e) => { e.preventDefault(); for (const f of [...(e.dataTransfer?.files ?? [])].filter((f) => /\.(mp3|wav|ogg|m4a|flac|aac)$/i.test(f.name))) void handleFileImport(f); }}>
   377→      {/* Transport */}
   378→      <div className="flex shrink-0 items-center gap-1 border-b border-border bg-bg-2 px-2 py-1.5">
   379→        <button onClick={() => seek(0)} className="flex h-7 w-7 items-center justify-center rounded text-text-mid hover:bg-bg-3" title="Start"><SkipBack size={14} /></button>
   380→        <button onClick={togglePlay} className={`flex h-9 w-9 items-center justify-center rounded-full text-white ${playing ? 'bg-warn' : 'bg-accent'} hover:opacity-80`} title="Play (Space)">{playing ? <Pause size={18} /> : <Play size={18} />}</button>
   381→        <button onClick={() => void toggleRecord()} className={`flex h-8 w-8 items-center justify-center rounded-full border ${recording ? 'border-danger bg-danger/20 text-danger animate-pulse' : 'border-border text-text-mid hover:bg-bg-3'}`} title="Record (R)"><Mic size={15} /></button>

   458→const TrackHeader = memo(function TrackHeader({ track, hasSolo, slate, engineRef }: {
   459→  track: AudioTrack; hasSolo: boolean;
   460→  slate: ReturnType<typeof useRoom>['slate'];
   461→  engineRef: React.RefObject<AudioEngine | null>;
   462→}) {
   463→  const [vol, setVol] = useState(track.volume);
   464→  const [pan, setPan] = useState(track.pan);
   465→  useEffect(() => { setVol(track.volume); }, [track.volume]);
   466→  useEffect(() => { setPan(track.pan); }, [track.pan]);
   467→
   468→  const update = (patch: Partial<AudioTrack>) => { updateAudioTrack(slate, track.id, patch); engineRef.current?.updateTracks(slate); };
   469→  const onVol = (v: number) => { setVol(v); engineRef.current?.updateTracks(slate); };
   470→  const onVolEnd = () => updateAudioTrack(slate, track.id, { volume: vol });
   471→  const onPan = (p: number) => { setPan(p); engineRef.current?.updateTracks(slate); };
   472→  const onPanEnd = () => updateAudioTrack(slate, track.id, { pan: pan });
   473→
   474→  return (
   475→    <div className="border-b border-border/15 px-2 py-1" style={{ height: TRACK_H }}>
   476→      <div className="flex items-center gap-0.5">
   477→        <span className="h-2 w-2 shrink-0 rounded-full" style={{ backgroundColor: track.color }} />
   478→        <input type="text" value={track.name} onChange={(e) => update({ name: e.target.value })} className="min-w-0 flex-1 bg-transparent text-[11px] font-medium text-text outline-none" />
   479→        <button onClick={() => update({ muted: !track.muted })} className={`flex h-4 w-4 items-center justify-center rounded ${track.muted && !hasSolo ? 'bg-warn/30 text-warn' : 'text-text-mid hover:bg-bg-3'}`} title="M">{track.muted ? <VolumeX size={9} /> : <Volume2 size={9} />}</button>
   480→        <button onClick={() => update({ solo: !track.solo })} className={`flex h-4 w-4 items-center justify-center rounded ${track.solo ? 'bg-accent/30 text-accent' : 'text-text-mid hover:bg-bg-3'}`} title="S"><Headphones size={9} /></button>
   481→        <button onClick={() => update({ armed: !track.armed, input: !track.armed ? 'mic' : 'none' })} className={`flex h-4 w-4 items-center justify-center rounded ${track.armed ? 'bg-danger/30 text-danger' : 'text-text-mid hover:bg-bg-3'}`} title="Arm"><div className="h-1.5 w-1.5 rounded-full" style={{ backgroundColor: track.armed ? 'currentColor' : 'transparent', border: '1px solid currentColor' }} /></button>
   482→        <button onClick={() => deleteAudioTrack(slate, track.id)} className="flex h-4 w-4 items-center justify-center rounded text-text-mid hover:bg-bg-3 hover:text-danger" title="Del"><Trash2 size={9} /></button>
   483→      </div>
   484→      <div className="mt-0.5 flex items-center gap-1">
   485→        <Volume2 size={8} className="shrink-0 text-text-dim" />
   486→        <input type="range" min={0} max={1} step={0.01} value={vol} onChange={(e) => onVol(Number(e.target.value))} onPointerUp={onVolEnd} className="h-0.5 flex-1 accent-accent" />
   487→        <input type="range" min={-1} max={1} step={0.01} value={pan} onChange={(e) => onPan(Number(e.target.value))} onPointerUp={onPanEnd} className="h-0.5 w-8 accent-accent" />
   488→      </div>
   489→    </div>
   490→  );
   491→});

   495→const ClipBlock = memo(function ClipBlock({ clip, pxPerSec, selected, onSelect, onDragStart, onTrimStart }: {
   496→  clip: AudioClip; pxPerSec: number; selected: boolean;
   497→  onSelect: (id: string) => void;
   498→  onDragStart: (clip: AudioClip, e: React.PointerEvent, el: HTMLElement, waveEl: HTMLElement | null) => void;
   499→  onTrimStart: (clip: AudioClip, side: 'left' | 'right', e: React.PointerEvent, el: HTMLElement, waveEl: HTMLElement | null) => void;
   500→}) {
   501→  const left = clip.start * pxPerSec;
   502→  const width = Math.max(4, clip.duration * pxPerSec);
   503→  const elRef = useRef<HTMLDivElement | null>(null);
   504→  const waveRef = useRef<HTMLDivElement | null>(null);
   505→  const w = Math.max(2, Math.floor(width));
   506→
   507→  return (
   508→    <div ref={elRef} onPointerDown={(e) => { if (elRef.current) { e.stopPropagation(); onSelect(clip.id); window.dispatchEvent(new CustomEvent('slate:audio-clip-select', { detail: clip.id })); onDragStart(clip, e, elRef.current, waveRef.current); } }}
   509→      className={`group pointer-events-auto absolute top-0.5 bottom-0.5 cursor-grab overflow-hidden rounded border ${selected ? 'border-warn' : 'border-black/30'} active:cursor-grabbing`} style={{ left, width, backgroundColor: `${clip.color}20` }}>
   510→      {/* Fixed-width waveform layer — clipped by the box so trimming cuts the
   511→          audio rather than squashing the whole wave into a smaller space. */}
   512→      <div ref={waveRef} className="pointer-events-none absolute top-0 bottom-0 left-0" style={{ width }}>
   513→        {clip.sampleKey && <WaveformImg clipId={clip.id} sampleKey={clip.sampleKey} channels={clip.channels} sampleRate={clip.sampleRate} offset={clip.offset} duration={clip.duration} speed={clip.speed ?? 1} width={w} color={clip.color} />}
   514→      </div>
   515→      <span className="absolute left-1 top-0 truncate text-[7px] font-medium text-text-mid/70 pointer-events-none">{clip.name}{clip.mute ? ' (muted)' : ''}</span>
   516→      <div onPointerDown={(e) => { if (elRef.current) onTrimStart(clip, 'left', e, elRef.current, waveRef.current); }} className="absolute left-0 top-0 bottom-0 w-1.5 cursor-ew-resize bg-white/40 opacity-0 group-hover:opacity-100" />
   517→      <div onPointerDown={(e) => { if (elRef.current) onTrimStart(clip, 'right', e, elRef.current, waveRef.current); }} className="absolute right-0 top-0 bottom-0 w-1.5 cursor-ew-resize bg-white/40 opacity-0 group-hover:opacity-100" />
   518→    </div>
   519→  );
   520→});
