     1→     1→/**
     2→     2→ * DocEditor — collaborative rich-text document surface for 'doc' boards.
     3→     3→ *
     4→     4→ * TipTap (ProseMirror) bound to the board's shared Y.XmlFragment via the
     5→     5→ * Collaboration extension (y-prosemirror underneath), so every keystroke
     6→     6→ * merges through the same Yjs pipeline as the other modes — offline edits
     7→     7→ * included. CollaborationCursor broadcasts carets/selections over the room's
     8→     8→ * existing Hocuspocus awareness, colored with the same per-peer palette as
     9→     9→ * the People widget.
    10→    10→ *
    11→    11→ * Undo/redo is the Collaboration extension's Yjs undo manager (scoped to
    12→    12→ * YOUR edits — undoing won't revert a collaborator's typing), deliberately
    13→    13→ * separate from the room-level UndoManager which tracks the other modes'
    14→    14→ * containers.
    15→    15→ *
    16→    16→ * Node set: headings 1-3, bold/italic/strike/underline/inline code, links,
    17→    17→ * text color + highlight, bullet/ordered/task lists, blockquote, syntax-
    18→    18→ * highlighted code blocks (lowlight/common grammars), images (downscaled to
    19→    19→ * a data URL small enough for a single Yjs update — same importer the 2D
    20→    20→ * canvas uses), horizontal rule, alignment (left/center/right), tables
    21→    21→ * (insert/add-row/add-col/delete). Export to Markdown from the toolbar
    22→    22→ * (and the Export dialog). Find via prompt + ProseMirror text search.
    23→    23→ */

   155→   155→
   156→   156→  // The AI assistant edits this document by dispatching a window event with the

   302→   302→  const findInDoc = () => {
   303→   303→    if (!editor) return;

   305→   305→    const term = window.prompt('Find in document');
   306→   306→    if (term === null || term === '') return;
   307→   307→    const needle = term.toLowerCase();
   308→   308→    let found = false;
   309→   309→    editor.state.doc.descendants((node, pos) => {
   310→   310→      if (found) return false; // stop walking once we've landed on a match
   311→   311→      // Non-text nodes (paragraphs, list items, table cells, …) — descend
   312→   312→      // into their children so the search reaches text nested anywhere.
   313→   313→      if (!node.isText || !node.text) return;

   315→   315→      if (idx === -1) return;

   320→   320→      // asks the view to bring the new selection into view.

   322→   322→        .setSelection(TextSelection.create(editor.state.doc, from, to))
   323→   323→        .scrollIntoView();
   324→   324→      editor.view.dispatch(tr);
   325→   325→      editor.view.focus();
   326→   326→      found = true;
   327→   327→      return false;
   328→   328→    });

   364→   364→            className={`grid h-7 w-7 place-items-center rounded transition-colors ${
   365→   365→              colorOpen ? 'bg-accent/15 text-accent' : 'text-text-mid hover:bg-bg-3 hover:text-text'
   366→   366→            }`}
   367→   367→          >
