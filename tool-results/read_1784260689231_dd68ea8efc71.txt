    46→// ── Waveform cache: pre-computed PNG data URLs ───────────────────────────────
    47→// Key: `${clipId}:${sampleCount}:${width}` → data URL
    48→const waveformPNGCache = new Map<string, string>();
    49→
    50→// Draw only the window [startFrame, endFrame) of the sample across `width`
    51→// pixels. Rendering the *window* (not the whole sample stretched) is what makes
    52→// trimming look like cutting the audio away rather than squashing it.
    53→function computeWaveformPNG(samples: Float32Array, channels: number, width: number, color: string, height: number, startFrame: number, endFrame: number): string {
    54→  const canvas = document.createElement('canvas');
    55→  const dpr = window.devicePixelRatio || 1;
    56→  const w = Math.max(2, Math.floor(width));
    57→  canvas.width = w * dpr;
    58→  canvas.height = height * dpr;
    59→  const ctx = canvas.getContext('2d')!;
    60→  ctx.scale(dpr, dpr);
    61→  ctx.clearRect(0, 0, w, height);
    62→  const totalFrames = samples.length / channels;
    63→  const s = Math.max(0, Math.min(totalFrames, Math.floor(startFrame)));
    64→  const e = Math.max(s, Math.min(totalFrames, Math.floor(endFrame)));
    65→  const span = e - s;
    66→  if (span <= 0) return canvas.toDataURL();
    67→  const mid = height / 2;
    68→  ctx.fillStyle = color;
    69→  for (let x = 0; x < w; x++) {
    70→    const f0 = s + Math.floor((x / w) * span);
    71→    const f1 = Math.min(e, s + Math.floor(((x + 1) / w) * span) + 1);
    72→    let peak = 0;
    73→    for (let i = f0; i < f1; i++) { const v = Math.abs(samples[i * channels] ?? 0); if (v > peak) peak = v; }
    74→    const bh = Math.max(1, peak * mid * 0.85);
    75→    ctx.fillRect(x, mid - bh, 1, bh * 2);
    76→  }
    77→  return canvas.toDataURL();
    78→}

    80→/** Waveform image for the buffer window the clip actually plays. With speed s,
    81→ *  a clip of `duration` timeline seconds consumes `duration*s` buffer seconds.
    82→ *
    83→ *  Two robustness fixes:
    84→ *  1. If `loadSamples` returns an EMPTY Float32Array (length 0) — which happens
    85→ *     when a freshly-split/created clip's IndexedDB write hasn't landed yet —
    86→ *     we DON'T cache the resulting blank PNG. Instead we show the `···`
    87→ *     placeholder and retry a few times (500ms apart) so the real waveform
    88→ *     appears once the samples are available.
    89→ *  2. We listen for `slate:audio-clip-changed` (fired by Normalize/Reverse/split)
    90→ *     and invalidate our cache entry for the current clipId, then force a
    91→ *     recompute via a `bust` counter — otherwise the memoised component would
    92→ *     keep showing the stale PNG even after the cache was cleared. */
    93→const WaveformImg = memo(function WaveformImg({ clipId, sampleKey, channels, sampleRate, offset, duration, speed, width, color }: {
    94→  clipId: string; sampleKey: string; channels: number; sampleRate: number;
    95→  offset: number; duration: number; speed: number; width: number; color: string;
    96→}) {
    97→  const height = TRACK_H - 6;
    98→  const [imgUrl, setImgUrl] = useState<string>('');
    99→  const [bust, setBust] = useState(0);
   100→  const retryRef = useRef(0);
   101→
   102→  // Invalidate cached PNGs for this clip when its samples change (normalize /
   103→  // reverse / split). The parent AudioEditor also invalidates + bumps version,
   104→  // but the memoised WaveformImg wouldn't otherwise recompute (its primitive
   105→  // props are unchanged) — the `bust` counter forces the load effect to re-run.
   106→  useEffect(() => {
   107→    const onChanged = (e: Event) => {
   108→      const detail = (e as CustomEvent<string>).detail;
   109→      // Match by clipId (normalize/reverse/split events) OR by sampleKey
   110→      // (multiplayer sample-arrival events from registerSampleSyncMap).
   111→      if (detail !== clipId && detail !== sampleKey) return;
   112→      for (const key of waveformPNGCache.keys()) {
   113→        if (key.startsWith(`${clipId}:`)) waveformPNGCache.delete(key);
   114→      }
   115→      retryRef.current = 0;
   116→      setBust((n) => n + 1);
   117→    };
   118→    window.addEventListener('slate:audio-clip-changed', onChanged as EventListener);
   119→    return () => window.removeEventListener('slate:audio-clip-changed', onChanged as EventListener);
   120→  }, [clipId, sampleKey]);
   121→
   122→  useEffect(() => {
   123→    const startFrame = Math.round(offset * sampleRate);
   124→    const endFrame = Math.round((offset + duration * speed) * sampleRate);
   125→    const cacheKey = `${clipId}:${sampleKey}:${startFrame}:${endFrame}:${Math.floor(width)}`;
   126→    const cached = waveformPNGCache.get(cacheKey);
   127→    if (cached) { setImgUrl(cached); return; }
   128→    let cancelled = false;
   129→    let retryTimer: ReturnType<typeof setTimeout> | null = null;
   130→
   131→    const attempt = () => {
   132→      if (cancelled) return;
   133→      void loadSamples(sampleKey).then((samples) => {
   134→        if (cancelled) return;
   135→        // Empty samples = the IndexedDB write for this sampleKey hasn't landed
   136→        // yet (e.g. brand-new clip from a split, OR a remote peer's clip whose
   137→        // sample blob is still in flight via the Yjs sync map). DON'T cache a
   138→        // blank PNG — retry for up to 10s (20 × 500ms) so the real waveform
   139→        // appears once the samples arrive. The previous 5-retry / 2.5s budget
   140→        // was too short for cross-peer sample sync, which can take several
   141→        // seconds for a 5MB blob over a slow link. The `slate:audio-clip-changed`
   142→        // event (dispatched by onClipsAdded when a remote clip is added, and by
   143→        // registerSampleSyncMap when samples arrive) resets `retryRef` to 0 so
   144→        // the WaveformImg re-tries immediately on the signal — this retry
   145→        // budget is the safety net for the case where the event fires before
   146→        // the WaveformImg mounts, or doesn't fire at all.
   147→        if (samples.length === 0) {
   148→          if (retryRef.current < 20) {
   149→            retryRef.current += 1;
   150→            retryTimer = setTimeout(attempt, 500);
   151→          }
   152→          return;
   153→        }
   154→        retryRef.current = 0;
   155→        const url = computeWaveformPNG(samples, channels, width, color, height, startFrame, endFrame);
   156→        waveformPNGCache.set(cacheKey, url);
   157→        setImgUrl(url);
   158→      });
   159→    };
   160→    attempt();
   161→
   162→    return () => {
   163→      cancelled = true;
   164→      if (retryTimer) clearTimeout(retryTimer);
   165→    };
   166→  }, [clipId, sampleKey, channels, sampleRate, offset, duration, speed, width, color, height, bust]);
   167→
   168→  if (!imgUrl) return <div className="flex h-full items-center justify-center text-[7px] text-text-dim">···</div>;
   169→  return <img src={imgUrl} alt="" className="pointer-events-none h-full w-full" style={{ objectFit: 'fill' }} />;
   170→});
   171→
   172→/** Drop cached waveform PNGs for a clip (call when its samples change). */
   173→function invalidateWaveform(clipId: string): void {
   174→  for (const key of waveformPNGCache.keys()) if (key.startsWith(`${clipId}:`)) waveformPNGCache.delete(key);
   175→}

   193→  const engineRef = useRef<AudioEngine | null>(null);
   194→  const stopRecRef = useRef<(() => Promise<{ samples: number[]; sampleRate: number; channels: number; duration: number }>) | null>(null);
   195→  const rafRef = useRef(0);
   196→  const positionRef = useRef(0);
   197→  // Refs mirror the latest `playing` and `slate` so the long-lived Yjs / event
   198→  // listeners (which close over the initial render) can read fresh values
   199→  // without re-subscribing on every state change.
   200→  const playingRef = useRef(false);
   201→  playingRef.current = playing;
   202→  const hoveredRef = audioEditorHovered;
   203→  const slateRef = useRef(slate);
   204→  slateRef.current = slate;
   205→  /** Debounce timer for restartPlayback — coalesces rapid clip/sample changes
   206→   *  so we don't tear down and rebuild the scheduler 5× in a row when a peer
   207→   *  adds several clips at once. */
   208→  const restartTimerRef = useRef<ReturnType<typeof setTimeout> | null>(null);
   209→  const playheadRef = useRef<HTMLDivElement | null>(null);
   210→  const posDisplayRef = useRef<HTMLSpanElement | null>(null);
   211→  const scrollRef = useRef<HTMLDivElement | null>(null);
   212→  const dragRef = useRef<{
   213→    clipId: string; el: HTMLElement; waveEl: HTMLElement | null;
   214→    os: number; od: number; oo: number; speed: number; sx: number;
   215→    leftLimit: number; rightLimit: number; mode: 'drag' | 'trimL' | 'trimR';
   216→    neighbours: { start: number; end: number }[];
   217→  } | null>(null);
   218→  /** Latest pointer `clientX` during an active drag — written by every
   219→   *  `pointermove` and read inside a `requestAnimationFrame` callback. Decouples
   220→   *  the high-frequency pointermove stream (~60-120Hz on modern pointers) from
   221→   *  the actual DOM-mutation work, so multiple moves within the same frame
   222→   *  coalesce into a single `style.left`/`style.width` write. Null when no
   223→   *  drag is in progress. */
   224→  const moveXRef = useRef<number | null>(null);
   225→  /** Pending rAF id for processing the next drag move. Zero when no frame is
   226→   *  scheduled. Lets `onMove` bail out cheaply (one ref read + one rAF check)
   227→   *  when a frame is already queued. */
   228→  const moveRafRef = useRef(0);
   229→  const selectedRef = useRef<string | null>(null);
   230→  selectedRef.current = selectedClipId;
   231→  const clipsRef = useRef<AudioClip[]>([]);
   232→  const pxRef = useRef(pxPerSec);
   233→  pxRef.current = pxPerSec;
   234→  /** Desired scrollLeft to apply after the next pxPerSec commit (used by
   235→   *  zoomAtPlayhead so the playhead stays at the same screen position when
   236→   *  zooming). Cleared in the layout effect below once applied. */
   237→  const pendingScrollRef = useRef<number | null>(null);
   238→  /** Timestamp (performance.now ms) of the last awareness publish of our audio
   239→   *  playhead position. The tick rAF loop throttles publishes to ~7 Hz (150ms)
   240→   *  so a long play session doesn't saturate the awareness broadcast — peers
   241→   *  render at display rate via their own rAF, so 7 Hz is plenty for smooth
   242→   *  remote-playhead motion. */
   243→  const lastAudioPublishRef = useRef(0);
   244→  /** Snapshot of remote peer awareness states — passed to RemotePlayheads.
   245→   *  The diffing setPeerStates below skips updates that only change high-
   246→   *  frequency fields (audio.pos), so we don't re-render AudioEditor at 7 Hz. */
   247→  const [peerStates, setPeerStates] = useState<AwarenessState[]>([]);
   248→
   249→  /** Schedule a debounced restartPlayback (500ms). Called when the clip set
   250→   *  changes mid-playback (remote peer adds a clip) or when a clip's samples
   251→   *  arrive after play started (remote sample blob landed via the sync map).
   252→   *  The debounce coalesces rapid bursts so we don't tear down and rebuild
   253→   *  the scheduler repeatedly. No-op when not playing. Reads only refs, so the
   254→   *  callback identity is stable for the lifetime of the component. */
   255→  const scheduleRestart = useCallback(() => {
   256→    if (!playingRef.current) return;
   257→    if (restartTimerRef.current) clearTimeout(restartTimerRef.current);
   258→    restartTimerRef.current = setTimeout(() => {
   259→      restartTimerRef.current = null;
   260→      const eng = engineRef.current;
   261→      if (eng && playingRef.current) eng.restartPlayback(slateRef.current, positionRef.current);
   262→    }, 500);
   263→  }, []);
   264→
   265→  // Subscribe to awareness changes. Diffing skips updates that only change
   266→  // high-frequency fields (audio.pos) so the parent doesn't re-render at the
   267→  // 7 Hz publish rate — RemotePlayheads subscribes itself for live positions.
   268→  useEffect(
   269→    () =>
   270→      room.onAwarenessChange((states) => {
   271→        setPeerStates((prev) => {
   272→          if (prev.length !== states.length) return states;
   273→          // Stable peer-set key — id + name + color + audio-presence.
   274→          // Position/playing/cursor/cam/voiceLevel changes are intentionally
   275→          // excluded so they don't trigger a re-render.
   276→          const key = (s: AwarenessState) =>
   277→            `${s.id}|${s.name}|${s.color}|${s.audio ? 1 : 0}`;
   278→          const prevKey = prev.map(key).sort().join(';');
   279→          const nextKey = states.map(key).sort().join(';');
   280→          return prevKey === nextKey ? prev : states;
   281→        });
   282→      }),
   283→    [room],
   284→  );
   285→
   286→  // Publish our audio playhead state whenever transport play/pause changes.
   287→  // Reads positionRef.current at effect-run time so the published position is
   288→  // always the latest. Also resets the throttle stamp so the rAF loop doesn't
   289→  // immediately re-publish.
   290→  useEffect(() => {
   291→    room.setLocalAwareness({ audio: { pos: positionRef.current, playing } });
   292→    lastAudioPublishRef.current = performance.now();
   293→  }, [playing, room]);
   294→
   295→  // Clear our audio playhead on unmount so other peers stop seeing us in the
   296→  // audio editor. (audio: null = "not in the audio editor".)
   297→  useEffect(() => {
   298→    return () => {
   299→      room.setLocalAwareness({ audio: null });
   300→    };
   301→  }, [room]);
   302→
   303→  // Yjs subscription.
   304→  useEffect(() => {
   305→    // NOTE: registerSampleSyncMap(room) is now called from useSlateRoom's
   306→    // attach() — registering here only on AudioEditor mount meant a peer on
   307→    // a 2D/3D board (audio panel closed) never registered the sync map and
   308→    // never received remote sample blobs. See useSlateRoom.ts.
   309→    const tracks = slate.audioTracks();
   310→    const clips = slate.audioClips();
   311→    const audioMap = slate.doc.getMap('audio');
   312→    let pending = false;
   313→    const bump = () => { if (pending) return; pending = true; requestAnimationFrame(() => { pending = false; setVersion((v) => v + 1); }); };
   314→    tracks.observeDeep(bump); clips.observeDeep(bump); audioMap.observe(bump); bump();
   315→    // Detect NEW clips being added while playing and schedule a debounced
   316→    // restartPlayback so the new clip is picked up mid-playback without a
   317→    // manual stop/play. We watch the shallow clips map (not observeDeep) so
   318→    // we only react to key additions, not to every property edit on an
   319→    // existing clip (which would restart playback every time a peer nudged
   320→    // a clip's volume, fighting the user). The restart is debounced 500ms so
   321→    // a burst of additions (e.g. a peer imports 5 files) coalesces into one.
   322→    //
   323→    // LIVE WAVEFORM REFRESH: when a remote peer adds a clip, its metadata
   324→    // arrives via Yjs immediately but the sample blob travels separately via
   325→    // the audioSampleSync Y.Map (see sampleStore.ts). If we don't ping the
   326→    // WaveformImg, it might have already exhausted its empty-samples retry
   327→    // budget (now 10s, but a slow link can still exceed that) and given up
   328→    // showing a blank "···" placeholder until the user refreshes. So for
   329→    // each newly-added clip we:
   330→    //   1. clearCache(clipId) on the engine so the next play re-loads samples
   331→    //      (the cache might hold a null/empty entry from a prior failed getBuffer)
   332→    //   2. dispatch `slate:audio-clip-changed` with the clipId — the
   333→    //      AudioEditor's listener (below) re-invalidates the waveform cache +
   334→    //      clears the engine cache + bumps version, AND the WaveformImg's own
   335→    //      listener resets its retryRef to 0 and bumps `bust` to immediately
   336→    //      re-attempt loadSamples (which will now find the synced blob once
   337→    //      it lands).
   338→    //   3. scheduleRestart() so if we're currently playing, the new clip is
   339→    //      picked up atomically (debounced 500ms to coalesce bursts).
   340→    const onClipsAdded = (event: Y.YMapEvent<Y.Map<unknown>>) => {
   341→      const addedIds: string[] = [];
   342→      for (const key of event.keysChanged) {
   343→        const change = event.changes.keys.get(key);
   344→        if (change?.action === 'add') addedIds.push(key);
   345→      }
   346→      if (addedIds.length === 0) return;
   347→      for (const id of addedIds) {
   348→        engineRef.current?.clearCache(id);
   349→        window.dispatchEvent(new CustomEvent('slate:audio-clip-changed', { detail: id }));
   350→      }
   351→      scheduleRestart();
   352→    };
   353→    clips.observe(onClipsAdded);
   354→    const lateRead = setTimeout(bump, 200);
   355→    return () => { clearTimeout(lateRead); tracks.unobserveDeep(bump); clips.unobserveDeep(bump); audioMap.unobserve(bump); clips.unobserve(onClipsAdded); };
   356→    // eslint-disable-next-line react-hooks/exhaustive-deps
   357→  }, [slate]);
   358→
   359→  useEffect(() => {
   360→    engineRef.current = new AudioEngine();
   361→    engineRef.current.setMasterVolume(masterVol);
   362→    return () => {
   363→      // Cancel any pending restart so it doesn't fire after dispose().
   364→      if (restartTimerRef.current) { clearTimeout(restartTimerRef.current); restartTimerRef.current = null; }
   365→      engineRef.current?.dispose();
   366→      engineRef.current = null;
   367→    };
   368→    // eslint-disable-next-line react-hooks/exhaustive-deps
   369→  }, []);
   370→
   371→  // Playhead — direct DOM.
   372→  useEffect(() => {
   373→    if (!playing) {
   374→      // Playback just stopped — cancel any pending mid-playback restart so
   375→      // it doesn't fire and call restartPlayback (which would re-start audio
   376→      // after the user pressed pause).
   377→      if (restartTimerRef.current) { clearTimeout(restartTimerRef.current); restartTimerRef.current = null; }
   378→      return;
   379→    }
   380→    const tick = () => {
   381→      rafRef.current = requestAnimationFrame(tick);
   382→      const eng = engineRef.current; if (!eng) return;
   383→      const pos = eng.getPosition();
   384→      positionRef.current = pos;
   385→      if (playheadRef.current) playheadRef.current.style.transform = `translateX(${pos * pxRef.current}px)`;
   386→      if (posDisplayRef.current) posDisplayRef.current.textContent = `${pos.toFixed(1)}s`;
   387→      // Throttled awareness publish at ~7 Hz (150ms) — peers render at display
   388→      // rate via their own rAF, so 7 Hz is plenty for smooth remote playhead
   389→      // motion while keeping the awareness broadcast off the hot path.
   390→      const now = performance.now();
   391→      if (now - lastAudioPublishRef.current >= 150) {
   392→        lastAudioPublishRef.current = now;
   393→        room.setLocalAwareness({ audio: { pos, playing: true } });
   394→      }
   395→      if (pos > timelineDuration + 2) { eng.stop(); setPlaying(false); }
   396→    };
   397→    rafRef.current = requestAnimationFrame(tick);
   398→    return () => cancelAnimationFrame(rafRef.current);
   399→    // eslint-disable-next-line react-hooks/exhaustive-deps
   400→  }, [playing]);
   401→
   402→  // Ctrl+scroll zoom — centred on the playhead so the screen position under
   403→  // the cursor/playhead stays put instead of zooming toward the left edge.
   404→  /** Zoom while keeping the playhead at the same screen offset. Records the
   405→   *  playhead's offset from the left edge of the visible viewport (relative to
   406→   *  the timeline content), changes `pxPerSec`, then schedules a scroll
   407→   *  correction so the playhead lands at the same offset in the new zoom.
   408→   *  The actual `scrollLeft` write happens in the layout effect below (after
   409→   *  React commits the new `minWidth` on the timeline div — setting it earlier
   410→   *  would be clamped by the stale `scrollWidth`). */
   411→  const zoomAtPlayhead = useCallback((newPxPerSec: number) => {
   412→    const el = scrollRef.current;
   413→    const oldPxPerSec = pxRef.current;
   414→    if (!el || oldPxPerSec === newPxPerSec) {
   415→      setPxPerSec(newPxPerSec);
   416→      return;
   417→    }
   418→    const scrollLeft = el.scrollLeft;
   419→    const playheadX = positionRef.current * oldPxPerSec;
   420→    const playheadOffset = playheadX - scrollLeft;
   421→    const newPlayheadX = positionRef.current * newPxPerSec;
   422→    const newScrollLeft = newPlayheadX - playheadOffset;
   423→    pendingScrollRef.current = newScrollLeft;
   424→    setPxPerSec(newPxPerSec);
   425→  }, []);
   426→
   427→  // Apply any pending scroll correction after pxPerSec commits to the DOM.
   428→  useLayoutEffect(() => {
   429→    if (pendingScrollRef.current !== null && scrollRef.current) {
   430→      scrollRef.current.scrollLeft = pendingScrollRef.current;
   431→      pendingScrollRef.current = null;
   432→    }
   433→  }, [pxPerSec]);
   434→
   435→  useEffect(() => {
   436→    const el = scrollRef.current; if (!el) return;

   438→      if (!e.ctrlKey && !e.metaKey) return;

   441→      zoomAtPlayhead(next);

   444→    return () => el.removeEventListener('wheel', onWheel);

   447→  // Global pointermove/up — pure DOM, zero React state.
   448→  // rAF-throttled: pointermove fires at the hardware's report rate (often
   449→  // 60-120Hz), and the move handler does a neighbours-overlap scan on every
   450→  // event. Coalescing multiple moves into a single rAF callback caps the work
   451→  // at the display refresh rate (~60fps) so a fast pointer drag never stalls
   452→  // the main thread. A no-op skip (cursor hasn't moved since the last
   453→  // processed frame) avoids redundant DOM writes for stationary pointers.
   454→  useEffect(() => {
   455→    /** Last clientX we actually applied to the DOM. Lets us skip work when
   456→     *  pointermove fires with the same x (e.g. a touchpad reporting pressure
   457→     *  changes without movement, or sub-pixel jitter that rounds to the same
   458→     *  px). */
   459→    let lastProcessedX: number | null = null;
   460→
   461→    const applyMove = () => {
   462→      moveRafRef.current = 0;
   463→      const d = dragRef.current;
   464→      if (!d) return;
   465→      const clientX = moveXRef.current;
   466→      if (clientX === null) return;
   467→      // Skip no-op moves — cursor hasn't moved to a new pixel since the last
   468→      // frame we processed. Saves a neighbours scan + 1-2 style writes.
   469→      if (lastProcessedX === clientX) return;
   470→      lastProcessedX = clientX;
   471→
   472→      const pps = pxRef.current;
   473→      const dt = (clientX - d.sx) / pps;
   474→      const oldEnd = d.os + d.od;
   475→      if (d.mode === 'drag') {
   476→        // Snap-to-neighbour drag: the clip follows the mouse, but if the new
   477→        // position would overlap another clip on the same track, instead of
   478→        // stopping dead at the neighbour's edge we JUMP to the free side of
   479→        // that neighbour (right of it when dragging right, left of it when
   480→        // dragging left). Iterates so a chain of back-to-back clips resolves
   481→        // to the next free slot. A small overlap threshold avoids accidental
   482→        // snaps when the cursor barely crosses the boundary.
   483→        const duration = d.od;
   484→        const rawStart = d.os + dt;
   485→        let start = rawStart;
   486→        const overlapThreshold = 0.05; // s — ignore near-boundary jitter
   487→        for (let iter = 0; iter < d.neighbours.length + 1; iter++) {
   488→          const blocker = d.neighbours.find((n) => {
   489→            const ovStart = Math.max(start, n.start);
   490→            const ovEnd = Math.min(start + duration, n.end);
   491→            return ovEnd - ovStart > overlapThreshold;
   492→          });
   493→          if (!blocker) break;
   494→          if (dt >= 0) start = blocker.end;          // snap to RIGHT of blocker
   495→          else start = blocker.start - duration;     // snap to LEFT of blocker
   496→        }
   497→        // Fallback clamp: never let the clip start before t=0.
   498→        start = Math.max(0, start);
   499→        d.el.style.left = `${start * pps}px`;
   500→      } else if (d.mode === 'trimL') {
   501→        // Cut from the left: never past the left neighbour, the source start
   502→        // (offset ≥ 0 → limited by the trimmed head in timeline seconds), or a
   503→        // minimum width.
   504→        const minStart = Math.max(d.leftLimit, d.os - d.oo / d.speed);
   505→        const start = Math.min(oldEnd - 0.1, Math.max(minStart, d.os + dt));
   506→        d.el.style.left = `${start * pps}px`;
   507→        d.el.style.width = `${(oldEnd - start) * pps}px`;
   508→        // Shift the (fixed-width) waveform the opposite way so the audio stays
   509→        // anchored — the newly-trimmed head is clipped, not squashed.
   510→        if (d.waveEl) d.waveEl.style.left = `${-(start - d.os) * pps}px`;
   511→      } else if (d.mode === 'trimR') {
   512→        // Cut from the right: never past the right neighbour or a min width.
   513→        const end = Math.min(d.rightLimit, Math.max(d.os + 0.1, oldEnd + dt));
   514→        d.el.style.width = `${(end - d.os) * pps}px`;
   515→      }
   516→    };
   517→
   518→    const onMove = (ev: PointerEvent) => {
   519→      const d = dragRef.current; if (!d) return;
   520→      // Stash the latest clientX — the rAF callback reads it. Cheap (one ref
   521→      // write) so even 120Hz pointermove streams don't pile up work.
   522→      moveXRef.current = ev.clientX;
   523→      // Schedule a frame if one isn't already pending. The guard is what
   524→      // coalesces multiple moves within the same frame into one DOM write.
   525→      if (!moveRafRef.current) {
   526→        moveRafRef.current = requestAnimationFrame(applyMove);
   527→      }
   528→    };
   529→    const onUp = () => {
   530→      const d = dragRef.current; if (!d) return;
   531→      // Cancel any pending rAF — we'll process the final position
   532→      // SYNCHRONOUSLY here so the committed Yjs value reflects the exact
   533→      // pointer-up location (otherwise the last pointermove before pointerup
   534→      // could be dropped, leaving the committed position ~1 frame behind the
   535→      // cursor). The `force` flag bypasses the no-op skip in case the final
   536→      // pointermove had the same clientX as the prior processed frame but we
   537→      // still need to read the just-written style to commit to Yjs.
   538→      if (moveRafRef.current) {
   539→        cancelAnimationFrame(moveRafRef.current);
   540→        moveRafRef.current = 0;
   541→      }
   542→      if (moveXRef.current !== null) {
   543→        lastProcessedX = null; // bypass the no-op skip on the final flush
   544→        applyMove();
   545→      }
   546→      const pps = pxRef.current;
   547→      const left = parseFloat(d.el.style.left) / pps;
   548→      const width = parseFloat(d.el.style.width) / pps;
   549→      if (d.mode === 'drag') updateAudioClip(slate, d.clipId, { start: Math.max(0, left) });
   550→      else if (d.mode === 'trimL') updateAudioClip(slate, d.clipId, { start: left, duration: width, offset: Math.max(0, d.oo + (left - d.os) * d.speed) });
   551→      else if (d.mode === 'trimR') updateAudioClip(slate, d.clipId, { duration: width });
   552→      moveXRef.current = null;
   553→      lastProcessedX = null;
   554→      dragRef.current = null;
   555→    };
   556→    window.addEventListener('pointermove', onMove);
   557→    window.addEventListener('pointerup', onUp);
   558→    return () => {
   559→      window.removeEventListener('pointermove', onMove);
   560→      window.removeEventListener('pointerup', onUp);
   561→      // Don't leak a pending rAF if the component unmounts mid-drag.
   562→      if (moveRafRef.current) {
   563→        cancelAnimationFrame(moveRafRef.current);
   564→        moveRafRef.current = 0;
   565→      }
   566→    };
   567→  }, [slate]);

   896→const TrackHeader = memo(function TrackHeader({ track, hasSolo, slate, engineRef }: {
   897→  track: AudioTrack; hasSolo: boolean;
   898→  slate: ReturnType<typeof useRoom>['slate'];
   899→  engineRef: React.RefObject<AudioEngine | null>;
   900→}) {
   901→  const [vol, setVol] = useState(track.volume);
   902→  const [pan, setPan] = useState(track.pan);
   903→  // Separate drag flags per slider — the previous single shared `isDraggingRef`
   904→  // meant starting a volume drag also blocked the pan prop-sync effect (and
   905→  // vice versa), which could cause one slider's committed Yjs value to be
   906→  // momentarily ignored by the other's gating. Per-slider flags keep each
   907→  // slider's prop-sync independent.
   908→  const isDraggingVolRef = useRef(false);
   909→  const isDraggingPanRef = useRef(false);
   910→  useEffect(() => { if (!isDraggingVolRef.current) setVol(track.volume); }, [track.volume]);
   911→  useEffect(() => { if (!isDraggingPanRef.current) setPan(track.pan); }, [track.pan]);
   912→
   913→  // Audibility — mirrors engine.setupTrackNodes so the live gain we write
   914→  // during a volume drag respects mute/solo (don't briefly un-mute a muted
   915→  // track just because the user is dragging its volume slider).
   916→  const audible = hasSolo ? track.solo : !track.muted;
   917→
   918→  // `update` is for non-slider track edits (name/mute/solo/arm) — these DO
   919→  // need a full updateTracks so the audio graph reflects the new state
   920→  // (e.g. toggling solo rebalances every track's gain).
   921→  const update = (patch: Partial<AudioTrack>) => { updateAudioTrack(slate, track.id, patch); engineRef.current?.updateTracks(slate); };
   922→
   923→  // Volume slider: update local state + the engine's gain node DIRECTLY for
   924→  // immediate audio feedback. Do NOT call updateTracks (which re-reads every
   925→  // track from Yjs and rebuilds the audio graph — far too expensive per
   926→  // onChange event, and the source of the reported lag). Commit to Yjs once
   927→  // on pointerup.
   928→  const onVolDown = () => { isDraggingVolRef.current = true; };
   929→  const onVol = (v: number) => { setVol(v); engineRef.current?.setTrackVolume(track.id, v, audible); };
   930→  const onVolEnd = () => { isDraggingVolRef.current = false; updateAudioTrack(slate, track.id, { volume: vol }); };
   931→
   932→  // Pan slider: same pattern — set the panner node directly, commit on
   933→  // pointerup. The pan slider is smaller than the volume slider (secondary
   934→  // control) but still usable, with L/R labels so its function is obvious
   935→  // (the user complaint was "two sliders, only one works" — partly because
   936→  // neither slider was labelled, so it wasn't clear what each did).
   937→  const onPanDown = () => { isDraggingPanRef.current = true; };
   938→  const onPan = (p: number) => { setPan(p); engineRef.current?.setTrackPan(track.id, p); };
   939→  const onPanEnd = () => { isDraggingPanRef.current = false; updateAudioTrack(slate, track.id, { pan: pan }); };
   940→
   941→  return (
   942→    <div className="border-b border-border/15 px-2 py-1" style={{ height: TRACK_H }}>
   943→      <div className="flex items-center gap-0.5">
   944→        <span className="h-2 w-2 shrink-0 rounded-full" style={{ backgroundColor: track.color }} />
   945→        <input type="text" value={track.name} onChange={(e) => update({ name: e.target.value })} className="min-w-0 flex-1 bg-transparent text-[11px] font-medium text-text outline-none" />
   946→        <button onClick={() => update({ muted: !track.muted })} className={`flex h-4 w-4 items-center justify-center rounded ${track.muted && !hasSolo ? 'bg-warn/30 text-warn' : 'text-text-mid hover:bg-bg-3'}`} title="M">{track.muted ? <VolumeX size={9} /> : <Volume2 size={9} />}</button>
   947→        <button onClick={() => update({ solo: !track.solo })} className={`flex h-4 w-4 items-center justify-center rounded ${track.solo ? 'bg-accent/30 text-accent' : 'text-text-mid hover:bg-bg-3'}`} title="S"><Headphones size={9} /></button>
   948→        <button onClick={() => update({ armed: !track.armed, input: !track.armed ? 'mic' : 'none' })} className={`flex h-4 w-4 items-center justify-center rounded ${track.armed ? 'bg-danger/30 text-danger' : 'text-text-mid hover:bg-bg-3'}`} title="Arm"><div className="h-1.5 w-1.5 rounded-full" style={{ backgroundColor: track.armed ? 'currentColor' : 'transparent', border: '1px solid currentColor' }} /></button>
   949→        <button onClick={() => deleteAudioTrack(slate, track.id)} className="flex h-4 w-4 items-center justify-center rounded text-text-mid hover:bg-bg-3 hover:text-danger" title="Del"><Trash2 size={9} /></button>
   950→      </div>
   951→      <div className="mt-0.5 flex items-center gap-1">
   952→        <Volume2 size={9} className="shrink-0 text-text-dim" aria-hidden />
   953→        <input type="range" min={0} max={1} step={0.01} value={vol} aria-label="Volume" title="Volume" onPointerDown={onVolDown} onChange={(e) => onVol(Number(e.target.value))} onPointerUp={onVolEnd} className="h-1 flex-1 accent-accent" />
   954→        <span className="shrink-0 text-[8px] font-medium leading-none text-text-dim" aria-hidden>L</span>
   955→        <input type="range" min={-1} max={1} step={0.01} value={pan} aria-label="Pan" title="Pan" onPointerDown={onPanDown} onChange={(e) => onPan(Number(e.target.value))} onPointerUp={onPanEnd} className="h-1 w-10 accent-accent" />
   956→        <span className="shrink-0 text-[8px] font-medium leading-none text-text-dim" aria-hidden>R</span>
   957→      </div>
   958→    </div>
   959→  );
   960→});

   964→const ClipBlock = memo(function ClipBlock({ clip, pxPerSec, selected, onSelect, onDragStart, onTrimStart }: {
   965→  clip: AudioClip; pxPerSec: number; selected: boolean;
   966→  onSelect: (id: string) => void;
   967→  onDragStart: (clip: AudioClip, e: React.PointerEvent, el: HTMLElement, waveEl: HTMLElement | null) => void;
   968→  onTrimStart: (clip: AudioClip, side: 'left' | 'right', e: React.PointerEvent, el: HTMLElement, waveEl: HTMLElement | null) => void;
   969→}) {
   970→  const left = clip.start * pxPerSec;
   971→  const width = Math.max(4, clip.duration * pxPerSec);
   972→  const elRef = useRef<HTMLDivElement | null>(null);
   973→  const waveRef = useRef<HTMLDivElement | null>(null);
   974→  const w = Math.max(2, Math.floor(width));
   975→  // Fade overlay widths (clamped to the clip box so a too-long fade doesn't
   976→  // spill past the opposite edge).
   977→  const fadeInW = clip.fadeIn > 0 ? Math.min(width, clip.fadeIn * pxPerSec) : 0;
   978→  const fadeOutW = clip.fadeOut > 0 ? Math.min(width, clip.fadeOut * pxPerSec) : 0;
   979→
   980→  return (
   981→    <div ref={elRef} onPointerDown={(e) => { if (elRef.current) { e.stopPropagation(); onSelect(clip.id); window.dispatchEvent(new CustomEvent('slate:audio-clip-select', { detail: clip.id })); onDragStart(clip, e, elRef.current, waveRef.current); } }}
   982→      className={`group pointer-events-auto absolute top-0.5 bottom-0.5 cursor-grab overflow-hidden rounded border ${selected ? 'border-warn' : 'border-black/30'} active:cursor-grabbing`} style={{ left, width, backgroundColor: `${clip.color}20` }}>
   983→      {/* Fixed-width waveform layer — clipped by the box so trimming cuts the
   984→          audio rather than squashing the whole wave into a smaller space. */}
   985→      <div ref={waveRef} className="pointer-events-none absolute top-0 bottom-0 left-0" style={{ width }}>
   986→        {clip.sampleKey && <WaveformImg clipId={clip.id} sampleKey={clip.sampleKey} channels={clip.channels} sampleRate={clip.sampleRate} offset={clip.offset} duration={clip.duration} speed={clip.speed ?? 1} width={w} color={clip.color} />}
   987→      </div>
   988→      {/* Fade-in overlay: triangle from full height at the outer (left) edge
   989→          tapering to 0 at the inner edge — the dark wedge represents the
   990→          portion of the clip still under the fade. */}
   991→      {fadeInW > 0 && (
   992→        <div className="pointer-events-none absolute top-0 bottom-0 left-0 bg-black/30" style={{ width: fadeInW, clipPath: 'polygon(0% 0%, 0% 100%, 100% 50%)' }} />
   993→      )}
   994→      {/* Fade-out overlay: mirrored on the right side. */}
   995→      {fadeOutW > 0 && (
   996→        <div className="pointer-events-none absolute top-0 bottom-0 right-0 bg-black/30" style={{ width: fadeOutW, clipPath: 'polygon(100% 0%, 100% 100%, 0% 50%)' }} />
   997→      )}
   998→      <span className="absolute left-1 top-0 truncate text-[7px] font-medium text-text-mid/70 pointer-events-none">{clip.name}{clip.mute ? ' (muted)' : ''}</span>
   999→      <div onPointerDown={(e) => { if (elRef.current) onTrimStart(clip, 'left', e, elRef.current, waveRef.current); }} className="absolute left-0 top-0 bottom-0 w-1.5 cursor-ew-resize bg-white/40 opacity-0 group-hover:opacity-100" />
  1000→      <div onPointerDown={(e) => { if (elRef.current) onTrimStart(clip, 'right', e, elRef.current, waveRef.current); }} className="absolute right-0 top-0 bottom-0 w-1.5 cursor-ew-resize bg-white/40 opacity-0 group-hover:opacity-100" />
  1001→    </div>
  1002→  );
  1003→});
