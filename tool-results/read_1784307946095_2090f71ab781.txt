   708тЖТ   708тЖТ  const m = Math.floor(s / 60);
   709тЖТ   709тЖТ  if (m < 60) return `${m} min ago`;
   710тЖТ   710тЖТ  const h = Math.floor(m / 60);
   711тЖТ   711тЖТ  if (h < 24) return `${h} h ago`;
   712тЖТ   712тЖТ  const d = Math.floor(h / 24);
