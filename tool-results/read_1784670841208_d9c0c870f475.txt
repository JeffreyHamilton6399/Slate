   742тЖТ   742тЖТfunction timeAgo(t: number): string {
   743тЖТ   743тЖТ  const s = Math.max(1, Math.floor((Date.now() - t) / 1000));
   744тЖТ   744тЖТ  if (s < 60) return 'just now';
   745тЖТ   745тЖТ  const m = Math.floor(s / 60);
   746тЖТ   746тЖТ  if (m < 60) return `${m} min ago`;
   747тЖТ   747тЖТ  const h = Math.floor(m / 60);
   748тЖТ   748тЖТ  if (h < 24) return `${h} h ago`;
   749тЖТ   749тЖТ  const d = Math.floor(h / 24);
   750тЖТ   750тЖТ  return d === 1 ? 'yesterday' : `${d} days ago`;
   751тЖТ   751тЖТ}
